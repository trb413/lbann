83 bool data_reader_triplet::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
84 
85  std::vector<::Mat> X_v = create_datum_views(X, mb_idx);
86 
87  sample_t sample = m_samples.get_sample(data_id);
88  for(size_t i=0u; i < m_num_img_srcs; ++i) {
89  int width=0, height=0, img_type=0;
90  const std::string imagepath = get_file_dir() + sample.first[i];
91  bool ret = true;
92  if (m_data_store != nullptr) {
93  std::vector<unsigned char> *image_buf;
94  m_data_store->get_data_buf(data_id, image_buf, tid);
95  // This could probably have used image_utils::import_image()
96  ret = lbann::image_utils::load_image(*image_buf, width, height, img_type, *(m_pps[tid]), X_v[i]);
97  } else {
98  ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v[i]);
99  }
100 
101  if(!ret) {
102  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
103  + get_type() + ": image_utils::load_image failed to load - "
104  + imagepath);
105  }
106  if((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
107  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
108  + get_type() + ": mismatch data size -- either width, height or channel - "
109  + imagepath + " [w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
110  + "x" + std::to_string(CV_MAT_CN(img_type)) + "] != " + std::to_string(m_image_linearized_size));
111  }
112  }
113  return true;
114 }

148 void data_reader_triplet::load() {
149  const std::string data_filename = get_data_filename();
150 
151  // To support m_first_n semantic, m_samples.load() takes m_first_n
152  // as an argument and attempt to shrink the CNPY arrays loaded as needed
153  if (!m_samples.load(data_filename, m_first_n)) {
154  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
155  + get_type() + ": failed to load the file " + data_filename);
156  }
157 
158  size_t num_samples = m_samples.get_num_samples();
159 
160  if (m_first_n > 0) {
161  num_samples = (static_cast<size_t>(m_first_n) <= num_samples)?
162  static_cast<size_t>(m_first_n) : num_samples;
163 
164  m_first_n = num_samples;
165  set_use_percent(1.0);
166  set_absolute_sample_count(0u);
167  }
168 
169  // reset indices
170  m_shuffled_indices.clear();
171 
172  m_shuffled_indices.resize(num_samples);
173  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
174 
175  select_subset_of_data();
176 }
