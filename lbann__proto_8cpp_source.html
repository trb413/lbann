61 void lbann_proto::readPrototextFile(const char *fn)
62 {
63  stringstream err;
64  int fd = open(fn, O_RDONLY);
65  if (fd == -1) {
66  err << __FILE__ << " " << __LINE__ << " :: failed to open " << fn << " for reading";
67  throw lbann_exception(err.str());
68  }
69  google::protobuf::io::FileInputStream* input = new google::protobuf::io::FileInputStream(fd);
70  bool success = google::protobuf::TextFormat::Parse(input, &m_pb);
71  if (not success and m_master) {
72  err << __FILE__ << " " << __LINE__ << " :: failed to read or parse prototext file: " << fn << endl;
73  throw lbann_exception(err.str());
74  }
75 }

77 bool lbann_proto::writePrototextFile(const char *fn)
78 {
79  int fd = open(fn, O_WRONLY | O_CREAT | O_TRUNC, 0644);
80  if (fd == -1) {
81  return false;
82  }
83  FileOutputStream* output = new FileOutputStream(fd);
84  if (not google::protobuf::TextFormat::Print(m_pb, output)) {
85  close(fd);
86  delete output;
87  return false;
88  }
89  delete output;
90  close(fd);
91  return true;
92 }

203 void lbann_proto::add_layer(const layer_params &p)
204 {
205  lbann_data::Model *model = m_pb.mutable_model();
206  lbann_data::Layer *layer = model->add_layer();
207  string name = p.name;
208 
209  if (name == "input_distributed_minibatch_parallel_io") {
210  lbann_data::InputDistributedMiniBatchParallelIO *real_layer = layer->mutable_input_distributed_minibatch_parallel_io();
211  real_layer->set_num_parallel_readers(PB_FIX(p.num_parallel_readers));
212  real_layer->set_mini_batch_size(PB_FIX(p.mini_batch_size));
213  }
214 
215  else if (name == "fully_connected") {
216  lbann_data::FullyConnected *real_layer = layer->mutable_fully_connected();
217  real_layer->set_num_prev_neurons(PB_FIX(p.num_prev_neurons));
218  real_layer->set_num_neurons(PB_FIX(p.num_neurons));
219  real_layer->set_mini_batch_size(PB_FIX(p.mini_batch_size));
220  real_layer->set_activation_type( get_activation_type(p.activation) );
221  real_layer->set_weight_initialization( get_weight_initialization_type(p.weight_init) );
222  for (size_t j=0; j<p.regularizers.size(); j++) {
223  //XX real_layer->add_regularizer(p.regularizers[j]);
224  }
225 
226  } else if (name == "softmax") {
227  lbann_data::Softmax *real_layer = layer->mutable_softmax();
228  real_layer->set_num_prev_neurons(PB_FIX(p.num_prev_neurons));
229  real_layer->set_num_neurons(PB_FIX(p.num_neurons));
230  real_layer->set_activation_type( get_activation_type(p.activation) );
231  real_layer->set_weight_initialization( get_weight_initialization_type(p.weight_init) );
232  }
233 
234  else if (name == "target_distributed_minibatch_parallel_io") {
235  lbann_data::TargetDistributedMinibatchParallelIO *real_layer = layer->mutable_target_distributed_minibatch_parallel_io();
236  real_layer->set_num_parallel_readers(p.num_parallel_readers);
237  real_layer->set_mini_batch_size(p.mini_batch_size);
238  real_layer->set_shared_data_reader(p.shared_data_reader);
239  real_layer->set_for_regression(p.for_regression);
240  }
241 
242  else {
243  stringstream err;
244  err << __FILE__ << " " << __LINE__ << " :: lbann_proto::add_layer: not implemented or nothing known about layer type: " << name;
245  throw lbann_exception(err.str());
246  }
247 }
