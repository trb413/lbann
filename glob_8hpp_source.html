44  int r = glob(pattern.c_str(), 0, nullptr, &pglob);
45  if (r != 0) {
46  // Either an error or no match.
47  if (r == GLOB_NOMATCH) {
48  return {};
49  } else {
50  throw lbann_exception("glob error");
51  }
52  }
53  std::vector<std::string> results(pglob.gl_pathc);
54  for (size_t i = 0; i < pglob.gl_pathc; ++i) {
55  results[i] = std::string(pglob.gl_pathv[i]);
56  }
