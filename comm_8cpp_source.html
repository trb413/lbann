94 void lbann_comm::split_models(int ppm) {
95  int world_size = El::mpi::Size(get_world_comm());
96  procs_per_model = ppm;
97  if (ppm == 0) {
98  procs_per_model = world_size;
99  }
100  // Check if parameters are valid
101  if (procs_per_model > world_size) {
102  throw lbann_exception(
103  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
104  " :: Not enough processes to create one model; procs_per_model: " +
105  std::to_string(procs_per_model) + " is larger than world_size: " +
106  std::to_string(world_size));
107  }
108  if (world_size % procs_per_model != 0) {
109  throw lbann_exception(
110  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
111  " :: Procs per model does not divide total number of procs; procs_per_model: " +
112  std::to_string(procs_per_model) + " total number of procs (world size): " +
113  std::to_string(world_size));
114  }
115 
116  num_models = world_size / procs_per_model;
117  model_rank = El::mpi::Rank(get_world_comm()) / procs_per_model;
118  rank_in_model = El::mpi::Rank(get_world_comm()) % procs_per_model;
119 
120  // Initialize model and intermodel communicators
121  El::mpi::Split(get_world_comm(), model_rank, rank_in_model, model_comm);
122  El::mpi::Split(get_world_comm(), rank_in_model, model_rank,
123  intermodel_comm);
124 
125  // Initialize Elemental grid
126  if (grid != nullptr) {
127  delete grid;
128  }
129  grid = new Grid(model_comm);
130 }

142 void lbann_comm::allreduce(AbsDistMat& m,
143  const El::mpi::Comm c,
144  El::mpi::Op op,
145  std::type_index t) {
146  const int local_size = m.LocalHeight() * m.LocalWidth();
147  bytes_sent += sizeof(DataType) * local_size;
148 #ifdef LBANN_HAS_ALUMINUM
149  if (m.LocalHeight() != m.LDim()) {
150  throw lbann_exception("Aluminum does not support allreduces on"
151  " non-contiguous matrices");
152  }
153  auto&& comm = get_al_comm(c, t);
154  if (t == std::type_index(typeid(::Al::MPIBackend))) {
155  ::Al::Allreduce<::Al::MPIBackend>(
156  m.Buffer(),
157  local_size,
158  mpi_op_to_al_op(op),
159  *comm);
160  }
162 #ifdef LBANN_HAS_NCCL2
163  if (t == std::type_index(typeid(::Al::NCCLBackend))) {
164  ::Al::Allreduce<::Al::NCCLBackend>(
165  m.Buffer(),
166  local_size,
167  mpi_op_to_al_op(op),
168  *static_cast<::Al::NCCLCommunicator*>(comm));
169  }
170 #endif // LBANN_HAS_NCCL2
171 #else
172  El::AllReduce(m, c, op);
173 #endif
174  bytes_received += sizeof(DataType) * local_size * (El::mpi::Size(c) - 1);
175 }

332 void lbann_comm::intermodel_allreduce(
333  Mat& mat, int max_recv_count,
334  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
335  std::function<int(uint8_t *, Mat&)> recv_transform,
336  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
337  const lbann_comm::allreduce_options opts) {
338  // Determine which algorithm to actually use.
339  lbann_comm::allreduce_algorithm algo = opts.algo;
340  if (algo == allreduce_algorithm::DEFAULT) {
341  algo = get_default_allreduce_algorithm();
342  }
343  const int nprocs = get_num_models();
344  const El::Int small_message_threshold = 64*64;
345  if (algo == allreduce_algorithm::DYNAMIC) {
346  // For small messages and power-of-2 number of processes, use RD.
347  if (!(nprocs & (nprocs - 1)) &&
348  mat.Height() * mat.Width() <= small_message_threshold) {
349  algo = allreduce_algorithm::RECURSIVE_DOUBLING;
350  } else {
351  algo = allreduce_algorithm::PAIRWISE_EXCHANGE_RING;
352  }
353  }
354  switch (algo) {
355  case allreduce_algorithm::RECURSIVE_DOUBLING:
356  recursive_doubling_allreduce_pow2(
357  intermodel_comm, mat, max_recv_count,
358  send_transform, recv_apply_transform, opts);
359  break;
360  case allreduce_algorithm::PAIRWISE_EXCHANGE_RING:
361  pe_ring_allreduce(
362  intermodel_comm, mat, max_recv_count, send_transform,
363  recv_transform, recv_apply_transform, opts);
364  break;
365  case allreduce_algorithm::RING:
366  ring_allreduce(
367  intermodel_comm, mat, max_recv_count, send_transform,
368  recv_transform, recv_apply_transform, opts);
369  break;
370  case allreduce_algorithm::RABENSEIFNER:
371  rabenseifner_allreduce(
372  intermodel_comm, mat, max_recv_count, send_transform,
373  recv_transform, recv_apply_transform, opts);
374  break;
375  case allreduce_algorithm::DEFAULT:
376  case allreduce_algorithm::DYNAMIC:
377  default:
378  throw lbann_exception("intermodel_allreduce: bad algorithm");
379  break;
380  }
381 }

383 void lbann_comm::recursive_doubling_allreduce_pow2(
384  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
385  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
386  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
387  const lbann_comm::allreduce_options opts) {
388  double ar_start = get_time();
389  const int rank = El::mpi::Rank(comm);
390  const unsigned int nprocs = El::mpi::Size(comm);
391  if (nprocs == 1) {
392  return; // Nothing to do.
393  }
394  // This implementation requires a power-of-2 number of processes.
395  if (nprocs & (nprocs - 1)) {
396  throw lbann_exception("lbann_comm: recursive doubling allreduce requires"
397  " a power-of-2 number of participating processes");
398  }
399  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
400  uint8_t *recv_buf = max_recv_buf;
401  unsigned int mask = 1;
402  while (mask < nprocs) {
403  int partner = rank ^ mask; // The rank we exchange with this step.
404  const bool is_local = opts.no_local_trans &&
405  is_rank_node_local(partner, comm);
406  // Transform the data we want to send.
407  double send_trans_start = get_time();
408  int send_size;
409  int recv_size = max_recv_count;
410  uint8_t *send_buf = nullptr;
411  if (is_local) {
412  send_buf = (uint8_t *) mat.Buffer();
413  send_size = sizeof(DataType) * mat.Height() * mat.Width();
414  recv_size = send_size;
415  recv_buf = get_collective_buffer(recv_size);
416  } else {
417  send_buf = send_transform(mat, El::ALL, El::ALL, send_size, false, 0);
418  recv_buf = max_recv_buf;
419  }
420  ar_send_transform_time += get_time() - send_trans_start;
421  bytes_sent += send_size;
422  ar_bytes_sent += send_size;
423  double sendrecv_start = get_time();
424  El::mpi::SendRecv(send_buf, send_size, partner,
425  recv_buf, recv_size, partner, comm);
426  double sendrecv_tot = get_time() - sendrecv_start;
427  ar_send_time += sendrecv_tot;
428  ar_recv_time += sendrecv_tot;
429  // Transform and reduce the received data.
430  double recv_apply_trans_start = get_time();
431  recv_size = recv_apply_transform(recv_buf, mat, is_local);
432  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
433  bytes_received += recv_size;
434  ar_bytes_received += recv_size;
435  mask <<= 1;
436  }
437  ar_time += get_time() - ar_start;
438 }

440 void lbann_comm::pe_ring_allreduce(
441  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
442  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
443  std::function<int(uint8_t *, Mat&)> recv_transform,
444  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
445  const lbann_comm::allreduce_options opts) {
446  double ar_start = get_time();
447  const int rank = El::mpi::Rank(comm);
448  const int nprocs = El::mpi::Size(comm);
449  if (nprocs == 1) {
450  return; // Nothing to do.
451  }
452  // Compute the number of columns each processor sends.
453  // If it doesn't divide evenly, give one extra to the earlier ranks.
454  const El::Int cols_per_proc = mat.Width() / nprocs;
455  const El::Int cols_remainder = mat.Width() % nprocs;
456  // Compute the lengths/ends of the slices.
457  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
458  for (int i = 0; i < cols_remainder; ++i) {
459  slice_lengths[i] += 1;
460  }
461  std::vector<El::Int> slice_ends(nprocs);
462  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
463  slice_ends.begin());
464  std::vector<uint8_t *> max_recv_buffers(opts.max_reduces, nullptr);
465  for (size_t i = 0; i < max_recv_buffers.size(); ++i) {
466  max_recv_buffers[i] = get_collective_buffer(max_recv_count, i);
467  }
468  // Local slice of our accumulated data.
469  auto accum_view = mat(El::ALL, El::IR(slice_ends[rank] - slice_lengths[rank],
470  slice_ends[rank]));
471  // Do a pairwise-exchange reduce-scatter.
472  double rs_start = get_time();
473  for (int outer_step = 1;
474  outer_step < nprocs;
475  outer_step += opts.max_reduces) {
476  const int reduces_this_step = std::min(opts.max_reduces,
477  nprocs - outer_step);
478  std::vector<El::mpi::Request<uint8_t>> send_reqs(reduces_this_step);
479  std::vector<El::mpi::Request<uint8_t>> recv_reqs(reduces_this_step);
480  std::vector<uint8_t *> recv_buffers(max_recv_buffers);
481  int num_local_recvs = 0;
482  std::vector<bool> local_recvs(reduces_this_step, false);
483  for (int step = outer_step; step < outer_step + reduces_this_step; ++step) {
484  const int reduce_idx = step - outer_step;
485  // Compute where we send to/receive from.
486  const int dst = (rank + step) % nprocs;
487  const int src = (rank - step + nprocs) % nprocs;
488  const bool is_send_local = opts.no_local_trans &&
489  is_rank_node_local(dst, comm);
490  const bool is_recv_local = opts.no_local_trans &&
491  is_rank_node_local(src, comm);
492  // Post the receive.
493  double recv_start = get_time();
494  int recv_size = max_recv_count;
495  if (is_recv_local) {
496  recv_size = sizeof(DataType) * accum_view.Height() * accum_view.Width();
497  recv_buffers[reduce_idx] = get_collective_buffer(recv_size,
498  num_local_recvs);
499  ++num_local_recvs;
500  local_recvs[reduce_idx] = is_recv_local;
501  }
502  El::mpi::IRecv(recv_buffers[reduce_idx], recv_size, src, comm,
503  recv_reqs[reduce_idx]);
504  double recv_tot = get_time() - recv_start;
505  ar_recv_time += recv_tot;
506  ar_rs_recv_time += recv_tot;
507  // Transform the data we send. We do not look at the same chunk of data
508  // twice.
509  double send_trans_start = get_time();
510  int send_size;
511  uint8_t *send_buf = nullptr;
512  if (is_send_local) {
513  auto send_view = mat(El::ALL,
514  El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
515  send_buf = (uint8_t *) send_view.Buffer();
516  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
517  } else {
518  send_buf = send_transform(
519  mat, El::ALL, El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]),
520  send_size, true, reduce_idx);
521  }
522  ar_send_transform_time += get_time() - send_trans_start;
523  bytes_sent += send_size;
524  ar_bytes_sent += send_size;
525  ar_rs_bytes_sent += send_size;
526  // Post the send.
527  double send_start = get_time();
528  El::mpi::ISend(send_buf, send_size, dst, comm, send_reqs[reduce_idx]);
529  double send_tot = get_time() - send_start;
530  ar_send_time += send_tot;
531  ar_rs_send_time += send_tot;
532  }
533  // Complete the receives (in any order).
534  // We need to extract the raw MPI_Request because Elemental does not support
535  // MPI_Waitany.
536  std::vector<MPI_Request> raw_reqs(reduces_this_step);
537  for (int i = 0; i < reduces_this_step; ++i) {
538  raw_reqs[i] = recv_reqs[i].backend;
539  }
540  for (int i = 0; i < reduces_this_step; ++i) {
541  int completed_idx;
542  double recv_start = get_time();
543  MPI_Waitany(reduces_this_step, raw_reqs.data(), &completed_idx,
544  MPI_STATUS_IGNORE);
545  double recv_tot = get_time() - recv_start;
546  ar_recv_time += recv_tot;
547  ar_rs_recv_time += recv_tot;
548  double recv_apply_trans_start = get_time();
549  int recv_size = recv_apply_transform(
550  recv_buffers[completed_idx], accum_view, local_recvs[completed_idx]);
551  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
552  bytes_received += recv_size;
553  ar_bytes_received += recv_size;
554  ar_rs_bytes_received += recv_size;
555  }
556  // Complete all the sends.
557  double send_start = get_time();
558  El::mpi::WaitAll(reduces_this_step, send_reqs.data(), MPI_STATUSES_IGNORE);
559  double send_tot = get_time() - send_start;
560  ar_send_time += send_tot;
561  ar_rs_send_time += send_tot;
562  }
563  uint8_t *recv_buf = max_recv_buffers[0]; // Get a regular recv buffer.
564  ar_rs_time += get_time() - rs_start;
565  // Do a ring allgather.
566  double ag_start = get_time();
567  const int src = (rank - 1 + nprocs) % nprocs;
568  const int dst = (rank + 1) % nprocs;
569  // Apply the transform to our locally-accumulated slice of the data.
570  // Since the same data is cycled to every process, we do not do the
571  // no_local_trans here.
572  int send_size;
573  // Do the first step where we forward our local data.
574  {
575  double send_trans_start = get_time();
576  uint8_t *send_buf = send_transform(
577  mat, El::ALL, El::IR(slice_ends[rank] - slice_lengths[rank], slice_ends[rank]),
578  send_size, false, 0);
579  ar_send_transform_time += get_time() - send_trans_start;
580  const int data_src = (rank - 1 + nprocs) % nprocs;
581  bytes_sent += send_size;
582  ar_bytes_sent += send_size;
583  ar_ag_bytes_sent += send_size;
584  auto recv_view = mat(El::ALL,
585  El::IR(slice_ends[data_src] - slice_lengths[data_src],
586  slice_ends[data_src]));
587  // If we can, receive directly into the destination matrix.
588  if (opts.id_recv) {
589  recv_buf = (uint8_t *) recv_view.Buffer();
590  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
591  }
592  double sendrecv_start = get_time();
593  El::mpi::SendRecv(send_buf, send_size, dst,
594  recv_buf, max_recv_count, src, comm);
595  double sendrecv_tot = get_time() - sendrecv_start;
596  ar_send_time += sendrecv_tot;
597  ar_recv_time += sendrecv_tot;
598  ar_ag_send_time += sendrecv_tot;
599  ar_ag_recv_time += sendrecv_tot;
600  double recv_trans_start = get_time();
601  int recv_size = 0;
602  if (opts.id_recv) {
603  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
604  } else {
605  recv_size = recv_transform(recv_buf, recv_view);
606  }
607  ar_recv_transform_time += get_time() - recv_trans_start;
608  bytes_received += recv_size;
609  ar_bytes_received += recv_size;
610  ar_ag_bytes_received += send_size;
611  send_size = recv_size;
612  }
613  // Now do the remaining nprocs - 2 steps.
614  // We always send from recv_buf and receive to recv_buf2, swapping
615  // pointers to avoid copying.
616  uint8_t *recv_buf2 = nullptr;
617  if (!opts.id_recv) {
618  recv_buf2 = get_collective_buffer(max_recv_count, 1);
619  }
620  for (int step = 1; step < nprocs - 1; ++step) {
621  // Compute where the data we get is coming from.
622  const int data_src = (rank - step - 1 + nprocs) % nprocs;
623  auto recv_view = mat(El::ALL,
624  El::IR(slice_ends[data_src] - slice_lengths[data_src],
625  slice_ends[data_src]));
626  if (opts.id_recv) {
627  recv_buf2 = (uint8_t *) recv_view.Buffer();
628  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
629  }
630  bytes_sent += send_size;
631  ar_bytes_sent += send_size;
632  ar_ag_bytes_sent += send_size;
633  double sendrecv_start = get_time();
634  El::mpi::SendRecv(recv_buf, send_size, dst,
635  recv_buf2, max_recv_count, src, comm);
636  double sendrecv_tot = get_time() - sendrecv_start;
637  ar_send_time += sendrecv_tot;
638  ar_recv_time += sendrecv_tot;
639  ar_ag_send_time += sendrecv_tot;
640  ar_ag_recv_time += sendrecv_tot;
641  double recv_trans_start = get_time();
642  int recv_size = 0;
643  if (opts.id_recv) {
644  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
645  } else {
646  recv_size = recv_transform(recv_buf2, recv_view);
647  }
648  ar_recv_transform_time += get_time() - recv_trans_start;
649  bytes_received += recv_size;
650  // Swap the send and receive buffers.
651  std::swap(recv_buf, recv_buf2);
652  send_size = recv_size;
653  ar_bytes_received += recv_size;
654  ar_ag_bytes_received += send_size;
655  }
656  ar_ag_time += get_time() - ag_start;
657  ar_time += get_time() - ar_start;

660 void lbann_comm::ring_allreduce(
661  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
662  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
663  std::function<int(uint8_t *, Mat&)> recv_transform,
664  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
665  const lbann_comm::allreduce_options opts) {
666  double ar_start = get_time();
667  const int rank = El::mpi::Rank(comm);
668  const int nprocs = El::mpi::Size(comm);
669  if (nprocs == 1) {
670  return; // Nothing to do.
671  }
672  // Compute the number of columns each processor sends.
673  const El::Int cols_per_proc = mat.Width() / nprocs;
674  const El::Int cols_remainder = mat.Width() % nprocs;
675  // Compute the lengths/ends of the slices.
676  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
677  for (int i = 0; i < cols_remainder; ++i) {
678  slice_lengths[i] += 1;
679  }
680  std::vector<El::Int> slice_ends(nprocs);
681  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
682  slice_ends.begin());
683  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
684  uint8_t *recv_buf = max_recv_buf;
685  // Compute source/destination in the ring.
686  const int src = (rank - 1 + nprocs) % nprocs;
687  const int dst = (rank + 1) % nprocs;
688  const bool is_send_local = opts.no_local_trans &&
689  is_rank_node_local(dst, comm);
690  const bool is_recv_local = opts.no_local_trans &&
691  is_rank_node_local(src, comm);
692  // Do a ring-based reduce-scatter.
693  // This is like the pairwise-exchange reduce-scatter except instead of
694  // rank i accumulating only slice i, the slices are cycled around and
695  // each node accumulates its portion into the slice when it passes
696  // through. After the nprocs-1 steps slice k will be on rank
697  // (k + nprocs - 1) % nprocs.
698  double rs_start = get_time();
699  for (int step = 0; step < nprocs - 1; ++step) {
700  // Compute the slices to send/recv.
701  const int send_slice = (rank - step + nprocs) % nprocs;
702  const int recv_slice = (rank - step - 1 + nprocs) % nprocs;
703  // Transform the data to send.
704  double send_trans_start = get_time();
705  int send_size;
706  int recv_size = max_recv_count;
707  uint8_t *send_buf = nullptr;
708  if (is_send_local) {
709  auto send_view = mat(El::ALL,
710  El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
711  send_buf = (uint8_t *) send_view.Buffer();
712  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
713  } else {
714  send_buf = send_transform(mat, El::ALL,
715  El::IR(slice_ends[send_slice] - slice_lengths[send_slice],
716  slice_ends[send_slice]), send_size, false, 0);
717  }
718  auto recv_view = mat(El::ALL,
719  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice], slice_ends[recv_slice]));
720  if (is_recv_local) {
721  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
722  recv_buf = get_collective_buffer(recv_size);
723  } else {
724  recv_buf = max_recv_buf;
725  }
726  ar_send_transform_time += get_time() - send_trans_start;
727  bytes_sent += send_size;
728  ar_bytes_sent += send_size;
729  ar_rs_bytes_sent += send_size;
730  double sendrecv_start = get_time();
731  El::mpi::SendRecv(send_buf, send_size, dst,
732  recv_buf, recv_size, src, comm);
733  double sendrecv_tot = get_time() - sendrecv_start;
734  ar_send_time += sendrecv_tot;
735  ar_recv_time += sendrecv_tot;
736  ar_rs_send_time += sendrecv_tot;
737  ar_rs_recv_time += sendrecv_tot;
738  double recv_apply_trans_start = get_time();
739  recv_size = recv_apply_transform(recv_buf, recv_view, is_recv_local);
740  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
741  bytes_received += recv_size;
742  ar_bytes_received += recv_size;
743  ar_rs_bytes_received += recv_size;
744  }
745  recv_buf = max_recv_buf; // Ensure we're back to the original.
746  ar_rs_time += get_time() - rs_start;
747  // Do a ring allgather, first applying the transform to local data.
748  double ag_start = get_time();
749  int send_size;
750  {
751  const int send_slice = (rank + 1) % nprocs;
752  const int recv_slice = rank;
753  double send_trans_start = get_time();
754  uint8_t *send_buf = send_transform(
755  mat, El::ALL, El::IR(slice_ends[send_slice] - slice_lengths[send_slice],
756  slice_ends[send_slice]), send_size, false, 0);
757  ar_send_transform_time += get_time() - send_trans_start;
758  bytes_sent += send_size;
759  ar_bytes_sent += send_size;
760  ar_ag_bytes_sent += send_size;
761  auto recv_view = mat(El::ALL,
762  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
763  slice_ends[recv_slice]));
764  // If we can, receive directly into the destination matrix.
765  if (opts.id_recv) {
766  recv_buf = (uint8_t *) recv_view.Buffer();
767  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
768  }
769  double sendrecv_start = get_time();
770  El::mpi::SendRecv(send_buf, send_size, dst,
771  recv_buf, max_recv_count, src, comm);
772  double sendrecv_tot = get_time() - sendrecv_start;
773  ar_send_time += sendrecv_tot;
774  ar_recv_time += sendrecv_tot;
775  ar_ag_send_time += sendrecv_tot;
776  ar_ag_recv_time += sendrecv_tot;
777  double recv_trans_start = get_time();
778  int recv_size = 0;
779  if (opts.id_recv) {
780  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
781  } else {
782  recv_size = recv_transform(recv_buf, recv_view);
783  }
784  ar_recv_transform_time += get_time() - recv_trans_start;
785  send_size = recv_size;
786  bytes_received += recv_size;
787  ar_bytes_received += recv_size;
788  ar_ag_bytes_received += recv_size;
789  }
790  uint8_t *recv_buf2 = nullptr;
791  if (!opts.id_recv) {
792  recv_buf2 = get_collective_buffer(max_recv_count, 1);
793  }
794  for (int step = 1; step < nprocs - 1; ++step) {
795  const int recv_slice = (rank - step + nprocs) % nprocs;
796  auto recv_view = mat(El::ALL,
797  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
798  slice_ends[recv_slice]));
799  if (opts.id_recv) {
800  recv_buf2 = (uint8_t *) recv_view.Buffer();
801  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
802  }
803  bytes_sent += send_size;
804  ar_bytes_sent += send_size;
805  ar_ag_bytes_sent += send_size;
806  double sendrecv_start = get_time();
807  El::mpi::SendRecv(recv_buf, send_size, dst,
808  recv_buf2, max_recv_count, src, comm);
809  double sendrecv_tot = get_time() - sendrecv_start;
810  ar_send_time += sendrecv_tot;
811  ar_recv_time += sendrecv_tot;
812  ar_ag_send_time += sendrecv_tot;
813  ar_ag_recv_time += sendrecv_tot;
814  double recv_trans_start = get_time();
815  int recv_size = 0;
816  if (opts.id_recv) {
817  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
818  } else {
819  recv_size = recv_transform(recv_buf2, recv_view);

821  ar_recv_transform_time += get_time() - recv_trans_start;

833 void lbann_comm::rabenseifner_allreduce(
834  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
835  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
836  std::function<int(uint8_t *, Mat&)> recv_transform,
837  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
838  const lbann_comm::allreduce_options opts) {
839  double ar_start = get_time();
840  const int rank = El::mpi::Rank(comm);
841  const unsigned int nprocs = El::mpi::Size(comm);
842  if (nprocs == 1) {
843  return; // Nothing to do.
844  }
845  // This implementation requires a power-of-2 number of processes.
846  if (nprocs & (nprocs - 1)) {
847  throw lbann_exception("lbann_comm: Rabenseifner allreduce requires"
848  " a power-of-2 number of participating processes");
849  }
850  // Compute the slices on each processor.
851  const El::Int cols_per_proc = mat.Width() / nprocs;
852  const El::Int cols_remainder = mat.Width() % nprocs;
853  // Compute the lengths/ends of the slices.
854  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
855  for (int i = 0; i < cols_remainder; ++i) {
856  slice_lengths[i] += 1;
857  }
858  std::vector<El::Int> slice_ends(nprocs);
859  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
860  slice_ends.begin());
861  // Do a recursive-halving reduce-scatter.
862  // In each step here a process sends all the data needed for the other
863  // "half" of the processes. i.e. each process sends half their data in the
864  // first step, a quarter in the second step, etc.
865  double rs_start = get_time();
866  unsigned int partner_mask = nprocs >> 1;
867  unsigned int slice_mask = 1;
868  unsigned int send_idx = 0;
869  unsigned int recv_idx = 0;
870  unsigned int last_idx = nprocs;
871  uint8_t *recv_buf = get_collective_buffer(max_recv_count);
872  while (partner_mask > 0) {
873  int partner = rank ^ partner_mask; // The rank we exchange with this step.
874  const bool is_local = opts.no_local_trans &&
875  is_rank_node_local(partner, comm);
876  // Determine the range of data to send/recv.
877  El::IR send_range, recv_range;
878  if (rank < partner) {
879  send_idx = recv_idx + nprocs / (slice_mask*2);
880  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
881  slice_ends[last_idx-1]);
882  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
883  slice_ends[send_idx-1]);
884  } else {
885  recv_idx = send_idx + nprocs / (slice_mask*2);
886  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
887  slice_ends[recv_idx-1]);
888  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
889  slice_ends[last_idx-1]);
890  }
891  auto recv_view = mat(El::ALL, recv_range);
892  // Transform the data to send.
893  double send_trans_start = get_time();
894  int send_size;
895  int recv_size = max_recv_count;
896  uint8_t *send_buf = nullptr;
897  if (is_local) {
898  auto send_view = mat(El::ALL, send_range);
899  send_buf = (uint8_t *) send_view.Buffer();
900  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
901  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
902  } else {
903  send_buf = send_transform(mat, El::ALL, send_range, send_size, false, 0);
904  }
905  ar_send_transform_time += get_time() - send_trans_start;
906  bytes_sent += send_size;
907  ar_bytes_sent += send_size;
908  ar_rs_bytes_sent += send_size;
909  double sendrecv_start = get_time();
910  El::mpi::SendRecv(send_buf, send_size, partner,
911  recv_buf, recv_size, partner, comm);
912  double sendrecv_tot = get_time() - sendrecv_start;
913  ar_send_time += sendrecv_tot;
914  ar_recv_time += sendrecv_tot;
915  ar_rs_send_time += sendrecv_tot;
916  ar_rs_recv_time += sendrecv_tot;
917  // Transform the received data.
918  double recv_apply_trans_start = get_time();
919  recv_size = recv_apply_transform(recv_buf, recv_view, is_local);
920  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
921  bytes_received += recv_size;
922  ar_bytes_received += recv_size;
923  ar_rs_bytes_received += send_size;
924  // Update info for next iteration.
925  // Except last_idx when needed for the allgather.
926  send_idx = recv_idx;
927  partner_mask >>= 1;
928  slice_mask <<= 1;
929  if (partner_mask > 0) {
930  last_idx = recv_idx + nprocs / (slice_mask);
931  }
932  }
933  ar_rs_time += get_time() - rs_start;
934  // Do a recursive-doubling algather.
935  double ag_start = get_time();
936  slice_mask >>= 1;
937  partner_mask = 1;
938  // Now do the remaining steps.
939  while (partner_mask < nprocs) {
940  int partner = rank ^ partner_mask;
941  const bool is_local = opts.no_local_trans &&
942  is_rank_node_local(partner, comm);
943  // Determine range to send/recv.
944  El::IR send_range, recv_range;
945  if (rank < partner) {
946  if (slice_mask != nprocs / 2) {
947  last_idx = last_idx + nprocs / (slice_mask*2);
948  }
949  recv_idx = send_idx + nprocs / (slice_mask*2);
950  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
951  slice_ends[recv_idx-1]);
952  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
953  slice_ends[last_idx-1]);
954  } else {
955  recv_idx = send_idx - nprocs / (slice_mask*2);
956  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
957  slice_ends[last_idx-1]);
958  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
959  slice_ends[send_idx-1]);
960  }
961  auto recv_view = mat(El::ALL, recv_range);
962  // Transform the data to send.
963  double send_trans_start = get_time();
964  int send_size;
965  int recv_size = max_recv_count;
966  uint8_t *send_buf = nullptr;
967  if (is_local) {
968  auto send_view = mat(El::ALL, send_range);
969  send_buf = (uint8_t *) send_view.Buffer();
970  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
971  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
972  } else {
973  send_buf = send_transform(mat, El::ALL, send_range, send_size, false, 0);
974  }
975  ar_send_transform_time += get_time() - send_trans_start;
976  if (opts.id_recv || is_local) {
977  recv_buf = (uint8_t *) recv_view.Buffer();
978  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
979  }
980  bytes_sent += send_size;
981  ar_bytes_sent += send_size;
982  ar_ag_bytes_sent += send_size;
983  double sendrecv_start = get_time();
984  El::mpi::SendRecv(send_buf, send_size, partner,
985  recv_buf, recv_size, partner, comm);
986  double sendrecv_tot = get_time() - sendrecv_start;
987  ar_send_time += sendrecv_tot;
988  ar_recv_time += sendrecv_tot;
989  ar_ag_send_time += sendrecv_tot;
990  ar_ag_recv_time += sendrecv_tot;
991  double recv_trans_start = get_time();
992  if (opts.id_recv) {
993  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
994  } else {
995  recv_size = recv_transform(recv_buf, recv_view);
996  }
997  ar_recv_transform_time += get_time() - recv_trans_start;
998  bytes_received += recv_size;
999  ar_bytes_received += recv_size;
1000  ar_ag_bytes_received += send_size;
1001  // Update for the next iteration.
1002  if (rank > partner) {
1003  send_idx = recv_idx;
1004  }
1005  partner_mask <<= 1;
1006  slice_mask >>= 1;
1007  }
1008  ar_ag_time += get_time() - ag_start;
1009  ar_time += get_time() - ar_start;
1010 }
