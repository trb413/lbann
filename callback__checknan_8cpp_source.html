35 void lbann_callback_checknan::on_forward_prop_end(model *m, Layer *l) {
36  if (dynamic_cast<generic_target_layer*>(l) != nullptr) {
37  return;
38  }
39  const AbsDistMat& acts = l->get_activations();
40  if (!is_good(acts)) {
41  dump_network(m);
42  std::stringstream ss;
43  ss << name() << ": "
44  << "[" << std::to_string(m->get_comm()->get_rank_in_world()) << "]: "
45  << "error in activations of " << l->get_name() << " "
46  << "(step=" << std::to_string(m->get_cur_step()) << ")";
47  throw lbann_exception(ss.str());
48  }
49 }

80 bool lbann_callback_checknan::is_good(const AbsDistMat& m) {
81  const Mat& lm = m.LockedMatrix();
82  const El::Int height = lm.Height();
83  const El::Int width = lm.Width();
84  for (El::Int col = 0; col < width; ++col) {
85  for (El::Int row = 0; row < height; ++row) {
86  const DataType val = lm(row, col);
87  if (std::isnan(val)) {
88  std::cout << "Found NaN at (" << row << ", " << col << ")!" << std::endl;
89  return false;
90  } else if (std::isinf(val)) {
91  std::cout << "Found inf (" << row << ", " << col << ")!" << std::endl;
92  return false;
93  }
94  }
95  }
96  return true;
97 }

99 void lbann_callback_checknan::dump_network(model *m) {
100  // Dump only the local matrices because not every rank will necessarily
101  // have bad data, and the check is purely local.
102  for (const Layer *layer : m->get_layers()) {
103  const std::string prefix
104  = ("model" + std::to_string(m->get_comm()->get_model_rank())
105  + "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
106  + "-epoch" + std::to_string(m->get_cur_epoch())
107  + "-step" + std::to_string(m->get_cur_step())
108  + "-" + layer->get_name()
109  + "-");
110  El::Write(layer->get_activations().LockedMatrix(),
111  prefix + "Activations",
112  El::ASCII);
113  }
114  for (weights *w : m->get_weights()) {
115  const std::string prefix
116  = ("model" + std::to_string(m->get_comm()->get_model_rank())
117  + "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
118  + "-epoch" + std::to_string(m->get_cur_epoch())
119  + "-step" + std::to_string(m->get_cur_step())
120  + "-" + w->get_name()
121  + "-");
122  El::Write(w->get_values().LockedMatrix(),
123  prefix + "Weights",
124  El::ASCII);
125  optimizer *opt = w->get_optimizer();
126  if (opt != nullptr) {
127  El::Write(opt->get_gradient().LockedMatrix(),
128  prefix + "Gradient",
129  El::ASCII);
130  }
131  }
132 }
