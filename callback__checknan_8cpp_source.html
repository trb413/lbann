34 void lbann_callback_checknan::on_forward_prop_end(model *m, Layer *l) {
35  if (dynamic_cast<generic_target_layer*>(l) != nullptr) {
36  return;
37  }
38  const AbsDistMat& acts = l->get_activations();
39  if (!is_good(acts)) {
40  dump_network(m);
41  std::stringstream ss;
42  ss << name() << ": "
43  << "[" << std::to_string(m->get_comm()->get_rank_in_world()) << "]: "
44  << "error in activations of " << l->get_name() << " "
45  << "(step=" << std::to_string(m->get_cur_step()) << ")";
46  throw lbann_exception(ss.str());
47  }
48 }

79 bool lbann_callback_checknan::is_good(const AbsDistMat& m) {
80  const Mat& lm = m.LockedMatrix();
81  const El::Int height = lm.Height();
82  const El::Int width = lm.Width();
83  for (El::Int col = 0; col < width; ++col) {
84  for (El::Int row = 0; row < height; ++row) {
85  const DataType val = lm(row, col);
86  if (std::isnan(val)) {
87  std::cout << "Found NaN at (" << row << ", " << col << ")!" << std::endl;
88  return false;
89  } else if (std::isinf(val)) {
90  std::cout << "Found inf (" << row << ", " << col << ")!" << std::endl;
91  return false;
92  }
93  }
94  }
95  return true;
96 }

98 void lbann_callback_checknan::dump_network(model *m) {
99  // Dump only the local matrices because not every rank will necessarily
100  // have bad data, and the check is purely local.
101  for (const Layer *layer : m->get_layers()) {
102  const std::string prefix
103  = ("model" + std::to_string(m->get_comm()->get_model_rank())
104  + "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
105  + "-epoch" + std::to_string(m->get_cur_epoch())
106  + "-step" + std::to_string(m->get_cur_step())
107  + "-" + layer->get_name()
108  + "-");
109  El::Write(layer->get_activations().LockedMatrix(),
110  prefix + "Activations",
111  El::ASCII);
112  }
113  for (weights *w : m->get_weights()) {
114  const std::string prefix
115  = ("model" + std::to_string(m->get_comm()->get_model_rank())
116  + "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
117  + "-epoch" + std::to_string(m->get_cur_epoch())
118  + "-step" + std::to_string(m->get_cur_step())
119  + "-" + w->get_name()
120  + "-");
121  El::Write(w->get_values().LockedMatrix(),
122  prefix + "Weights",
123  El::ASCII);
124  optimizer *opt = w->get_optimizer();
125  if (opt != nullptr) {
126  El::Write(opt->get_gradient().LockedMatrix(),
127  prefix + "Gradient",
128  El::ASCII);
129  }
130  }
131 }
