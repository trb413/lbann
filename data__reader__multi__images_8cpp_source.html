91 std::vector<::Mat> data_reader_multi_images::create_datum_views(::Mat& X, const int mb_idx) const {
92  std::vector<::Mat> X_v(m_num_img_srcs);
93  El::Int h = 0;
94  for(unsigned int i=0u; i < m_num_img_srcs; ++i) {
95  El::View(X_v[i], X, El::IR(h, h + m_image_linearized_size), El::IR(mb_idx, mb_idx + 1));
96  h = h + m_image_linearized_size;
97  }
98  return X_v;
99 }

101 bool data_reader_multi_images::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
102 
103  std::vector<::Mat> X_v = create_datum_views(X, mb_idx);
104 
105  const img_src_t& img_src = m_image_list[data_id].first;
106  for(size_t i=0u; i < m_num_img_srcs; ++i) {
107  int width=0, height=0, img_type=0;
108  const std::string imagepath = get_file_dir() + img_src[i];
109  bool ret = true;
110  if (m_data_store != nullptr) {
111  std::vector<unsigned char> *image_buf;
112  m_data_store->get_data_buf(data_id, image_buf, i);
113  ret = lbann::image_utils::load_image(*image_buf, width, height, img_type, *(m_pps[tid]), X_v[i]);
114  } else {
115  ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v[i]);
116  }
117 
118  if(!ret) {
119  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
120  + get_type() + ": image_utils::load_image failed to load - "
121  + imagepath);
122  }
123  if((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
124  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
125  + get_type() + ": mismatch data size -- either width, height or channel - "
126  + imagepath + " [w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
127  + "x" + std::to_string(CV_MAT_CN(img_type)) + "] != " + std::to_string(m_image_linearized_size));
128  }
129  }
130  return true;
131 }

150 bool data_reader_multi_images::read_text_stream(std::istream& text_stream,
151  std::vector<data_reader_multi_images::sample_t>& list)
152 {
153  std::string line;
154 
155  if (!std::getline(text_stream, line)) {
156  return false;
157  }
158 
159  while (text_stream) {
160  img_src_t img_srcs;
161  {
162  std::stringstream sstr(line.c_str());
163  for (unsigned int i=0u; i < m_num_img_srcs; ++i) {
164  std::string path;
165  sstr >> path;
166  img_srcs.push_back(path);
167  }
168  label_t label;
169  sstr >> label;
170  if (sstr.bad()) {
171  return false;
172  }
173  list.emplace_back(img_srcs, label);
174  }
175  std::getline(text_stream, line);
176  }
177  list.shrink_to_fit();
178  return true;
179 }
