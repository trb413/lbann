64 cv::Mat cv_subtractor::read_binary_image_file(const std::string filename) {
65  std::vector<int> tokens;
66  { // Extract the information on the image from the file name
67  const std::vector<char> delims = {'-', 'x','x','-','.'};
68  std::string dir;
69  std::string basename;
70 
71  parse_path(filename, dir, basename);
72  tokens = get_tokens(basename, delims);
73  if (tokens.size() != delims.size()) {
74  return cv::Mat();
75  }
76  }
77 
78  std::ifstream file(filename, std::ios::binary);
79  if (!file.good()) {
80  return cv::Mat();
81  }
82  file.unsetf(std::ios::skipws);
83 
84  { // Check file size
85  const size_t image_byte_size
86  = tokens[1] * tokens[2] * tokens[3] * CV_ELEM_SIZE(tokens[4]);
87 
88  file.seekg(0, std::ios::end);
89  const size_t file_size = static_cast<size_t>(file.tellg());
90  if (image_byte_size != file_size) {
91  return cv::Mat();
92  }
93  }
94 
95  // Construct an image data structure
96  cv::Mat image(tokens[1], tokens[2], CV_MAKETYPE(tokens[4], tokens[3]));
97 
98  // Reset the file pointer
99  file.seekg(0, std::ios::beg);
100 
101  // Load the image from the file
102  std::copy(std::istream_iterator<unsigned char>(file),
103  std::istream_iterator<unsigned char>(),
104  reinterpret_cast<unsigned char*>(image.data));
105 
106  return image;
107 }

109 void cv_subtractor::set(const std::string name_of_img_to_sub, const int depth_code) {
110  cv::Mat img_to_sub;
111  std::string ext = get_ext_name(name_of_img_to_sub);
112  std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
113  if (ext == "bin") {
114  img_to_sub = read_binary_image_file(name_of_img_to_sub);
115  } else { // let OpenCV handle
116  img_to_sub = cv::imread(name_of_img_to_sub);
117  }
118  if (img_to_sub.empty()) {
119  std::stringstream err;
120  err << __FILE__ << " " << __LINE__ << " :: cv_subtractor: cannot load the image "
121  << name_of_img_to_sub << " to subtract.";
122  throw lbann_exception(err.str());
123  }
124  set(img_to_sub, depth_code);
125 }

127 void cv_subtractor::set(const cv::Mat& image, const int depth_code) {
128  reset();
129 
130  const double f = get_depth_normalizing_factor(image.depth());
131 
132  // Make sure that the image is set as a floating point type image
133  // Note that this is the only way to set m_img_to_sub. This means that
134  // m_img_to_sub will be of a floating point type unless it is empty.
135 
136  if ((depth_code != CV_32F) && (depth_code != CV_64F)) {
137  // If the depth_code does not indicate a floating point type, see if the
138  // image is already of a floating point type. If so, use the same type.
139  // Otherwise, use the type of LBANN's DataType.
140  if (check_if_cv_Mat_is_float_type(image)) {
141  image.convertTo(m_img_to_sub, image.depth(), f, 0.0);
142  } else {
143  image.convertTo(m_img_to_sub, cv_image_type<lbann::DataType>::T(), f, 0.0);
144  }
145  } else {
146  image.convertTo(m_img_to_sub, depth_code, f, 0.0);
147  }
148 }

163 bool cv_subtractor::apply(cv::Mat& image) {
164  m_enabled = false; // turn off as the transform is applied once
165  if (m_subtracted) { // inverse if applied already
166  const double f = get_depth_denormalizing_factor(CV_8U);
167  cv::Mat image_sub;
168  cv::addWeighted(m_img_to_sub, f, image, 1.0, 0.0, image_sub, CV_8U);
169  image = image_sub;
170  m_subtracted = false;
171  } else {
172  const double f = get_depth_normalizing_factor(image.depth());
173  cv::Mat image_sub;
174  cv::addWeighted(m_img_to_sub, -1.0, image, f, 0.0, image_sub, m_img_to_sub.depth());
175  image = image_sub;
176  m_subtracted = true;
177  }
178 
179  return true;
180 }
