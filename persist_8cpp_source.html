67 bool lbann::persist::write_rank_distmat(persist_type type, const char *name, const AbsDistMat& M) {
68  // TODO: store in network order
69  std::string filename = m_checkpoint_dir;
70  if (type == persist_type::train) {
71  filename += std::string("/train_") + name;
72  } else if (type == persist_type::model) {
73  filename += std::string("/model_") + name;
74  } else {
75  throw lbann_exception("persist: invalid persist_type");
76  }
77  // skip all of this if matrix is not held on rank
78  const El::Int localHeight = M.LocalHeight();
79  const El::Int localWidth = M.LocalWidth();
80  // If this is the case we will try to grab the matrix from model rank 0 on reload
81  if(localHeight * localWidth == 0) { return true; }
82 
83 
84  int fd = lbann::openwrite(filename.c_str());
85 
86  // build our header
87  struct layer_header header;
88  header.rank = (uint64_t) M.Grid().Rank();
89  header.width = (uint64_t) M.Width();
90  header.height = (uint64_t) M.Height();
91  header.localwidth = (uint64_t) M.LocalWidth();
92  header.localheight = (uint64_t) M.LocalHeight();
93  header.ldim = (uint64_t) M.LDim();
94 
95  // write the header to the file
96  ssize_t write_rc = write(fd, &header, sizeof(header));
97  if (write_rc != sizeof(header)) {
98  // error!
99  }
100  m_bytes += write_rc;
101 
102  // now write the data for our part of the distributed matrix
103  const El::Int lDim = M.LDim();
104  if(localHeight == lDim) {
105  // the local dimension in memory matches the local height,
106  // so we can write our data in a single shot
107  auto *buf = (void *) M.LockedBuffer();
108  El::Int bufsize = localHeight * localWidth * sizeof(DataType);
109  write_rc = write(fd, buf, bufsize);
110  if (write_rc != bufsize) {
111  // error!
112  }
113  m_bytes += write_rc;
114  } else {
115  // TODO: if this padding is small, may not be a big deal to write it out anyway
116  // we've got some padding along the first dimension
117  // while storing the matrix in memory, avoid writing the padding
118  for(El::Int j = 0; j < localWidth; ++j) {
119  auto *buf = (void *) M.LockedBuffer(0, j);
120  El::Int bufsize = localHeight * sizeof(DataType);
121  write_rc = write(fd, buf, bufsize);
122  if (write_rc != bufsize) {
123  // error!
124  }
125  m_bytes += write_rc;
126  }
127  }
128  return true;
129 }

133 bool lbann::persist::read_rank_distmat(persist_type type, const char *name, AbsDistMat& M) {
134  // read in the header
135  std::string filename = m_checkpoint_dir;
136  if (type == persist_type::train) {
137  filename += std::string("/train_") + name;
138  } else if (type == persist_type::model) {
139  filename += std::string("/model_") + name;
140  } else {
141  throw lbann_exception("persist: invalid persist_type");
142  }
143  int fd = openread(filename.c_str());
144  // file does not exist. we will try to grab matrix from rank 0
145  if( fd == -1 ) {return false;}
146 
147  struct layer_header header;
148  ssize_t read_rc = read(fd, &header, sizeof(header));
149  if (read_rc != sizeof(header)) {
150  // error!
151  throw lbann_exception("Failed to read layer header");
152  }
153  m_bytes += read_rc;
154 
155  // resize our global matrix
156  El::Int height = header.height;
157  El::Int width = header.width;
158  M.Resize(height, width);
159  // TODO: check that header values match up
160  const El::Int localheight = header.localheight;
161  const El::Int localwidth = header.localwidth;
162  if(M.ColStride() == 1 && M.RowStride() == 1) {
163  if(M.Height() == M.LDim()) {
164  auto *buf = (void *) M.Buffer();
165  El::Int bufsize = localheight * localwidth * sizeof(DataType);
166  read_rc = read(fd, buf, bufsize);
167  if (read_rc != bufsize) {
168  // error!
169  throw lbann_exception("Failed to read layer data");
170  }
171  m_bytes += read_rc;
172  } else {
173  for(El::Int j = 0; j < localwidth; ++j) {
174  auto *buf = (void *) M.Buffer(0, j);
175  El::Int bufsize = localheight * sizeof(DataType);
176  read_rc = read(fd, buf, bufsize);
177  if (read_rc != bufsize) {
178  // error!
179  throw lbann_exception("Failed to read layer data");
180  }
181  m_bytes += read_rc;
182  }
183  }
184  } else {
185  const El::Int lDim = M.LDim();
186  if(localheight == lDim) {
187  auto *buf = (void *) M.Buffer();
188  El::Int bufsize = localheight * localwidth * sizeof(DataType);
189  read_rc = read(fd, buf, bufsize);
190  if (read_rc != bufsize) {
191  // error!
192  throw lbann_exception("Failed to read layer data");
193  }
194  m_bytes += read_rc;
195  } else {
196  for(El::Int jLoc = 0; jLoc < localwidth; ++jLoc) {
197  auto *buf = (void *) M.Buffer(0, jLoc);
198  El::Int bufsize = localheight * sizeof(DataType);
199  read_rc = read(fd, buf, bufsize);
200  if (read_rc != bufsize) {
201  // error!
202  throw lbann_exception("Failed to read layer data");
203  }
204  m_bytes += read_rc;
205  }
206  }
207  }
208  return true;
209 }
