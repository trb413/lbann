69 bool lbann::writeDist(int fd, const char *filename, const DistMat& M, uint64_t *bytes) {
70  // TODO: store in network order
71 
72  // build our header
73  struct layer_header header;
74  header.rank = (uint64_t) M.Grid().Rank();
75  header.width = (uint64_t) M.Width();
76  header.height = (uint64_t) M.Height();
77  header.localwidth = (uint64_t) M.LocalWidth();
78  header.localheight = (uint64_t) M.LocalHeight();
79  header.ldim = (uint64_t) M.LDim();
80 
81  // write the header to the file
82  ssize_t write_rc = write(fd, &header, sizeof(header));
83  if (write_rc != sizeof(header)) {
84  // error!
85  }
86  *bytes += write_rc;
87 
88  // now write the data for our part of the distributed matrix
89  const Int localHeight = M.LocalHeight();
90  const Int localWidth = M.LocalWidth();
91  const Int lDim = M.LDim();
92  if(localHeight == lDim) {
93  // the local dimension in memory matches the local height,
94  // so we can write our data in a single shot
95  void *buf = (void *) M.LockedBuffer();
96  Int bufsize = localHeight * localWidth * sizeof(DataType);
97  write_rc = write(fd, buf, bufsize);
98  if (write_rc != bufsize) {
99  // error!
100  }
101  *bytes += write_rc;
102  } else {
103  // TODO: if this padding is small, may not be a big deal to write it out anyway
104  // we've got some padding along the first dimension
105  // while storing the matrix in memory, avoid writing the padding
106  for(Int j = 0; j < localWidth; ++j) {
107  void *buf = (void *) M.LockedBuffer(0, j);
108  Int bufsize = localHeight * sizeof(DataType);
109  write_rc = write(fd, buf, bufsize);
110  if (write_rc != bufsize) {
111  // error!
112  }
113  *bytes += write_rc;
114  }
115  }
116 
117  return true;
118 }

122 bool lbann::readDist(int fd, const char *filename, DistMat& M, uint64_t *bytes) {
123  // read in the header
124  struct layer_header header;
125  ssize_t read_rc = read(fd, &header, sizeof(header));
126  if (read_rc != sizeof(header)) {
127  // error!
128  throw lbann_exception("Failed to read layer header");
129  }
130  *bytes += read_rc;
131 
132  // resize our global matrix
133  Int height = header.height;
134  Int width = header.width;
135  M.Resize(height, width);
136 
137  // TODO: check that header values match up
138 
139  if(M.ColStride() == 1 && M.RowStride() == 1) {
140  if(M.Height() == M.LDim()) {
141  void *buf = (void *) M.Buffer();
142  Int bufsize = height * width * sizeof(DataType);
143  read_rc = read(fd, buf, bufsize);
144  if (read_rc != bufsize) {
145  // error!
146  throw lbann_exception("Failed to read layer data");
147  }
148  *bytes += read_rc;
149  } else {
150  for(Int j = 0; j < width; ++j) {
151  void *buf = (void *) M.Buffer(0, j);
152  Int bufsize = height * sizeof(DataType);
153  read_rc = read(fd, buf, bufsize);
154  if (read_rc != bufsize) {
155  // error!
156  throw lbann_exception("Failed to read layer data");
157  }
158  *bytes += read_rc;
159  }
160  }
161  } else {
162  const Int localHeight = M.LocalHeight();
163  const Int localWidth = M.LocalWidth();
164  const Int lDim = M.LDim();
165  if(localHeight == lDim) {
166  void *buf = (void *) M.Buffer();
167  Int bufsize = localHeight * localWidth * sizeof(DataType);
168  read_rc = read(fd, buf, bufsize);
169  if (read_rc != bufsize) {
170  // error!
171  throw lbann_exception("Failed to read layer data");
172  }
173  *bytes += read_rc;
174  } else {
175  for(Int jLoc = 0; jLoc < localWidth; ++jLoc) {
176  void *buf = (void *) M.Buffer(0, jLoc);
177  Int bufsize = localHeight * sizeof(DataType);
178  read_rc = read(fd, buf, bufsize);
179  if (read_rc != bufsize) {
180  // error!
181  throw lbann_exception("Failed to read layer data");
182  }
183  *bytes += read_rc;
184  }
185  }
186  }
187  return true;
188 }
