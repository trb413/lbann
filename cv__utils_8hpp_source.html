117 template<typename T, int NCh>
118 inline bool cv_utils::copy_cvMat_to_buf_with_full_info(
119  const cv::Mat& image, std::vector<uint8_t>& buf, const cv_process& pp) {
120  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
121 
122  const int Width = image.cols;
123  const int Height = image.rows;
124  const int sz = Height*Width;
125 
126  buf.resize(sz*NCh*sizeof(T));
127  auto *Pixels = reinterpret_cast<T *>(&(buf[0]));
128 
129  if (pp.to_split()) {
130  // TODO: like the case with the output in El::Matrixi type, branch on whether the
131  // input channel type T is same as that of the output (especially ::DataType)
132  std::vector<cv_normalizer::channel_trans_t> trans = pp.get_transform_normalize();
133  if (trans.size() == 0u) {
134  trans.assign(NCh, cv_normalizer::channel_trans_t(1.0, 0.0));
135  }
136  _LBANN_MILD_EXCEPTION((trans.size() != NCh),
137  "Incorrect number of channels in transform", false);
138  std::vector<cv::Mat> channels(NCh);
139 
140  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
141  channels[ch] = cv::Mat(Height, Width, CV_MAKETYPE(image.depth(),1), Pixels);
142  }
143  cv::split(image, channels);
144 
145  Pixels = reinterpret_cast<T *>(&(buf[0]));
146 
147  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
148  cv_normalizer::
149  scale(Pixels, Pixels + sz, Pixels, {trans[ch]});
150  }
151  } else {
152  if (image.isContinuous()) {
153  cv_normalizer::
154  scale(reinterpret_cast<const T *>(image.datastart),
155  reinterpret_cast<const T *>(image.dataend),
156  Pixels, pp.get_transform_normalize());
157  } else {
158  const int stride = Width*NCh;
159  for (int i = 0; i < Height; ++i, Pixels += stride) {
160  const auto *ptr = reinterpret_cast<const T *>(image.ptr<const T>(i));
161  cv_normalizer::
162  scale(ptr, ptr+stride, Pixels, pp.get_transform_normalize());
163  }
164  }
165  }
166 
167  return true;
168 }

198 inline cv::Mat cv_utils::copy_buf_to_cvMat_with_full_info(
199  const std::vector<uint8_t>& buf, const int Width, const int Height, const cv_process& pp) {
200 
201  const int sz = Height*Width;
202 
203  _LBANN_MILD_EXCEPTION(sz*NCh*sizeof(T) != buf.size(), \
204  "Size mismatch. Buffer has " << buf.size() << " items when " \
205  << sz*NCh*sizeof(T) << " are expected.", \
206  cv::Mat())
207 
208  const auto *Pixels = reinterpret_cast<const T *>(&(buf[0]));
209 
210  cv::Mat image = cv::Mat(Height, Width, CV_MAKETYPE(cv::DataType<T>::depth, NCh));
211 
212  if (pp.to_split()) {
213  // TODO: like the case with the output of El::Matrix type, branch on whether the
214  // input channel type T is same as that of the output (especially ::DataType)
215  std::vector<cv_normalizer::channel_trans_t> trans = pp.get_transform_normalize();
216  if (trans.size() == 0u) {
217  trans.assign(NCh, cv_normalizer::channel_trans_t(1.0, 0.0));
218  }
219  _LBANN_MILD_EXCEPTION((trans.size() != NCh),
220  "Incorrect number of channels in transform", cv::Mat());
221  std::vector<cv::Mat> channels(NCh);
222 
223  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
224  channels[ch] = cv::Mat(Height, Width, CV_MAKETYPE(image.depth(),1), const_cast<T *>(Pixels));
225  }
226 
227  cv::merge(channels, image);
228  auto *optr = reinterpret_cast<T *>(image.data);
229  for(size_t ch=0; ch < NCh; ++ch, optr += sz) {
230  cv_normalizer::
231  scale(reinterpret_cast<const T *>(image.datastart),
232  reinterpret_cast<const T *>(image.dataend),
233  optr, {trans[ch]});
234  }
235  } else {
236  cv_normalizer::
237  scale(Pixels, Pixels + sz*NCh, reinterpret_cast<T *>(image.data), pp.get_transform_normalize());
238  }
239 
240  return image;
241 }

287 template<typename T, int NCh>
288 inline bool cv_utils::copy_cvMat_to_buf_with_full_info(
289  const cv::Mat& image, ::Mat& buf, const cv_process& pp) {
290  // NCh need not be a template parameter here. It can be a function argument.
291  // However, keeping it as a static parameter enables custom accesses on pixels
292  // For example,
293  // using Vec_T = cv::Vec<T, NCh>;
294  // image.at<Vec_T>(y, x) = newPixel;
295  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
296 
297  const int Width = image.cols;
298  const int Height = image.rows;
299  const int sz = Height*Width;
300 
301  if (buf.Height() != sz*NCh) {
302 #if 0
303  return false;
304 #else
305  //_LBANN_DEBUG_MSG("Resizing buffer height to " << sz*NCh);
306  buf.Resize(sz*NCh, ((buf.Width()<1)? 1 : buf.Width()));
307 #endif
308  }
309 
310  DataType *Pixels = buf.Buffer();
311 
312  if (pp.to_split()) {
313  std::vector<cv_normalizer::channel_trans_t> trans = pp.get_transform_normalize();
314  if (trans.size() == 0u) {
315  trans.assign(NCh, cv_normalizer::channel_trans_t(1.0, 0.0));
316  }
317  _LBANN_MILD_EXCEPTION((trans.size() != NCh),
318  "Incorrect number of channels in transform", false);
319  std::vector<cv::Mat> channels(NCh);
320 
321  if (std::is_same<DataType, T>::value) {
322  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
323  // create a separate image per channel aliasing the memory of buf
324  channels[ch] = cv::Mat(Height, Width, CV_MAKETYPE(image.depth(),1), Pixels);
325  }
326  Pixels = buf.Buffer();
327 
328  cv::split(image, channels);
329 
330  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
331  cv_normalizer::
332  scale(Pixels, Pixels + sz, Pixels, {trans[ch]});
333  }
334  } else {
335  cv::split(image, channels);
336 
337  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
338  cv_normalizer::
339  scale(reinterpret_cast<const T *>(channels[ch].datastart),
340  reinterpret_cast<const T *>(channels[ch].dataend),
341  Pixels, {trans[ch]});
342  }
343  }
344  } else {
345  if (image.isContinuous()) {
346  cv_normalizer::
347  scale(reinterpret_cast<const T *>(image.datastart),
348  reinterpret_cast<const T *>(image.dataend),
349  Pixels, pp.get_transform_normalize());
350  } else {
351  const int stride = Width*NCh;
352  for (int i = 0; i < Height; ++i, Pixels += stride) {
353  const auto *ptr = reinterpret_cast<const T *>(image.ptr<const T>(i));
354  cv_normalizer::
355  scale(ptr, ptr+stride, Pixels, pp.get_transform_normalize());
356  }
357  }
358  }
359 
360  return true;
361 }

393 template<typename T, int NCh>
394 inline cv::Mat cv_utils::copy_buf_to_cvMat_with_full_info(
395  const ::Mat& buf, const int Width, const int Height, const cv_process& pp) {
396 
397  const int sz = Height*Width;
398  _LBANN_MILD_EXCEPTION(sz*NCh != buf.Height(), \
399  "Size mismatch. Buffer has " << buf.Height() << " items in a column when " \
400  << sz*NCh << " are expected.", \
401  cv::Mat())
402 
403  const DataType *Pixels = buf.LockedBuffer();
404 
405  cv::Mat image = cv::Mat(Height, Width, CV_MAKETYPE(cv::DataType<T>::depth, NCh));
406 
407  if (pp.to_split()) {
408  std::vector<cv_normalizer::channel_trans_t> trans = pp.get_transform_normalize();
409  if (trans.size() == 0u) {
410  trans.assign(NCh, cv_normalizer::channel_trans_t(1.0, 0.0));
411  }
412  _LBANN_MILD_EXCEPTION((trans.size() != NCh),
413  "Incorrect number of channels in transform", cv::Mat());
414  std::vector<cv::Mat> channels(NCh);
415 
416  if (std::is_same<DataType, T>::value) {
417  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
418  channels[ch] = cv::Mat(Height, Width, CV_MAKETYPE(image.depth(),1),
419  const_cast<DataType *>(Pixels));
420  }
421 
422  cv::merge(channels, image);
423  const auto *iptr = reinterpret_cast<const T *>(image.data);
424  auto *optr = reinterpret_cast<T *>(image.data);
425 
426  cv_normalizer::
427  scale(iptr, iptr+sz*NCh, optr, trans);
428  } else {
429  for(size_t ch=0; ch < NCh; ++ch, Pixels += sz) {
430  channels[ch] = cv::Mat(Height, Width, CV_MAKETYPE(image.depth(),1));
431  cv_normalizer::
432  scale(Pixels, Pixels+sz,
433  reinterpret_cast<T *>(channels[ch].data), {trans[ch]});
434  }
435  cv::merge(channels, image);
436  }
437  } else {
438  cv_normalizer::
439  scale(Pixels, Pixels + sz*NCh,
440  reinterpret_cast<T *>(image.data),
441  pp.get_transform_normalize());
442  }
443 
444  return image;
445 }
