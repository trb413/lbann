35 void setup_parents_and_children(lbann_comm* comm,
36  std::vector<Layer*>& layers,
37  std::unordered_map<std::string, Layer*>& names_to_layers,
38  const lbann_data::Model& proto_model) {
39  std::stringstream err;
40  for (int i=0; i<proto_model.layer_size(); ++i) {
41  const auto& proto_layer = proto_model.layer(i);
42  const auto& parents = parse_list<std::string>(proto_layer.parents());
43  const auto& children = parse_list<std::string>(proto_layer.children());
44  for (const auto& parent : parents) {
45  if (names_to_layers.count(parent) == 0) {
46  err << "could not find parent layer " << parent << " "
47  << "for layer " << layers[i]->get_name();
48  LBANN_ERROR(err.str());
49  }
50  layers[i]->add_parent_layer(names_to_layers[parent]);
51  }
52  for (const auto& child : children) {
53  if (names_to_layers.count(child) == 0) {
54  err << "could not find child layer " << child << " "
55  << "for layer " << layers[i]->get_name();
56  LBANN_ERROR(err.str());
57  }
58  layers[i]->add_child_layer(names_to_layers[child]);
59  }
60  }
61 }

64 void setup_target_pointers(lbann_comm* comm,
65  std::vector<Layer*>& layers,
66  std::unordered_map<std::string, Layer*>& names_to_layers,
67  const lbann_data::Model& proto_model) {
68  std::stringstream err;
69  for (int i=0; i<proto_model.layer_size(); ++i) {
70  generic_target_layer* target = dynamic_cast<generic_target_layer*>(layers[i]);
71  if (target != nullptr) {
72  generic_input_layer* input = nullptr;
73  const auto& input_name = proto_model.layer(i).target().paired_input_layer();
74  if (!input_name.empty()) {
75  input = dynamic_cast<generic_input_layer*>(names_to_layers[input_name]);
76  } else {
77  for (auto&& other : layers) {
78  input = dynamic_cast<generic_input_layer*>(other);
79  if (input != nullptr) { break; }
80  }
81  }
82  if (input == nullptr) {
83  err << "could not find input layer " << input_name << " "
84  << "to pair with target layer " << target->get_name();
85  LBANN_ERROR(err.str());
86  }
87  if (input->is_for_regression() != target->is_for_regression()) {
88  err << "target layer " << target->get_name() << " "
89  << "and its paired input layer " << input->get_name()
90  << "are not consistent regarding regression/classification";
91  LBANN_ERROR(err.str());
92  }
93  target->set_paired_input_layer(input);
94  }
95  }
96 }

99 void setup_reconstruction_pointers(lbann_comm* comm,
100  std::vector<Layer*>& layers,
101  std::unordered_map<std::string, Layer*>& names_to_layers,
102  const lbann_data::Model& proto_model) {
103  std::stringstream err;
104  for (int i=0; i<proto_model.layer_size(); ++i) {
105  const auto& proto_layer = proto_model.layer(i);
106  Layer* l = layers[i];
107  if (proto_layer.has_reconstruction()) {
108  Layer* original = nullptr;
109  const auto& original_name = proto_layer.reconstruction().original_layer();
110  if (!original_name.empty()) {
111  original = names_to_layers[original_name];
112  } else {
113  for (auto&& other : layers) {
114  original = dynamic_cast<generic_input_layer*>(other);
115  if (original != nullptr) { break; }
116  }
117  }
118  if (original == nullptr) {
119  err << "could not find original layer " << original_name << " "
120  << "for reconstruction layer " << l->get_name();
121  LBANN_ERROR(err.str());
122  }
123  auto&& recon_dp = dynamic_cast<reconstruction_layer<data_layout::DATA_PARALLEL>*>(l);
124  auto&& recon_mp = dynamic_cast<reconstruction_layer<data_layout::MODEL_PARALLEL>*>(l);
125  if (recon_dp != nullptr) { recon_dp->set_original_layer(original); }
126  if (recon_mp != nullptr) { recon_mp->set_original_layer(original); }
127  }
128  }
129 }

156 std::vector<Layer*> construct_layer_graph(lbann_comm* comm,
157  std::map<execution_mode, generic_data_reader *>& data_readers,
158  cudnn::cudnn_manager* cudnn,
159  const lbann_data::Model& proto_model) {
160  std::stringstream err;
161 
162  // List of layers
163  std::vector<Layer*> layers;
164 
165  // Map from names to layer pointers
166  std::unordered_map<std::string, Layer*> names_to_layers;
167 
168  // Create each layer in prototext
169  for (int i=0; i<proto_model.layer_size(); ++i) {
170  const auto& proto_layer = proto_model.layer(i);
171 
172  // Check that layer name is valid
173  auto name = proto_layer.name();
174  const auto& parsed_name = parse_list<std::string>(name);
175  if (!name.empty()) {
176  if (parsed_name.empty() || parsed_name.front() != name) {
177  err << "weights name \"" << name << "\" is invalid since it "
178  << "contains whitespace";
179  LBANN_ERROR(err.str());
180  }
181  if (names_to_layers.count(name) != 0) {
182  err << "layer name \"" << name << "\" is not unique";
183  LBANN_ERROR(err.str());
184  }
185  }
186 
187  // Get parameters from prototext
188  const auto& layout_str = proto_layer.data_layout();
189  data_layout layout = data_layout::invalid;
190  if (layout_str.empty()) { layout = data_layout::DATA_PARALLEL; }
191  if (layout_str == "data_parallel") { layout = data_layout::DATA_PARALLEL; }
192  if (layout_str == "model_parallel") { layout = data_layout::MODEL_PARALLEL; }
193  const auto& num_parallel_readers = proto_model.num_parallel_readers();
194 
195  // Construct layer
196  Layer* l = nullptr;
197  switch (layout) {
198  case data_layout::DATA_PARALLEL:
199  l = construct_layer<data_layout::DATA_PARALLEL>(
200  comm,
201  data_readers,
202  num_parallel_readers,
203  cudnn,
204  proto_layer
205  );
206  break;
207  case data_layout::MODEL_PARALLEL:
208  l = construct_layer<data_layout::MODEL_PARALLEL>(
209  comm,
210  data_readers,
211  num_parallel_readers,
212  cudnn,
213  proto_layer
214  );
215  break;
216  case data_layout::invalid:
217  default:
218  err << "layer " << name << " has an invalid data layout "
219  << "(" << layout_str << ")";
220  LBANN_ERROR(err.str());
221  }
222 
223  // Check that layer has been constructed
224  if (l == nullptr) {
225  err << "could not construct layer " << name;
226  LBANN_ERROR(err.str());
227  }
228 
229  // Initialize layer name and check it is unique
230  if (!name.empty()) {
231  l->set_name(name);
232  }
233  name = l->get_name();
234  if (names_to_layers.count(name) != 0) {
235  err << "layer name \"" << name << "\" is not unique";
236  LBANN_ERROR(err.str());
237  }
238  names_to_layers[name] = l;
239 
240  if (proto_layer.freeze()) {
241  if (comm->am_world_master()) {
242  std::cout << "freezing " << l->get_name() << std::endl;
243  }
244  l->freeze();
245  }
246  // Add layer to list
247  layers.push_back(l);
248 
249  }
250 
251  // Setup pointers between layers
252  setup_parents_and_children(comm, layers, names_to_layers, proto_model);
253  setup_target_pointers(comm, layers, names_to_layers, proto_model);
254  setup_reconstruction_pointers(comm, layers, names_to_layers, proto_model);
255  setup_unpooling_pointers(comm, layers, names_to_layers, proto_model);
256 
257  // Return layer list
258  return layers;
259 
260 }
