41 void siamese_model::setup_layer_topology() {
42 
44  m_follower_to_master_layer.clear();
45 
46  // Initialize network with master head
47  directed_acyclic_graph_model::setup_layer_topology();
48  setup_layer_execution_order();
49 
50  // Determine layers in master head
51  int heads_start = m_layers.size();
52  int heads_end = -1;
53  for (size_t i = 0; i < m_layers.size(); ++i) {
54  if (m_layers[i]->get_expected_num_child_layers() < 0) {
55  heads_start = i + 1;
56  break;
57  }
58  }
59  for (int i = m_layers.size() - 1; i >= 0; --i) {
60  if (m_layers[i]->get_expected_num_parent_layers() < 0) {
61  heads_end = i;
62  break;
63  }
64  }
65  if (heads_start > heads_end) {
66  std::stringstream err;
67  err << __FILE__ << " " << __LINE__ << " :: "
68  << "siamese models must have a fan-out layer before a fan-in layer "
69  << "(found fan-out layer at position " << heads_start - 1
70  << " and fan-in layer at position " << heads_end << ")";
71  throw lbann_exception(err.str());
72  }
73  std::vector<Layer*> master_head(m_layers.begin() + heads_start,
74  m_layers.begin() + heads_end);
75 
76  // Construct map from follower layers to master layers
77  for (const auto& layer : m_layers) {
78  m_follower_to_master_layer[layer] = layer;
79  }
80 
81  // Duplicate master head
82  for (int i = 1; i < m_num_heads; ++i) {
83  std::string name_suffix = "_head" + std::to_string(i);
84 
85  // Construct map from master layers to follower layers
86  std::unordered_map<Layer*,Layer*> master_to_follower_layer;
87  for (const auto& master_layer : m_layers) {
88  master_to_follower_layer[master_layer] = master_layer;
89  }
90 
91  // Construct follower head by copying layers in master head
92  std::vector<Layer*> follower_head;
93  for (const auto& master_layer : master_head) {
94  Layer* follower_layer = master_layer->copy();
95  follower_layer->set_name(follower_layer->get_name() + name_suffix);
96  follower_head.push_back(follower_layer);
97  master_to_follower_layer[master_layer] = follower_layer;
98  m_follower_to_master_layer[follower_layer] = master_layer;
99  }
100 
101  // Fix pointers in follower head
102  for (const auto& follower_layer : follower_head) {
103  auto layer_pointers = follower_layer->get_layer_pointers();
104  for (auto& layer_pointer : layer_pointers) {
105  layer_pointer = master_to_follower_layer[layer_pointer];
106  }
107  follower_layer->set_layer_pointers(layer_pointers);
108  }
109 
110  // Fix pointers at start and end of head
111  m_layers[heads_start-1]->add_child_layer(follower_head.front());
112  m_layers[heads_end]->add_parent_layer(follower_head.back());
113 
114  // Add follower head to model
115  m_layers.insert(m_layers.begin() + heads_end,
116  follower_head.begin(),
117  follower_head.end());
118  heads_end += follower_head.size();
119 
120  }
121 
122  // Rename layers in master head
123  for (const auto& layer : master_head) {
124  layer->set_name(layer->get_name() + "_head0");
125  }
126 
127  // Make sure all parent/child relationships are reciprocated
128  directed_acyclic_graph_model::setup_layer_topology();
129 
130 }
