56 void siamese_model::setup_layer_topology() {
57 
59  m_follower_to_master_layer.clear();
60 
61  // Initialize network with master head
62  directed_acyclic_graph_model::setup_layer_topology();
63  setup_layer_execution_order();
64 
65  // Determine layers in master head
66  int heads_start = m_layers.size();
67  int heads_end = -1;
68  for (size_t i = 0; i < m_layers.size(); ++i) {
69  if (m_layers[i]->get_expected_num_child_layers() < 0) {
70  heads_start = i + 1;
71  break;
72  }
73  }
74  for (int i = m_layers.size() - 1; i >= 0; --i) {
75  if (m_layers[i]->get_expected_num_parent_layers() < 0) {
76  heads_end = i;
77  break;
78  }
79  }
80  if (heads_start > heads_end) {
81  std::stringstream err;
82  err << __FILE__ << " " << __LINE__ << " :: "
83  << "siamese models must have a fan-out layer before a fan-in layer "
84  << "(found fan-out layer at position " << heads_start - 1
85  << " and fan-in layer at position " << heads_end << ")";
86  throw lbann_exception(err.str());
87  }
88  std::vector<Layer*> master_head(m_layers.begin() + heads_start,
89  m_layers.begin() + heads_end);
90 
91  // Construct map from follower layers to master layers
92  for (const auto& layer : m_layers) {
93  m_follower_to_master_layer[layer] = layer;
94  }
95 
96  // Duplicate master head
97  for (int i = 1; i < m_num_heads; ++i) {
98  std::string name_suffix = "_head" + std::to_string(i);
99 
100  // Construct map from master layers to follower layers
101  std::unordered_map<Layer*,Layer*> master_to_follower_layer;
102  for (const auto& master_layer : m_layers) {
103  master_to_follower_layer[master_layer] = master_layer;
104  }
105 
106  // Construct follower head by copying layers in master head
107  std::vector<Layer*> follower_head;
108  for (const auto& master_layer : master_head) {
109  Layer* follower_layer = master_layer->copy();
110  follower_layer->set_name(follower_layer->get_name() + name_suffix);
111  follower_head.push_back(follower_layer);
112  master_to_follower_layer[master_layer] = follower_layer;
113  m_follower_to_master_layer[follower_layer] = master_layer;
114  }
115 
116  // Fix pointers in follower head
117  for (const auto& follower_layer : follower_head) {
118  auto layer_pointers = follower_layer->get_layer_pointers();
119  for (auto& layer_pointer : layer_pointers) {
120  layer_pointer = master_to_follower_layer[layer_pointer];
121  }
122  follower_layer->set_layer_pointers(layer_pointers);
123  }
124 
125  // Fix pointers at start and end of head
126  m_layers[heads_start-1]->add_child_layer(follower_head.front());
127  m_layers[heads_end]->add_parent_layer(follower_head.back());
128 
129  // Add follower head to model
130  m_layers.insert(m_layers.begin() + heads_end,
131  follower_head.begin(),
132  follower_head.end());
133  heads_end += follower_head.size();
134 
135  }
136 
137  // Rename layers in master head
138  for (const auto& layer : master_head) {
139  layer->set_name(layer->get_name() + "_head0");
140  }
141 
142  // Make sure all parent/child relationships are reciprocated
143  directed_acyclic_graph_model::setup_layer_topology();
144 
145  freeze_layers_under_frozen_surface();
146 }
