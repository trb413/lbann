154 void adam::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
155 
156  // Precompute the bias correction and learning rate.
157  m_current_beta1 *= m_beta1;
158  m_current_beta2 *= m_beta2;
159  const DataType correction = m_learning_rate *
160  (std::sqrt(DataType(1) - m_current_beta2)
161  / (DataType(1) - m_current_beta1));
162 
163  // Get local matrix data
164  const int local_height = values.LocalHeight();
165  const int local_width = values.LocalWidth();
166  DataType* __restrict__ values_buffer = values.Buffer();
167  const int values_ldim = values.LDim();
168  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
169  const int gradient_ldim = gradient.LDim();
170  DataType* __restrict__ moment1_buffer = m_moment1->Buffer();
171  const int moment1_ldim = m_moment1->LDim();
172  DataType* __restrict__ moment2_buffer = m_moment2->Buffer();
173  const int moment2_ldim = m_moment2->LDim();
174 
175  // Check if matrix data is contiguous
176  if (values_ldim != local_height
177  || gradient_ldim != local_height
178  || moment1_ldim != local_height
179  || moment2_ldim != local_height) {
180  // Update with non-contiguous data
181  #pragma omp parallel for collapse(2)
182  for (int j=0; j<local_width; ++j) {
183  for (int i=0; i<local_height; ++i) {
184  DataType& x = values_buffer[i+j*values_ldim];
185  // See below; avoid denormalization.
186  const DataType g = gradient_buffer[i+j*gradient_ldim] + m_eps;
187  DataType& m1 = moment1_buffer[i+j*moment1_ldim];
188  DataType& m2 = moment2_buffer[i+j*moment2_ldim];
189  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
190  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
191  x -= correction * m1 / (std::sqrt(m2) + m_eps);
192  }
193  }
194  } else {
195  // Update with contiguous data
196  #pragma omp parallel for
197  for (int i=0; i<local_height*local_width; ++i) {
198  DataType& x = values_buffer[i];
199  // We add eps here because sometimes the gradient is small enough that
200  // g*g can become denormalized, which can significantly impact the
201  // performance.
202  const DataType g = gradient_buffer[i] + m_eps;
203  DataType& m1 = moment1_buffer[i];
204  DataType& m2 = moment2_buffer[i];
205  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
206  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
207  x -= correction * m1 / (std::sqrt(m2) + m_eps);
208  }
209  }
210 }
