90  void fp_compute() {
91  // Compute cost will be sum of squared error of fp_input (linearly transformed to m_activations)
92  // and original layer fp_input/original input
93  DataType avg_error = this->m_neural_network_model->m_obj_fn->compute_obj_fn(*this->m_prev_activations_v, original_layer_act_v);

127  void epoch_print() const {
128  double avg_cost = average_cost();
129  if (this->m_comm->am_world_master()) {
130  std::vector<double> avg_costs(this->m_comm->get_num_models());
131  this->m_comm->intermodel_gather(avg_cost, avg_costs);
132  for (size_t i = 0; i < avg_costs.size(); ++i) {
133  std::cout << "model " << i << " average reconstruction cost: " << avg_costs[i] << std::endl;
134  }
135  } else {
136  this->m_comm->intermodel_gather(avg_cost, this->m_comm->get_world_master());
137  }
138  }
