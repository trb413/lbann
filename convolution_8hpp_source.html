149  void setup_dims() override {
150 
151  // Initialize previous neuron tensor dimensions
152  base_convolution_layer::setup_dims();
153 
154  // Initialize convolution kernel dimensions
155  this->m_kernel_dims.insert(this->m_kernel_dims.begin() + 1,
156  this->m_prev_neuron_dims[0]);
157 
158  // Check if previous neuron tensor dimensions are valid
159  #ifdef LBANN_DEBUG
160  if(this->m_num_neuron_dims != (int) this->m_kernel_dims.size() - 1) {
161  throw lbann_exception("convolution_layer: neuron tensor dimensions are unexpected");
162  }
163  #endif
164 
165  // Initialize neuron tensor dimensions
166  this->m_neuron_dims[0] = this->m_kernel_dims[0];
167  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
168  const int effective_dim = (this->m_prev_neuron_dims[i+1]
169  + 2 * this->m_pads[i]
170  - this->m_kernel_dims[i+2] + 1);
171  this->m_neuron_dims[i+1]= ((effective_dim + this->m_strides[i] - 1)
172  / this->m_strides[i]);
173  }
174  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),
175  this->m_neuron_dims.end(),
176  1,
177  std::multiplies<int>());
178 
179  // Get size of convolutional kernel
180  this->m_kernel_size = std::accumulate(m_kernel_dims.begin(),
181  m_kernel_dims.end(),
182  1,
183  std::multiplies<int>());
184 
185  }
