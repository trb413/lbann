68 void loss_function::setup(model& m) {
69  objective_function_term::setup(m);
70 
71  // Check that loss function has one target layer and no weights
72  // Note: if target layer is not specified, choose the latest target
73  // layer in the model.
74  if (m_layers.size() > 1) {
75  std::stringstream err;
76  err << __FILE__ << " " << __LINE__ << " :: "
77  << "attempted to setup loss function with invalid number of target layers "
78  << "(expected 1, found " << m_layers.size() << ")";
79  throw lbann_exception(err.str());
80  }
81  if (m_layers.empty()) {
82  const std::vector<Layer*> layers = m.get_layers();
83  for (int i = layers.size() - 1; i >= 0; --i) {
84  if (dynamic_cast<generic_target_layer*>(layers[i]) != nullptr) {
85  m_layers.push_back(layers[i]);
86  break;
87  }
88  }
89  }
90  if (m_layers.size() != 1
91  || dynamic_cast<generic_target_layer*>(m_layers[0]) == nullptr) {
92  std::stringstream err;
93  err << __FILE__ << " " << __LINE__ << " :: "
94  << "could not setup loss function with target layer";
95  throw lbann_exception(err.str());
96  }
97  if (m_weights.size() != 0) {
98  std::stringstream err;
99  err << __FILE__ << " " << __LINE__ << " :: "
100  << "attempted to setup loss function with invalid number of weights "
101  << "(expected 0, found " << m_weights.size() << ")";
102  throw lbann_exception(err.str());
103  }
104 
105  // Initialize gradient matrix
106  auto *target = (generic_target_layer*) m_layers[0];
107  const AbsDistMat& ground_truth = target->get_ground_truth();
108  m_gradient = ground_truth.Construct(ground_truth.Grid(),
109  ground_truth.Root());
110  El::Zeros(*m_gradient, ground_truth.Height(), ground_truth.Width());
111 
112 }

129 void loss_function::differentiate() {
130  if (m_scale_factor == EvalType(0)) { return; }
131  auto *target = (generic_target_layer*) m_layers[0];
132  const AbsDistMat& prediction = target->get_prediction();
133  const AbsDistMat& ground_truth = target->get_ground_truth();
134  El::Zeros(*m_gradient, prediction.Height(), prediction.Width());
135  differentiate_compute(prediction, ground_truth, *m_gradient);
136  target->add_to_error_signal(*m_gradient, DataType(m_scale_factor));
