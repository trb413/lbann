138  static unsigned long long djb2_hash(void *func, int table_size)
139  {
140  char *str = (char*)&func;
141  unsigned long long hash = 5381;
142  int d = sizeof(func);
143  for (int j=0; j<d; j++) {
144  hash = ((hash << 5) + hash) + (int)str[j]; /* hash * 33 + c */
145  }
146  return hash;
147  }

165  static void c_hash_insert(void *func)
166  {
167  c_hash_table *h = c_hash_the_hash_table;
168  h->count += 1;
169  if (1000 + h->count == h->size) {
170  printf("out of slots! Please reimplement rehash\n");
171  exit(9);
172  }
173 
174  dladdr(func, &c_hash_info);
175  if (c_hash_info.dli_sname != NULL) {
176  int success = 0;
177  char *demangled_name = abi::__cxa_demangle(c_hash_info.dli_sname, nullptr, nullptr, nullptr);
178  if (demangled_name != NULL) {
179  unsigned long long key = djb2_hash(func, h->size);
180  unsigned long start, tmp;
181  HASH_1(key, h->size, &start);
182  for (long i=0; i<h->size; i++) {
183  HASH_2(key, h->size, &tmp);
184  long idx = (start + i*tmp) % h->size;
185  if (h->data[idx].mark < h->cur_mark) {
186  h->data[idx].key = key;
187  h->data[idx].mark = h->cur_mark;
188  h->data[idx].count = 1;
189  h->data[idx].name = (char*)malloc(strlen(demangled_name)+1);
190  strcpy(h->data[idx].name, demangled_name);
191  h->data[idx].func = func;
192  h->data[idx].id = c_hash_func_id;
193  if (c_hash_fp_full_stack_trace_metadata != NULL) {
194  fprintf(c_hash_fp_full_stack_trace_metadata, "%hd %s\n", c_hash_func_id, demangled_name);
195  fwrite(&c_hash_func_id, sizeof(short), 1, c_hash_fp_full_stack_trace);
196  fwrite(&c_hash_depth, sizeof(short), 1, c_hash_fp_full_stack_trace);
197  ++c_hash_func_id;
198  }
199  success = 1;
200  break;
201  }
202  }
203  if (success != 1) {
204  printf("\n\nERROR: success = %d\n\n", success);
205  exit(9);
206  }
207  }
208  free(demangled_name);
209  }
210  }

214  c_hash_node * c_hash_lookup(void *func)
215  {
216  ++c_hash_lookups;
217  c_hash_node * retval = NULL;
218  c_hash_table *h = c_hash_the_hash_table;
219  unsigned long long key = djb2_hash(func, h->size);
220 
221  unsigned long start, tmp;
222  HASH_1(key, h->size, &start);
223  for (long i=0; i<h->size; i++) {
224  if (i > 0) ++c_hash_collisions;
225  HASH_2(key, h->size, &tmp);
226  long idx = (start + i*tmp) % h->size;
227  if (h->data[idx].mark != h->cur_mark) {
228  break;
229  } else {
230  if (func == h->data[idx].func) {
231  retval = &h->data[idx];
232  }
233  }
234  }
235  return retval;
236  }
