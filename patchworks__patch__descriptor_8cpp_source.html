96 bool patch_descriptor::get_first_patch(ROI& patch) {
97  int x_center = (m_sample_area.width()+1)/2 + m_sample_area.left();
98  int y_center = (m_sample_area.height()+1)/2 + m_sample_area.top();
99  int x_margin = 0;
100  int y_margin = 0;
101 
102  if (m_mode_center == 0u) {
103  // Consider the jitter for a patch at the boundary of an image
104  x_margin = (m_width+1)/2 + m_jitter;
105  y_margin = (m_height+1)/2 + m_jitter;
106  } else if (m_mode_center == 1u) {
107  // The jitter for the center patch is a part of gap.
108  //if (m_jitter > m_gap) return false;
109  x_margin = m_width + (m_width+1)/2 + 2*m_jitter + m_gap;
110  y_margin = m_height + (m_height+1)/2 + 2*m_jitter + m_gap;
111  }
112 
113  ::lbann::rng_gen& gen = ::lbann::get_generator();
114 
115  if ((m_mode_center == 0u || m_mode_center == 1u)) {
116  // area where the center of a center patch can be in
117  ROI center_patch_area;
118  bool ok = center_patch_area.set_by_corners(x_margin + m_sample_area.left(),
119  y_margin + m_sample_area.top(),
120  m_sample_area.width() - x_margin,
121  m_sample_area.height() - y_margin);
122  if (!ok) {
123  std::cout << "invalid center patch area: " << center_patch_area << std::endl;
124  return false;
125  }
126  if (!center_patch_area.is_valid()) {
127  return false;
128  }
129 
130  // randomly generate the center coordinate within the center patch area
131  std::uniform_int_distribution<int> rg_center_x(0, center_patch_area.width()-1);
132  std::uniform_int_distribution<int> rg_center_y(0, center_patch_area.height()-1);
133  x_center = rg_center_x(gen) + center_patch_area.left();
134  y_center = rg_center_y(gen) + center_patch_area.top();
135  }
136 
137  if (m_jitter > 0u) { // apply position jitter if enabled
138  std::uniform_int_distribution<int> rg_jitter_x(0, 2*m_jitter);
139  std::uniform_int_distribution<int> rg_jitter_y(0, 2*m_jitter);
140  x_center += rg_jitter_x(gen) - m_jitter;
141  y_center += rg_jitter_y(gen) - m_jitter;
142  }
143 
144  // set the center patch
145  ROI p;
146  if (!p.set_by_center(x_center, y_center, m_width, m_height) ||
147  !(m_sample_area >= p)) {
148  return false;
149  }
150 
151  m_patch_center = p;
152  patch = p;
153  m_positions.clear();
154  m_cur_patch_idx = 0u;
155  m_positions.push_back(patch);
156 
157  return true;
158 }

160 bool patch_descriptor::get_next_patch(ROI& patch) {
161  bool got_one = false;
162 
163  ::lbann::rng_gen& gen = ::lbann::get_generator();
164 
165  do {
166  ROI p = m_patch_center;
167 
168  if (m_cur_patch_idx >= m_displacements.size()) {
169  return false;
170  }
171  p.move(m_displacements[m_cur_patch_idx++]);
172 
173  if (m_jitter > 0u) {
174  std::uniform_int_distribution<int> rg_jitter_x(0, 2*m_jitter);
175  std::uniform_int_distribution<int> rg_jitter_y(0, 2*m_jitter);
176  const int x_jitter = rg_jitter_x(gen) - m_jitter;
177  const int y_jitter = rg_jitter_y(gen) - m_jitter;
178  p.move(displacement_type(x_jitter, y_jitter));
179  }
180 
181  if (p.is_valid() && (m_sample_area >= p)) {
182  patch = p;
183  got_one = true;
184  }
185  } while (!got_one);
186 
187  m_positions.push_back(patch);
188  return true;
189 }

192  patches.clear();
193  if (img.data == nullptr) {

195  }

204 
205 #if 0 // to generate all the patches defined in the set
206  unsigned int i = 1u;

209  patches.push_back(img(roi.rect()).clone());
