59 void generic_data_store::get_my_datastore_indices() {
60  //compute storage
61  size_t n = 0;
62  int stride = m_comm->get_procs_per_model();
63  for (size_t j=m_rank; j<m_num_global_indices; j+=stride) {
64  ++n;
65  }
66  //get the indices
67  m_my_datastore_indices.reserve(n); //these are the indices passed to data_reader::fetch_data
68  m_my_global_indices.reserve(n); //these are the shuffled indices
69 
70  for (size_t j=m_rank; j<m_num_global_indices; j+=stride) {
71  m_my_datastore_indices.push_back(j);
72  m_my_global_indices.push_back((*m_shuffled_indices)[j]);
73  }
74 }

76 void generic_data_store::setup() {
77  set_shuffled_indices( &(m_reader->get_shuffled_indices()) );
78  set_num_global_indices(); //virtual override in child classes
79  m_num_readers = m_reader->get_num_parallel_readers();
80 
81  // get the set of global indices used by this processor in
82  // generic_data_reader::fetch_data(). Note that these are
83  // "original' indices, not shuffled indices, i.e, these indices
84  // remain constant through all epochs
85  if (m_collect_minibatch_indices) {
86  if (m_master) { std::cerr << "calling m_model->collect_indices\n"; }
87  m_reader->set_save_minibatch_entries(true);
88  if (m_reader->get_role() == "train") {
89  m_model->collect_indices(execution_mode::training);
90  } else if (m_reader->get_role() == "validate") {
91  m_model->collect_indices(execution_mode::validation);
92  } else if (m_reader->get_role() == "test") {
93  m_model->collect_indices(execution_mode::testing);
94  } else {
95  std::stringstream s2;
96  s2 << __FILE__ << " " << __LINE__ << " :: "
97  << " bad role; should be train, test, or validate;"
98  << " we got: " << m_reader->get_role();
99  throw lbann_exception(s2.str());
100  }
101  m_reader->set_save_minibatch_entries(false);
102  }
103 
104  m_minibatch_indices = &(m_reader->get_minibatch_indices());
105 }

108 size_t generic_data_store::get_file_size(std::string dir, std::string fn) {
109  std::string imagepath;
110  if (m_dir == "") {
111  imagepath = fn;
112  } else {
113  imagepath = dir + fn;
114  }
115  struct stat st;
116  if (stat(imagepath.c_str(), &st) != 0) {
117  std::stringstream err;
118  err << __FILE__ << " " << __LINE__ << " :: "
119  << "stat failed for dir: " << dir
120  << " and fn: " << fn;
121  throw lbann_exception(err.str());
122  }
123  return st.st_size;
124 }
