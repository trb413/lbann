107 void generic_data_store::setup() {
108  set_shuffled_indices( &(m_reader->get_shuffled_indices()) );
109  set_num_global_indices();
110  m_num_readers = m_reader->get_num_parallel_readers();
111  if (m_master) {
112  std::cerr << "data_reader type is: " << m_reader->get_type() << "\n";
113  }
114 
115  if (is_subsidiary_store()) {
116  return;
117  }
118 
119  // get the set of global indices used by this processor in
120  // generic_data_reader::fetch_data(). Note that these are
121  // "original' indices, not shuffled indices, i.e, these indices
122  // remain constant through all epochs
123  if (m_master) { std::cerr << "calling m_model->collect_indices\n"; }
124  m_reader->set_save_minibatch_entries(true);
125  if (m_reader->get_role() == "train") {
126  m_model->collect_indices(execution_mode::training);
127  } else if (m_reader->get_role() == "validate") {
128  m_model->collect_indices(execution_mode::validation);
129  } else if (m_reader->get_role() == "test") {
130  m_model->collect_indices(execution_mode::testing);
131  } else {
132  std::stringstream s2;
133  s2 << __FILE__ << " " << __LINE__ << " :: "
134  << " bad role; should be train, test, or validate;"
135  << " we got: " << m_reader->get_role();
136  throw lbann_exception(s2.str());
137  }
138  m_reader->set_save_minibatch_entries(false);
139  m_my_minibatch_indices = &(m_reader->get_minibatch_indices());
140 }

193 void generic_data_store::exchange_mb_indices() {
194  exchange_mb_counts();
195  //setup data structures to exchange minibatch indices with all processors
196  //displacement vector
197  std::vector<int> displ(m_np);
198  displ[0] = 0;
199  for (size_t j=1; j<m_mb_counts.size(); j++) {
200  displ[j] = displ[j-1] + m_mb_counts[j-1];
201  }
202 
203  //recv vector
204  int n = std::accumulate(m_mb_counts.begin(), m_mb_counts.end(), 0);
205  std::vector<int> all_indices(n);
206 
207  //receive the indices
208  m_comm->all_gather<int>(m_my_minibatch_indices_v, all_indices, m_mb_counts, displ, m_comm->get_world_comm());
209 
210  //fill in the final data structure
211  m_all_minibatch_indices.resize(m_np);
212  for (int j=0; j<m_np; j++) {
213  m_all_minibatch_indices[j].reserve(m_mb_counts[j]);
214  for (int i=displ[j]; i<displ[j]+m_mb_counts[j]; i++) {
215  m_all_minibatch_indices[j].push_back(all_indices[i]);
216  }
217  }
218 }

220 void generic_data_store::exchange_partitioned_indices() {
221  //determine the largest number of minibatches over all processors
222  std::vector<int> counts(m_np);
223  int my_num_mb = m_my_minibatch_indices->size();
224  m_comm->model_all_gather<int>(my_num_mb, counts);
225  m_num_minibatches = 0;
226  for (auto t : counts) {
227  m_num_minibatches = (size_t)t > m_num_minibatches ? t : m_num_minibatches;
228  }
229  if (m_master) std::cerr << "num minibatches: " << m_num_minibatches << "\n";
230 
231  //pack m_my_minibatch_indices into a single vector;
232  //first, compute vector size, and exchange size with all procs
233  std::vector<int> v;
234  int count = m_my_minibatch_indices->size() + 1;
235  for (auto t : (*m_my_minibatch_indices)) {
236  count += t.size();
237  }
238  m_comm->model_all_gather<int>(count, counts);
239 
240 
241  //now, fill in the vector
242  std::vector<int> w;
243  w.reserve(count);
244  w.push_back(m_my_minibatch_indices->size());
245  for (auto t : (*m_my_minibatch_indices)) {
246  w.push_back(t.size());
247  for (size_t h=0; h<t.size(); h++) {
248  w.push_back(t[h]);
249  }
250  }
251  if (w.size() != (size_t)count) {
252  std::stringstream err;
253  err << "count: " << count << " w.size: " << w.size();
254  throw lbann_exception(err.str());
255  }
256 
257  // exchange the vectors
258  std::vector<int> displ(m_np);
259  displ[0] = 0;
260  for (size_t k=1; k<counts.size(); k++) {
261  displ[k] = displ[k-1] + counts[k-1];
262  }
263 
264  //construct recv vector
265  int n = std::accumulate(counts.begin(), counts.end(), 0);
266  std::vector<int> all_w(n);
267 
268  //exchange the indices
269  m_comm->all_gather<int>(w, all_w, counts, displ, m_comm->get_world_comm());
270 
271  //fill in the final data structure
272  m_all_partitioned_indices.resize(m_np);
273  for (size_t p=0; p<(size_t)m_np; p++) {
274  int *ww = all_w.data() + displ[p];
275  //note: it's possible that m_num_minibatches > num_minibatches;
276  // that's OK; for simplicity elsewhere in the code we want
277  // all procs to have the same number of minibatches
278  m_all_partitioned_indices[p].resize(m_num_minibatches);
279  size_t num_minibatches = *ww++;
280  for (size_t i=0; i<num_minibatches; i++) {
281  int mb_size = *ww++;
282  m_all_partitioned_indices[p][i].reserve(mb_size);
283  for (int j=0; j<mb_size; j++) {
284  m_all_partitioned_indices[p][i].push_back(*ww++);
285  }
286  }
287  }
288 }
