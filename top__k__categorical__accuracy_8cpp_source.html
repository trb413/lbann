35 EvalType top_k_categorical_accuracy_metric::evaluate_compute(const AbsDistMat& prediction,
36  const AbsDistMat& ground_truth) {
37  // This first computes the top k predictions within each column locally,
38  // then each column master gathers these, computes the global top k, and
39  // determines if an error was made.
40  El::Int num_errors = 0;
41  // Note: assumes structure is packed.
42  struct top_k_ele {
43  DataType val; // Predicted value.
44  DataType gt; // Ground truth.
45  };
46  const El::Int local_width = prediction.LocalWidth(); // minibatch dim
47  const El::Int local_height = prediction.LocalHeight(); // class dim
48  // Pack the top k predictions for each local column together.
49  std::vector<top_k_ele> local_top_k(m_top_k * local_width);
50  // Compute the top k entries locally.
51  std::vector<El::Int> local_indices(local_height);
52  std::iota(local_indices.begin(), local_indices.end(), 0);
53  for (El::Int mb_idx = 0; mb_idx < local_width; ++mb_idx) {
54  // Determine the top k local entries in this column.
55  std::partial_sort(
56  local_indices.begin(), local_indices.begin() + m_top_k,
57  local_indices.end(),
58  [mb_idx, &prediction] (El::Int a, El::Int b) -> bool {
59  return prediction.GetLocal(a, mb_idx) >
60  prediction.GetLocal(b, mb_idx); });
61  for (El::Int i = 0; i < m_top_k; ++i) {
62  El::Int idx = mb_idx*m_top_k + i;
63  local_top_k[idx].val = prediction.GetLocal(local_indices[i], mb_idx);
64  local_top_k[idx].gt = ground_truth.GetLocal(local_indices[i], mb_idx);
65  }
66  }
67  // Gather the data for each column to rank 0 within that column.
68  El::mpi::Comm col_comm = prediction.ColComm();
69  int col_comm_size = El::mpi::Size(col_comm);
70  if (El::mpi::Rank(col_comm) == 0) {
71  // This vector ends up being the concatenation of each local_top_k, and
72  // therefore accessing data for a single mini-batch requires computing the
73  // appropriate strides.
74  std::vector<top_k_ele> global_top_k(
75  m_top_k * local_width * col_comm_size);
76  get_comm().gather((DataType*) local_top_k.data(), 2*local_top_k.size(),
77  (DataType*) global_top_k.data(), col_comm);
78  // Compute the global top k elements in each column.
79  std::vector<El::Int> global_indices(m_top_k * col_comm_size);
80  std::iota(global_indices.begin(), global_indices.end(), 0);
81  for (El::Int mb_idx = 0; mb_idx < local_width; ++mb_idx) {
82  std::partial_sort(
83  global_indices.begin(), global_indices.begin() + m_top_k,
84  global_indices.end(),
85  [mb_idx, col_comm_size, &global_top_k, this]
86  (El::Int a, El::Int b) -> bool {
87  El::Int mb_offset = mb_idx * m_top_k;
88  El::Int a_proc_offset = (a/m_top_k) * m_top_k * col_comm_size;
89  El::Int a_idx = a_proc_offset + mb_offset + (a%m_top_k);
90  El::Int b_proc_offset = (b/m_top_k) * m_top_k * col_comm_size;
91  El::Int b_idx = b_proc_offset + mb_offset + (b%m_top_k);
92  return global_top_k[a_idx].val > global_top_k[b_idx].val;
93  });
94  // Check if there is a 1 ground truth label in the top k.
95  bool found = false;
96  for (El::Int i = 0; i < m_top_k; ++i) {
97  El::Int idx = global_indices[i];
98  idx = mb_idx*m_top_k + (i/m_top_k)*m_top_k*col_comm_size + (i%m_top_k);
99  if (global_top_k[idx].gt == DataType(1)) {
100  found = true;
101  break;
102  }
103  }
104  if (!found) {
105  ++num_errors;
106  }
107  }
108  } else {
109  get_comm().gather((DataType*) local_top_k.data(), 2*local_top_k.size(), 0,
110  col_comm);
111  }
112  num_errors = get_comm().model_allreduce(num_errors);
113  const int mini_batch_size = prediction.Width();
114  return (mini_batch_size - num_errors) * 100;
115 }
