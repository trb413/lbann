34 void lbann_callback_check_init::on_train_begin(model *m) {
35  // Skip after the first epoch.
36  if (m->get_cur_epoch() != 0) {
37  return;
38  }
39  lbann_comm *comm = m->get_comm();
40  if (comm->am_world_master()) {
41  std::cout << "Checking all model initial weights match..." << std::endl;
42  }
43  if (comm->get_num_models() == 1) {
44  return;
45  }
46 
47  std::vector<Layer *>& layers = m->get_layers();
48  // Skip the input/output layers.
49  for (size_t l = 1; l < layers.size() - 1; ++l) {
50  if (comm->am_world_master()) {
51  std::cout << "Checking layer " << l << std::endl;
52  }
53  // Skip non-learning layers.
54  learning *learning_layer = (learning *) dynamic_cast<learning *> (layers[l]);
55  if(learning_layer == NULL) {
56  continue;
57  }
58  // Model 0 holds the master copy, it gathers the values from other models
59  // and compares them.
60  const ElMat& weights = learning_layer->get_weights();
61  const Mat& local_weights = weights.LockedMatrix();
62  for (int model = 1; model < comm->get_num_models(); ++model) {
63  comm->global_barrier();
64  if (comm->get_model_rank() == 0) {
65  Mat remote_weights(local_weights.Height(), local_weights.Width());
66  comm->recv(remote_weights, model);
67  if (!check_equal(local_weights, remote_weights)) {
68  throw lbann_exception(
69  "check_init: model " + std::to_string(model) + " rank in model " +
70  std::to_string(comm->get_rank_in_model()) +
71  " does not match model 0");
72  }
73  } else if (comm->get_model_rank() == model) {
74  comm->send(local_weights, 0);
75  }
76  }
77  }
78 }

80 bool lbann_callback_check_init::check_equal(const Mat& x, const Mat& y) const {
81  const Int height = x.Height();
82  const Int width = x.Width();
83  if (height != y.Height() || width != y.Width() || x.LDim() != y.LDim()) {
84  return false;
85  }
86  const DataType *x_buf = x.LockedBuffer();
87  const DataType *y_buf = y.LockedBuffer();
88  for (Int i = 0; i < height * width; ++i) {
89  if (x_buf[i] != y_buf[i]) {
90  return false;
91  }
92  }
93  return true;
94 }
