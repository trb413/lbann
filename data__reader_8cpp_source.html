69 int lbann::generic_data_reader::fetch_data(Mat& X) {
70  int nthreads = omp_get_max_threads();
71  if(!position_valid()) {
72  throw lbann_exception(
73  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
74  + " :: generic data reader load error: !position_valid"
75  + " -- current pos = " + std::to_string(m_current_pos)
76  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
77  }
78 
79  if (!m_save_minibatch_indices) {
82  #pragma omp parallel for schedule(static, 1)
83  for (int t = 0; t < nthreads; t++) {
84  preprocess_data_source(omp_get_thread_num());
85  }
86  }
87 
88  int loaded_batch_size = get_loaded_mini_batch_size();
89  const int end_pos = std::min(static_cast<size_t>(m_current_pos+loaded_batch_size),
90  m_shuffled_indices.size());
91  const int mb_size = std::min(
92  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
93  X.Width());
94 
95  if (!m_save_minibatch_indices) {
96  El::Zeros(X, X.Height(), X.Width());
97  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
98  }
99 
100  if (m_save_minibatch_indices) {
101  m_my_minibatch_indices.resize(m_my_minibatch_indices.size() + 1);
102  for (int s = 0; s < mb_size; s++) {
103  int n = m_current_pos + (s * m_sample_stride);
104  m_my_minibatch_indices.back().push_back(n);
105  }
106  }
107 
108  else {
109  #pragma omp parallel for
110  for (int s = 0; s < mb_size; s++) {
111  // Catch exceptions within the OpenMP thread.
112  try {
113  int n = m_current_pos + (s * m_sample_stride);
114  int index = m_shuffled_indices[n];
115  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
116  if (!valid) {
117  throw lbann_exception(
118  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
119  " :: generic data reader load error: datum not valid");
120  }
121  m_indices_fetched_per_mb.Set(s, 0, index);
122  } catch (lbann_exception& e) {
123  lbann_report_exception(e);
124  } catch (std::exception& e) {
125  El::ReportException(e);
126  }
127  }
128 
131  #pragma omp parallel for schedule(static, 1)
132  for (int t = 0; t < nthreads; t++) {
133  postprocess_data_source(omp_get_thread_num());
134  }
135  }
136 
137  return mb_size;
138 }

232 bool generic_data_reader::update(bool is_active_reader) {
233  bool reader_not_done = true; // BVE The sense of this should be fixed
234  m_current_mini_batch_idx++;
235 
236  if(is_active_reader) {
237  m_current_pos = get_next_position();
238 
240  if (!m_save_minibatch_indices) {
241  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Height(), 1);
242  }
243 
244  m_loaded_mini_batch_idx += m_iteration_stride;
245  }
246  if (m_loaded_mini_batch_idx >= m_num_iterations_per_epoch) {
247  reader_not_done = false;
248  }
249  if ((size_t)m_current_pos >= m_shuffled_indices.size()) {
250  reader_not_done = false;
251  }
252  if (m_current_mini_batch_idx == m_num_iterations_per_epoch) {
253  if ((get_rank() < m_num_parallel_readers) && (m_current_pos < (int)m_shuffled_indices.size())) {
254  throw lbann_exception(
255  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
256  + " :: generic data reader update error: the epoch is complete,"
257  + " but not all of the data has been used -- current pos = " + std::to_string(m_current_pos)
258  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices"
259  + " : iteration="
260  + std::to_string(m_current_mini_batch_idx) + "C ["
261  + std::to_string(m_loaded_mini_batch_idx) +"L] of "
262  + std::to_string(m_num_iterations_per_epoch) + "+"
263  + std::to_string(m_iteration_stride) + " : "
264  + " index stride="
265  + std::to_string(m_stride_to_next_mini_batch) + "/"
266  + std::to_string(m_stride_to_last_mini_batch));
267  }
268 
269  if (!m_save_minibatch_indices) {
270  if (m_shuffle) {
271  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
272  get_data_seq_generator());
273  }
274  }
275 
276  set_initial_position();
277 
278  if (!m_save_minibatch_indices) {
279  if (m_data_store) {
280  m_data_store->set_shuffled_indices(&m_shuffled_indices);
281  }
282  }
283  }
284  return reader_not_done;
285 }

334 void generic_data_reader::select_subset_of_data() {
335  shuffle_indices();
336 
337  size_t count = get_absolute_sample_count();
338  double use_percent = get_use_percent();
339  if (count == 0 and use_percent == 0.0) {
340  throw lbann_exception(
341  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
342  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
343  + "and get_absolute_sample_count() are both zero; exactly one "
344  + "must be zero");
345  }
346  if (!(count == 0 or use_percent == 0.0)) {
347  throw lbann_exception(
348  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
349  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
350  "and get_absolute_sample_count() are both non-zero; exactly one "
351  "must be zero");
352  }
353 
354 
355  if (count != 0) {
356  if(count > static_cast<size_t>(get_num_data())) {
357  throw lbann_exception(
358  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
359  " :: generic_data_reader::select_subset_of_data() - absolute_sample_count=" +
360  std::to_string(count) + " is > get_num_data=" +
361  std::to_string(get_num_data()));
362  }
363  m_shuffled_indices.resize(get_absolute_sample_count());
364  }
365 
366  if (use_percent) {
367  m_shuffled_indices.resize(get_use_percent()*get_num_data());
368  }
369 
370  long unused = get_validation_percent()*get_num_data(); //get_num_data() = m_shuffled_indices.size()
371  long use_me = get_num_data() - unused;
372  if (unused > 0) {
373  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
374  m_shuffled_indices.resize(use_me);
375  }
376 
377  if(!m_shuffle) {
378  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
379  std::sort(m_unused_indices.begin(), m_unused_indices.end());
380  }
381 }

399 bool lbann::generic_data_reader::load_from_checkpoint_shared(persist& p, const char *name) {
400  // rank 0 reads the training state file
401  struct packing_header header;
402  if (p.get_rank() == 0) {
403  unpack_scalars(p,&header,name);
404  }
405  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
406  unpack_header(header);
407  if(p.get_rank() ==0){
408  m_shuffled_indices.resize(header.data_size);
409  m_unused_indices.resize(header.unused_data_size);
410  }
411  MPI_Bcast(&m_shuffled_indices[0], header.data_size, MPI_INT, 0, MPI_COMM_WORLD);
412  MPI_Bcast(&m_unused_indices[0], header.unused_data_size, MPI_INT, 0, MPI_COMM_WORLD);
413  return true;
414 }
