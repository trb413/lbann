65 int lbann::generic_data_reader::fetch_data(Mat& X) {
66  int nthreads = omp_get_max_threads();
67  if(!position_valid()) {
68  throw lbann_exception(
69  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
70  + " :: generic data reader load error: !position_valid"
71  + " -- current pos = " + std::to_string(m_current_pos)
72  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
73  }
74 
75  if (!m_save_minibatch_indices) {
78  #pragma omp parallel for schedule(static, 1)
79  for (int t = 0; t < nthreads; t++) {
80  preprocess_data_source(omp_get_thread_num());
81  }
82  }
83 
84  int loaded_batch_size = get_loaded_mini_batch_size();
85  const int end_pos = std::min(static_cast<size_t>(m_current_pos+loaded_batch_size),
86  m_shuffled_indices.size());
87  const int mb_size = std::min(
88  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
89  X.Width());
90 
91  if (!m_save_minibatch_indices) {
92  El::Zeros(X, X.Height(), X.Width());
93  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
94  }
95 
96  if (m_save_minibatch_indices) {
97  m_my_minibatch_indices.resize(m_my_minibatch_indices.size() + 1);
98  for (int s = 0; s < mb_size; s++) {
99  int n = m_current_pos + (s * m_sample_stride);
100  m_my_minibatch_indices.back().push_back(n);
101  }
102  }
103 
104  else {
105 
106  #pragma omp parallel for
107  for (int s = 0; s < mb_size; s++) {
108  // Catch exceptions within the OpenMP thread.
109  try {
110  int n = m_current_pos + (s * m_sample_stride);
111  int index = m_shuffled_indices[n];
112  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
113  if (!valid) {
114  throw lbann_exception(
115  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
116  " :: generic data reader load error: datum not valid");
117  }
118  m_indices_fetched_per_mb.Set(s, 0, index);
119  } catch (lbann_exception& e) {
120  lbann_report_exception(e);
121  } catch (std::exception& e) {
122  El::ReportException(e);
123  }
124  }
125 
128  #pragma omp parallel for schedule(static, 1)
129  for (int t = 0; t < nthreads; t++) {
130  postprocess_data_source(omp_get_thread_num());
131  }
132  }
133 
134  return mb_size;
135 }

229 bool generic_data_reader::update(bool is_active_reader) {
230  bool reader_not_done = true; // BVE The sense of this should be fixed
231  m_current_mini_batch_idx++;
232 
233  if(is_active_reader) {
234  m_current_pos = get_next_position();
235 
237  if (!m_save_minibatch_indices) {
238  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Height(), 1);
239  }
240 
241  m_loaded_mini_batch_idx += m_iteration_stride;
242  }
243  if (m_loaded_mini_batch_idx >= m_num_iterations_per_epoch) {
244  reader_not_done = false;
245  }
246  if ((size_t)m_current_pos >= m_shuffled_indices.size()) {
247  reader_not_done = false;
248  }
249  if (m_current_mini_batch_idx == m_num_iterations_per_epoch) {
250  if ((get_rank() < m_num_parallel_readers) && (m_current_pos < (int)m_shuffled_indices.size())) {
251  throw lbann_exception(
252  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
253  + " :: generic data reader update error: the epoch is complete,"
254  + " but not all of the data has been used -- current pos = " + std::to_string(m_current_pos)
255  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices"
256  + " : iteration="
257  + std::to_string(m_current_mini_batch_idx) + "C ["
258  + std::to_string(m_loaded_mini_batch_idx) +"L] of "
259  + std::to_string(m_num_iterations_per_epoch) + "+"
260  + std::to_string(m_iteration_stride) + " : "
261  + " index stride="
262  + std::to_string(m_stride_to_next_mini_batch) + "/"
263  + std::to_string(m_stride_to_last_mini_batch));
264  }
265 
266  if (!m_save_minibatch_indices) {
267  if (m_shuffle) {
268  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
269  get_data_seq_generator());
270  }
271  }
272 
273  set_initial_position();
274 
275  if (!m_save_minibatch_indices) {
276  if (m_data_store) {
277  m_data_store->set_shuffled_indices(&m_shuffled_indices);
278  }
279  }
280  }
281  return reader_not_done;
282 }

319 void generic_data_reader::select_subset_of_data() {
320  shuffle_indices();
321 
322  size_t count = get_absolute_sample_count();
323  double use_percent = get_use_percent();
324  if (count == 0 and use_percent == 0.0) {
325  throw lbann_exception(
326  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
327  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
328  + "and get_absolute_sample_count() are both zero; exactly one "
329  + "must be zero");
330  }
331  if (!(count == 0 or use_percent == 0.0)) {
332  throw lbann_exception(
333  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
334  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
335  "and get_absolute_sample_count() are both non-zero; exactly one "
336  "must be zero");
337  }
338 
339 
340  if (count != 0) {
341  if(count > static_cast<size_t>(get_num_data())) {
342  throw lbann_exception(
343  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
344  " :: generic_data_reader::select_subset_of_data() - absolute_sample_count=" +
345  std::to_string(count) + " is > get_num_data=" +
346  std::to_string(get_num_data()));
347  }
348  m_shuffled_indices.resize(get_absolute_sample_count());
349  }
350 
351  if (use_percent) {
352  m_shuffled_indices.resize(get_use_percent()*get_num_data());
353  }
354 
355  long unused = get_validation_percent()*get_num_data(); //get_num_data() = m_shuffled_indices.size()
356  long use_me = get_num_data() - unused;
357  if (unused > 0) {
358  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
359  m_shuffled_indices.resize(use_me);
360  }
361 
362  if(!m_shuffle) {
363  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
364  std::sort(m_unused_indices.begin(), m_unused_indices.end());
365  }
366 }
