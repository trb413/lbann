58 int lbann::generic_data_reader::fetch_data(Mat& X) {
59  int nthreads = omp_get_max_threads();
60  if(!position_valid()) {
61  throw lbann_exception(
62  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
63  + " :: generic data reader load error: !position_valid"
64  + " -- current pos = " + std::to_string(m_current_pos)
65  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
66  }
67 
70  #pragma omp parallel for schedule(static, 1)
71  for (int t = 0; t < nthreads; t++) {
72  preprocess_data_source(omp_get_thread_num());
73  }
74 
75  int current_batch_size = get_current_mini_batch_size();
76  const int end_pos = std::min(static_cast<size_t>(m_current_pos+current_batch_size),
77  m_shuffled_indices.size());
78  const int mb_size = std::min(
79  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
80  X.Width());
81 
82  El::Zeros(X, X.Height(), X.Width());
83  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
84  #pragma omp parallel for
85  for (int s = 0; s < mb_size; s++) {
86  // Catch exceptions within the OpenMP thread.
87  try {
88  int n = m_current_pos + (s * m_sample_stride);
89  int index = m_shuffled_indices[n];
90  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
91  if (!valid) {
92  throw lbann_exception(
93  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
94  " :: generic data reader load error: datum not valid");
95  }
96  m_indices_fetched_per_mb.Set(s, 0, index);
97  } catch (lbann_exception& e) {
98  lbann_report_exception(e);
99  } catch (exception& e) {
100  El::ReportException(e);
101  }
102  }
103 
106  #pragma omp parallel for schedule(static, 1)
107  for (int t = 0; t < nthreads; t++) {
108  postprocess_data_source(omp_get_thread_num());
109  }
110 
111  return mb_size;
112 }

188 bool generic_data_reader::update() {
189  m_current_pos = get_next_position();
190 
192  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Width(), 1);
193 
194  m_current_mini_batch_idx += m_iteration_stride;
195  if (m_current_mini_batch_idx >= m_num_iterations_per_epoch) {
196  if (m_current_pos < (int)m_shuffled_indices.size()) {
197  throw lbann_exception(
198  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
199  + " :: generic data reader update error: the epoch is complete,"
200  + " but not all of the data has been used -- current pos = " + std::to_string(m_current_pos)
201  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
202  }
203  if (not m_first_n) {
204  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
205  get_data_seq_generator());
206  }
207  set_initial_position();
208  return false;
209  }else {
210  return true;
211  }

239 void generic_data_reader::select_subset_of_data() {
240  if(!get_firstN()) {
241  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(), get_data_seq_generator());
242  }
243 
244  if (not (has_max_sample_count() or has_use_percent() or has_validation_percent())) {
245  return;
246  }
247 
248  if (has_max_sample_count()) {
249  size_t count = get_max_sample_count();
250  if(count > static_cast<size_t>(get_num_data())) {
251  throw lbann_exception(
252  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
253  " :: generic_data_reader::select_subset_of_data() - max_sample_count=" +
254  std::to_string(count) + " is > get_num_data=" +
255  std::to_string(get_num_data()));
256  }
257  m_shuffled_indices.resize(get_max_sample_count());
258  } else if (has_use_percent()) {
259  m_shuffled_indices.resize(get_use_percent()*get_num_data());
260  }
261 
262  if (has_validation_percent()) {
263  long unused = get_validation_percent()*get_num_data(); //get_num_data() = m_shuffled_indices.size()
264  long use_me = get_num_data() - unused;
265  if (unused > 0) {
266  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
267  m_shuffled_indices.resize(use_me);
268  }
269  }
270 
271  if(get_firstN()) {
272  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
273  std::sort(m_unused_indices.begin(), m_unused_indices.end());
274  }
275 }

284 bool generic_data_reader::saveToCheckpointShared(persist& p, const char *name) {
285  // rank 0 writes the training state file
286  if (p.get_rank() == 0) {
287  char fieldname[1024];
288 
289  // record minibatch index
290  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
291  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_mini_batch_idx);
292 
293  // get size of list of training examples
294  int size = m_shuffled_indices.size();
295 
296  // record size of ShuffleIndices
297  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
298  p.write_uint64(persist_type::train, fieldname, (uint64_t) size);
299 
300  // TODO: each model may have a different position, need to gather and write these
301  // record current position within training data
302  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
303  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_pos);
304 
305  // write list of indices
306  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
307  p.write_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
308  }
309 
310  return true;

314 bool lbann::generic_data_reader::loadFromCheckpointShared(persist& p, const char *name) {
315  // rank 0 reads the training state file
316  if (p.get_rank() == 0) {
317  char fieldname[1024];
318 
319  // record minibatch index
320  uint64_t val;
321  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
322  p.read_uint64(persist_type::train, fieldname, &val);
323  m_current_mini_batch_idx = (int) val;
324 
325  // get size of ShuffleIndices
326  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
327  p.read_uint64(persist_type::train, fieldname, &val);
328  int size = (int) val;
329 
330  // get current position within data
331  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
332  p.read_uint64(persist_type::train, fieldname, &val);
333  m_current_pos = (int) val;
334 
335  // resize shuffled index array to hold values
336  m_shuffled_indices.resize(size);
337 
338  // read list of indices
339  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
340  p.read_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
341  }
342 
343  // broadcast minibatch index
344  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
345 
346  // TODO: with multiple readers, make this a scatter
347  // broadcast current position
348  MPI_Bcast(&m_current_pos, 1, MPI_INT, 0, MPI_COMM_WORLD);
349 
350  // broadcast values from rank 0
351  int size = m_shuffled_indices.size();
352  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
353 
354  // resize shuffled index array to hold values
355  if (p.get_rank() != 0) {
356  m_shuffled_indices.resize(size);
357  }
358 
359  // broadcast index array
360  MPI_Bcast(&m_shuffled_indices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
361 
362  return true;
363 }
