58 EvalType l1_weight_regularization::finish_evaluation() {
59  if (m_scale_factor == EvalType(0)) { return EvalType(0); }
60  EvalType value = EvalType(0);
61  for (weights* w : m_weights) {
62 
63  // Get matrices
64  const AbsDistMat& values = w->get_values();
65  const Mat& values_local = values.LockedMatrix();
66  const int local_height = values_local.Height();
67  const int local_width = values_local.Width();
68 
69  // Compute L1 regularization term
70  EvalType sum = 0;
71  #pragma omp parallel for reduction(+:sum) collapse(2)
72  for (int col = 0; col < local_width; ++col) {
73  for (int row = 0; row < local_height; ++row) {
74  const EvalType val = values_local(row, col);
75  sum += val >= EvalType(0) ? val : - val;
76  }
77  }
78  value += get_comm().allreduce(sum, values.DistComm());
79 
80  }
81  return m_scale_factor * value;
82 }

84 void l1_weight_regularization::compute_weight_regularization() {
85  if (m_scale_factor == EvalType(0)) { return; }
86  AbsDistMat* gradient;
87  for (weights* w : m_weights) {
88 
89  // Get matrices
90  const AbsDistMat& values = w->get_values();
91  const Mat& values_local = values.LockedMatrix();
92  const int local_height = values_local.Height();
93  const int local_width = values_local.Width();
94 
95  // Compute gradient
96  gradient = values.Copy();
97  Mat& gradient_local = gradient->Matrix();
98  #pragma omp parallel for collapse(2)
99  for (int col = 0; col < local_width; ++col) {
100  for (int row = 0; row < local_height; ++row) {
101  const DataType val = values_local(row, col);
102  DataType& grad = gradient_local(row, col);
103  if (val > DataType(0)) {
104  grad = DataType(1);
105  } else if (val < DataType(0)) {
106  grad = DataType(-1);
107  } else {
108  grad = DataType(0);
109  }
110  }
111  }
112  w->get_optimizer()->add_to_gradient(*gradient, m_scale_factor);
113  delete gradient;
114 
115  }
