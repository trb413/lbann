40 void lbann_callback_variable_minibatch::on_train_begin(model *m) {
41  // Avoid issues with the train method being called multiple times.
42  if (m->get_cur_epoch() != 0) { return; }
43 
44  // Get first input layer in model
45  generic_input_layer* input = nullptr;
46  for (auto&& l : m->get_layers()) {
47  input = dynamic_cast<generic_input_layer*>(l);
48  if (input != nullptr) { break; }
49  }
50  if (input == nullptr) { LBANN_ERROR("could not get input layer"); }
51 
52  if (m_starting_mbsize > m->get_max_mini_batch_size()) {
53  throw lbann_exception(
54  "variable_minibatch: starting mini-batch size is larger than max");
55  }
56  if (m_starting_mbsize == m->get_max_mini_batch_size()) {
57  if (m->get_comm()->am_world_master()) {
58  std::cout << "WARNING: starting mini-batch size equals max mini-batch "
59  << "size and using variable-sized mini-batches" << std::endl;
60  }
61  }
62  input->calculate_num_iterations_per_epoch_training_spans_models(
63  m_starting_mbsize);
64 }

66 void lbann_callback_variable_minibatch::on_epoch_end(model *m) {
67 
68  // Get first input layer in model
69  generic_input_layer* input = nullptr;
70  for (auto&& l : m->get_layers()) {
71  input = dynamic_cast<generic_input_layer*>(l);
72  if (input != nullptr) { break; }
73  }
74  if (input == nullptr) { LBANN_ERROR("could not get input layer"); }
75 
76  lbann_comm *comm = m->get_comm();
77  int new_mbsize = 0;
78  float new_lr = 0.0f;
79  int ramp_time = 0;
80  if (schedule(m, new_mbsize, new_lr, ramp_time)) {
81  if (new_mbsize > m->get_max_mini_batch_size()) {
82  if (comm->am_model_master()) {
83  std::cout << "Model " << comm->get_model_rank() << ": WARNING " <<
84  "requested new mini-batch size " << new_mbsize <<
85  " is greater than the model maximum mini-batch size " <<
86  m->get_max_mini_batch_size() << std::endl;
87  }
88  new_mbsize = m->get_max_mini_batch_size();
89  }
90  input->calculate_num_iterations_per_epoch_training_spans_models(new_mbsize);
91  m_current_mini_batch_size = new_mbsize;
92  m_ramp_count = ramp_time;
93  if (new_lr != 0.0f) {
94  if (ramp_time == 0) {
95  // Change learning rate immediately.
96  change_learning_rate(m, new_lr);
97  } else {
98  // Compute the per-epoch learning rate increment.
99  float cur_lr = get_current_learning_rate(m);
100  m_lr_incr = (new_lr - cur_lr) / ramp_time;
101  }
102  if (comm->am_model_master()) {
103  std::cout << "Model " << comm->get_model_rank() <<
104  ": Changing mini-batch size to " << new_mbsize <<
105  " and learning rate to " << new_lr << " at epoch " <<
106  m->get_cur_epoch() << std::endl;
107  }
108  } else if (comm->am_model_master()) {
109  std::cout << "Model " << comm->get_model_rank() <<
110  ": Changing mini-batch size to " << new_mbsize <<
111  " at epoch " << m->get_cur_epoch() << std::endl;
112  }
113  }
114  // Ramp the learning rate, if needed.
115  if (m_ramp_count > 0) {
116  --m_ramp_count;
117  float target_lr = get_current_learning_rate(m) + m_lr_incr;
118  change_learning_rate(m, target_lr);
119  if (comm->am_model_master()) {
120  std::cout << "Model " << comm->get_model_rank() <<
121  ": Variable-size mini-batch ramping learning rate to " <<
122  target_lr << std::endl;
123  }
124  }
125 }

153 bool lbann_callback_step_minibatch::schedule(
154  model *m, int& new_mbsize, float& new_lr, int& ramp_time) {
155  if (m->get_cur_epoch() % m_step == 0) {
156  new_mbsize = m_current_mini_batch_size * 2;
157  new_lr = get_current_learning_rate(m) * 2;
158  ramp_time = m_ramp_time;
159  return true;
160  } else {
161  return false;
162  }
163 }

165 lbann_callback_minibatch_schedule::lbann_callback_minibatch_schedule(
166  int starting_mbsize, std::vector<minibatch_step> steps) :
167  lbann_callback_variable_minibatch(starting_mbsize), m_steps(std::move(steps)) {
168  std::sort(m_steps.rbegin(), m_steps.rend(),

174 bool lbann_callback_minibatch_schedule::schedule(
175  model *m, int& new_mbsize, float& new_lr, int& ramp_time) {
176  if (!m_steps.empty() && m->get_cur_epoch() == m_steps.back().epoch) {
177  new_mbsize = m_steps.back().mbsize;
178  new_lr = m_steps.back().lr;
179  ramp_time = m_steps.back().ramp_time;
180  m_steps.pop_back();
181  return true;
182  }
183  return false;
184 }
