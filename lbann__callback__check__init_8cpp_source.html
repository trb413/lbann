34 void lbann_callback_check_init::on_train_begin(model* m) {
35  // Skip after the first epoch.
36  if (m->get_cur_epoch() != 0) return;
37  lbann_comm* comm = m->get_comm();
38  if (comm->am_world_master()) {
39  std::cout << "Checking all model initial weights match..." << std::endl;
40  }
41  if (comm->get_num_models() == 1) return;
42 
43  std::vector<Layer*>& layers = m->get_layers();
44  // Skip the input/output layers.
45  for (size_t l = 1; l < layers.size() - 1; ++l) {
46  if (comm->am_world_master()) {
47  std::cout << "Checking layer " << l << std::endl;
48  }
49  // Model 0 holds the master copy, it gathers the values from other models
50  // and compares them.
51  const ElMat& weights = layers[l]->get_weights_biases();
52  const Mat& local_weights = weights.LockedMatrix();
53  for (int model = 1; model < comm->get_num_models(); ++model) {
54  comm->global_barrier();
55  if (comm->get_model_rank() == 0) {
56  Mat remote_weights(local_weights.Height(), local_weights.Width());
57  comm->recv(remote_weights, model);
58  if (!check_equal(local_weights, remote_weights)) {
59  throw lbann_exception(
60  "check_init: model " + std::to_string(model) + " rank in model " +
61  std::to_string(comm->get_rank_in_model()) +
62  " does not match model 0");
63  }
64  } else if (comm->get_model_rank() == model) {
65  comm->send(local_weights, 0);
66  }
67  }
68  }
69 }

71 bool lbann_callback_check_init::check_equal(const Mat& x, const Mat& y) const {
72  const Int height = x.Height();
73  const Int width = x.Width();
74  if (height != y.Height() || width != y.Width() || x.LDim() != y.LDim()) {
75  return false;
76  }
77  const DataType* x_buf = x.LockedBuffer();
78  const DataType* y_buf = y.LockedBuffer();
79  for (Int i = 0; i < height * width; ++i) {
80  if (x_buf[i] != y_buf[i]) return false;
81  }
82  return true;
83 }
