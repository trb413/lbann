62 void lbann_callback_io::on_epoch_end(Model* m) {
63  lbann_comm* comm = m->get_comm();
64  std::vector<Layer*>& layers = m->get_layers();
65  for (size_t l = 0; l < layers.size(); ++l) {
66  Layer* layer = layers[l];
67  uint idx = layer->get_index();
68  if (layer_indices.size() == 0 ||
69  layer_indices.find(idx) != layer_indices.end()) {
70 
71  input_layer *input = (input_layer*) dynamic_cast<input_layer*> (layer);
72  if(input != NULL) {
73  cout << "Rank " << comm->get_rank_in_model() << " processed "
74  << input->get_num_samples_trained() << " training samples of "
75  << input->get_total_num_training_samples() << " ("
76  << input->get_num_samples_trained() / m->get_cur_epoch() << " per epoch)" << endl;
77  }
78  target_layer *target = (target_layer*) dynamic_cast<target_layer*> (layer);
79  if(target != NULL) {
80  cout << "Rank " << comm->get_rank_in_model() << " processed "
81  << target->get_num_samples_trained() << " training labels of "
82  << target->get_total_num_training_samples() << " ("
83  << target->get_num_samples_trained() / m->get_cur_epoch() << " per epoch)" << endl;
84  }
85  }
