58 void lbann::SoftmaxLayer::setup(int numPrevNeurons) {
59  Layer::setup(numPrevNeurons);
60  if(optimizer != NULL) {
61  optimizer->setup(numPrevNeurons, NumNeurons);
62  }
63 
64  // Initialize weight-bias matrix
65  Zeros(*m_weights, NumNeurons, numPrevNeurons);
66 
67  // Initialize weights
68  DistMat weights;
69  View(weights, *m_weights, IR(0,NumNeurons), IR(0,numPrevNeurons));
70  switch(m_weight_initialization) {
71  case weight_initialization::uniform:
72  uniform_fill(weights, weights.Height(), weights.Width(),
73  DataType(0), DataType(1));
74  break;
75  case weight_initialization::normal:
76  gaussian_fill(weights, weights.Height(), weights.Width(),
77  DataType(0), DataType(1));
78  break;
79  case weight_initialization::glorot_normal: {
80  const DataType var = 2.0 / (numPrevNeurons + NumNeurons);
81  gaussian_fill(weights, weights.Height(), weights.Width(),
82  DataType(0), sqrt(var));
83  break;
84  }
85  case weight_initialization::glorot_uniform: {
86  const DataType var = 2.0 / (numPrevNeurons + NumNeurons);
87  uniform_fill(weights, weights.Height(), weights.Width(),
88  DataType(0), sqrt(3*var));
89  break;
90  }
91  case weight_initialization::he_normal: {
92  const DataType var = 1.0 / numPrevNeurons;
93  gaussian_fill(weights, weights.Height(), weights.Width(),
94  DataType(0), sqrt(var));
95  break;
96  }
97  case weight_initialization::he_uniform: {
98  const DataType var = 1.0 / numPrevNeurons;
99  uniform_fill(weights, weights.Height(), weights.Width(),
100  DataType(0), sqrt(3*var));
101  break;
102  }
103  case weight_initialization::zero: // Zero initialization is default
104  default:
105  Zero(weights);
106  break;
107  }
108 
109  // Initialize other matrices
110  Zeros(*m_weights_gradient, NumNeurons, numPrevNeurons);
111  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size);
112  Zeros(*m_error_signal, numPrevNeurons, m_mini_batch_size); // m_error_signal holds the product of m_weights^T * m_prev_error_signal
113  Zeros(*m_weighted_sum, NumNeurons, m_mini_batch_size);
114  Zeros(*m_activations, NumNeurons, m_mini_batch_size);
115  Zeros(*m_prev_activations, numPrevNeurons, m_mini_batch_size);
116 }

122 void lbann::SoftmaxLayer::fp_linearity()
123 {
124  // _Z = m_weights * Xs -- Xs is previous layer Activations
125  // ZsColMax[c,0] = max(_Z[0..numNeurons-1, c]) -- (m_mini_batch_size x 1)
126  // ZsNorm[r,c] = _Z[r,c] - ZsColMax[c,0] -- Column-wise normalized Zs matrix: _Z[r,c] - ZsColMax[1,c]
127  // ZsNormExp[r,c] = exp(ZsNorm[r,c])
128  // ZsNormExpSum[c,0] = sum(ZsNormExp[0..numNeurons-1, c]) -- Column-wise sum over normalized, exponentiated _Z
129  // _Y[r,c] = ZsNormExp[r,c] / ZsNormExpSum[c,0] -- exp(norm(_Z[r,c])) = Sum(exp(norm(Zs[r,c])))
130 
131  // Apply linear transform
132  Gemm(NORMAL, NORMAL, (DataType) 1.0, *m_weights, *m_prev_activations_v, (DataType) 0.0, *m_weighted_sum_v);
133 
134  // For each minibatch (column) find the maximimum value
135  Zeros(ZsColMax, m_mini_batch_size, 1); // Clear the entire matrix
136  ColumnMax((DistMat&) *m_weighted_sum_v, ZsColMax);
137 
138  // Redistribute the per-minibatch maximum values
139  Copy(ZsColMax, ZsColMaxStar);
140 
142 
143  // Compute exp(z) of each entry. Subtract the max of each column from its
144  // entries to prevent the exp from blowing up. Large negative values are
145  // expected to underflow to 0.
146  IndexDependentMap(
147  *m_weighted_sum_v,
148  (std::function<DataType(El::Int,El::Int,const DataType&)>)
149  ([this](El::Int r, El::Int c, const DataType& z)->DataType {
150  El::Int rL = this->ZsColMaxStar.LocalRow(c);
151  // if(isnan(std::exp(z - this->ZsColMaxStar.GetLocal(rL, 0)))) { cout << "[" << comm->get_rank_in_world() << "] has a nan "<<std::exp(z - this->ZsColMaxStar.GetLocal(rL, 0)) << " at [" << r << ", " << c << "]="<<z<<endl;throw(new lbann_exception("Foo"));}
152  return std::exp(z - this->ZsColMaxStar.GetLocal(rL, 0));
153  }));
154 
155  // For each minibatch (column) sum up the exponentiated values
156  Zeros(ZsNormExpSum, m_mini_batch_size, 1); // Clear the entire matrix
157  // ColSumMat ZsNormExpSum;
158  ColumnSum((DistMat&) *m_weighted_sum_v, ZsNormExpSum);
159  Copy(ZsNormExpSum, ZsNormExpSumStar);
160 
161  // Divide each entry: exp(x_ij) / Sum_i(exp(x_ij))
162  Copy(*m_weighted_sum_v, *m_activations_v);
163 
164  IndexDependentMap(*m_activations_v,
165  (std::function<DataType(Int,Int,const DataType&)>)([this /*ZsNormExpSum*/](Int r, Int c, const DataType& z)->
166  DataType{Int rL = this->ZsNormExpSumStar.LocalRow(c); return z/this->ZsNormExpSumStar.GetLocal(rL,0);}));
167 
168 #if 0

191 void lbann::SoftmaxLayer::bp_linearity()
192 {
193 
195 
196  // Compute error signal from nonlinearity (categorical cross entropy case)
197  // Note: error signal is already computed in objective function object
198  if(neural_network_model->obj_fn->type == objective_functions::obj_fn_type::categorical_cross_entropy
199  && (m_next_layer_type == layer_type::target_distributed_minibatch
200  || m_next_layer_type == layer_type::target_distributed_minibatch_parallel_io
201  // || m_next_layer_type == layer_type::target_unsupervised
202  )) {}
203 
204  // Compute error signal from nonlinearity (default case)
205  // Note: error_signal = (prev_error_signal - prev_error_signal^T activations) * activations
206  else {
207  StarMat prev_error_signal_dot_activations(get_effective_minibatch_size(), 1);
208  DistMat curr_prev_error_signal, curr_activations;
209  DataType curr_dot_product;
210  for(Int c = 0; c < get_effective_minibatch_size(); c++) {
211  LockedView(curr_prev_error_signal, *m_prev_error_signal, ALL, IR(c));
212  LockedView(curr_activations, *m_activations, ALL, IR(c));
213  curr_dot_product = Dot(curr_prev_error_signal, curr_activations);
214  prev_error_signal_dot_activations.SetLocal(c, 0, curr_dot_product);
215  }
216  IndexDependentMap(*m_prev_error_signal_v,
217  (std::function<DataType(Int,Int,const DataType&)>)
218  ([&prev_error_signal_dot_activations](Int r, Int c, const DataType& z)->DataType {
219  return z - prev_error_signal_dot_activations.GetLocal(c,0);
220  }));
221  Hadamard(*m_prev_error_signal_v, *m_activations_v, *m_prev_error_signal_v);
222  }
223 
224  // Compute the partial delta update for the next lower layer (delta * activation_prev^T)
225  Gemm(TRANSPOSE, NORMAL, (DataType) 1., *m_weights, *m_prev_error_signal_v, (DataType) 0., *m_error_signal_v);
226 
227  // Compute update for weights - include division by mini-batch size
228  Gemm(NORMAL, TRANSPOSE, (DataType) 1.0/get_effective_minibatch_size(), *m_prev_error_signal_v,
229  *m_prev_activations_v, (DataType) 0., *m_weights_gradient);
230 }
