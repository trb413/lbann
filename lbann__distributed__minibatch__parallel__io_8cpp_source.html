115 int lbann::distributed_minibatch_parallel_io::fetch_to_local_matrix(Mat& M_local) {
116  int num_parallel_readers = get_num_parallel_readers();
117 
120  if (m_root == 0) {
121  if (comm->get_rank_in_model() < num_parallel_readers && !m_local_reader_done) {
122  Zero(M_local);
123 
126  m_num_samples_in_batch = fetch_from_data_reader(M_local);
127  bool data_valid = (m_num_samples_in_batch > 0);
128  if(data_valid) {
129  m_num_data_per_epoch+=m_num_samples_in_batch;
130  preprocess_data_samples(M_local, m_num_samples_in_batch);
131  }
132  m_local_data_valid = data_valid;
133  }
134  m_num_valid_readers = comm->model_allreduce((int) m_local_data_valid, mpi::SUM);
135  }
136  return m_num_samples_in_batch;

139 void lbann::distributed_minibatch_parallel_io::distribute_from_local_matrix(Mat& M_local, CircMat& Ms) {
140  int num_parallel_readers = m_num_valid_readers;
141  Ms.SetRoot(m_root);
142 
143  comm->model_barrier();
144 
145  if (comm->get_rank_in_model() == m_root) {
146  if(!m_local_data_valid) {
147  throw lbann_exception("lbann_distributed_minibatch_parallel_io: No valid data for this step -- local data was invalid");
148  }
149  CopyFromRoot(M_local, Ms);
150  m_local_data_valid = false;
151  m_num_samples_in_batch = 0;
152  }else {
153  CopyFromNonRoot(Ms);
154  }
155 
156  comm->model_barrier();
157 
158  m_root = (m_root + 1) % num_parallel_readers;
159  return;

162 bool lbann::distributed_minibatch_parallel_io::is_data_set_processed() {
163  int num_readers_done = 0;
164  int max_active_parallel_readers = get_num_parallel_readers(); // When calculating if all parallel readers are done, include the maximum number,
165  // not just the ones in the last round. This will ensure that all readers, that had data
166  // will have distributed it.
167  int num_parallel_readers = m_num_valid_readers;
168 
169  if(comm->get_rank_in_model() < num_parallel_readers) {
170  if((comm->get_rank_in_model()+1)%num_parallel_readers == m_root) {
171  if(m_local_data_valid) {
172  throw lbann_exception("lbann_input_layer_distributed_minibatch_parallel_io: all valid data was not processed.");
173  }
174  m_local_reader_done = !update_data_reader();
175  }
176  }
177 
179  if(m_local_reader_done) {
180  num_readers_done = 1;
181  }
182 
184  num_readers_done = comm->model_allreduce(num_readers_done);
185  if(num_readers_done == max_active_parallel_readers) {
186  m_local_reader_done = false;
187  m_root = 0;
188  m_num_data_per_epoch = 0;
191  return true;
192  }else {
193  return false;
194  }
