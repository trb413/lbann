93 bool cv_cropper::determine_transform(const cv::Mat& image) {
94  m_enabled = false; //sufficient for now in place of reset();
95 
96  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
97 
98  double zoom_h = 1.0;
99  double zoom_v = 1.0;
100  if (m_is_roi_set) {
101  zoom_h = image.cols / static_cast<double>(m_roi_size.first);
102  zoom_v = image.rows / static_cast<double>(m_roi_size.second);
103  }
104 
105  m_zoom = std::min(zoom_h, zoom_v);
106 
107  if (m_zoom > 1.0) { // rescales the image by the factor of 1/m_zoom (shrink)
108  m_interpolation = m_interpolation_choices[static_cast<int>(m_adaptive_interpolation)];
109  } else {
110  m_interpolation = m_interpolation_choices[static_cast<int>(m_adaptive_interpolation) << 1];
111  }
112 
113  return (m_enabled = true);
114 }

140 bool cv_cropper::apply(cv::Mat& image) {
141  m_enabled = false; // turn off as it is applied
142 
143  //_LBANN_SILENT_EXCEPTION(image.empty(), "", false); // redundant
144 
145  const double zoomed_roi_width = m_roi_size.first * m_zoom;
146  const double zoomed_roi_height = m_roi_size.second * m_zoom;
147  const double zoomed_width = m_width * m_zoom;
148  const double zoomed_height = m_height * m_zoom;
149 
150  int crop_x_start = 0;
151  int crop_y_start = 0;
152 
153  // Get random crop of image
154  if(m_rand_crop) {
155  const int rnd_dw = fast_rand_int(get_fast_generator(), static_cast<int>(2*(zoomed_roi_width - zoomed_width)) + 1);
156  const int rnd_dh = fast_rand_int(get_fast_generator(), static_cast<int>(2*(zoomed_roi_height - zoomed_height)) + 1);
157  crop_x_start = static_cast<int>(image.cols - zoomed_roi_width + rnd_dw + 1) / 2;
158  crop_y_start = static_cast<int>(image.rows - zoomed_roi_height + rnd_dh + 1) / 2;
159  } else {
160  crop_x_start = static_cast<int>(image.cols - zoomed_width + 1) / 2;
161  crop_y_start = static_cast<int>(image.rows - zoomed_height + 1) / 2;
162  }
163 
164  cv::Mat zoomed_crop = image(cv::Rect(crop_x_start, crop_y_start, zoomed_width, zoomed_height));
165  cv::Mat crop;
166  cv::resize(zoomed_crop, crop, cv::Size(m_width,m_height), 0, 0, m_interpolation);
167  image = crop;
168 
169  return true;
170 }
