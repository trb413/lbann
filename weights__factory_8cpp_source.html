35 weights_initializer* construct_initializer(lbann_comm* comm,
36  const lbann_data::Weights& proto_weights) {
37 
38  // Random initialization
39  if (proto_weights.has_uniform_initializer()) {
40  const auto& params = proto_weights.uniform_initializer();
41  const auto& min = params.min();
42  const auto& max = params.max();
43  if (min != 0.0 || max != 0.0) {
44  return new uniform_initializer(comm, min, max);
45  } else {
46  return new uniform_initializer(comm);
47  }
48  }
49  if (proto_weights.has_normal_initializer()) {
50  const auto& params = proto_weights.normal_initializer();
51  const auto& mean = params.mean();
52  const auto& standard_deviation = params.standard_deviation();
53  if (mean != 0.0 || standard_deviation != 0.0) {
54  return new normal_initializer(comm, mean, standard_deviation);
55  } else {
56  return new normal_initializer(comm);
57  }
58  }
59 
60  // Glorot and He initialization
61  if (proto_weights.has_glorot_normal_initializer()) {
62  return new glorot_normal_initializer(comm);
63  }
64  if (proto_weights.has_glorot_uniform_initializer()) {
65  return new glorot_uniform_initializer(comm);
66  }
67  if (proto_weights.has_he_normal_initializer()) {
68  return new he_normal_initializer(comm);
69  }
70  if (proto_weights.has_he_uniform_initializer()) {
71  return new he_uniform_initializer(comm);
72  }
73 
74  // Constant initialization
75  // Note: default is zero-initialization
76  const auto& params = proto_weights.constant_initializer();
77  return new constant_initializer(comm, params.value());
78 
79 }
