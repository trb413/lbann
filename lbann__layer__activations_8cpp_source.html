97 // There are three mechanisms for updating all fields of a distributed matrix
98 // 1) EntrywiseMap + independent reset of the bias row - Fastest (about ~30% faster)
99 // 2) IndexDependentMap (conditional application of function to mask bias row) - Slow
100 // IndexDependentMap(m,
101 // (std::function<double(int,int,double)>)[m](int r, int c, double z)->
102 // double{int bias_row = m.Height();
103 // if(r == bias_row - 1){return 1.0;}else{return sigmoid(z);}
104 // });
105 // 3) Dual nested loop over local matrix indices with global matrix lookups - Slow
107 void sigmoid_layer::forwardProp(ElMat& m)
108 {
109  const Int ngrows = m.Height();
110  const Int nrows = m.LocalHeight();
111  const Int ncols = m.LocalWidth();
112  EntrywiseMap(m, std::function<DataType(DataType)>(sigmoid));
113 
114  const Int r = nrows-1;
115  const Int gr = m.GlobalRow(r);
116  if(gr == ngrows - 1) { // Bias row
117  for (int c = 0; c < ncols; c++) {
118  m.SetLocal(r, c, 1.0); // Set the bias row back to 1.0
119  }
120  }
121 }
