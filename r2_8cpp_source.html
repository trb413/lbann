32 EvalType r2_metric::evaluate_compute(const AbsDistMat& prediction,
33  const AbsDistMat& ground_truth) {
34 
35  // Get matrix dimensions
36  const int local_height = prediction.LocalHeight();
37  const int local_width = prediction.LocalWidth();
38 
39  // Get local matrices
40  const Mat& prediction_local = prediction.LockedMatrix();
41  const Mat& ground_truth_local = ground_truth.LockedMatrix();
42 
43  DataType gt_mean, gt_std;
44  // Entry-wise mean of ground truth
45  //@todo fix stat class not to compute stdev if not needed
46  entrywise_mean_and_stdev(ground_truth, gt_mean, gt_std);
47 
48  // Compute residual sum of squares ss_res
49  // and sum of squares ss_tot as sum(square(ground_truth - mean(ground_truth)))
50  EvalType ss_res = 0;
51  EvalType ss_tot = 0;
52  #pragma omp parallel for reduction(+:ss_res) collapse(2)
53  for(El::Int col = 0; col < local_width; ++col) {
54  for(El::Int row = 0; row < local_height; ++row) {
55  const EvalType true_val = ground_truth_local(row, col);
56  const EvalType pred_val = prediction_local(row, col);
57  const EvalType val1 = true_val - pred_val;
58  const EvalType val2 = true_val - gt_mean;
59  ss_res += val1 * val1;
60  ss_tot += val2 * val2;
61  }
62  }
63 
64  EvalType res_tot[2] = {ss_res, ss_tot}; // Pack to do one allreduce.
65  El::mpi::AllReduce(res_tot, 2, prediction.DistComm());
66  //Keras and TF add epsilon (1e-07) to denominator to avoid inf score
67  //We might actually need to do this here and other places too
68  EvalType ss_tot_eps = res_tot[1] + 0.0000001;
69  return (1-(res_tot[0]/ss_tot_eps));
70 }
