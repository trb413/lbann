97  // fully_connected_layer : single network layer class

250  void setup_data() {
251  learning::setup_data();
252  // Initialize matrices
253  // Note: the weights-bias matrix has an extra column so it includes bias term
254  El::Zeros(*this->m_weights, this->m_num_neurons, this->m_num_prev_neurons+1);
255  El::Zeros(*this->m_weights_gradient, this->m_num_neurons,
256  this->m_num_prev_neurons + 1);
257 
258  // Initialize the activations part of the weight matrix -- leave the bias term weights zero
259  El::View(*this->m_activation_weights_v, *this->m_weights, El::ALL, El::IR(0, this->m_num_prev_neurons));
260  initialize_matrix(*this->m_activation_weights_v, m_weight_initialization, this->m_num_prev_neurons, this->m_num_neurons);
261 
262  // Initialize optimizer
263  if (this->m_optimizer != NULL) {
264 #if !(defined(__LIB_CUDA) && defined(LBANN_FULLY_CONNECTED_CUDA))
265  this->m_optimizer->setup(this->m_weights);
266 #else
267  // setup_gpu is used when using GPUs
268  if (!this->m_using_gpus) {
269  this->m_optimizer->setup(this->m_weights);
270  }
271 #endif
272  }

279  El::View(*this->m_activation_weights_v, *this->m_weights, El::ALL, El::IR(0, this->m_num_prev_neurons));

289  void setup_gpu() {
290  learning::setup_gpu();
291 #if !(defined(__LIB_CUDA) && defined(LBANN_FULLY_CONNECTED_CUDA))
292  throw lbann_exception("fully_connected_layer: CUDA not detected");
293 #else
294  // DATA_PARALLEL is assumed
295  // Allocate GPU memory
296  this->m_cudnn->allocate_on_gpus(m_weights_d,
297  m_weights->Height(),
298  m_weights->Width());
299  this->m_cudnn->broadcast_to_gpus(m_weights_d,
300  m_weights->LockedMatrix());
301 
302  this->m_cudnn->allocate_on_gpus(m_weights_gradient_d,
303  m_weights_gradient->Height(),
304  m_weights_gradient->Width());
305  m_activation_weights_d = m_weights_d;
306  m_activation_weights_gradient_d = m_weights_gradient_d;
307 
308  for (int i = 0; i < this->m_cudnn->get_num_gpus(); ++i) {
309  // point to the last column
310  m_bias_weights_d.push_back(m_weights_d[i] +
311  m_weights->Height() * (m_weights->Width() - 1));
312  m_bias_weights_gradient_d.push_back(
313  m_weights_gradient_d[i] +
314  m_weights_gradient->Height() * (m_weights_gradient->Width() - 1));
315  }
316 
317  this->m_cudnn->allocate_on_gpus(m_work_column_d, this->m_num_neurons, 1);
318 
319 
320  // CUDNN setup
321  FORCE_CHECK_CUDNN(cudnnCreateTensorDescriptor(&m_bias_weights_desc));
322 
323  // Two ways to create tensor descriptor. Should be identical.
324 #if 1
325  std::vector<int> bias_dims(4, 1);
326  bias_dims[3] = this->m_bias_weights_v->Height();
327  std::vector<int> bias_strides(4, 1);
328  for (int i = 2; i >=0; --i) {
329  bias_strides[i] = bias_strides[i+1] * bias_dims[i+1];
330  }
331  FORCE_CHECK_CUDNN(cudnnSetTensorNdDescriptor(m_bias_weights_desc,
332  this->m_cudnn->get_cudnn_data_type(),
333  bias_dims.size(),
334  bias_dims.data(),
335  bias_strides.data()));
336 #else
337  FORCE_CHECK_CUDNN(cudnnSetTensor4dDescriptor(m_bias_weights_desc,
338  CUDNN_TENSOR_NCHW,
339  this->m_cudnn->get_cudnn_data_type(),
340  1, 1, 1, m_bias_weights_v->Height()));
341 #endif
342  FORCE_CHECK_CUDNN(cudnnCreateTensorDescriptor(&m_activations_desc));
343 #if 1
344  bias_dims[3] = this->m_bias_weights_v->Height();
345  bias_dims[2] = m_mini_batch_size_per_gpu;
346  for (int i = 2; i >=0; --i) {
347  bias_strides[i] = bias_strides[i+1] * bias_dims[i+1];
348  }
349  FORCE_CHECK_CUDNN(cudnnSetTensorNdDescriptor(m_activations_desc,
350  this->m_cudnn->get_cudnn_data_type(),
351  bias_dims.size(),
352  bias_dims.data(),
353  bias_strides.data()));
354 #else
355  FORCE_CHECK_CUDNN(cudnnSetTensor4dDescriptor(m_activations_desc,
356  CUDNN_TENSOR_NCHW,
357  this->m_cudnn->get_cudnn_data_type(),
358  1, 1, m_mini_batch_size_per_gpu,
359  m_bias_weights_v->Height()));
360 #endif
361 
362  if (this->m_optimizer != NULL) {
363  this->m_optimizer->setup_gpu(this->m_weights, this->m_weights_d);
364  }
365 
366 #endif // __LIB_CUDA
367  }

389  void fp_compute_cpu() {
390  // Apply weight matrix
391  fp_compute_weights<device::CPU>();
392 
393  // Apply bias if needed
394  if(m_bias_scaling_factor != DataType(0)) {
395  El::Copy(*m_bias_weights_v, *m_bias_weights_repl);
396  const Mat& local_bias_weights = m_bias_weights_repl->LockedMatrix();
397  El::IndexDependentMap(this->m_activations_v->Matrix(),
398  (std::function<DataType(El::Int,El::Int,const DataType&)>)
399  ([this,&local_bias_weights](El::Int r, El::Int c,const DataType& z)->DataType {
400  return z + m_bias_scaling_factor * local_bias_weights.Get(r, 0);
401  }));
402  }

473  void bp_compute_cpu() {
474  // Compute the error signal and gradients.
475  bp_compute_weights<device::CPU>();
476 
477  // Compute bias update if needed
478  if(m_bias_scaling_factor != DataType(0)) {
479  El::RowSum(*this->m_prev_error_signal,

481  El::Scale(m_bias_scaling_factor /
482  this->m_neural_network_model->get_effective_mini_batch_size(),
483  *m_bias_weights_gradient_repl);
484  El::Copy(*m_bias_weights_gradient_repl, *m_bias_weights_gradient_v);

514  DataType computeCost(DistMat& deltas) {
515  DataType avg_error = 0.0, total_error = 0.0;
516  // Compute the L2 norm on the deltas (activation - y)
517  ColSumMat norms;
518  ColumnTwoNorms(deltas, norms);
519  int c = 0;
520  // Sum the local, total error
521  for(int r = 0; r < norms.LocalHeight(); r++) {
522  total_error += norms.GetLocal(r,c);
523  }
524  mpi::AllReduce(total_error, norms.DistComm());
525  avg_error = total_error / norms.Height();
526  return avg_error;
527  }
