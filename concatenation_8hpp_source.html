120  void setup_dims() override {
121 
122  // Initialize previous layer dimensions with first parent layer
123  transform_layer::setup_dims();
124 
125  // Check if concatenation axis is valid
126  if(m_concatenation_axis < 0
127  || m_concatenation_axis >= this->m_num_neuron_dims) {
128  throw lbann_exception("concatenation_layer: invalid concatenation axis");
129  }
130 
131  // Get concatenation axis indices corresponding to each parent layer
132  m_concatenation_points.empty();
133  m_concatenation_points.push_back(0);
134  m_concatenation_points.push_back(this->m_neuron_dims[m_concatenation_axis]);
135  for(size_t i=1; i<this->m_parent_layers.size(); ++i) {
136 
137  // Get parent layer dimensions
138  std::vector<int> parent_dims = this->m_parent_layers[i]->fp_output_dims(this);
139 
140  // Check if parent layer has valid dimensions
141  if((int) parent_dims.size() != this->m_num_neuron_dims) {
142  throw lbann_exception("concatenation_layer: parent layer has invalid number of dimensions");
143  }
144  for(size_t d=0; d<parent_dims.size(); ++d) {
145  if((int) d != m_concatenation_axis
146  && this->m_neuron_dims[d] != parent_dims[d]) {
147  throw lbann_exception("concatenation_layer: parent layer has invalid dimensions");
148  }
149  }
150 
151  // Get concatentation axis upper bound for parent layer
152  m_concatenation_points.push_back(m_concatenation_points.back()
153  + parent_dims[m_concatenation_axis]);
154 
155  }
156 
157  // Update neuron dimensions
158  this->m_neuron_dims[m_concatenation_axis] = m_concatenation_points.back();
159  this->m_num_neurons = std::accumulate(m_neuron_dims.begin(),
160  m_neuron_dims.end(),
161  1,
162  std::multiplies<int>());
163 
164  }

166  void setup_gpu() override {
167  transform_layer::setup_gpu();
168  #ifndef LBANN_HAS_CUDNN
169  throw lbann_exception("concatenation_layer: cuDNN not detected");
170  #else
171 
172  // Copy backward propagation output from GPUs if a parent layer is
173  // not using GPU implementation
174  for(size_t i=1; i<this->m_parent_layers.size(); ++i) {
175  if(!this->m_parent_layers[i]->using_gpus()) {
176  m_copy_bp_output_from_gpus = true;
177  }
178  }
179 
180  // Allocate workspace if needed
181  if(m_copy_bp_output_from_gpus) {
182  int max_slice_dim = 0;
183  for(size_t parent_index = 1;
184  parent_index < this->m_parent_layers.size();
185  ++parent_index) {
186  if(!this->m_parent_layers[parent_index]->using_gpus()) {
187  max_slice_dim = std::max(max_slice_dim,
188  m_concatenation_points[parent_index+1]
189  - m_concatenation_points[parent_index]);
190  }
191  }
192  int max_slice_size = (this->m_num_neurons
193  / this->m_neuron_dims[m_concatenation_axis]
194  * max_slice_dim);
195  size_t required_work_space = (max_slice_size
196  * this->m_mini_batch_size_per_gpu
197  * sizeof(DataType));
198  for(int i=0; i<this->m_cudnn->get_num_gpus(); ++i) {
199  if(required_work_space > this->m_cudnn->get_work_space_size(i)) {
200  this->m_cudnn->set_work_space_size(i, required_work_space);

202  }

234  void fp_compute_gpu() {
235  #ifndef LBANN_HAS_CUDNN
236  throw lbann_exception("concatenation_layer: cuDNN not detected");
237  #else
238 
239  // Split the activations tensor into slices of width 1 along the
240  // concatenation axis
241  const int num_slices
242  = std::accumulate(this->m_neuron_dims.begin(),
243  this->m_neuron_dims.begin() + m_concatenation_axis,
244  1,
245  std::multiplies<int>());
246  const int slice_unit_size
247  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
248  this->m_neuron_dims.end(),
249  1,
250  std::multiplies<int>());
251  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
252  const int output_slice_size = output_slice_dim * slice_unit_size;
253 
254  // Copy entries in each child to error signal tensor
255  const int num_gpus = this->m_cudnn->get_num_gpus();
256  for(size_t parent_index = 0;
257  parent_index < this->m_parent_layers.size();
258  ++parent_index) {
259  const Layer* parent = this->m_parent_layers[parent_index];
260 
261  // Get child error signal on GPUs
262  std::vector<DataType*> input;
263  if(parent_index == 0) {
264  input = this->m_prev_activations_dv;
265  }
266  else {
268  std::vector<DataType*> work_spaces;
269  for(int i=0; i<num_gpus; ++i) {
270  work_spaces.push_back((DataType*)this->m_cudnn->get_work_space(i));
271  }
272  if(parent->using_gpus()) {
273  parent->get_gpu_fp_output(input, work_spaces, this);
274  }
275  else {
276  parent->get_fp_output(*this->m_prev_activations_v, this);
277  this->m_cudnn->scatter_to_gpus(work_spaces,
278  this->m_prev_activations_v->LockedMatrix(),
279  this->m_mini_batch_size_per_gpu);
280  input = work_spaces;
281  }
282  }
283 
284  // Split previous error signal tensor into slices
285  const int input_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
286  const int input_slice_size = input_slice_dim * slice_unit_size;
287  const int input_size = num_slices * input_slice_size;
288  const int slice_offset = m_concatenation_points[parent_index] * slice_unit_size;
289 
290  // Copy slices from previous error signal tensor into error signal tensor
291  for(int slice = 0; slice < num_slices; ++slice) {
292  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
293  for(int i = 0; i < num_gpus; ++i) {
294  input_slice[i] = input[i] + slice * input_slice_size;
295  output_slice[i] = this->m_activations_d[i] + slice * output_slice_size + slice_offset;
296  }
297  this->m_cudnn->copy_on_gpus(output_slice,
298  input_slice,
299  input_slice_size,
300  this->m_mini_batch_size_per_gpu,
301  input_size,
302  this->m_num_neurons);
303  }
304 
305  }
306 
307  #endif // #ifndef LBANN_HAS_CUDNN
308  }

310  void fp_compute_cpu() {
311 
312  // Split the neuron tensor into slices of width 1 along the
313  // concatenation axis
314  const int num_slices
315  = std::accumulate(this->m_neuron_dims.begin(),
316  this->m_neuron_dims.begin() + m_concatenation_axis,
317  1,
318  std::multiplies<int>());
319  const int slice_unit_size
320  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
321  this->m_neuron_dims.end(),
322  1,
323  std::multiplies<int>());
324  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
325  const int output_slice_size = output_slice_dim * slice_unit_size;
326 
327  // Copy entries in each parent to neuron tensor
328  for(size_t i = 0; i < this->m_parent_layers.size(); ++i) {
329 
330  // Split previous neuron tensor into slices
331  this->m_parent_layers[i]->get_fp_output(*this->m_prev_activations_v, this);
332  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
333  const int input_slice_size = input_slice_dim * slice_unit_size;
334  const int slice_offset_start = m_concatenation_points[i] * slice_unit_size;
335  const int slice_offset_end = m_concatenation_points[i+1] * slice_unit_size;
336 
337  // Copy slices from previous neuron tensor into neuron tensor
338  for(int slice = 0; slice < num_slices; ++slice) {
339  El::LockedView(*m_input_slice_v,
340  *this->m_prev_activations_v,
341  El::IR(slice * input_slice_size,
342  (slice+1) * input_slice_size),
343  El::ALL);
344  El::View(*m_output_slice_v,
345  *this->m_activations_v,
346  El::IR(slice * output_slice_size + slice_offset_start,
347  slice * output_slice_size + slice_offset_end),
348  El::ALL);
349  El::Copy(*m_input_slice_v, *m_output_slice_v);
350  }
351 
352  }

356  void get_bp_output(AbsDistMat& bp_output, const Layer* prev_layer) const override {
357 
358  // Check if input is in the list of parent layers
359  const int parent_index = (std::find(this->m_parent_layers.begin(),
360  this->m_parent_layers.end(),
361  prev_layer)
362  - this->m_parent_layers.begin());
363  if(parent_index >= (int) this->m_parent_layers.size()) {
364  transform_layer::get_bp_output(bp_output, prev_layer);
365  }
366 
367  // Split the error signal tensor into slices of width 1 along the
368  // concatenation axis
369  const int num_slices
370  = std::accumulate(this->m_neuron_dims.begin(),
371  this->m_neuron_dims.begin() + m_concatenation_axis,
372  1,
373  std::multiplies<int>());
374  const int slice_unit_size
375  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
376  this->m_neuron_dims.end(),
377  1,
378  std::multiplies<int>());
379  const int input_slice_dim = this->m_neuron_dims[m_concatenation_axis];
380  const int output_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
381  const int input_slice_size = input_slice_dim * slice_unit_size;
382  const int output_slice_size = output_slice_dim * slice_unit_size;
383  const int slice_offset_start = m_concatenation_points[parent_index] * slice_unit_size;
384  const int slice_offset_end = m_concatenation_points[parent_index+1] * slice_unit_size;
385 
386  if(num_slices == 1
387  && m_error_signal_v->DistData() == bp_output.DistData()) {
388  // Return view of error signal slice
389  El::LockedView(bp_output,
390  *this->m_error_signal_v,
391  El::IR(slice_offset_start, slice_offset_end),
392  El::ALL);
393  }
394  else {
395  // Copy slices from error signal tensor into output
396  bp_output.Empty(false);
397  bp_output.Resize(output_slice_size, m_error_signal_v->Width());
398  AbsDistMat* output_slice_v
399  = bp_output.Construct(bp_output.Grid(), bp_output.Root());
400  for(int slice = 0; slice < num_slices; ++slice) {
401  El::LockedView(*m_input_slice_v,
402  *this->m_error_signal_v,
403  El::IR(slice * input_slice_size + slice_offset_start,
404  slice * input_slice_size + slice_offset_end),
405  El::ALL);
406  El::View(*output_slice_v,
407  bp_output,
408  El::IR(slice * output_slice_size,
409  (slice+1) * output_slice_size),
410  El::ALL);
411  El::Copy(*m_input_slice_v, *output_slice_v);
412  }
413  delete output_slice_v;
414  }
