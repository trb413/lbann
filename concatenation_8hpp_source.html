175  // Check that dimensions are valid

177  expected_dims[m_concatenation_axis] = parent_dims[m_concatenation_axis];
178  }
179  if (parent_dims != expected_dims) {
180  err << __FILE__ << " " << __LINE__ << " :: "

184  err << (i > 0 ? "x" : "") << expected_dims[i];

226  void fp_compute_cpu() {
227 
228  // Gradient w.r.t. input
229  auto& output = get_activations();
230 
231  // Get number of contiguous regions in a tensor slice of width 1
232  const auto& output_dims = this->m_neuron_dims;
233  const int num_regions = std::accumulate(output_dims.begin(),
234  output_dims.begin() + m_concatenation_axis,
235  1,
236  std::multiplies<int>());
237  const int unit_region_size = std::accumulate(output_dims.begin() + m_concatenation_axis + 1,
238  output_dims.end(),
239  1,
240  std::multiplies<int>());
241 
242  // Get stride between contiguous regions in output tensor slices
243  const int output_slice_dim = output_dims[m_concatenation_axis];
244  const int output_region_stride = output_slice_dim * unit_region_size;
245 
246  // Populate output with slices of inputs
247  for (int i = 0; i < get_num_parents(); ++i) {
248  const auto& input = get_prev_activations(i);
249 
250  // Get stride between contiguous regions in input tensor slices
251  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
252  const int input_region_stride = input_slice_dim * unit_region_size;
253 
254  // Get position of first contiguous region in output tensor
255  const int output_region_start = m_concatenation_points[i] * unit_region_size;
256  const int output_region_end = m_concatenation_points[i+1] * unit_region_size;
257 
258  // Populate concatenation of gradient w.r.t. input
259  for (int region = 0; region < num_regions; ++region) {
260  El::LockedView(*m_input_region_v,
261  input,
262  El::IR(region * input_region_stride,
263  (region+1) * input_region_stride),
264  El::ALL);
265  El::View(*m_output_region_v,
266  output,
267  El::IR(output_region_start + region * output_region_stride,
268  output_region_end + region * output_region_stride),
269  El::ALL);
270  El::Copy(*m_input_region_v, *m_output_region_v);
271  }
272 
273  }

277  void bp_compute_cpu() {
278 
279  // Gradient w.r.t. output
280  const auto& gradient_wrt_output = get_prev_error_signals();
281 
282  // Get number of contiguous regions in a tensor slice of width 1
283  const auto& output_dims = this->m_neuron_dims;
284  const int num_regions = std::accumulate(output_dims.begin(),
285  output_dims.begin() + m_concatenation_axis,
286  1,
287  std::multiplies<int>());
288  const int unit_region_size = std::accumulate(output_dims.begin() + m_concatenation_axis + 1,
289  output_dims.end(),
290  1,
291  std::multiplies<int>());
292 
293  // Get stride between contiguous regions in input tensor slice
294  const int output_slice_dim = output_dims[m_concatenation_axis];
295  const int output_region_stride = output_slice_dim * unit_region_size;
296 
297  // Populate output tensors with concatenations of input tensor
298  for (int i = 0; i < get_num_parents(); ++i) {
299  auto& gradient_wrt_input = get_error_signals(i);
300 
301  // Get stride between contiguous regions in output tensor concatenations
302  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
303  const int input_region_stride = input_slice_dim * unit_region_size;
304 
305  // Get position of first contiguous region in output tensor
306  const int output_region_start = m_concatenation_points[i] * unit_region_size;
307  const int output_region_end = m_concatenation_points[i+1] * unit_region_size;
308 
309  // Populate current output tensor
310  for (int region = 0; region < num_regions; ++region) {
311  El::LockedView(*m_output_region_v,
312  gradient_wrt_output,
313  El::IR(output_region_start + region * output_region_stride,
314  output_region_end + region * output_region_stride),
315  El::ALL);
316  El::View(*m_input_region_v,
317  gradient_wrt_input,
318  El::IR(region * input_region_stride,
319  (region+1) * input_region_stride),
320  El::ALL);
321  El::Axpy(DataType(1), *m_output_region_v, *m_input_region_v);
322  }
323  }

327  void fp_compute_gpu() {
328  #ifndef LBANN_HAS_CUDNN
329  std::stringstream err;
330  err << __FILE__ << " " << __LINE__ << " :: concatenation_layer: cuDNN not detected";
331  throw lbann_exception(err.str());
332  #else
333 
334  // Gradient w.r.t. input
335  auto& output_d = m_activations_d[0];
336 
337  // Get number of contiguous regions in a tensor slice of width 1
338  const auto& output_dims = this->m_neuron_dims;
339  const int num_regions = std::accumulate(output_dims.begin(),
340  output_dims.begin() + m_concatenation_axis,
341  1,
342  std::multiplies<int>());
343  const int unit_region_size = std::accumulate(output_dims.begin() + m_concatenation_axis + 1,
344  output_dims.end(),
345  1,
346  std::multiplies<int>());
347 
348  // Get stride between contiguous regions in output tensor slices
349  const int output_slice_dim = output_dims[m_concatenation_axis];
350  const int output_region_stride = output_slice_dim * unit_region_size;
351 
352  // Populate output with slices of inputs
353  cudnn::matrix input_region_d(m_cudnn), output_region_d(m_cudnn);
354  for (int i = 0; i < get_num_parents(); ++i) {
355  const auto& input_d = m_prev_activations_d[i];
356 
357  // Get stride between contiguous regions in input tensor slices
358  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
359  const int input_region_stride = input_slice_dim * unit_region_size;
360 
361  // Get position of first contiguous region in output tensor
362  const int output_region_start = m_concatenation_points[i] * unit_region_size;
363 
364  // Populate slice of output
365  for (int region = 0; region < num_regions; ++region) {
366  auto input_ptrs = input_d.get_locked_data();
367  auto output_ptrs = output_d.get_data();
368  for (auto& ptr : input_ptrs) {
369  ptr += region * input_region_stride;
370  }
371  for (auto& ptr : output_ptrs) {
372  ptr += output_region_start + region * output_region_stride;
373  }
374  input_region_d.locked_attach(input_ptrs,
375  input_region_stride,
376  this->m_mini_batch_size_per_gpu,
377  input_d.get_leading_dim());
378  output_region_d.attach(output_ptrs,
379  input_region_stride,
380  this->m_mini_batch_size_per_gpu,
381  output_d.get_leading_dim());
382  output_region_d.copy(input_region_d);
383  }
384 
385  }
386 
387  #endif // LBANN_HAS_CUDNN

416  cudnn::matrix gradient_wrt_input_region_d(m_cudnn);

427 
428  // Populate current output tensor
429  for (int region = 0; region < num_regions; ++region) {
430  auto gradient_wrt_input_ptrs = gradient_wrt_input_d.get_data();
431  auto gradient_wrt_output_ptrs = gradient_wrt_output_d.get_locked_data();
432  for (auto& ptr : gradient_wrt_input_ptrs) {
433  ptr += region * input_region_stride;
434  }
435  for (auto& ptr : gradient_wrt_output_ptrs) {
436  ptr += output_region_start + region * output_region_stride;
437  }
438  gradient_wrt_input_region_d.attach(gradient_wrt_input_ptrs,
439  input_region_stride,
440  this->m_mini_batch_size_per_gpu,
441  gradient_wrt_input_d.get_leading_dim());
