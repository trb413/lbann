165  virtual void fp_compute_cpu() {
166 
167  // Local matrices
168  const auto& local_input = get_local_prev_activations();
169  auto& local_output = get_local_activations();
170  auto& local_workspace = m_workspace->Matrix();
171 
172  // Matrix parameters
173  const El::Int local_height = local_input.Height();
174  const El::Int local_width = local_input.Width();
175 
176  // Find maximum entry in each column
177  #pragma omp parallel for
178  for(El::Int col = 0; col < local_width; ++col) {
179  DataType max_entry = local_input(0, col);
180  for(El::Int row = 1; row < local_height; ++row) {
181  max_entry = std::max(max_entry, local_input(row, col));
182  }
183  local_workspace(0, col) = max_entry;
184  }
185  m_comm->allreduce(*m_workspace, m_workspace->RedundantComm(),
186  El::mpi::MAX);
187 
188  // Exponentiate activations and compute column sums
189  // Note: Subtracting by the column max prevents activations from
190  // blowing up. Large negative values underflow to 0.
191  #pragma omp parallel for
192  for (El::Int col = 0; col < local_width; ++col) {
193  const DataType shift = local_workspace(0, col);
194  DataType sum = 0;
195  for (El::Int row = 0; row < local_height; ++row) {
196  const DataType x = local_input(row, col);
197  const DataType y = std::exp(x - shift);
198  local_output(row, col) = y;
199  sum += y;
200  }
201  local_workspace(0, col) = sum;
202  }
203  m_comm->allreduce(*m_workspace, m_workspace->RedundantComm());
204 
205  // Divide activations by column sums
206  // Note: Small values are rounded to minimum output value to avoid
207  // denormalized floats.
208  #pragma omp parallel for
209  for (El::Int col = 0; col < local_width; ++col) {
210  const DataType scale = DataType(1) / local_workspace(0, col);
211  for (El::Int row = 0; row < local_height; ++row) {
212  DataType& y = local_output(row, col);
213  y *= scale;
214  #ifdef LBANN_ENABLE_SOFTMAX_CUTOFF
215  y = std::max(y, m_min_output);
216  #endif // LBANN_ENABLE_SOFTMAX_CUTOFF
217  }
218  }
219 
220  }

222  virtual void bp_compute_cpu() {
223 
224  // Local matrices
225  const auto& local_output = get_local_activations();
226  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
227  auto& local_gradient_wrt_input = get_local_error_signals();
228  auto& local_workspace = m_workspace->Matrix();
229 
230  // Matrix parameters
231  const El::Int local_height = local_output.Height();
232  const El::Int local_width = local_output.Width();
233 
234  // Compute dot products between output and gradient w.r.t. output
235  for (El::Int col = 0; col < local_width; ++col) {
236  const auto& y = local_output(El::ALL, El::IR(col));
237  const auto& dy = local_gradient_wrt_output(El::ALL, El::IR(col));
238  local_workspace(0, col) = El::Dot(y, dy);
239  }
240  m_comm->allreduce(*m_workspace, m_workspace->RedundantComm());
241 
242  // Compute gradient w.r.t. input
243  #pragma omp parallel for
244  for (El::Int col = 0; col < local_width; ++col) {
245  const DataType y_dot_dy = local_workspace(0, col);
246  for (El::Int row = 0; row < local_height; ++row) {
247  const DataType y = local_output(row, col);
248  const DataType dy = local_gradient_wrt_output(row, col);
249  DataType dx = y * (dy - y_dot_dy);
250  #ifdef LBANN_ENABLE_SOFTMAX_CUTOFF
251  if (y <= m_min_output) { dx = DataType(0); }
252  #endif
253  local_gradient_wrt_input(row, col) += dx;
254  }
255  }

259  void fp_compute_cudnn() {
260  #ifndef LBANN_HAS_CUDNN
261  throw lbann_exception("softmax_layer: cuDNN not detected");
262  #else
263 
264  // Useful constants
265  const DataType one = 1;
266  const DataType zero = 0;
267 
268  // Matrices
269  const auto& prev_activations_d = this->m_prev_activations_d[0];
270  auto& activations_d = this->m_activations_d[0];
271 
272  // Apply softmax on each GPU
273  const int num_gpus = this->m_cudnn->get_num_gpus();
274  for(int i = 0; i < num_gpus; ++i) {
275  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
276  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
277  this->m_cudnn->get_stream(i)));
278  CHECK_CUDNN(cudnnSoftmaxForward(this->m_cudnn->get_handle(i),
279  CUDNN_SOFTMAX_ACCURATE,
280  CUDNN_SOFTMAX_MODE_INSTANCE,
281  &one,
282  this->m_prev_activations_cudnn_desc,
283  prev_activations_d.get_locked_data(i),
284  &zero,
285  this->m_activations_cudnn_desc,
286  activations_d.get_data(i)));
287  }
288 
289  #ifdef LBANN_ENABLE_SOFTMAX_CUTOFF
290  // Round to minimum value to avoid denormalized floats
291  softmax_cuda::fp_cutoff(*this->m_cudnn,
292  activations_d.get_data(),
293  get_num_neurons(),
294  this->m_mini_batch_size_per_gpu,
295  m_min_output);
296  #endif // LBANN_ENABLE_SOFTMAX_CUTOFF
297 
298  #endif // LBANN_HAS_CUDNN
