32 template <data_layout layout>
33 Layer* construct_layer(lbann_comm* comm,
34  std::map<execution_mode, generic_data_reader*>& data_readers,
35  int num_parallel_readers,
36  cudnn::cudnn_manager* cudnn,
37  const lbann_data::Layer& proto_layer) {
38  std::stringstream err;
39 
40  // Currently only data-parallel layers have GPU support
42  if (layout == data_layout::MODEL_PARALLEL) { cudnn = nullptr; }
43 
44  // Input layers
45  if (proto_layer.has_input()) {
46  const auto& params = proto_layer.input();
47  const auto& io_buffer = params.io_buffer();
48  if (io_buffer == "distributed") {
49  return new input_layer<distributed_io_buffer, layout>(comm,
50  num_parallel_readers,
51  data_readers,
52  !params.data_set_per_model(),
53  params.for_regression());
54  }
55  if (io_buffer == "partitioned") {
56  return new input_layer<partitioned_io_buffer, layout>(comm,
57  num_parallel_readers,
58  data_readers,
59  !params.data_set_per_model(),
60  params.for_regression());
61  }
62  }
63 
64  // Target layers
65  if (proto_layer.has_target()) {
66  const auto& params = proto_layer.target();
67  const auto& io_buffer = params.io_buffer();
68  if (io_buffer == "distributed") {
69  return new target_layer<distributed_io_buffer, layout>(comm,
70  nullptr,
71  num_parallel_readers,
72  data_readers,
73  params.shared_data_reader(),
74  params.for_regression());
75  }
76  if (io_buffer == "partitioned") {
77  return new target_layer<partitioned_io_buffer, layout>(comm,
78  nullptr,
79  num_parallel_readers,
80  data_readers,
81  params.shared_data_reader(),
82  params.for_regression());
83  }
84  }
85  if (proto_layer.has_reconstruction()) {
86  return new reconstruction_layer<layout>(comm, nullptr);
87  }
88 
89  // Fully connected layer
90  if (proto_layer.has_fully_connected()) {
91  const auto& params = proto_layer.fully_connected();
92  int num_neurons = params.num_neurons();
93  if (proto_layer.num_neurons_from_data_reader()) {
94  num_neurons = data_readers[execution_mode::training]->get_linearized_data_size();
95  }
96  return new fully_connected_layer<layout>(comm,
97  num_neurons,
98  nullptr,
99  params.has_bias(),
100  cudnn);
101  }
102 
103  // Convolution and deconvolution layer
104  if (proto_layer.has_convolution()) {
105  const auto& params = proto_layer.convolution();
106  const auto& num_output_channels = params.num_output_channels();
107  const auto& bias = params.has_bias();
108  if (params.has_vectors()) {
109  const auto& dims = parse_list<int>(params.conv_dims());
110  const auto& pads = parse_list<int>(params.conv_pads());
111  const auto& strides = parse_list<int>(params.conv_strides());
112  if (layout == data_layout::DATA_PARALLEL) {
113  return new convolution_layer<data_layout::DATA_PARALLEL>(
114  comm, dims.size(), num_output_channels,
115  dims, pads, strides, bias, cudnn
116  );
117  }
118  } else {
119  const auto& num_dims = params.num_dims();
120  const auto& dim = params.conv_dims_i();
121  const auto& pad = params.conv_pads_i();
122  const auto& stride = params.conv_strides_i();
123  if (layout == data_layout::DATA_PARALLEL) {
124  return new convolution_layer<data_layout::DATA_PARALLEL>(
125  comm, num_dims, num_output_channels,
126  dim, pad, stride, bias, cudnn
127  );
128  }
129  }
130  }
131  if (proto_layer.has_deconvolution()) {
132  const auto& params = proto_layer.deconvolution();
133  const auto& bias = params.has_bias();
134  int num_output_channels = params.num_output_channels();
135  if (proto_layer.num_neurons_from_data_reader()) {
136  num_output_channels = data_readers[execution_mode::training]->get_linearized_data_size();
137  }
138  if (params.has_vectors()) {
139  const auto& dims = parse_list<int>(params.conv_dims());
140  const auto& pads = parse_list<int>(params.conv_pads());
141  const auto& strides = parse_list<int>(params.conv_strides());
142  if (layout == data_layout::DATA_PARALLEL) {
143  return new deconvolution_layer<data_layout::DATA_PARALLEL>(
144  comm, dims.size(), num_output_channels,
145  dims, pads, strides, bias, cudnn
146  );
147  }
148  } else {
149  const auto& num_dims = params.num_dims();
150  const auto& dim = params.conv_dims_i();
151  const auto& pad = params.conv_pads_i();
152  const auto& stride = params.conv_strides_i();
153  if (layout == data_layout::DATA_PARALLEL) {
154  return new deconvolution_layer<data_layout::DATA_PARALLEL>(
155  comm, num_dims, num_output_channels,
156  dim, pad, stride, bias, cudnn
157  );
158  }
159  }
160  }
161 
162  // Transform layers
163  if (proto_layer.has_reshape()) {
164  const auto& params = proto_layer.reshape();
165  std::vector<int> dims = parse_list<int>(params.dims());
166  if (proto_layer.num_neurons_from_data_reader()) {
167  dims.clear();
168  if (params.reshape_to_flattened_conv_format()) {
169  dims.push_back(1);
170  }
171  dims.push_back(data_readers[execution_mode::training]->get_linearized_data_size());
172  }
173  return new reshape_layer<layout>(comm, dims.size(), dims.data());
174  }
175  if (proto_layer.has_sum()) {
176  const auto& scaling_factors = parse_list<DataType>(proto_layer.sum().scaling_factors());
177  return new sum_layer<layout>(comm, scaling_factors, cudnn);
178  }
179  if (proto_layer.has_split()) {
180  return new split_layer<layout>(comm, cudnn);
181  }
182  if (proto_layer.has_concatenation()) {
183  const auto& axis = proto_layer.concatenation().concatenation_axis();
184  return new concatenation_layer<layout>(comm, axis, cudnn);
185  }
186  if (proto_layer.has_slice()) {
187  const auto& params = proto_layer.slice();
188  const auto& slice_points = parse_list<int>(params.slice_points());
189  return new slice_layer<layout>(comm,
190  params.slice_axis(),
191  slice_points,
192  cudnn);
193  }
194  if (proto_layer.has_hadamard()) {
195  return new hadamard_layer<layout>(comm, cudnn);
196  }
197  if (proto_layer.has_constant()) {
198  const auto& params = proto_layer.constant();
199  const auto& dims = parse_list<int>(params.num_neurons());
200  return new constant_layer<layout>(comm, params.value(), dims, cudnn);
201  }
202  if (proto_layer.has_gaussian()) {
203  const auto& params = proto_layer.gaussian();
204  const auto& dims = parse_list<int>(params.neuron_dims());
205  return new gaussian_layer<layout>(comm,
206  dims,
207  params.mean(),
208  params.stdev(),
209  cudnn);
210  }
211  if (proto_layer.has_bernoulli()) {
212  const auto& params = proto_layer.bernoulli();
213  const auto& dims = parse_list<int>(params.neuron_dims());
214  return new bernoulli_layer<layout>(comm,
215  dims,
216  params.prob(),
217  cudnn);
218  }
219  if (proto_layer.has_uniform()) {
220  const auto& params = proto_layer.uniform();
221  const auto& dims = parse_list<int>(params.neuron_dims());
222  return new uniform_layer<layout>(comm,
223  dims,
224  params.min(),
225  params.max(),
226  cudnn);
227  }
228  if (proto_layer.has_zero()) {
229  const auto& params = proto_layer.zero();
230  return new zero_layer<layout>(comm, params.first_half(), params.second_half(), cudnn);
231  }
232  if (proto_layer.has_pooling()) {
233  const auto& params = proto_layer.pooling();
234  const auto& mode_str = params.pool_mode();
235  pool_mode mode = pool_mode::invalid;
236  if (mode_str == "max" ) { mode = pool_mode::max; }
237  if (mode_str == "average" ) { mode = pool_mode::average; }
238  if (mode_str == "average_no_pad" ) { mode = pool_mode::average_no_pad; }
239  if (params.has_vectors()) {
240  const auto& dims = parse_list<int>(params.pool_dims());
241  const auto& pads = parse_list<int>(params.pool_pads());
242  const auto& strides = parse_list<int>(params.pool_strides());
243  if (layout == data_layout::DATA_PARALLEL) {
244  return new pooling_layer<data_layout::DATA_PARALLEL>(
245  comm, dims.size(), dims, pads, strides, mode, cudnn
246  );
247  }
248  } else {
249  const auto& num_dims = params.num_dims();
250  const auto& dim = params.pool_dims_i();
251  const auto& pad = params.pool_pads_i();
252  const auto& stride = params.pool_strides_i();
253  if (layout == data_layout::DATA_PARALLEL) {
254  return new pooling_layer<data_layout::DATA_PARALLEL>(
255  comm, num_dims, dim, pad, stride, mode, cudnn
256  );
257  }
258  }
259  }
260  if (proto_layer.has_unpooling()) {
261  if (layout == data_layout::DATA_PARALLEL) {
262  return new unpooling_layer<data_layout::DATA_PARALLEL>(comm);
263  }
264  }
265  if (proto_layer.has_reduction()) {
266  const auto& params = proto_layer.reduction();
267  const auto& mode_str = params.mode();
268  reduction_mode mode = reduction_mode::INVALID;
269  if (mode_str == "sum" || mode_str.empty()) { mode = reduction_mode::SUM; }
270  if (mode_str == "average") { mode = reduction_mode::AVERAGE; }
271  if (layout == data_layout::DATA_PARALLEL) {
272  return new reduction_layer<data_layout::DATA_PARALLEL>(comm, mode, cudnn);
273  }
274  }
275  if (proto_layer.has_evaluation()) {
276  return new evaluation_layer<layout>(comm, cudnn);
277  }
278 
279  // Regularizer layers
280  if (proto_layer.has_batch_normalization()) {
281  const auto& params = proto_layer.batch_normalization();
282  if (layout == data_layout::DATA_PARALLEL) {
283  return new batch_normalization<data_layout::DATA_PARALLEL>(comm,
284  params.decay(),
285  params.epsilon(),
286  params.global_stats(),
287  cudnn);
288  }
289  }
290  if (proto_layer.has_dropout()) {
291  const auto& params = proto_layer.dropout();
292  return new dropout<layout>(comm, params.keep_prob(), cudnn);
293  }
294  if (proto_layer.has_local_response_normalization()) {
295  const auto& params = proto_layer.local_response_normalization();
296  if (layout == data_layout::DATA_PARALLEL) {
297  return new local_response_normalization_layer<data_layout::DATA_PARALLEL>(comm,
298  params.window_width(),
299  params.lrn_alpha(),
300  params.lrn_beta(),
301  params.lrn_k(),
302  cudnn);
303  }
304  }
305  if (proto_layer.has_selu_dropout()) {
306  const auto& params = proto_layer.selu_dropout();
307  const auto& keep_prob = params.keep_prob();
308  const auto& alpha = params.alpha();
309  const auto& scale = params.scale();
310  if (alpha != 0.0 && scale != 0.0) {
311  return new selu_dropout<layout>(comm, keep_prob, alpha, scale);
312  } else {
313  return new selu_dropout<layout>(comm, keep_prob);
314  }
315  }
316 
317  // Activation layers
318  if (proto_layer.has_softmax()) {
319  return new softmax_layer<layout>(comm, cudnn);
320  }
321  if (proto_layer.has_relu()) {
322  return new relu_layer<layout>(comm, cudnn);
323  }
324  if (proto_layer.has_sigmoid()) {
325  return new sigmoid_layer<layout>(comm, cudnn);
326  }
327  if (proto_layer.has_tanh()) {
328  return new tanh_layer<layout>(comm);
329  }
330  if (proto_layer.has_atan()) {
331  return new atan_layer<layout>(comm);
332  }
333  if (proto_layer.has_exponential()) {
334  return new exponential_layer<layout>(comm);
335  }
336  if (proto_layer.has_identity()) {
337  return new identity_layer<layout>(comm);
338  }
339  if (proto_layer.has_bent_identity()) {
340  return new bent_identity_layer<layout>(comm);
341  }
342  if (proto_layer.has_softplus()) {
343  return new softplus_layer<layout>(comm);
344  }
345  if (proto_layer.has_smooth_relu()) {
346  return new smooth_relu_layer<layout>(comm);
347  }
348  if (proto_layer.has_leaky_relu()) {
349  return new leaky_relu_layer<layout>(comm);
350  }
351  if (proto_layer.has_swish()) {
352  return new swish_layer<layout>(comm);
353  }
354  if (proto_layer.has_elu()) {
355  const auto& params = proto_layer.elu();
356  return new elu_layer<layout>(comm, params.alpha());
357  }
358  if (proto_layer.has_selu()) {
359  const auto& params = proto_layer.selu();
360  const auto& alpha = params.alpha();
361  const auto& scale = params.scale();
362  if (alpha != 0.0 && scale != 0.0) {
363  return new selu_layer<layout>(comm, alpha, scale);
364  } else {
365  return new selu_layer<layout>(comm);
366  }
367  }
368  if (proto_layer.has_power()) {
369  const auto& params = proto_layer.power();
370  return new power_layer<layout>(comm, params.exponent());
371  }
372 
373  // Throw exception if layer has not been constructed
374  err << "could not construct layer " << proto_layer.name();
375  LBANN_ERROR(err.str());
376  return nullptr;
377
