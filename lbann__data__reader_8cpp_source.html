59 bool DataReader::update() {
61  if(m_use_alt_last_mini_batch_size && ((m_current_mini_batch_idx+1) >= (m_num_mini_batches_per_reader-1))) {
62  CurrentPos += m_last_mini_batch_stride;
63  }else {
64  CurrentPos += m_stride;
65  }
66  if (CurrentPos < (int)ShuffledIndices.size()) {
67  m_current_mini_batch_idx++;
68  return true;
69  } else {
70  if (m_shuffle) {
71  std::shuffle(ShuffledIndices.begin(), ShuffledIndices.end(),
72  get_generator());
73  }
74  m_current_mini_batch_idx = 0;
75  CurrentPos = m_base_offset + m_model_offset;
76  return false;
77  }

166 void DataReader::calculate_multi_model_data_distribution(lbann_comm *comm) {
167  int max_mini_batch_size = BatchSize;
168  int num_parallel_readers_per_model = (m_stride / comm->get_num_models()) / max_mini_batch_size;
169  int min_stride_across_models = max_mini_batch_size * comm->get_num_models();
170 
171  m_last_mini_batch_size = max_mini_batch_size;
172 
173  int num_whole_mini_batches_per_model = floor(getNumData() / min_stride_across_models);
174  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
175  int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
176  int per_model_partial_mini_batch_size = (getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(comm->get_num_models());
177  int world_master_remainder_data = 0;
178 
179  // Compute how many full "parallel" mini-batches are available
180  m_last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
181 
182  if(comm->get_rank_in_model() < parallel_readers_with_extra_mini_batch) {
183  num_whole_mini_batches_per_reader += 1;
184  m_last_mini_batch_size = max_mini_batch_size;
185  }
186 
187  m_num_mini_batches_per_reader = num_whole_mini_batches_per_reader;
188 
189  int world_master_remainder_adjustment = getNumData()
190  - (num_whole_mini_batches_per_model * min_stride_across_models)
191  - (per_model_partial_mini_batch_size * comm->get_num_models());
192  if(comm->am_world_master()) {
193  world_master_remainder_data = world_master_remainder_adjustment;
194  world_master_remainder_adjustment = 0;
195  }
196  per_model_partial_mini_batch_size += world_master_remainder_data;
197 
199  if(comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
200  m_num_mini_batches_per_reader++;
201  m_last_mini_batch_size = per_model_partial_mini_batch_size;
202  }
203 
204  if(m_last_mini_batch_size > max_mini_batch_size) { throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size"); }
205 
208 
210  m_last_mini_batch_stride = m_stride;
211 
212  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_reader - 1) * m_stride;
213 
215  if(comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
216  m_last_mini_batch_stride = (m_last_mini_batch_threshold - m_base_offset - m_model_offset - last_mini_batch_offset)
217  + comm->get_model_rank() * per_model_partial_mini_batch_size + world_master_remainder_adjustment;
218  }
219 
220  // cout << "[" << comm->get_rank_in_world() << "] " << comm->get_model_rank() << " model rank, "<< comm->get_rank_in_model() << " rank in model, num_whole_mini_batches_per_model " << num_whole_mini_batches_per_model << " num_whole_mini_batches_per_reader " << num_whole_mini_batches_per_reader << "(m_num_mini_batches_per_reader=" << m_num_mini_batches_per_reader << ") parallel_readers_with_extra_mini_batch " << parallel_readers_with_extra_mini_batch << " partial_mini_batch_size=" << per_model_partial_mini_batch_size << " last mini bath size=" << m_last_mini_batch_size << " world_master_remainder_data=" << world_master_remainder_data << " threshold " << m_last_mini_batch_threshold << " with a last stride of " << m_last_mini_batch_stride << " and stride of " << m_stride << " and there are " << num_parallel_readers_per_model << " parallel readers per model" << " last mini batch offset = " << last_mini_batch_offset << " parallel reader with extra minibatch = " << parallel_readers_with_extra_mini_batch << " model bracket = " << (parallel_readers_with_extra_mini_batch * max_mini_batch_size + per_model_partial_mini_batch_size + world_master_remainder_data) <<" base ofset "<< m_base_offset << " model offset " << m_model_offset <<endl;
221 
222  return;
223 }

226 bool DataReader::saveToCheckpointShared(persist& p, const char* name)
227 {
228  // rank 0 writes the training state file
229  if (p.m_rank == 0) {
230  char fieldname[1024];
231 
232  // record minibatch index
233  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
234  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_mini_batch_idx);
235 
236  // get size of list of training examples
237  int size = ShuffledIndices.size();
238 
239  // record size of ShuffleIndices
240  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
241  p.write_uint64(persist_type::train, fieldname, (uint64_t) size);
242 
243  // TODO: each model may have a different position, need to gather and write these
244  // record current position within training data
245  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
246  p.write_uint64(persist_type::train, fieldname, (uint64_t) CurrentPos);
247 
248  // write list of indices
249  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
250  p.write_int32_contig(persist_type::train, fieldname, &ShuffledIndices[0], (uint64_t) size);
251  }
252 
253  return true;
254 }

257 bool lbann::DataReader::loadFromCheckpointShared(persist& p, const char* name)
258 {
259  // rank 0 reads the training state file
260  if (p.m_rank == 0) {
261  char fieldname[1024];
262 
263  // record minibatch index
264  uint64_t val;
265  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
266  p.read_uint64(persist_type::train, fieldname, &val);
267  m_current_mini_batch_idx = (int) val;
268 
269  // get size of ShuffleIndices
270  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
271  p.read_uint64(persist_type::train, fieldname, &val);
272  int size = (int) val;
273 
274  // get current position within data
275  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
276  p.read_uint64(persist_type::train, fieldname, &val);
277  CurrentPos = (int) val;
278 
279  // resize shuffled index array to hold values
280  ShuffledIndices.resize(size);
281 
282  // read list of indices
283  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
284  p.read_int32_contig(persist_type::train, fieldname, &ShuffledIndices[0], (uint64_t) size);
285  }
286 
287  // broadcast minibatch index
288  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
289 
290  // TODO: with multiple readers, make this a scatter
291  // broadcast current position
292  MPI_Bcast(&CurrentPos, 1, MPI_INT, 0, MPI_COMM_WORLD);
293 
294  // broadcast values from rank 0
295  int size = ShuffledIndices.size();
296  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
297 
298  // resize shuffled index array to hold values
299  if (p.m_rank != 0) {
300  ShuffledIndices.resize(size);
301  }
302 
303  // broadcast index array
304  MPI_Bcast(&ShuffledIndices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
305 
306  return true;
307 }
