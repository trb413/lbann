33 void mean_squared_error::compute_value(const AbsDistMat& predictions,
34  const AbsDistMat& ground_truth) {
35 
36  // Get local matrices and matrix parameters
37  const Mat& predictions_local = predictions.LockedMatrix();
38  const Mat& ground_truth_local = ground_truth.LockedMatrix();
39  const El::Int height = predictions.Height();
40  const El::Int width = predictions.Width();
41  const El::Int local_height = predictions_local.Height();
42  const El::Int local_width = predictions_local.Width();
43 
44  // Compute sum of squared errors with Kahan summation
45  double sum = 0;
46  double correction = 0;
47  for(El::Int col = 0; col < local_width; ++col) {
48  for(El::Int row = 0; row < local_height; ++row) {
49  const double true_val = ground_truth_local(row, col);
50  const double pred_val = predictions_local(row, col);
51  const double error = pred_val - true_val;
52  double term = error * error;
53  term += correction;
54  const double next_sum = sum + term;
55  correction = term - (next_sum - sum);
56  sum = next_sum;
57  }
58  }
59 
60  // Compute mean squared error
61  double mse = sum / (height * width);
62  mse = El::mpi::AllReduce(mse, predictions.DistComm());
63 
64  // Update objective function value
65  add_to_value(mse);
66 
67 }
