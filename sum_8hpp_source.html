126  void fp_compute() override {
127  if(this->m_using_gpus) {
128  #ifndef LBANN_HAS_CUDNN
129  throw lbann_exception("sum_layer: cuDNN not detected");
130  #else
131  const int num_gpus = m_cudnn->get_num_gpus();
132  auto& output_d = this->m_activations_d[0];
133  this->m_cudnn->clear_on_gpus(output_d.get_data(),
134  output_d.get_height(),
135  this->m_mini_batch_size_per_gpu,
136  output_d.get_leading_dim());
137  for (int parent = 0; parent < get_num_parents(); ++parent) {
138  const auto& input_d = this->m_prev_activations_d[parent];
139  for (int gpu = 0; gpu < num_gpus; ++gpu) {
140  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(gpu)));
141  cublas::geam(this->m_cudnn->get_cublas_handle(gpu),
142  CUBLAS_OP_N,
143  CUBLAS_OP_N,
144  input_d.get_height(),
145  this->m_mini_batch_size_per_gpu,
146  m_scaling_factors[parent],
147  input_d.get_locked_data(gpu),
148  input_d.get_leading_dim(),
149  DataType(1),
150  output_d.get_locked_data(gpu),
151  output_d.get_leading_dim(),
152  output_d.get_data(gpu),
153  output_d.get_leading_dim());
154  }
155  }
156  #endif // LBANN_HAS_CUDNN
157  } else {
158  auto& output = get_activations();
159  El::Zero(output);
160  for (int parent = 0; parent < get_num_parents(); ++parent) {
161  const auto& input = get_prev_activations(parent);
162  El::Axpy(m_scaling_factors[parent], input, output);
163  }
164  }
