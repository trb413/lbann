72 bool greedy_layerwise_autoencoder::load_from_checkpoint_shared(persist& p) {
73  // have rank 0 read whether we have a mirror layer inserted
74  struct lbann_model_greedy_layerwise_autoencoder_header header;
75  if (p.get_rank() == 0) {
76  p.read_uint32(persist_type::train, "gla_phase_index", &header.phase_index);
77  p.read_uint32(persist_type::train, "gla_have_mirror", &header.have_mirror);
78  }
79 
80  // TODO: this assumes homogeneous processors
81  // broadcast state from rank 0
82  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
83 
84  // insert the mirror layer if needed
85  uint32_t phase_index = header.phase_index;
86  uint32_t have_mirror = header.have_mirror;
87  if (have_mirror) {
88  // note that this calls setup on the layers,
89  // and setup reinitializes a bunch of values like data reader positions
90  // and optimization layer cache values that we'll overwrite
91  // in load_from_checkpoint_shared below
92  insert_mirror(phase_index);
93  }
94 
95  // read parameters from base class first
96  sequential_model::load_from_checkpoint_shared(p);
97 
98  return true;
99 }

113 // inserts a mirror layer for specified layer index
114 void greedy_layerwise_autoencoder::insert_mirror(uint32_t layer_index) {
115  // compute layer index for mirrror
116  size_t mirror_index = layer_index + 2;
117 
118  // build mirror layer
119  Layer *original_layer = m_layers[layer_index];
120  Layer *mirror_layer = NULL;
121  switch(original_layer->get_data_layout()){
122  case data_layout::MODEL_PARALLEL:
123  mirror_layer = new reconstruction_layer<data_layout::MODEL_PARALLEL>(mirror_index, m_comm, original_layer);
124  break;
125  case data_layout::DATA_PARALLEL:
126  mirror_layer = new reconstruction_layer<data_layout::DATA_PARALLEL>(mirror_index, m_comm, original_layer);
127  break;
128  default:
129  break;
130  }
131 
132  // insert mirror layer into model
133  insert(mirror_index, mirror_layer);
134 
135  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
136  //assume that necessary layer parameters are set e.g., m_num_neurons when layers were constructed
137  setup(layer_index, mirror_index+1); //set up all active layers
138 
139  // set flag to indicate that we have a mirror layer inserted
140  m_have_mirror = 1;
141 }

144 void greedy_layerwise_autoencoder::remove_mirror(uint32_t layer_index) {
145  if (m_have_mirror) {
146  // compute layer index for mirrror
147  size_t mirror_index = layer_index + 2;
148 
149  // drop the mirror layer from the model
150  remove(mirror_index);
151 
152  // call base model setup again to reindex and set appropriate fp and bp input
153  if (m_comm->am_world_master()) {
154  std::cout << "Phase [" << layer_index << "] Done, Reset Layers " << std::endl;
155  for(auto& l:m_layers) {
156  std::cout << "Layer [ " << l->get_index() << "] #NumNeurons: " << l->get_num_neurons() << std::endl;
157  }
158  }
159  setup();
160 
161  // set flag to indicate we've deleted our mirror layer
162  m_have_mirror = 0;
163  }
164 }

179 void greedy_layerwise_autoencoder::train(int num_epochs, int evaluation_frequency) {
180  while(m_end_index < m_layers.size()-1) {
181  set_end_index();
182  train_phase(num_epochs, evaluation_frequency);
183  m_start_index = m_end_index;
184  m_reconstruction_layers.insert(m_reconstruction_layers.begin(),m_layers[m_end_index]);
185  // move on to the next (layerwise) phase
186  ++m_current_phase;
187  }
188  //evaluate and save all layers for i.e., (1) global cost (2) image reconstruction
189  evaluate(execution_mode::testing);
190  reset_phase();
191 }

194 void greedy_layerwise_autoencoder::train_phase(int num_epochs, int evaluation_frequency) {
195  do_train_begin_cbs();
196 
197  // Epoch main loop
198  while (get_cur_epoch() < num_epochs) {
199  // Check if training has been terminated
200  if (get_terminate_training()) {
201  break;
202  }
203 
204  // due to restart, may not always be at start of epoch
205  // use mini batch index in data reader to signify start of epoch
206  if (at_epoch_start()) {
207  ++m_current_epoch;
208  do_epoch_begin_cbs(); // needed for selected callback e.g., dump matrices
209  }
210 
211  //Overide default print callback
212  if (m_comm->am_world_master()) {
213  //std::cout << "-----------------------------------------------------------" << std::endl;
214  //std::cout << "Phase [" << m_current_phase << "] Epoch [" << m_current_epoch << "]" << std::endl;
215  std::cout << "\n Training hidden layer [" << m_current_phase+1 << "] at layer-wise epoch [" << m_current_epoch << "]" << std::endl;
216  std::cout << "-----------------------------------------------------------" << std::endl;
217  }
218 
219  //Print(m_layers[m_start_index]->get_activations());
221  m_execution_mode = execution_mode::training;
222  for (size_t l =0; l < m_layers.size(); l++) {
223  m_layers[l]->set_execution_mode(execution_mode::training);
224  }
225 
226  // Train on mini-batches until data set is traversed
227  // Note: The data reader shuffles the data after each epoch
228  for (auto&& m : m_metrics) {
229  m->reset_metric();
230  }
231  bool finished_epoch;
232  do {
233  finished_epoch = train_mini_batch();
234 
235  // save a checkpoint if needed
236  if (need_checkpoint()) {
237  checkpointShared();
238  }
239  } while(!finished_epoch);
240 
241 
242  //print training reconstruction cost
243  if (m_comm->am_world_master()) {
244  std::cout << "Layer-wise training ";
245  }
246  m_layers[m_end_index]->epoch_print();
247 
248 
249  do_epoch_end_cbs(); //needed for selected callback e.g., dump matrices
250 
251  for (Layer *layer : m_layers) {
252  layer->epoch_reset();
253  } // train epoch end, this reset cost
254 
255  evaluate_phase(execution_mode::validation);
256 
257  //print validation reconstruction cost
258  if (m_comm->am_world_master()) {
259  std::cout << "Layer-wise validation ";
260  }
261  m_layers[m_end_index]->epoch_print();
262 
263  //Reset cost again
264  for (Layer *layer : m_layers) {
265  layer->epoch_reset();
266  } // validation epoch
267 
268  // Reset execution mode back to training
269  m_execution_mode = execution_mode::training;
270  for (Layer *layer : m_layers) {
271  layer->set_execution_mode(execution_mode::training);
272  }
273 
274  // save checkpoint after epoch
275  if (need_checkpoint()) {
276  checkpointShared();
277  }
278  }
279 
280  do_train_end_cbs();
281  m_current_epoch = 0; //reset epoch counter
282 }

286  do_batch_begin_cbs();

288  // Forward propagation

298

301  // Backward propagation
302  do_model_backward_prop_begin_cbs();
303  //@todo optimize: optimize considering input layer
304  for (size_t l = m_end_index+1; l-- > m_start_index;) {

307  do_layer_backward_prop_end_cbs(m_layers[l]);
308  }

310 
313  for (size_t l = m_end_index; l > m_start_index; --l) {
314  m_layers[l]->update();
315  }
316  const bool data_set_processed = m_layers[0]->update();
317 
318 
319  do_batch_end_cbs();
