254 bool lbann::model::need_checkpoint()
255 {
256  /* TODO: since we're using clocks, this requires a bcast for each call,
257  * we could use number of samples processed to make a local decision */
258 
259  // if none of our checkpoint conditions are set, assume we're not checkpointing
260  if (m_checkpoint_epochs == 0 &&
261  m_checkpoint_steps == 0 &&
262  m_checkpoint_secs == 0.0)
263  {
264  return false;
265  }
266 
267  // assume that we won't checkpoint
268  int flag = 0;
269 
270  // if at start of epoch and evenly divide
271  if (flag == 0 && m_checkpoint_epochs > 0) {
272  if (at_epoch_start()) {
273  flag = (int) (m_current_epoch % m_checkpoint_epochs == 0);
274  }
275  }
276 
277  // if our current step is evenly divisable by checkpoint steps,
278  // take a checkpoint
279  if (flag == 0 && m_checkpoint_steps > 0) {
280  flag = (int) (m_current_step % m_checkpoint_steps == 0);
281  }
282 
283  // check the clock if time-based checkpoint is enabled
284  if (flag == 0 && m_checkpoint_secs != 0.0) {
285  // have rank 0 determine whether we should checkpoint
286  // to avoid issues with clock skew, we rely on rank 0 to make decision
287  if (m_rank == 0) {
288  // get the current time
289  double current = MPI_Wtime();
290 
291  // compute time next checkpoint is due
292  double next = m_checkpoint_last + m_checkpoint_secs;
293 
294  // determine whether it's time for a checkpoint
295  flag = (current >= next);
296  }
297 
298  // get flag from rank 0
299  MPI_Bcast(&flag, 1, MPI_INT, 0, MPI_COMM_WORLD);
300  }
301 
302  return (bool)flag;
303 }

306 static bool write_latest(const char* dir, const char* name, int epoch, int train)
307 {
308  // define filename
309  char filename[1024];
310  sprintf(filename, "%s/%s", dir, name);
311 
312  // open the file for writing
313  int fd = lbann::openwrite(filename);
314  if (fd != -1) {
315  lbann::write_uint32(fd, "epoch", (uint32_t)epoch);
316  lbann::write_uint32(fd, "train", (uint32_t)train);
317 
318  // close our file
319  lbann::closewrite(fd, filename);
320  }
321 
322  return true;

326 static bool read_latest(const char* dir, const char* name, int* epochLast, int* trainLast)
327 {
328  // assume we don't have a file, we'll return -1 in that case
329  *epochLast = -1;
330  *trainLast = -1;
331 
332  // define filename
333  char filename[1024];
334  sprintf(filename, "%s/%s", dir, name);
335 
336  // open the file for reading
337  int fd = lbann::openread(filename);
338  if (fd != -1) {
339  // read epoch from file
340  uint32_t epoch;

343 
344  // read epoch from file

346  lbann::read_uint32(fd, "train", &train);
347  *trainLast = train;
348 
349  // close our file
350  lbann::closeread(fd, filename);
351  }
352 
353  return true;

363 bool lbann::model::checkpointShared()
364 {
365  // if the checkpoint directory is not defined, bail
366  if (m_checkpoint_dir.length() == 0) {
367  return false;
368  }
369 
370  // time how long this takes
371  Timer timer;
372 
373  // get checkpoint directory
374  const char* dir = m_checkpoint_dir.c_str();
375 
376  // read current epoch and step counters from model
377  int64_t epoch = m_current_epoch;
378  int64_t step = m_current_step;
379 
380  // let user know we're saving a checkpoint
381  MPI_Barrier(MPI_COMM_WORLD);
382  if (m_rank == 0) {
383  timer.Start();
384  printf("Checkpoint: epoch %d step %d ...\n", epoch, step);
385  fflush(stdout);
386  }
387 
388  // create top level directory
389  //const char* dir = trainParams.ParameterDir.c_str();
390  lbann::makedir(dir);
391 
392  // create subdirectory for this epoch
393  char epochdir[1024];
394  snprintf(epochdir, sizeof(epochdir), "%s/shared.epoch.%d.step.%d", dir, epoch, step);
395 
396  // start our checkpoint
397  lbann::persist p;
398  p.open_checkpoint(epochdir);
399 
400  // call virtual function to checkpoint model state
401  this->save_to_checkpoint_shared(p);
402 
403  // close our checkpoint
404  p.close_checkpoint();
405 
406  uint64_t bytes_count = p.get_bytes();
407 
408  // write epoch number to current file, we do this at the end so as to only update
409  // this file when we know we have a new valid checkpoint
410  if (m_rank == 0) {
411  write_latest(dir, "shared.last", epoch, step);
412  }
413 
414  // stop timer and report cost
415  MPI_Barrier(MPI_COMM_WORLD);
416  if (m_rank == 0) {
417  double secs = timer.Stop();
418  double bw = 0.0;
419  if (secs > 0.0) {
420  bw = ((double) bytes_count) / (secs * 1024.0 * 1024.0);
421  }
422  printf("Checkpoint complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
423  epoch, step, secs, (unsigned long long) bytes_count, bw
424  );
425  fflush(stdout);
426  }
427 
428  // saved a checkpoint, update our last checkpoint time
429  m_checkpoint_last = MPI_Wtime();
430 
431  return true;
432 }

434 bool lbann::model::restartShared()
435 {
436  // if the checkpoint directory is not defined, bail
437  if (m_checkpoint_dir.length() == 0) {
438  return false;
439  }
440 
441  // get top level directory
442  const char* dir = m_checkpoint_dir.c_str();
443 
444  // read epoch number from current file
445  int epoch, step;
446  if (m_rank == 0) {
447  read_latest(dir, "shared.last", &epoch, &step);
448  }
449  MPI_Bcast(&epoch, 1, MPI_INT, 0, MPI_COMM_WORLD);
450  MPI_Bcast(&step, 1, MPI_INT, 0, MPI_COMM_WORLD);
451 
452  // if we couldn't find the latest epoch, just return
453  if (epoch < 0) {
454  return false;
455  }
456 
457  // time how long this takes
458  Timer timer;
459 
460  // let user know we're restarting from a checkpoint
461  MPI_Barrier(MPI_COMM_WORLD);
462  if (m_rank == 0) {
463  timer.Start();
464  printf("Restart: epoch %d ...\n", epoch);
465  fflush(stdout);
466  }
467 
468  // get subdirectory for this epoch
469  char epochdir[1024];
470  sprintf(epochdir, "%s/shared.epoch.%d.step.%d", dir, epoch, step);
471 
472  // open our checkpoint
473  lbann::persist p;
474  p.open_restart(epochdir);
475 
476  // call virtual function to restore model from checkpoint
477  this->load_from_checkpoint_shared(p);
478 
479  // close our checkpoint
480  p.close_restart();
481 
482  uint64_t bytes_count = p.get_bytes();
483 
484  // let user know we've completed reading our restart
485  MPI_Barrier(MPI_COMM_WORLD);
486  if (m_rank == 0) {
487  double secs = timer.Stop();
488  double bw = 0.0;
489  if (secs > 0.0) {
490  bw = ((double) bytes_count) / (secs * 1024.0 * 1024.0);
491  }
492  printf("Restart complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
493  epoch, step, secs, (unsigned long long) bytes_count, bw
494  );
495  fflush(stdout);
496  }
497 
498  return true;
