86  void fp_compute() override {
87  if(this->m_using_gpus) {
88  #ifndef LBANN_HAS_CUDNN
89  LBANN_ERROR("cuDNN not detected");
90  #else
91 
92  // GPU data
93  const auto& input_d = this->m_prev_activations_d[0];
94  auto& output_d = this->m_activations_d[0];
95  const auto& input_size = get_num_prev_neurons();
96  const auto& mini_batch_size = m_mini_batch_size_per_gpu;
97  const auto& input_ldim = input_d.get_leading_dim();
98  const int num_gpus = m_cudnn->get_num_gpus();
99 
100  // Stop early if possible
101  if (mini_batch_size == 0) { return; }
102 
103  // Apply reduction on GPU
104  switch (m_mode) {
105  case reduction_mode::SUM:
106  {
107  cudnn::matrix ones_d(this->m_cudnn);
108  ones_d.attach_to_work_spaces(input_size);
109  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), input_size);
110  for (int i = 0; i < num_gpus; ++i) {
111  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
112  cublas::gemv(this->m_cudnn->get_cublas_handle(i),
113  CUBLAS_OP_T,
114  input_size, mini_batch_size,
115  DataType(1),
116  input_d.get_locked_data(i), input_ldim,
117  ones_d.get_locked_data(i), 1,
118  DataType(0),
119  output_d.get_data(i), 1);
120  }
121  }
122  break;
123  case reduction_mode::AVERAGE:
124  {
125  cudnn::matrix ones_d(this->m_cudnn);
126  ones_d.attach_to_work_spaces(input_size);
127  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), input_size);
128  for (int i = 0; i < num_gpus; ++i) {
129  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
130  cublas::gemv(this->m_cudnn->get_cublas_handle(i),
131  CUBLAS_OP_T,
132  input_size, mini_batch_size,
133  DataType(1) / input_size,
134  input_d.get_locked_data(i), input_ldim,
135  ones_d.get_locked_data(i), 1,
136  DataType(0),
137  output_d.get_data(i), 1);
138  }
139  }
140  break;
141  default:
142  LBANN_ERROR("invalid reduction mode");
143  }
144  #endif // LBANN_HAS_CUDNN
145  } else {
146  // Apply reduction on CPU
147  const auto& local_input = get_local_prev_activations();
148  auto& local_output = get_local_activations();
149  switch (m_mode) {
150  case reduction_mode::SUM:
151  El::ColumnSum(local_input, local_output);
152  break;
153  case reduction_mode::AVERAGE:
154  El::ColumnSum(local_input, local_output);
155  local_output *= DataType(1) / get_num_prev_neurons();
156  break;
157  default:
158  LBANN_ERROR("invalid reduction mode");
159  }
160  }
161  }

163  void bp_compute() override {
164  if(this->m_using_gpus) {
165  #ifndef LBANN_HAS_CUDNN
166  LBANN_ERROR("cuDNN not detected");
167  #else
168 
169  // GPU data
170  const auto& gradient_wrt_output_d = this->m_prev_error_signals_d[0];
171  auto& gradient_wrt_input_d = this->m_error_signals_d[0];
172  const auto& input_size = get_num_prev_neurons();
173  const auto& mini_batch_size = m_mini_batch_size_per_gpu;
174  const auto& gradient_wrt_input_ldim = gradient_wrt_input_d.get_leading_dim();
175  const int num_gpus = m_cudnn->get_num_gpus();
176 
177  // Stop early if possible
178  if (mini_batch_size == 0) { return; }
179 
180  // Apply reduction on GPU
181  switch (m_mode) {
182  case reduction_mode::SUM:
183  {
184  cudnn::matrix ones_d(this->m_cudnn);
185  ones_d.attach_to_work_spaces(input_size);
186  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), input_size);
187  for (int i = 0; i < num_gpus; ++i) {
188  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
189  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
190  CUBLAS_OP_N, CUBLAS_OP_N,
191  input_size, mini_batch_size, 1,
192  DataType(1),
193  ones_d.get_locked_data(i), input_size,
194  gradient_wrt_output_d.get_locked_data(i), 1,
195  DataType(0),
196  gradient_wrt_input_d.get_data(i),
197  gradient_wrt_input_ldim);
198  }
199  }
200  break;
201  case reduction_mode::AVERAGE:
202  {
203  cudnn::matrix ones_d(this->m_cudnn);
204  ones_d.attach_to_work_spaces(input_size);
205  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), input_size);
206  for (int i = 0; i < num_gpus; ++i) {
207  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
208  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
209  CUBLAS_OP_N, CUBLAS_OP_T,
210  input_size, mini_batch_size, 1,
211  DataType(1) / mini_batch_size,
212  ones_d.get_locked_data(i), input_size,
213  gradient_wrt_output_d.get_locked_data(i), 1,
214  DataType(0),
215  gradient_wrt_input_d.get_data(i),
216  gradient_wrt_input_ldim);
217  }
218  }
219  break;
220  default:
221  LBANN_ERROR("invalid reduction mode");
222  }
223 
224  #endif // LBANN_HAS_CUDNN
225  } else {
226  // Apply reduction on CPU
227  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
228  auto& local_gradient_wrt_input = get_local_error_signals();
229  switch (m_mode) {
230  case reduction_mode::SUM:
231  El::IndexDependentMap(local_gradient_wrt_input,
232  (std::function<DataType(El::Int,El::Int,const DataType&)>)
233  ([this,&local_gradient_wrt_output]
234  (El::Int r, El::Int c,const DataType& dx)
235  ->DataType {
236  return dx + local_gradient_wrt_output(0, c);
237  }));
238  break;
239  case reduction_mode::AVERAGE:
240  {
241  const DataType scale = DataType(1) / get_num_prev_neurons();
242  El::IndexDependentMap(local_gradient_wrt_input,
243  (std::function<DataType(El::Int,El::Int,const DataType&)>)
244  ([this,&local_gradient_wrt_output,scale]
245  (El::Int r, El::Int c,const DataType& dx)
246  ->DataType {
247  return dx + scale * local_gradient_wrt_output(0, c);
248  }));
249  }
250  break;
251  default:
252  LBANN_ERROR("invalid reduction mode");
253  }
254  }
255  }
