83 void metric::setup(model& m) {
84 
85  // Set target layer if needed
86  if (m_target_layer == nullptr) {
87  const std::vector<Layer*> layers = m.get_layers();
88  for (int i = layers.size() - 1; i >= 0; --i) {
89  const generic_target_layer *target = dynamic_cast<const generic_target_layer*>(layers[i]);
90  if (target != nullptr) {
91  m_target_layer = target;
92  break;
93  }
94  }
95  if (m_target_layer == nullptr) {
96  std::stringstream err;
97  err << __FILE__ << " " << __LINE__ << " :: "
98  << "could not setup metric with target layer";
99  throw lbann_exception(err.str());
100  }
101  }
102 
103 }

105 EvalType metric::evaluate(execution_mode mode,
106  int mini_batch_size) {
107 
108  // Check if target layer pointer has been setup
109  if (m_target_layer == nullptr) {
110  std::stringstream err;
111  err << __FILE__ << " " << __LINE__ << " :: "
112  << "attempted to evaluate metric without setting a target layer";
113  throw lbann_exception(err.str());
114  }
115 
116  // Evaluate objective function
117  const auto& start = get_time();
118  const EvalType total_value = evaluate_compute(m_target_layer->get_prediction(),
119  m_target_layer->get_ground_truth());
120  m_evaluate_time += get_time() - start;
121 
122  // Record result in statistics and return
123  m_statistics[mode].add_value(total_value, mini_batch_size);
124  return total_value / mini_batch_size;
125 
126 }
