387 uint lbann::sequential_model::add(const std::string layer_name,
388  data_layout data_dist,
389  const int layer_dim,
390  const activation_type activation,
391  const weight_initialization init,
392  std::vector<regularizer*> regularizers)
393 {
394  const int layer_index = m_layers.size();
395  optimizer *opt = create_optimizer();
396 
397  // Get properties of previous layer
398  int prev_layer_dim = -1;
399  int prev_layer_index = -1;
400  if(m_layers.size() != 0) {
401  Layer* prev_layer = m_layers.back();
402  prev_layer_dim = prev_layer->NumNeurons;
403  prev_layer_index = prev_layer->Index;
404  }
405 
406  if (comm->am_model_master()) {
407  std::cout << "Adding a layer with input " << prev_layer_dim

409  << " prev layer index " << prev_layer_index << std::endl;
410  }
411 
412  if(layer_name.compare("FullyConnected") == 0) {
413  Layer* new_layer
414  = layer_fac->create_layer<FullyConnectedLayer>("FullyConnected",
415  data_dist,
416  layer_index,
417  prev_layer_dim,
418  layer_dim,
419  m_mini_batch_size,
420  activation, init,
421  comm,
422  opt,
423  regularizers);
424  m_layers.push_back(new_layer);
425  } else if(layer_name.compare("Softmax") == 0) {
426  Layer* new_layer
427  = layer_fac->create_layer<SoftmaxLayer>("Softmax",
428  data_dist,
429  layer_index,
430  prev_layer_dim,
431  layer_dim,
432  m_mini_batch_size,
433  init,
434  comm,
435  opt);
436  m_layers.push_back(new_layer);
437  } else {
438  std::cout << "Unknown layer type " << layer_name << std::endl;
439  }
440 
441  return layer_index;
442 }

499 {

504  // Get properties from adjacent layers

506  set_bp_input(start_index, end_index);
507 
508  // Setup each layer
