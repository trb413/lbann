150 bool cv_process::postprocess(cv::Mat& image)
151 {
152  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
153 
154  bool ok = true;
155 
156  if (m_transform3 && m_transform3->is_enabled()) {
157  // must have been enabled by calling determine_inverse_transform()
158  ok = m_transform3->apply(image);
159  _LBANN_MILD_EXCEPTION(!ok, "inverse transform3 has failed!", false);
160  }
161 
162  // Inverse the early-normalization in case that there was something
163  // (transform3) had to be done after normalization during preprocessing.
164  // Otherwise, the transform will be done during copying via scaling.
165 
166  if (m_transform3) {
167  if (m_normalizer && m_normalizer->determine_inverse_transform()) {
168  ok = m_normalizer->apply(image);
169  _LBANN_MILD_EXCEPTION(!ok, "inverse normalization has failed!", false);
170  }
171  }
172 
173  #if 0
174  const std::vector<cv_normalizer::channel_trans_t> ntrans = get_transform_normalize();
175  for (size_t i=0u; i < ntrans.size(); ++i) {
176  std::cout << "postprocess scaling: " << ntrans[i].first << ' ' << ntrans[i].second << std::endl;
177  }
178  #endif
179 
180  if (m_transform2 && m_transform2->is_enabled()) {
181  ok = m_transform2->apply(image);
182  _LBANN_MILD_EXCEPTION(!ok, "inverse transform2 has failed!", false);
183  }
184 
185  if (to_flip())
186  cv::flip(image, image, how_to_flip());
187 
188  if (m_transform1 && m_transform1->is_enabled()) {
189  ok = m_transform1->apply(image);
190  _LBANN_MILD_EXCEPTION(!ok, "inverse transform1 has failed!", false);
191  }
192 
193  return ok;
194 }
