32 lbann::metrics::statistics* lbann::metrics::metric::get_statistics(execution_mode mode) {
33  statistics *stats;
34 
35  switch(mode) {
36  case execution_mode::training:
37  stats = &m_training_stats;
38  break;
39  case execution_mode::validation:
40  stats = &m_validation_stats;
41  break;
42  case execution_mode::testing:
43  stats = &m_testing_stats;
44  break;
45  default:
46  throw lbann_exception("Invalid execution mode");
47  };
48  return stats;
49 }
