37 void mesh_reader::load() {
38  if (m_data_height == 0 || m_data_width == 0) {
39  throw lbann_exception("mesh_reader: data shape must be non-zero");
40  }
41  // Compute total number of samples based on number of targets.
42  std::vector<std::string> matches = glob(
43  get_file_dir() + m_target_name + m_suffix + "/*.bin");
44  if (matches.size() == 0) {
45  throw lbann_exception("mesh_reader: could not find any targets");
46  }
47  m_num_samples = matches.size();
48  // Set up buffers to load data into.
49  m_load_bufs.resize(omp_get_max_threads());
50  for (auto&& buf : m_load_bufs) {
51  buf.resize(m_data_height * m_data_width);
52  }
53  // Set up the format string.
54  if (std::pow(10, m_index_length) <= m_num_samples) {
55  throw lbann_exception("mesh_reader: index length too small");
56  }
57  m_index_format_str = "%0" + std::to_string(m_index_length) + "d";
58  // Set up to record flipping if needed.
59  if (m_random_flips) {
60  m_flip_choices.resize(m_num_samples);
61  }
62  // Reset indices.
63  m_shuffled_indices.resize(m_num_samples);
64  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
65  select_subset_of_data();
66 }

68 bool mesh_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
69  if (m_random_flips) {
70  fast_rng_gen& gen = get_fast_generator();
71  std::uniform_int_distribution<int> dist(0, 1);
72  m_flip_choices[data_id].first = dist(gen);
73  m_flip_choices[data_id].second = dist(gen);
74  }
75  for (size_t i = 0; i < m_channels.size(); ++i) {
76  Mat X_view = El::View(
77  X, El::IR(i*m_data_height*m_data_width, (i+1)*m_data_height*m_data_width),
78  El::IR(mb_idx));
79  load_file(data_id, m_channels[i], X_view);
80  }
81  return true;
82 }

90 void mesh_reader::load_file(int data_id, const std::string channel, Mat& mat) {
91  const std::string filename = construct_filename(channel, data_id);
92  std::ifstream f(filename, std::ios::binary);
93  if (f.fail()) {
94  throw lbann_exception("mesh_reader: failed to open " + filename);
95  }
96  // Load into a local buffer.
97  float* buf = m_load_bufs[omp_get_thread_num()].data();
98  if (!f.read((char*) buf, m_data_height * m_data_width * sizeof(float))) {
99  throw lbann_exception("mesh_reader: failed to read " + filename);
100  }
101  if (std::is_same<float, DataType>::value) {
102  // Need to transpose from row-major to column-major order.
103  Mat tmp_mat(m_data_width, m_data_height, buf, m_data_width);
104  Mat mat_reshape(m_data_height, m_data_width, mat.Buffer(), m_data_height);
105  El::Transpose(tmp_mat, mat_reshape);
106  // Flip if needed.
107  if (m_random_flips) {
108  if (m_flip_choices[data_id].first) {
109  horizontal_flip(mat_reshape);
110  }
111  if (m_flip_choices[data_id].second) {
112  vertical_flip(mat_reshape);
113  }
114  }
115  } else {
116  // Need to transpose and convert from float. Not yet supported.
117  throw lbann_exception("mesh_reader: does not support DataType != float");
118  }
119 }

121 std::string mesh_reader::construct_filename(std::string channel, int data_id) {
122  std::string filename = get_file_dir() + channel + m_suffix + "/" + channel;
123  char idx[m_index_length + 1];
124  std::snprintf(idx, m_index_length + 1, m_index_format_str.c_str(), data_id);
125  return filename + std::string(idx) + ".bin";
126 }

128 void mesh_reader::horizontal_flip(Mat& mat) {
129  // TODO: Could probably optimize this for better locality.
130  const El::Int height = mat.Height();
131  const El::Int width = mat.Width();
132  for (El::Int row = 0; row < height; ++row) {
133  for (El::Int col = 0; col < (width / 2); ++col) {
134  DataType tmp = mat(row, col);
135  mat(row, col) = mat(row, width - col - 1);
136  mat(row, width - col - 1) = tmp;
137  }
138  }
139 }
