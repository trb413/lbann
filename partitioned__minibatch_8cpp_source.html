114  if(data_reader == NULL) { return; }
115  // If the data reader does not have any data bail out (e.g. unused validation reader)
116  if(data_reader->get_use_percent() == double(0.0)) { return; }
117 
119  if(max_mini_batch_size > data_reader->get_num_data()) {
120  max_mini_batch_size = data_reader->get_num_data();
121  }
122 
124  int num_parallel_readers_per_model = compute_max_num_parallel_readers(data_reader->get_num_data(), max_mini_batch_size, m_requested_max_num_parallel_readers);
125  data_reader->set_num_parallel_readers(num_parallel_readers_per_model);
126  if(num_parallel_readers_per_model == 0
127  || (num_parallel_readers_per_model != m_comm->get_procs_per_model() && num_parallel_readers_per_model != max_mini_batch_size)) {
128  throw lbann_exception(
129  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
130  + " :: partitioned_minibatch: number of parallel readers is " + std::to_string(num_parallel_readers_per_model)
131  + " and there are " + std::to_string(m_comm->get_procs_per_model()) + " processes in the model");
132  }
133 
135  int batch_stride = m_comm->get_num_models() * max_mini_batch_size;
136  int base_offset = m_comm->get_rank_in_model();
137  int model_offset = m_comm->get_model_rank() * max_mini_batch_size;
139  data_reader->set_mini_batch_size(max_mini_batch_size);
140  data_reader->set_mini_batch_stride(batch_stride);
141  data_reader->set_sample_stride(num_parallel_readers_per_model);
142  data_reader->set_iteration_stride(1);
144  data_reader->set_base_offset(base_offset);
145  data_reader->set_model_offset(model_offset);
146  data_reader->set_initial_position();
147 
148  int min_stride_across_models = max_mini_batch_size * m_comm->get_num_models();
149 
150  data_reader->set_global_mini_batch_size(min_stride_across_models);
151 
152  data_reader->set_last_mini_batch_size(max_mini_batch_size);
153  data_reader->set_global_last_mini_batch_size(min_stride_across_models);
154 
155  int num_whole_mini_batches_per_model = floor(data_reader->get_num_data() / min_stride_across_models);
156  int global_partial_mini_batch_size = data_reader->get_num_data() - (num_whole_mini_batches_per_model * min_stride_across_models);
157  int per_model_partial_mini_batch_size = global_partial_mini_batch_size / m_comm->get_num_models();
158  int world_master_remainder_data = 0;
159 
160  // Compute how many full "parallel" mini-batches are available
161  int last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
162 
163  int world_master_remainder_adjustment = data_reader->get_num_data()
164  - (num_whole_mini_batches_per_model * min_stride_across_models)
165  - (per_model_partial_mini_batch_size * m_comm->get_num_models());
166  if(m_comm->get_model_rank() == 0) {
167  world_master_remainder_data = world_master_remainder_adjustment;
168  world_master_remainder_adjustment = 0;
169  }
170  per_model_partial_mini_batch_size += world_master_remainder_data;
171 
172  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
173  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
174  data_reader->set_global_last_mini_batch_size(global_partial_mini_batch_size);
175  }
176 
177  if(global_partial_mini_batch_size != 0) {
178  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model+1);
179  }else {
180  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model);
181  }
182 
183  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
184  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
185  }
186 
189 
191  data_reader->set_last_mini_batch_stride(data_reader->get_batch_stride());
192 
193  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_model - 1) * data_reader->get_batch_stride();
194 
196  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
197  data_reader->set_last_mini_batch_stride((last_mini_batch_threshold - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset) + m_comm->get_model_rank() * per_model_partial_mini_batch_size + m_comm->get_rank_in_model());
198  }
199 
200  //cout << "[" << m_comm->get_rank_in_world() << "] " << m_comm->get_model_rank() << " model rank, "<< m_comm->get_rank_in_model() << " rank in model, num_whole_mini_batches_per_model " << num_whole_mini_batches_per_model << " num_whole_mini_batches_per_reader " << num_whole_mini_batches_per_reader << "(m_num_mini_batches_per_reader=" << data_reader->get_num_mini_batches_per_reader() << ") parallel_readers_with_extra_mini_batch " << /*parallel_readers_with_extra_mini_batch <<*/ " partial_mini_batch_size=" << per_model_partial_mini_batch_size << " last mini bath size=" << data_reader->get_last_mini_batch_size() << " world_master_remainder_data=" << world_master_remainder_data << " threshold " << data_reader->get_last_mini_batch_threshold() << " with a last stride of " << data_reader->get_last_mini_batch_stride() << " and stride of " << data_reader->get_batch_stride() << " and there are " << num_parallel_readers_per_model << " parallel readers per model" << " last mini batch offset = " << last_mini_batch_offset << " parallel reader with extra minibatch = " << /*parallel_readers_with_extra_mini_batch << */" model bracket = " << (/*parallel_readers_with_extra_mini_batch **/ max_mini_batch_size + per_model_partial_mini_batch_size + world_master_remainder_data) <<" base ofset "<< data_reader->get_base_offset() << " model offset " << data_reader->get_model_offset() <<endl;
