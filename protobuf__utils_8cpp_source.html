38 void protobuf_utils::parse_prototext_filenames_from_command_line(
39  bool master,
40  int argc,
41  char **argv,
42  std::vector<prototext_fn_triple> &names) {
43  std::vector<std::string> models;
44  std::vector<std::string> optimizers;
45  std::vector<std::string> readers;
46  for (int k=1; k<argc; k++) {
47  std::string s(argv[k]);
48  if (s[0] != '-' or s[1] != '-') {
49  std::cerr << "badly formed cmd line param; must begin with '--': " << s << std::endl;
50  exit(1);
51  }
52  if (s.find(',') != std::string::npos) {
53  std::stringstream err;
54  err << __FILE__ << __LINE__ << " :: "
55  << " badly formed param; contains ','; " << s << "\n"
56  << "possibly you left out '{' or '}' or both ??\n";
57  throw lbann_exception(err.str());
58  }
59 
60  size_t equal_sign = s.find("=");
61  if (equal_sign != std::string::npos) {
62  std::string which = s.substr(2, equal_sign-2);
63  std::string fn = s.substr(equal_sign+1);
64  if (which == "loadme") {
65  models.push_back(fn);
66  }
67  if (which == "model") {
68  models.push_back(fn);
69  }
70  if (which == "reader") {
71  readers.push_back(fn);
72  }
73  if (which == "optimizer") {
74  optimizers.push_back(fn);
75  }
76  }
77  }
78 
79  size_t n = models.size();
80  if (! (optimizers.size() == 1 || optimizers.size() == n)) {
81  std::stringstream err;
82  err << __FILE__ << " " << __LINE__ << " :: "
83  << " you specified " << n << " model filenames, and " << optimizers.size()
84  << " optimizer filenames; you must specify either one or "<< n
85  << " optimizer filenames";
86  throw lbann_exception(err.str());
87  }
88  if (! (readers.size() == 1 || readers.size() == n)) {
89  std::stringstream err;
90  err << __FILE__ << " " << __LINE__ << " :: "
91  << " you specified " << n << " model filenames, and " << readers.size()
92  << " reader filenames; you must specify either one or "<< n
93  << " reader filenames";
94  throw lbann_exception(err.str());
95  }
96 
97  names.clear();
98  for (size_t i=0; i<models.size(); i++) {
99  prototext_fn_triple t;
100  t.model = models[i];
101  if (readers.size() == 1) {
102  t.reader = readers[0];
103  } else {
104  t.reader = readers[i];
105  }
106  if (optimizers.size() == 1) {
107  t.optimizer = optimizers[0];
108  } else {
109  t.optimizer = optimizers[i];
110  }
111  names.push_back(t);
112  }
113 }

116 void protobuf_utils::read_in_prototext_files(
117  bool master,
118  std::vector<prototext_fn_triple> &names,
119  std::vector<lbann_data::LbannPB*> &models_out) {
120  models_out.clear();
121  for (auto t : names) {
122  lbann_data::LbannPB *pb = new lbann_data::LbannPB;
123  if (t.model != "none")
124  read_prototext_file(t.model.c_str(), *pb, master);
125  if (t.reader != "none") {
126  lbann_data::LbannPB p;
127  read_prototext_file(t.reader.c_str(), p, master);
128  pb->MergeFrom(p);
129  }
130  if (t.optimizer != "none") {
131  lbann_data::LbannPB p;
132  read_prototext_file(t.optimizer.c_str(), p, master);
133  pb->MergeFrom(p);
134  }
135  models_out.push_back(pb);
136  }
137 }

139 void protobuf_utils::load_prototext(

142  char **argv,
143  std::vector<lbann_data::LbannPB *> &models_out) {

151  << " failed to load any prototext files";
152  throw lbann_exception(err.str());
153  }
154  }
155  verify_prototext(master, models_out);
156 }

158 void protobuf_utils::verify_prototext(bool master, const std::vector<lbann_data::LbannPB *> &models) {
159  if (master) {
160  std::cout << "protobuf_utils::verify_prototext; starting verify for " << models.size() << " models\n";
161  }
162  for (size_t j=0; j<models.size(); j++) {
163  bool is_good = true;
164  lbann_data::LbannPB *t = models[j];
165  if (! t->has_data_reader()) {
166  is_good = false;
167  if (master) {
168  std::cerr << "model #" << j << " is missing data_reader\n";
169  }
170  }
171  if (! t->has_model()) {
172  is_good = false;
173  if (master) {
174  std::cerr << "model #" << j << " is missing model\n";
175  }
176  }
177  if (! t->has_optimizer()) {
178  is_good = false;
179  if (master) {
180  std::cerr << "model #" << j << " is missing optimizer\n";
181  }
182  }
183 
184  if (! is_good) {
185  if (master) {
186  std::stringstream err;
187  err << __FILE__ << __LINE__ << " :: "
188  << " prototext is missing reader, optimizer, and/or model;\n"
189  << " please check your command line\n";
190  throw lbann_exception(err.str());
191  }
192  }
193  }
194 }
