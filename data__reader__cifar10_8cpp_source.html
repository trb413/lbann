48 void cifar10_reader::load() {
49  //open data file
50  std::string image_dir = get_file_dir();
51  std::string filename = get_data_filename();
52  std::string path = image_dir + "/" + filename;
53  std::ifstream in(path, std::ios::binary);
54  if (!in.good()) {
55  throw lbann_exception(
56  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
57  " :: failed to open " + path + " for reading");
58  }
59 
60  //get number of images, with error checking
61  int len = get_linearized_data_size() + 1; //should be 3073
62  in.seekg(0, in.end);
63  std::streampos fs = in.tellg();
64  in.seekg(0, in.beg);
65  if (fs % len != 0) {
66  throw lbann_exception(
67  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
68  " :: fs % len != 0; fs: " + std::to_string(fs) + " len: " +
69  std::to_string(len));
70  }
71 
72  //reserve space for string images
73  int num_images = fs / len;
74  m_data.resize(num_images);
75  for (auto & h : m_data) {
76  h.resize(len);
77  }
78 
79  //read in the images; each image is 1 byte, which is the
80  //label (0-9), and 3072 pixels
81  for (auto & h : m_data) {
82  in.read((char *)&(h[0]), len);
83  }
84  in.close();
85 
86  m_shuffled_indices.resize(m_data.size());
87  for (size_t n = 0; n < m_data.size(); n++) {
88  m_shuffled_indices[n] = n;
89  }
90 
91  select_subset_of_data();
92 }

94 bool cifar10_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
95  for (size_t p = 1; p<m_data[data_id].size(); p++) {
96  X.Set(p-1, mb_idx, m_data[data_id][p]);
97  }
98 
99  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
100  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
101  normalize(pixel_col, m_image_num_channels);
102  pixel_noise(pixel_col); //add noise to image, disable by default
103  return true;
104 }
