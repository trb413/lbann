40 void cifar10_reader::load() {
41  //open data file
42  std::string image_dir = get_file_dir();
43  std::string filename = get_data_filename();
44  std::string path = image_dir + "/" + filename;
45  std::ifstream in(path, std::ios::binary);
46  if (not in.good()) {
47  throw lbann_exception(
48  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
49  " :: failed to open " + path + " for reading");
50  }
51 
52  //get number of images, with error checking
53  int len = get_linearized_data_size() + 1; //should be 3073
54  in.seekg(0, in.end);
55  std::streampos fs = in.tellg();
56  in.seekg(0, in.beg);
57  if (fs % len != 0) {
58  throw lbann_exception(
59  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
60  " :: fs % len != 0; fs: " + std::to_string(fs) + " len: " +
61  std::to_string(len));
62  }
63 
64  //reserve space for string images
65  int num_images = fs / len;
66  m_data.resize(num_images);
67  for (size_t h=0; h<m_data.size(); h++) {
68  m_data[h].resize(len);
69  }
70 
71  //read in the images; each image is 1 byte, which is the
72  //label (0-9), and 2072 pixels
73  for (size_t h=0; h<m_data.size(); h++) {
74  in.read((char *)&(m_data[h][0]), len);
75  }
76  in.close();
77 
78  m_shuffled_indices.resize(m_data.size());
79  for (size_t n = 0; n < m_data.size(); n++) {
80  m_shuffled_indices[n] = n;
81  }
82 
83  select_subset_of_data();
84 }

86 bool cifar10_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
87  for (size_t p = 1; p<m_data[data_id].size(); p++) {
88  X.Set(p-1, mb_idx, m_data[data_id][p]);
89  }
90 
91  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
92  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
93  normalize(pixel_col, m_image_num_channels);
94  pixel_noise(pixel_col); //add noise to image, disable by default
95  return true;
96 }
