61 EvalType layer_metric::evaluate(execution_mode mode,
62  int mini_batch_size) {
63 
64  // Check if evaluation layer pointer has been setup
65  if (m_evaluation_layer == nullptr) {
66  LBANN_ERROR("attempted to evaluate metric without setting a target layer");
67  }
68 
69  const auto& start = get_time();
70  EvalType total_value = 0;
71  auto&& eval_dp = dynamic_cast<evaluation_layer<data_layout::DATA_PARALLEL>*>(m_evaluation_layer);
72  auto&& eval_mp = dynamic_cast<evaluation_layer<data_layout::MODEL_PARALLEL>*>(m_evaluation_layer);
73  if (eval_dp) { total_value = eval_dp->get_value(true) * mini_batch_size; }
74  if (eval_mp) { total_value = eval_mp->get_value(true) * mini_batch_size; }
75  if (m_unit == "%") { total_value *= 100; }
76  get_evaluate_time() += get_time() - start;
77 
78  // Record result in statistics and return
79  get_statistics()[mode].add_value(total_value, mini_batch_size);
80  return total_value / mini_batch_size;
81 
82 }
