64 void lbann::lbann_callback_debug::on_batch_evaluate_begin(model *m) {
65  if(m_debug_phase == execution_mode::invalid || m_debug_phase == m->get_execution_mode()) {
66  int64_t step;
67  switch(m->get_execution_mode()) {
68  case execution_mode::validation:
69  step = m->get_cur_validation_step();
70  break;
71  case execution_mode::testing:
72  step = m->get_cur_testing_step();
73  break;
74  default:
75  throw lbann_exception("Illegal execution mode in evaluate forward prop function");

78  }
