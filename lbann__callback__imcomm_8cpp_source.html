48 void lbann_callback_imcomm::setup(model* m) {
49  if (ct != NONE) {
50  bool add = layer_indices.size() == 0;
51  std::vector<Layer*>& layers = m->get_layers();
52  for (Layer* layer : layers) {
53  uint idx = layer->get_index();
54  if (add || layer_indices.find(idx) != layer_indices.end()) {
55  // Ensure index is present (overwrites if already there).
56  layer_indices.insert(idx);
57  // Update the layer's effective mini-batch size so it averages properly.
58  layer->set_effective_minibatch_size(
59  layer->get_minibatch_size() * m->get_comm()->get_num_models());
60  // Skip adding matrices when we don't need to.
61  if (!ct_does_quantization()) continue;
62  // TODO: handle case where weights_gradient is in other matrix distribution
63  DistMat& weights_gradient = (DistMat&) layer->get_weights_biases_gradient();
64  quantization_errors.emplace(idx, Mat{});
65  im_quantization_errors.emplace(idx, Mat{});
66  if (ct == ONEBIT_QUANTIZATION) {
67  // Set up gradient history and SGD optimizer for one-bit quantization.
68  gradhistories.emplace(idx, Mat{});
69  if (layer->m_optimizer != nullptr) {
70  if (typeid(*(layer->m_optimizer)) != typeid(adagrad)) {
71  throw lbann_exception(
72  "lbann_callback_imcomm: Cannot do one-bit quantization for "
73  "layer that does not use Adagrad");
74  }
75  // TODO: This leaks the old optimizer.
76  layer->m_optimizer = new sgd(layer->comm,
77  layer->m_optimizer->get_learning_rate(),
78  0.0f,
79  0.0f,
80  false);
81  layer->m_optimizer->setup(layer->m_weights);
82  }
83  }
84  }
85  }
86  }

89 void lbann_callback_imcomm::on_epoch_end(model* m) {
90  lbann_comm* comm = m->get_comm();
91  if (comm->get_num_models() == 1 ||
92  m->get_execution_mode() != execution_mode::training) {
93  return; // No point with only one model.
94  }
95  if (ct_does_quantization()) {
96  std::vector<Layer*>& layers = m->get_layers();
97  for (size_t l = 0; l < layers.size(); ++l) {
98  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
99  continue;
100  }
101  comm->intermodel_sum_matrix(quantization_errors[l]);
102  // TODO: handle case where weights_gradient is in other matrix distribution
103  DistMat& weights_gradient = (DistMat&) layers[l]->get_weights_biases_gradient();
104  Mat& local_mat = weights_gradient.Matrix();
105  local_mat = quantization_errors[l];
106  // Apply optimizer update again.
107  layers[l]->update();
108  quantization_errors[l].Empty();
109  }
110  }
111 }

113 void lbann_callback_imcomm::on_backward_prop_end(model* m) {
114  lbann_comm* comm = m->get_comm();
115  if (comm->get_num_models() == 1 ||
116  m->get_execution_mode() != execution_mode::training) {
117  return; // No point with only one model.
118  }
119  std::vector<Layer*>& layers = m->get_layers();
120  for (size_t l = 0; l < layers.size(); ++l) {
121  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
122  continue;
123  }
124  double start_time = get_time();
125  // TODO: handle case where weights_gradient is in other matrix distribution
126  DistMat& weights_gradient = (DistMat&) layers[l]->get_weights_biases_gradient();
127  switch (ct) {
128  case NONE:
129  break;
130  case NORMAL:
131  comm->intermodel_sum_matrix(weights_gradient);
132  break;
133  case NORMAL_AR:
134  quantizer.intermodel_sum(comm, weights_gradient);
135  break;
136  case ONEBIT_QUANTIZATION:
137  quantizer.intermodel_sum_quantized(
138  comm, weights_gradient, quantization_errors[l], im_quantization_errors[l], true,
139  &(gradhistories[l]));
140  break;
141  case THRESH_QUANTIZATION:
142  // TODO: Don't hardcode thresholds.
143  quantizer.intermodel_sum_threshold_quantized(
144  comm, weights_gradient, quantization_errors[l], 0.01f, -0.01f,
145  im_quantization_errors[l], false);
146  break;
147  case COMPRESSED_THRESH_QUANTIZATION:
148  // TODO: Don't hardcode thresholds.
149  quantizer.intermodel_sum_threshold_quantized(
150  comm, weights_gradient, quantization_errors[l], 0.01f, -0.01f,
151  im_quantization_errors[l], true);
152  break;
153  case ADAPTIVE_THRESH_QUANTIZATION:
154  // TODO: Don't hardcode proportion.
155  quantizer.intermodel_sum_adaptive_threshold_quantized(
156  comm, weights_gradient, quantization_errors[l], 64,
157  im_quantization_errors[l]);
158  break;
159  /*case COMPRESSED_ADAPTIVE_THRESH_QUANTIZATION:
160  // TODO: Don't hardcode proportion.
161  quantizer.intermodel_sum_adaptive_threshold_quantized(
162  comm, weights_gradient, quantization_errors[l], 64,
163  im_quantization_errors[l], true);
164  break;*/
165  }
166  double im_time = get_time() - start_time;
167  if (summarizer != nullptr && ct != NONE) {
168  std::string prefix = "layer" + std::to_string(
169  static_cast<long long>(layers[l]->get_index())) + "/imcomm_";
170  summarizer->reduce_scalar(prefix + "time",
171  im_time, m->get_cur_step());
172  size_t bytes_sent = 0;
173  size_t bytes_received = 0;
174  if (ct_does_quantization()) {
175  bytes_sent = quantizer.get_bytes_sent();
176  bytes_received = quantizer.get_bytes_received();
177  } else {
178  // Use the same approximation the comm layer does.
179  bytes_sent = sizeof(DataType) * weights_gradient.LocalHeight() * weights_gradient.LocalWidth();
180  bytes_received = sizeof(DataType) * weights_gradient.LocalHeight() * weights_gradient.LocalWidth();
181  }
182  summarizer->reduce_scalar(prefix + "bytes_sent",
183  bytes_sent, m->get_cur_step());
184  summarizer->reduce_scalar(prefix + "bytes_received",
185  bytes_received, m->get_cur_step());
186  if (ct_does_quantization()) {
187  summarizer->reduce_scalar(prefix + "rs_bytes_sent",
188  quantizer.get_rs_bytes_sent(),
189  m->get_cur_step());
190  summarizer->reduce_scalar(prefix + "ag_bytes_sent",
191  quantizer.get_ag_bytes_sent(),
192  m->get_cur_step());
193  summarizer->reduce_scalar(prefix + "rs_bytes_received",
194  quantizer.get_rs_bytes_received(),
195  m->get_cur_step());
196  summarizer->reduce_scalar(prefix + "ag_bytes_received",
197  quantizer.get_ag_bytes_received(),
198  m->get_cur_step());
199  summarizer->reduce_scalar(prefix + "rs_send_trans_time",
200  quantizer.get_rs_send_trans_time(),
201  m->get_cur_step());
202  summarizer->reduce_scalar(prefix + "rs_recv_trans_time",
203  quantizer.get_rs_recv_trans_time(),
204  m->get_cur_step());
205  summarizer->reduce_scalar(prefix + "ag_recv_trans_time",
206  quantizer.get_ag_recv_trans_time(),
207  m->get_cur_step());
208  quantizer.reset_bytes_counters();
209  quantizer.reset_time_counters();
210  if (ct == ADAPTIVE_THRESH_QUANTIZATION) {
211  summarizer->reduce_scalar(prefix + "quantized_count",
212  quantizer.get_quantized_count(),
213  m->get_cur_step());
214  }
215  }
216  }
217  }
