87 void rmsprop::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
88 
89  // Get local matrix data
90  const int local_height = values.LocalHeight();
91  const int local_width = values.LocalWidth();
92  DataType* __restrict__ values_buffer = values.Buffer();
93  const int values_ldim = values.LDim();
94  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
95  const int gradient_ldim = gradient.LDim();
96  DataType* __restrict__ cache_buffer = m_cache->Buffer();
97  const int cache_ldim = m_cache->LDim();
98 
99  // Check if matrix data is contiguous
100  if (values_ldim != local_height
101  || gradient_ldim != local_height
102  || cache_ldim != local_height) {
103  // Update with non-contiguous data
104  #pragma omp parallel for collapse(2)
105  for (int j=0; j<local_width; ++j) {
106  for (int i=0; i<local_height; ++i) {
107  DataType& x = values_buffer[i+j*values_ldim];
108  const DataType g = gradient_buffer[i+j*gradient_ldim];
109  DataType& c = cache_buffer[i+j*cache_ldim];
110  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
111  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
112  }
113  }
114  } else {
115  // Update with contiguous data
116  #pragma omp parallel for
117  for (int i=0; i<local_height*local_width; ++i) {
118  DataType& x = values_buffer[i];
119  const DataType g = gradient_buffer[i];
120  DataType& c = cache_buffer[i];
121  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
122  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
123  }
124  }
125 }
