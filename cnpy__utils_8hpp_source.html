51 template<typename T>
52 inline T& data(const cnpy::NpyArray& na, const std::vector<size_t> indices) {
53  if ((sizeof(T) != na.word_size) && (sizeof(T) != 1u)) {
54  throw lbann_exception("cnpy_utils::data() : The data type is not consistent with the word size of the array.");
55  }
