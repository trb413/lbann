74 EvalType cross_entropy_with_uncertainty::finish_evaluate_compute(
75  const AbsDistMat& predictions, const AbsDistMat& ground_truth) {
76 
77  // Initialize workspace
78  m_prediction_sums->Resize(1, predictions.Width());
79  Mat& prediction_sums_local = m_prediction_sums->Matrix();
80 
81  // Local matrices
82  const Mat& predictions_local = predictions.LockedMatrix();
83  const Mat& ground_truth_local = ground_truth.LockedMatrix();
84 
85  // Matrix parameters
86  const int width = predictions.Width();
87  const int local_height = predictions_local.Height();
88  const int local_width = predictions_local.Width();
89 
90  // Compute sum of predictions
91  #pragma omp parallel for
92  for (int col = 0; col < local_width; ++col) {
93  EvalType pred_sum = EvalType(0);
94  for (int row = 0; row < local_height; ++row) {
95  if (ground_truth_local(row, col) != EvalType(0)) {
96  pred_sum += predictions_local(row, col);
97  }
98  }
99  prediction_sums_local(0, col) = pred_sum;
100  }
101  get_comm().allreduce(*m_prediction_sums,
102  m_prediction_sums->RedundantComm());
103 
104  // Compute mean objective function value
105  EvalType local_sum = EvalType(0);
106  for (int col = 0; col < local_width; ++col) {
107  local_sum += -std::log(prediction_sums_local(0, col));
108  }
109  return get_comm().allreduce(local_sum / width,
110  m_prediction_sums->DistComm());
111 
112 }

114 void cross_entropy_with_uncertainty::differentiate_compute(const AbsDistMat& predictions,
115  const AbsDistMat& ground_truth,

117 
118  // Local matrices
119  const Mat& ground_truth_local = ground_truth.LockedMatrix();

121  Mat& gradient_local = gradient.Matrix();
122 
123  // Matrix parameters
124  const El::Int local_height = gradient_local.Height();
125  const El::Int local_width = gradient_local.Width();
126 
127  // Compute gradient
128  #pragma omp parallel for collapse(2)
129  for (El::Int col = 0; col < local_width; ++col) {
130  for (El::Int row = 0; row < local_height; ++row) {
131  const DataType true_val = ground_truth_local(row, col);
132  DataType& grad_val = gradient_local(row, col);
133  if (true_val != DataType(0)) {
134  grad_val = -true_val / prediction_sums_local(0, col);
135  } else {
136  grad_val = DataType(0);
137  }
138  }
139  }
