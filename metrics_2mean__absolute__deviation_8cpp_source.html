31 EvalType mean_absolute_deviation_metric::evaluate_compute(const AbsDistMat& prediction,
32  const AbsDistMat& ground_truth) {
33 
34  // Get matrix dimensions
35  const int height = prediction.Height();
36  const int local_height = prediction.LocalHeight();
37  const int local_width = prediction.LocalWidth();
38 
39  // Get local matrices
40  const Mat& prediction_local = prediction.LockedMatrix();
41  const Mat& ground_truth_local = ground_truth.LockedMatrix();
42 
43  // Compute sum of squared errors
44  EvalType sum = 0;
45  #pragma omp parallel for reduction(+:sum) collapse(2)
46  for(El::Int col = 0; col < local_width; ++col) {
47  for(El::Int row = 0; row < local_height; ++row) {
48  const EvalType true_val = ground_truth_local(row, col);
49  const EvalType pred_val = prediction_local(row, col);
50  const EvalType error = true_val - pred_val;
51  sum += error >= DataType(0) ? error : - error;
52  }
53  }
54 
55  // Compute mean value across mini-batch
56  return get_comm().allreduce(sum / height, prediction.DistComm());
57 
58 }
