102  m_moment2 = new DistMat(m_comm->get_model_grid());

117  m_moment2 = new StarVCMat(m_comm->get_model_grid());

119  break;

123  }
124  El::Zeros(*m_moment1, m_height, m_width);
125  El::Zeros(*m_moment2, m_height, m_width);
126  El::Zeros(*m_old_gradient, m_height, m_width);

129 void hypergradient_adam::update(const AbsDistMat *gradient) {
130  m_current_beta1 *= m_beta1;
131  m_current_beta2 *= m_beta2;
132  // Precompute the bias correction.
133  const DataType correction = std::sqrt(DataType(1) - m_current_beta2) /
134  (DataType(1) - m_current_beta1);
135 
136  const int local_height = m_parameters->LocalHeight();
137  const int local_width = m_parameters->LocalWidth();
138  DataType *parameters_buffer = m_parameters->Buffer();
139  const int parameters_ldim = m_parameters->LDim();
140  const DataType *gradient_buffer = gradient->LockedBuffer();
141  const int gradient_ldim = gradient->LDim();
142  DataType *moment1_buffer = m_moment1->Buffer();
143  const int moment1_ldim = m_moment1->LDim();
144  DataType *moment2_buffer = m_moment2->Buffer();
145  const int moment2_ldim = m_moment2->LDim();
146  DataType *old_gradient_buffer = m_old_gradient->Buffer();
147  const int old_gradient_ldim = m_old_gradient->LDim();
148 
149  // Compute the learning rate update.
150  DataType lr_update = El::Dot(*gradient, *m_old_gradient);
151  m_learning_rate += m_hyper_learning_rate * lr_update;
152 
153  // Check if matrix data is contiguous.
154  if (parameters_ldim != local_height
155  || gradient_ldim != local_height
156  || moment1_ldim != local_height
157  || moment2_ldim != local_height
158  || old_gradient_ldim != local_height) {
159  // Non-contiguous data.
160  #pragma omp parallel for collapse(2)
161  for (int j = 0; j < local_width; ++j) {
162  for (int i = 0; i < local_height; ++i) {
163  DataType& x = parameters_buffer[i+j*parameters_ldim];
164  const DataType g = gradient_buffer[i+j*gradient_ldim] + m_eps;
165  DataType& m1 = moment1_buffer[i+j*moment1_ldim];
166  DataType& m2 = moment2_buffer[i+j*moment2_ldim];
167  DataType& old_c = old_gradient_buffer[i+j*old_gradient_ldim];
168  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
169  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
170  old_c = correction * m1 / (std::sqrt(m2) + m_eps);
171  x -= m_learning_rate * old_c;
172  }
173  }
174  } else {
175  // Contiguous data.
176  #pragma omp parallel for
177  for (int i = 0; i < local_height * local_width; ++i) {
178  DataType& x = parameters_buffer[i];
179  // Add eps here to avoid denormalized floats.
180  const DataType g = gradient_buffer[i] + m_eps;
181  DataType& m1 = moment1_buffer[i];
182  DataType& m2 = moment2_buffer[i];
183  DataType& old_c = old_gradient_buffer[i];
184  // Update the first/second moment estimates.
185  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
186  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
187  // Compute the unbiased gradient estimate.
188  old_c = correction * m1 / (std::sqrt(m2) + m_eps);
189  // Parameter update.
190  x -= m_learning_rate * old_c;
191  }
192  }
193 }
