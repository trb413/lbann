59  static const char* __attribute__((used)) _layer_type_to_string(layer_type l) {
60  switch(l) {
61  case layer_type::fully_connected:
62  return "fully_connected";
63  case layer_type::softmax:
64  return "softmax";
65  case layer_type::convolution:
66  return "convolution";
67  case layer_type::pooling:
68  return "pooling";
69  case layer_type::input_distributed_minibatch:
70  return "input_distributed_minibatch";
71  case layer_type::input_distributed_minibatch_parallel_io:
72  return "input_distributed_minibatch_parallel_io";
73  case layer_type::target_distributed_minibatch:
74  return "target_distributed_minibatch";
75  case layer_type::target_distributed_minibatch_parallel_io:
76  return "target_distributed_minibatch_parallel_io";
77  case layer_type::target_unsupervised:
78  return "target_unsupervised";
79  case layer_type::INVALID:
80  return "INVALID";
81  default:
82  throw(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " Invalid layer_type specified");
83  }
84  return NULL;
