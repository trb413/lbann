35 template<typename F>
36 void ColumnSum( const Matrix<F>& X, Matrix<F>& sums ) {
37 // DEBUG_ONLY(CSE cse("ColumnSum"))
38 
39  // Input matrix parameters
40  const Int m = X.Height();
41  const Int n = X.Width();
42  const F *XBuf = X.LockedBuffer();
43  const Int XLDim = X.LDim();
44 
45  // Initialize output
46  Zeros( sums, 1, n );
47  F *sumsBuf = sums.Buffer();
48  const Int sumsLDim = sums.LDim();
49 
50  // Compute sum over each column
51  EL_PARALLEL_FOR
52  for( Int j=0; j<n; ++j )
53  {
54  for( Int i=0; i<m; ++i )
55  {
56  sumsBuf[j*sumsLDim] += XBuf[i+j*XLDim];
57  }
58  }
59 
60 }

63 void ColumnSum
64 ( const AbstractDistMatrix<F>& A, AbstractDistMatrix<F>& sums ) {
65 // DEBUG_ONLY(CSE cse("ColumnSum"))
66 
67  // Check that distributed matrix formats are valid
68  if( A.DistData().rowDist != sums.DistData().rowDist
69  || sums.DistData().colDist != STAR
70  || A.DistData().blockHeight != sums.DistData().blockHeight
71  || A.DistData().blockWidth != sums.DistData().blockWidth)
72  {

84 template<typename F>
85 void RowSum(const Matrix<F>& X, Matrix<F>& sums) {
86 
87  // Input matrix parameters
88  const Int m = X.Height();
89  const Int n = X.Width();
90  const F *XBuf = X.LockedBuffer();
91  const Int XLDim = X.LDim();
92 
93  // Initialize output
94  Zeros( sums, m, 1 );
95  F *sumsBuf = sums.Buffer();
96 
97  // Iterate through row blocks
98  const Int bsize = Max( 64 / sizeof(F), 1 );
99  EL_PARALLEL_FOR
100  for( Int i=0; i<m; i+=bsize )
101  {
102  const Int mb = Min( bsize, m - i );
103  for( Int j=0; j<n; ++j )
104  {
105  for( Int ib=0; ib<mb; ++ib )
106  {
107  sumsBuf[i+ib] += XBuf[(i+ib)+j*XLDim];
108  }
109  }
110  }
111 
112 }
