89  if (is_subsidiary_store()) {

148 void data_store_csv::exchange_data() {
149  double tm1 = get_time();
150  std::stringstream err;
151 
152  //get indices I need for the next epoch, and start receives
153  std::unordered_set<int> indices;
154  get_indices(indices, m_rank);
155  std::vector<El::mpi::Request<DataType>> recv_req(indices.size());
156 
157  m_my_minibatch_data.clear();
158  size_t jj = 0;
159  for (auto data_id : indices) {
160  m_my_minibatch_data[data_id].resize(m_vector_size);
161  int owner = get_index_owner(data_id);
162  if (owner >= m_np or owner < 0) {
163  err << __FILE__ << " " << __LINE__ << " :: "
164  << " ERROR: bad rank for owner in nb_recv; owner: " << owner << " data_id: " << data_id << " jj: " << jj+1 << " of " << indices.size();
165  throw lbann_exception(err.str());
166  }
167  m_comm->nb_tagged_recv<DataType>(m_my_minibatch_data[data_id].data(), m_vector_size, owner, data_id, recv_req[jj++], m_comm->get_model_comm());
168  }
169 
170  //start sends to all processors
171  std::vector<std::vector<El::mpi::Request<DataType>>> send_req(m_np);
172  for (int p=0; p<m_np; p++) {
173  get_my_indices(indices, p);
174  send_req[p].resize(indices.size());
175  jj = 0;
176  for (auto data_id : indices) {
177  if (m_data.find(data_id) == m_data.end()) {
178  err << __FILE__ << " " << __LINE__ << " :: "
179  << " m_data.find(" << data_id << ") failed.";
180  throw lbann_exception(err.str());
181  }
182  if (m_data[data_id].size() != (size_t)m_vector_size) {
183  err << __FILE__ << " " << __LINE__ << " :: "
184  << " m_data[" << data_id << "].size = " << m_data[data_id].size()
185  << " should be: " << m_vector_size << "; " << jj+1
186  << " of " << indices.size()
187  << " m_reader->get_role: " << m_reader->get_role();
188  throw lbann_exception(err.str());
189  }
190  m_comm->nb_tagged_send<DataType>(m_data[data_id].data(), m_vector_size, p, data_id, send_req[p][jj++], m_comm->get_model_comm());
191  }
192  }
193 
194  //wait for sends to finish
195  if (m_master) {
196  for (size_t i=0; i<send_req.size(); i++) {
197  m_comm->wait_all(send_req[i]);
198  }
199  }
200 
201  //wait for recvs to finish
202  m_comm->wait_all(recv_req);
203 
204  if (m_master) {
205  std::cerr << "TIME for data_store_csv::exchange_data(): "
206  << get_time() - tm1 << "; role: " << m_reader->get_role() << "\n";
207  }
208 }
