83 void data_store_imagenet::test_data() {

85  const std::vector<std::pair<std::string, int> > & image_list = reader->get_image_list();

95  std::stringstream err;
96  err << __FILE__ << " " << __LINE__ << " :: "
97  << "failed to open " << imagepath << " for reading";
98  throw lbann_exception(err.str());
99  }
100 
101  in.seekg(0, std::ios::end);
102  size_t sz = in.tellg();
103  in.seekg(0, std::ios::beg);
104  b.resize(sz);
105  in.read((char*)&b[0], sz);
106  in.close();
107 
108  //get from datastore

111  std::stringstream err;

113  << " :: data_store_imagenet::test_data, b != v; b.size: "

115  throw lbann_exception(err.str());
116  }
117  }
118 
119  std::cerr << "rank: " << m_rank << " role: " << m_reader->get_role() << " :: data_store_imagenet::test_data: PASSES!\n";

142 void data_store_imagenet::read_files() {
143  imagenet_reader *reader = dynamic_cast<imagenet_reader*>(m_reader);
144  const std::vector<std::pair<std::string, int> > & image_list = reader->get_image_list();
145  for (size_t j=0; j<m_my_global_indices.size(); j++) {
146  size_t index = m_my_global_indices[j];
147  if (m_offsets.find(index) == m_offsets.end()) {
148  std::stringstream err;
149  err << __FILE__ << " " << __LINE__ << " :: "
150  << " m_offsets.find(index) failed for index: " << index;
151  throw lbann_exception(err.str());
152  }
153  size_t offset = m_offsets[index];
154  if (m_file_sizes.find(index) == m_file_sizes.end()) {
155  std::stringstream err;
156  err << __FILE__ << " " << __LINE__ << " :: "
157  << " m_file_sizes.find(index) failed for index: " << index;
158  throw lbann_exception(err.str());
159  }
160  size_t file_len = m_file_sizes[index];
161  if (offset + file_len > m_data.size()) {
162  std::stringstream err;
163  err << __FILE__ << " " << __LINE__ << " :: " << " j: " << j
164  << " of " << m_my_global_indices.size() << " offset: " << offset
165  << " file_len: " << file_len << " offset+file_len: "
166  << offset+file_len << " m_data.size(): " << m_data.size()
167  << "\noffset+file_len must be <= m_data.size()";
168  throw lbann_exception(err.str());
169  }
170  load_file(m_dir, image_list[index].first, &m_data[offset], file_len);
171  }

174 void data_store_imagenet::get_file_sizes() {
175  imagenet_reader *reader = dynamic_cast<imagenet_reader*>(m_reader);
176  const std::vector<std::pair<std::string, int> > & image_list = reader->get_image_list();
177  //construct a vector of Triples
178  std::vector<Triple> my_file_sizes(m_my_global_indices.size());
179  size_t cur_offset = 0;
180  for (size_t j=0; j<m_my_global_indices.size(); j++) {
181  size_t index = m_my_global_indices[j];
182  my_file_sizes[j].global_index = index;
183  my_file_sizes[j].num_bytes = get_file_size(m_dir, image_list[index].first);
184  my_file_sizes[j].offset = cur_offset;
185  my_file_sizes[j].rank = m_rank;
186  cur_offset += my_file_sizes[j].num_bytes;
187  if (my_file_sizes[j].num_bytes == 0) {
188  std::stringstream err;
189  err << __FILE__ << " " << __LINE__ << " :: " << " j: " << j
190  << " file size is 0 (" << m_dir << "/" + image_list[index].first;
191  throw lbann_exception(err.str());
192  }
193  }
194 
195  exchange_file_sizes(my_file_sizes, m_num_global_indices);
196 }
