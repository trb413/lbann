155 void data_store_imagenet::get_file_sizes() {
156  std::vector<int> num_images(m_num_readers, 0);
157  for (size_t j=0; j<m_num_global_indices; j++) {
158  num_images[j % m_num_readers] += 1;
159  }
160  //num_images[j] contains the number of image files owned by P_j
161 
162  //get the number of bytes for files owned by this processor;
163  //my_file_sizes[j] is a global index (wrt m_reader->get_image_list()),
164  //and my_file_sizes[j+1] is the corresponding file size
165  imagenet_reader *reader = dynamic_cast<imagenet_reader*>(m_reader);
166  const std::vector<std::pair<std::string, int> > & image_list = reader->get_image_list();
167  int my_num_images = num_images[m_rank];
168  std::string file_dir = m_reader->get_file_dir();
169 
170  //construct a vector or Triples
171  std::vector<Triple> my_file_sizes(my_num_images);
172  size_t cur_offset = 0;
173  for (size_t j=0; j<m_my_datastore_indices.size(); j++) {
174  if (j >= image_list.size()) {
175  std::stringstream err;
176  err << __FILE__ << " " << __LINE__ << " :: " << " j: " << j
177  << " >= image_list.size() [" << image_list.size() << "]";
178  throw lbann_exception(err.str());
179  }
180  size_t index = m_my_datastore_indices[j];
181  my_file_sizes[j].global_index = index;
182  my_file_sizes[j].num_bytes = get_file_size(file_dir, image_list[index].first);
183  my_file_sizes[j].offset = cur_offset;
184  cur_offset += my_file_sizes[j].num_bytes;
185  if (my_file_sizes[j].num_bytes ==0) {
186  std::stringstream err;
187  err << __FILE__ << " " << __LINE__ << " :: " << " j: " << j
188  << " file size is 0 (" << file_dir << "/" + image_list[index].first;
189  throw lbann_exception(err.str());
190  }
191  }
192 
193  //exchange files sizes
194  std::vector<Triple> global_file_sizes(m_num_global_indices);
195  std::vector<int> disp(m_num_readers);
196  disp[0] = 0;
197  for (int h=1; h<(int)m_num_readers; h++) {
198  disp[h] = disp[h-1] + num_images[h-1]*sizeof(Triple);
199  }
200 
201  for (size_t j=0; j<num_images.size(); j++) {
202  num_images[j] *= sizeof(Triple);
203  }
204 
205  //m_comm->model_gatherv(&my_file_sizes[0], my_file_sizes.size(),
206  // &global_file_sizes[0], &num_images[0], &disp[0]);
207  MPI_Allgatherv(&my_file_sizes[0], my_file_sizes.size()*sizeof(Triple), MPI_BYTE,
208  &global_file_sizes[0], &num_images[0], &disp[0], MPI_BYTE, m_comm->get_model_comm().comm);
209 
210 
211  for (auto t : global_file_sizes) {
212  m_file_sizes[t.global_index] = t.num_bytes;
213  m_offsets[t.global_index] = t.offset;
214  if (t.num_bytes <= 0) {
215  std::stringstream err;
216  err << __FILE__ << " " << __LINE__ << " :: "
217  << "num_bytes <= 0 (" << t.num_bytes << ")";
218  throw lbann_exception(err.str());
219  }
220  }
221 }

261 void data_store_imagenet::exchange_data() {
262  double tm1 = get_time();
263  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
264  m_my_data_hash.clear();
265  for (size_t j = 0; j<m_my_minibatch_indices.size(); j++) {
266  int idx = (*m_shuffled_indices)[m_my_minibatch_indices[j]];
267  int offset = m_offsets[idx];
268  int file_len = m_file_sizes[idx];
269  if (file_len <= 0) {
270  throw lbann_exception(
271  std::string{} + __FILE__ + " :: " + std::to_string(__LINE__) + " :: "
272  + " file_len: " + std::to_string(file_len) + " (is <= 0)");
273  }
274  if (file_len > 100000000) {
275  throw lbann_exception(
276  std::string{} + __FILE__ + " :: " + std::to_string(__LINE__) + " :: "
277  + " file_len: " + std::to_string(file_len) + " (is > 100000000)");
278  }
279  if (j >= m_my_data.size()) {
280  throw lbann_exception(
281  std::string{} + __FILE__ + " :: " + std::to_string(__LINE__) + " :: "
282  + " j: " + std::to_string(j) + " is >= m_my_data.size(): "
283  + std::to_string(m_my_data.size()));
284  }
285  m_my_data[j].resize(file_len);
286  m_my_data_hash[idx] = j;
287  int owner = m_owner_mapping[idx];
288  MPI_Get(&m_my_data[j][0], file_len, MPI_BYTE,
289  owner, offset, file_len, MPI_BYTE, m_win);
290  }
291  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
292  double tm2 = get_time();
293  if (m_rank == 0) {
294  std::cout << "data_store_imagenet::exchange_data() time: " << tm2 - tm1 << std::endl;
295  }
296 }
