61 void init_random(int seed, lbann_comm* comm) {
62  if (seed != -1) {
63  // Seed every OpenMP thread, if present.
64  // Note: Threadprivate OMP variables don't work with dynamic threads.
65 #ifdef _OPENMP
66  #pragma omp parallel
67  {
68  get_generator().seed((seed << 8) | (omp_get_thread_num() & 0xff));
69  get_fast_generator().seed((seed << 8) | (omp_get_thread_num() & 0xff));
70  }
71 #else
72  get_generator().seed(seed);
73  get_fast_generator().seed(seed);
74 #endif
75 #ifdef LBANN_SET_EL_RNG
76  if (comm != nullptr) {
77  El::Generator().seed(seed ^ comm->get_rank_in_model());
78  } else {
79  El::Generator().seed(seed ^ El::mpi::Rank(El::mpi::COMM_WORLD));
80  }
81 #endif
82  } else {
83  // Seed with a random value.
84  std::random_device rd;
85  unsigned rand_val = rd();
86 #ifdef _OPENMP
87  #pragma omp parallel
88  {
89  get_generator().seed((rand_val << 8) | (omp_get_thread_num() & 0xff));
90  get_fast_generator().seed((rand_val << 8) | (omp_get_thread_num() & 0xff));
91  }
92 #else
93  get_generator().seed(rand_val);
94  get_fast_generator().seed(rand_val);
95 #endif
96 #ifdef LBANN_SET_EL_RNG
97  El::Generator().seed(rand_val);
98 #endif
99  }
100 }

148 void gaussian_fill_procdet(ElMat& mat, El::Int m, El::Int n, DataType mean,
149  DataType stddev) {
150  Zeros(mat, m, n);
151  if (mat.Grid().Rank() == 0) {
152  mat.Reserve(n * m);
153  auto& gen = get_generator();
154  std::normal_distribution<DataType> dist(mean, stddev);
155  for (El::Int col = 0; col < n; ++col) {
156  for (El::Int row = 0; row < m; ++row) {
157  mat.QueueUpdate(row, col, dist(gen));
158  }
159  }
160  }
161  mat.ProcessQueues();
162 }

164 void bernoulli_fill_procdet(ElMat& mat, El::Int m, El::Int n, double p) {
165  Zeros(mat, m, n);
166  if (mat.Grid().Rank() == 0) {
167  mat.Reserve(m * n);
168  auto& gen = get_generator();
169  std::bernoulli_distribution dist(p);
170  for (El::Int col = 0; col < n; ++col) {
171  for (El::Int row = 0; row < m; ++row) {
172  mat.QueueUpdate(row, col, dist(gen) ? 1.0f : 0.0f);
173  }
174  }
175  }
176  mat.ProcessQueues();
177 }

179 void uniform_fill_procdet(ElMat& mat, El::Int m, El::Int n, DataType center,
180  DataType radius) {
181  Zeros(mat, m, n);
182  if (mat.Grid().Rank() == 0) {
183  mat.Reserve(n * m);
184  auto& gen = get_generator();
185  std::uniform_real_distribution<DataType> dist(center - radius,
186  center + radius);
187  for (El::Int col = 0; col < n; ++col) {
188  for (El::Int row = 0; row < m; ++row) {
189  mat.QueueUpdate(row, col, dist(gen));
190  }
191  }
192  }
193  mat.ProcessQueues();
194 }

197 void initialize_matrix(ElMat& matrix_v, weight_initialization initialization, Int fan_in, Int fan_out) {
198  switch(initialization) {
199  case weight_initialization::uniform:
200  uniform_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
201  DataType(0), DataType(1));
202  break;
203  case weight_initialization::normal:
204  gaussian_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
205  DataType(0), DataType(1));
206  break;
207  case weight_initialization::glorot_normal: {
208  const DataType var = 2.0 / (fan_in + fan_out);
209  gaussian_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
210  DataType(0), sqrt(var));
211  break;
212  }
213  case weight_initialization::glorot_uniform: {
214  const DataType var = 2.0 / (fan_in + fan_out);
215  uniform_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
216  DataType(0), sqrt(3*var));
217  break;
218  }
219  case weight_initialization::he_normal: {
220  const DataType var = 1.0 / fan_in;
221  gaussian_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
222  DataType(0), sqrt(var));
223  break;
224  }
225  case weight_initialization::he_uniform: {
226  const DataType var = 1.0 / fan_in;
227  uniform_fill(matrix_v, matrix_v.Height(), matrix_v.Width(),
228  DataType(0), sqrt(3*var));
229  break;
230  }
231  case weight_initialization::zero: // Zero initialization is default
232  default:
233  Zero(matrix_v);
234  break;
235  }
236 }
