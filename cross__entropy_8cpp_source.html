60 void cross_entropy::compute_value(const AbsDistMat& predictions,
61  const AbsDistMat& ground_truth) {
62 
63  // Get local matrices and matrix parameters
64  const Mat& predictions_local = predictions.LockedMatrix();
65  const Mat& ground_truth_local = ground_truth.LockedMatrix();
66  const El::Int width = predictions.Width();
67  const El::Int local_height = predictions_local.Height();
68  const El::Int local_width = predictions_local.Width();
69 
70  // Compute sum of cross entropy terms with Kahan summation
71  double sum = 0;
72  double correction = 0;
73  for(El::Int col = 0; col < local_width; ++col) {
74  for(El::Int row = 0; row < local_height; ++row) {
75  const double true_val = ground_truth_local(row, col);
76  const double pred_val = predictions_local(row, col);
77  double term = - true_val * std::log(pred_val);
78  term += correction;
79  const double next_sum = sum + term;
80  correction = term - (next_sum - sum);
81  sum = next_sum;
82  }
83  }
84 
85  // Compute mean cross entropy across mini-batch
86  double mean_cross_entropy = sum / width;
87  mean_cross_entropy = El::mpi::AllReduce(mean_cross_entropy,
88  predictions.DistComm());
89 
90  // Update objective function value
91  add_to_value(mean_cross_entropy);
92 
93 }

95 void cross_entropy::compute_gradient(const AbsDistMat& predictions,
96  const AbsDistMat& ground_truth,
97  AbsDistMat& gradient) {
98 
99  // Apply softmax-cross-entropy shortcut if activated
100  if(m_shortcut_softmax_layer != nullptr) {
101  El::LockedView(gradient, ground_truth);
102  return;
103  }
104 
105  // Get local matrices
106  const Mat& predictions_local = predictions.LockedMatrix();
107  const Mat& ground_truth_local = ground_truth.LockedMatrix();
108  Mat& gradient_local = gradient.Matrix();
109 
110  // Compute gradient
111  El::IndexDependentFill(gradient_local,
112  (std::function<DataType(El::Int,El::Int)>)
113  ([&predictions_local, &ground_truth_local]
114  (El::Int r, El::Int c) -> DataType {
115  const DataType true_val = ground_truth_local(r,c);
116  if(true_val != DataType(0)) {
117  const DataType pred_val = predictions_local(r,c);
118  return - true_val / pred_val;
119  }
120  else {
121  return DataType(0);
122  }
123  }));
