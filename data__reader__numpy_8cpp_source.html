61 void numpy_reader::load() {
62  std::string infile = get_data_filename();
63  // Ensure the file exists.
64  std::ifstream ifs(infile);
65  if (!ifs) {
66  throw lbann_exception(
67  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
68  " numpy_reader::load() - can't open file : " + infile);
69  }
70  ifs.close();
71 
72  m_data = cnpy::npy_load(infile);
73  m_num_samples = m_data.shape[0];
74  m_num_features = std::accumulate(
75  m_data.shape.begin() + 1, m_data.shape.end(), (unsigned) 1,
76  std::multiplies<unsigned>());
77 
78  // Ensure we understand the word size.
79  if (!(m_data.word_size == 4 || m_data.word_size == 8)) {
80  throw lbann_exception(
81  "numpy_reader: word size " + std::to_string(m_data.word_size) +
82  " not supported");
83  }
84  // Fortran order not yet supported.
85  if (m_data.fortran_order) {
86  throw lbann_exception(
87  "numpy_reader: fortran order not supported");
88  }
89  // Don't currently support both labels and responses.
90  if (m_has_labels && m_has_responses) {
91  throw lbann_exception(
92  "numpy_reader: labels and responses not supported at same time");
93  }
94 
95  if (m_has_labels) {
96  // Shift feature count because the last becomes the label.
97  m_num_features -= 1;
98  // Determine number of label classes.
99  std::unordered_set<int> label_classes;
100  for (int i = 0; i < m_num_samples; ++i) {
101  if (m_data.word_size == 4) {
102  float *data = m_data.data<float>() + i*(m_num_features+1);
103  label_classes.insert((int) data[m_num_features+1]);
104  } else if (m_data.word_size == 8) {
105  double *data = m_data.data<double>() + i*(m_num_features+1);
106  label_classes.insert((int) data[m_num_features+1]);
107  }
108  }
109  // Sanity checks.
110  auto minmax = std::minmax_element(label_classes.begin(), label_classes.end());
111  if (*minmax.first != 0) {
112  throw lbann_exception(
113  "numpy_reader: classes are not indexed from 0");
114  }
115  if (*minmax.second != (int) label_classes.size() - 1) {
116  throw lbann_exception(
117  "numpy_reader: label classes are not contiguous");
118  }
119  m_num_labels = label_classes.size();
120  }
121  if (m_has_responses) {
122  // Last feature becomes the response.
123  m_num_features -= 1;
124  }
125 
126  // Reset indices.
127  m_shuffled_indices.clear();
128  m_shuffled_indices.resize(m_num_samples);
129  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
130  select_subset_of_data();
131 }

133 bool numpy_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
134  int features_size = m_num_features;
135  if (m_has_labels || m_has_responses) {
136  features_size += 1;
137  }
138  if (m_data.word_size == 4) {
139  float *data = m_data.data<float>() + data_id * features_size;
140  for (int j = 0; j < m_num_features; ++j) {
141  X(j, mb_idx) = data[j];
142  }
143  } else if (m_data.word_size == 8) {
144  double *data = m_data.data<double>() + data_id * features_size;
145  for (int j = 0; j < m_num_features; ++j) {
146  X(j, mb_idx) = data[j];
147  }
148  }
149  return true;
150 }

152 bool numpy_reader::fetch_label(Mat& Y, int data_id, int mb_idx, int tid) {
153  if (!m_has_labels) {
154  throw lbann_exception("numpy_reader: do not have labels");
155  }
156  int label = 0;
157  if (m_data.word_size == 4) {
158  float *data = m_data.data<float>() + data_id*(m_num_features+1);
159  label = (int) data[m_num_features+1];
160  } else if (m_data.word_size == 8) {
161  double *data = m_data.data<double>() + data_id*(m_num_features+1);
162  label = (int) data[m_num_features+1];
163  }
164  Y(label, mb_idx) = 1;
165  return true;
166 }

168 bool numpy_reader::fetch_response(Mat& Y, int data_id, int mb_idx, int tid) {
169  if (!m_has_responses) {
170  throw lbann_exception("numpy_reader: do not have responses");
171  }
172  auto response = DataType(0);
173  if (m_data.word_size == 4) {
174  float *data = m_data.data<float>() + data_id*(m_num_features+1);
175  response = (DataType) data[m_num_features+1];
176  } else if (m_data.word_size == 8) {
177  double *data = m_data.data<double>() + data_id*(m_num_features+1);
178  response = (DataType) data[m_num_features+1];
179  }
180  Y(0, mb_idx) = response;
181  return true;
182 }
