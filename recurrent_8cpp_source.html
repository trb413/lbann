46 void recurrent_model::setup_layer_topology() {
47  model::setup_layer_topology();
48 
49  // Setup layer execution order
50  setup_layer_execution_order();

52  // We expect starting with input layer and ending with target layer
54  if (m_layers.size() < 4) {
55  std::stringstream err;
56  err << __FILE__ << " " << __LINE__ << " :: "
57  << "expected the first layer to be an input layer, "
58  << "the second to be a slice layer, "
59  << "the second to last to be a concatenation layer "
60  << "and the last to be a target layer";
61  throw lbann_exception(err.str());
62  }
63  auto input = *(m_layers.begin());
64  auto input_slice = *(m_layers.begin() + 1);
65  auto target_concat = *(m_layers.end() - 2);
66  auto target = *(m_layers.end() - 1);
67  if (dynamic_cast<generic_input_layer *>(m_layers.front()) == nullptr
68  || input_slice->get_type() != "slice"
69  || target_concat->get_type() != "concatenation"
70  || dynamic_cast<generic_target_layer *>(m_layers.back()) == nullptr) {
71  std::stringstream err;
72  err << __FILE__ << " " << __LINE__ << " :: "
73  << "expected the first layer to be an input layer, "
74  << "the second to be a slice layer, "
75  << "the second to last to be a concatenation layer "
76  << "and the last to be a target layer";
77  throw lbann_exception(err.str());
78  }
79 
80  // Initialize pointers for input and target layer
81  m_previous_roll_layer.clear();
82  m_next_roll_layer.clear();
83  m_previous_roll_layer[input] = input;
84  m_previous_roll_layer[input_slice] = input_slice;
85  m_previous_roll_layer[target_concat] = target_concat;
86  m_previous_roll_layer[target] = target;
87  m_next_roll_layer[input] = input;
88  m_next_roll_layer[input_slice] = input_slice;
89  m_next_roll_layer[target_concat] = target_concat;
90  m_next_roll_layer[target] = target;
91  std::vector<Layer *> input_children, target_parents;
92  for (const auto& child : input_slice->get_child_layers()) {
93  input_children.push_back(const_cast<Layer *>(child));
94  }
95  for (const auto& parent : target_concat->get_parent_layers()) {
96  target_parents.push_back(const_cast<Layer *>(parent));
97  }
98 
99  // Unroll network to desired depth
100  std::vector<Layer *> previous_roll_layers(m_layers.begin() + 2,
101  m_layers.end() - 2);
102  const int roll_size = previous_roll_layers.size();
103  for (int roll = 1; roll < m_unroll_depth; ++roll) {
104 
105  // Construct current roll by copying layers from previous roll
106  std::vector<Layer *> current_roll_layers;
107  for (const auto& previous_layer : previous_roll_layers) {
108  Layer *current_layer = previous_layer->copy();
109  current_roll_layers.push_back(current_layer);
110  m_previous_roll_layer[current_layer] = previous_layer;
111  m_next_roll_layer[previous_layer] = current_layer;
112  }
113 
114  // Fix pointers in current roll
115  for (const auto& current_layer : current_roll_layers) {
116  auto layer_pointers = current_layer->get_layer_pointers();
117  for (auto& layer_pointer : layer_pointers) {
118  layer_pointer = m_next_roll_layer[layer_pointer];
119  }
120  current_layer->set_layer_pointers(layer_pointers);
121  }
122 
123  // Fix pointers in input and target layers
124  for (auto& child : input_children) {
125  child = m_next_roll_layer[child];
126  input_slice->add_child_layer(child);
127  }
128  for (auto& parent : target_parents) {
129  parent = m_next_roll_layer[parent];
130  target_concat->add_parent_layer(parent);
131  }
132 
133  // Add current roll layers to model
134  m_layers.insert(m_layers.end() - 2,
135  current_roll_layers.begin(),
136  current_roll_layers.end());
137  previous_roll_layers = current_roll_layers;
138 
139  }
140 
141  // Rename layers
142  for (int roll = 0; roll < m_unroll_depth; ++roll) {
143  const std::string name_suffix = "_" + std::to_string(roll);
144  const int roll_start = 2 + roll * roll_size;
145  const int roll_end = 2 + (roll + 1) * roll_size;
146  for (int i = roll_start; i < roll_end; ++i) {
147  m_layers[i]->set_name(m_layers[i]->get_name() + name_suffix);
148  }
149  }
150 
151  // Fix pointers between adjacent rolls
152  std::unordered_map<Layer *,bool> is_visited;
153  std::vector<Layer *> placeholder_parents, placeholder_children;
154  for (const auto& layer : m_layers) {
155 
156  // Fix pointers to parent layers
157  std::vector<Layer *> parents;
158  for (const auto& parent : layer->get_parent_layers()) {
159  parents.push_back(const_cast<Layer *>(parent));
160  }
161  for (auto& parent : parents) {
162  if (!is_visited[parent]) {
163  if (m_previous_roll_layer[parent] != nullptr) {
164  parent = m_previous_roll_layer[parent];
165  } else {
166  if (parent->get_num_neurons() <= 0) {
167  std::stringstream err;
168  err << __FILE__ << " " << __LINE__ << " :: "
169  << layer->get_name() << " has ambiguous neuron "
170  << "dimensions since it depends on "
171  << parent->get_name() << ", which does not have "
172  << "specified neuron dimensions. Consider inserting a "
173  << "reshape layer after " << parent->get_name()
174  << "to explicitly specify neuron dimensions.";
175  throw lbann_exception(err.str());
176  }
177  Layer *placeholder;
178  switch (parent->get_data_layout()) {
179  case data_layout::DATA_PARALLEL:
180  placeholder = new constant_layer<data_layout::DATA_PARALLEL>
181  (get_comm(), DataType(0), parent->get_neuron_dims());
182  break;
183  case data_layout::MODEL_PARALLEL:
184  placeholder = new constant_layer<data_layout::MODEL_PARALLEL>
185  (get_comm(), DataType(0), parent->get_neuron_dims());
186  break;
187  default:
188  std::stringstream err;
189  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
190  throw lbann_exception(err.str());
191  }
192  placeholder->set_name(parent->get_name() + "_input_placeholder");
193  placeholder->add_child_layer(layer);
194  placeholder_parents.push_back(placeholder);
195  parent = placeholder;
196  }
197  }
198  }
199  layer->clear_parent_layers();
200  for (const auto& parent : parents) {
201  layer->add_parent_layer(parent);
202  }
203 
204  // Visit layer
205  is_visited[layer] = true;
206 
207  // Fix pointers to parent layers
208  std::vector<Layer *> children;
209  for (const auto& child : layer->get_child_layers()) {
210  children.push_back(const_cast<Layer *>(child));
211  }
212  for (auto& child : children) {
213  if (is_visited[child]) {
214  if (m_next_roll_layer[child] != nullptr) {
215  child = m_next_roll_layer[child];
216  } else {
217  Layer *placeholder;
218  switch (child->get_data_layout()) {
219  case data_layout::DATA_PARALLEL:
220  placeholder = new dummy_layer<data_layout::DATA_PARALLEL>(get_comm());
221  break;
222  case data_layout::MODEL_PARALLEL:
223  placeholder = new dummy_layer<data_layout::MODEL_PARALLEL>(get_comm());
224  break;
225  default:
226  std::stringstream err;
227  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
228  throw lbann_exception(err.str());
229  }
230  placeholder->set_name(child->get_name() + "_output_placeholder");
231  placeholder->add_parent_layer(layer);
232  placeholder_children.push_back(placeholder);
233  child = placeholder;
234  }
235  }
236  }
237  layer->clear_child_layers();
238  for (const auto& child : children) {
239  layer->add_child_layer(child);
240  }
241 
242  }
243 
244  // Add placeholder layers to model
245  m_layers.insert(m_layers.begin() + 2,
246  placeholder_parents.begin(),
247  placeholder_parents.end());
248  m_layers.insert(m_layers.end() - 2,
249  placeholder_children.begin(),
250  placeholder_children.end());

254 void recurrent_model::setup_layer_execution_order() {
255  model::setup_layer_execution_order();
256 
257  // Get layer graph
258  std::set<int> nodes;
259  std::map<int,std::set<int>> edges;
260  construct_layer_graph(nodes, edges);
261  const auto& edges_transpose = graph::transpose(nodes, edges);
262 
263  // Return immediately if no recurrence is detected
264  if (!graph::is_cyclic(nodes, edges)) {
265  return;
266  }
267 
268  // Get topologically sorted condensation graph
269  std::set<int> condensation_nodes;
270  std::map<int,std::set<int>> components, condensation_edges;
271  graph::condensation(nodes, edges,
272  components,
273  condensation_nodes, condensation_edges);
274 
275  // Sort nodes in each strongly connected component (SCC)
276  std::vector<int> order;
277  for (const auto& component : condensation_nodes) {
278  const auto& component_nodes = components[component];
279 
280  // Find a node that requires external input
281  // Note: Execution order is ambiguous if there are more than one
282  // such nodes.
283  int input_node = -1;
284  for (const auto& node : component_nodes) {
285  const auto& parents = graph::get_neighbors(node, edges_transpose);
286  for (const auto& parent : parents) {
287  if (!component_nodes.count(parent)) {
288  if (input_node != node && input_node >= 0) {
289  std::stringstream err;
290  err << __FILE__ << " " << __LINE__ << " :: "
291  << "recurrent network has ambiguous execution order "
292  << "(each strongly connected component in the layer "
293  << "graph can have at most one layer that requires "
294  << "external input)";
295  throw lbann_exception(err.str());
296  }
297  input_node = node;
298  }
299  }
300  }
301 
302  // Induce subgraph on SCC and delete edges toward input node
303  // Note: Execution order is ambiguous if this subgraph is cyclic.
304  auto&& component_edges = graph::induce_subgraph(component_nodes,
305  edges);
306  for (const auto& node : component_nodes) {
307  component_edges[node].erase(input_node);
308  }
309  if (graph::is_cyclic(component_nodes, component_edges)) {
310  std::stringstream err;
311  err << __FILE__ << " " << __LINE__ << " :: "
312  << "recurrent network has ambiguous execution order "
313  << "(each strongly connected component in the layer graph "
314  << "must be a DAG after removing edges toward input nodes)";
315  throw lbann_exception(err.str());
316  }
317 
318  // Sort SCC nodes topologically
319  const auto& component_order = graph::topological_sort(component_nodes,
320  component_edges);
321  order.insert(order.end(), component_order.begin(), component_order.end());
322 
323  }
324 
325  // Reorder layers
326  permute_layers(order);
327 
328 }
