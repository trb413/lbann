61  void setup(int num_prev_neurons) {
62  target_layer::setup(num_prev_neurons);
63  if(!this->m_shared_data_reader) {
64  if(io_layer::m_data_sets_span_models) {
65  io_layer::setup_data_readers_for_training(0, Layer::m_comm->get_num_models() * Layer::m_mini_batch_size,
66  Layer::m_comm->get_model_rank() * Layer::m_mini_batch_size);
67  io_layer::setup_data_readers_for_evaluation(0, this->m_mini_batch_size);
68  } else {
69  io_layer::setup_data_readers_for_training(0, this->m_mini_batch_size);
70  io_layer::setup_data_readers_for_evaluation(0, this->m_mini_batch_size);
71  }
72  }
73 
75  if(num_prev_neurons != this->m_num_neurons) {
76  throw -1;
77  }
78 
79  Zeros(*this->m_error_signal, this->m_num_neurons, this->m_mini_batch_size);
80  Zeros(Y_local, this->m_num_neurons, this->m_mini_batch_size);

82 
83  }

85  void fp_compute() {
86  generic_data_reader *data_reader = target_layer::select_data_reader();
87 
88  if (this->m_comm->get_rank_in_model() == m_root) {
89  Zero(Y_local);
90  data_reader->fetch_label(Y_local);
91  }
92 
93  if (this->m_comm->get_rank_in_model() == m_root) {
94  CopyFromRoot(Y_local, Ys);
95  } else {
96  CopyFromNonRoot(Ys);
97  }
98 
99  this->m_comm->model_barrier();
100  Copy(Ys, *this->m_activations);
101 
103  DataType avg_error = this->m_neural_network_model->m_obj_fn->compute_obj_fn(*this->m_prev_activations_v, *this->m_activations_v);
104  this->m_neural_network_model->m_obj_fn->record_obj_fn(this->m_execution_mode, avg_error);
105 
106  int curr_mini_batch_size = this->m_neural_network_model->get_current_mini_batch_size();
107  for (auto&& m : this->m_neural_network_model->m_metrics) {
108  double cur_num_errors = m->compute_metric(*this->m_prev_activations_v, *this->m_activations_v);
109  m->record_error(cur_num_errors, curr_mini_batch_size);
110  }
111 
112  return;
113  }
