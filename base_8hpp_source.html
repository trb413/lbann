136 /*
137  * endsWith: http://thispointer.com/c-how-to-check-if-a-string-ends-with-an-another-given-string/
138  * Case Sensitive Implementation of endsWith()
139  * It checks if the string 'mainStr' ends with given string
140  * 'toMatch'
141  */
142 static bool __attribute__((used)) endsWith(const std::string mainStr, const std::string &toMatch)
143 {
144  if(mainStr.size() >= toMatch.size() &&
145  mainStr.compare(mainStr.size() - toMatch.size(), toMatch.size(), toMatch) == 0)
146  return true;
147  else
148  return false;
149 }
