40 void lbann_callback_gradient_check::on_test_begin(model *m) {
41 
42  // Get model members
43  lbann_comm *comm = m->get_comm();
44  const std::vector<Layer*>& layers = m->get_layers();
45 
46  // Initialize network for testing
47  m->set_execution_mode(execution_mode::testing);
48  layers[0]->forward_prop();
49 
50  // Compute objective function
51  const DataType objective = compute_objective_function(m);
52 
53  // Choose finite difference step
54  // Note: Consider a central difference scheme:
55  // f'(x) ~ ( - f(x+2h) + 8 f(x+h) - 8 f(x-h) + f(x-2h) ) / 12h
56  // By Taylor's theorem, the truncation error is bounded by
57  // E_trunc <= | f'''''(xi) | / 18 * h^4
58  // Assuming f can be computed to a relative accuracy of epsilon,
59  // E_fl <= epsilon * | f(chi) | / h
60  // For simplicity, we assume f(chi) ~ f(x), and | f'''''(xi) | ~ 1.
61  // If step size is not specified, then we choose h so that
62  // E_fl <= sqrt(epsilon)
63  const DataType epsilon = std::pow(std::numeric_limits<DataType>::epsilon(), 0.9);
64  DataType step_size = m_step_size;
65  if (m_step_size <= DataType(0)) {
66  step_size = std::fabs(objective) * std::sqrt(epsilon);
67  }
68  DataType expected_error = (epsilon * objective / step_size
69  + std::pow(step_size, 4) / 18);
70  expected_error = std::pow(expected_error, 0.9);
71 
72  // Compute gradients
73  for (auto layer : layers) {
74  layer->clear_error_signal();
75  }
76  m->get_objective_function()->differentiate();
77  for (int l = layers.size() - 1; l > 0; --l) {
78  layers[l]->back_prop();
79  }
80  m->get_objective_function()->compute_weight_regularization();
81 
82  // Print objective function value
83  if (comm->am_world_master()) {
84  std::cout << "--------------------------------------------------------------------------------" << std::endl
85  << "Gradient checking..." << std::endl
86  << " Objective function value = " << objective << std::endl
87  << " Step size = " << step_size << std::endl
88  << " Expected gradient error = " << expected_error << std::endl;
89  }
90 
91  for (weights *w : m->get_weights()) {
92  if (w->get_optimizer() == nullptr) {
93  continue;
94  }
95  if (comm->am_world_master()) {
96  std::cout << "Checking " << w->get_name() << std::endl;
97  }
98 
99  // Get weights matrix and gradient
100  const AbsDistMat& weights_matrix = w->get_values();
101  const AbsDistMat& gradient = w->get_optimizer()->get_gradient();
102 
103  // Iterate through weights matrix entries
104  for (El::Int col = 0; col < weights_matrix.Width(); ++col) {
105  for (El::Int row = 0; row < weights_matrix.Height(); ++row) {
106  const bool weight_is_local = weights_matrix.IsLocal(row, col);
107  const El::Int local_row = (weight_is_local ?
108  weights_matrix.LocalRow(row) :
109  0);
110  const El::Int local_col = (weight_is_local ?
111  weights_matrix.LocalCol(col) :
112  0);
113  const DataType initial_weight = (weight_is_local ?
114  weights_matrix.GetLocal(local_row,
115  local_col) :
116  DataType(0));
117 
118  // Compute objective function values
119  // Note: matrix entry is reset after computing objective
120  // function values
121  w->set_value(initial_weight + 2 * step_size, row, col);
122  const DataType f_2h = compute_objective_function(m);
123  w->set_value(initial_weight + step_size, row, col);
124  const DataType f_h = compute_objective_function(m);
125  w->set_value(initial_weight - step_size, row, col);
126  const DataType f_nh = compute_objective_function(m);
127  w->set_value(initial_weight - 2 * step_size, row, col);
128  const DataType f_n2h = compute_objective_function(m);
129  w->set_value(initial_weight, row, col);
130 
131  // Compute relative error in gradient.
132  // Note: only weight owner participates
133  if (weight_is_local && weights_matrix.RedundantRank() == 0) {
134  const DataType analytical_gradient
135  = gradient.GetLocal(local_row, local_col);
136  const DataType numerical_gradient
137  = (- f_2h + 8 * f_h - 8 * f_nh + f_n2h) / (12 * step_size);
138  const DataType error = std::fabs(analytical_gradient - numerical_gradient);
139  auto relative_error = DataType(0);
140  if (error != DataType(0)) {
141  relative_error = error / std::max(std::fabs(analytical_gradient),
142  std::fabs(numerical_gradient));
143  }
144 
145  // Print warning if relative error is large
146  if (error > expected_error || std::isnan(error) || std::isinf(error)) {
147  std::cout << " GRADIENT ERROR: " << w->get_name() << ", "
148  << "entry (" << row << "," << col << ")" << std::endl;
149  std::cout << " Weight = " << initial_weight << std::endl
150  << " Analytical gradient = " << analytical_gradient << std::endl
151  << " Numerical gradient = " << numerical_gradient << std::endl
152  << " Error = " << error << std::endl
153  << " Relative error = " << relative_error << std::endl;
154  if (m_fail_on_error) {
155  throw lbann_exception("callback_gradient_check: found large error in gradient");
156  }
157  } else if (m_verbose) {
158  std::cout << " " << w->get_name() << ", "
159  << "entry (" << row << "," << col << ")" << std::endl;
160  std::cout << " Weight = " << initial_weight << std::endl
161  << " Analytical gradient = " << analytical_gradient << std::endl
162  << " Numerical gradient = " << numerical_gradient << std::endl
163  << " Error = " << error << std::endl
164  << " Relative error = " << relative_error << std::endl;
165  }
166  }
167 
168  }
169  }
170 
171  }
172 
173  if (comm->am_world_master()) {
174  std::cout << "--------------------------------------------------------------------------------" << std::endl;
175  }
176 
177 }

179 DataType lbann_callback_gradient_check::compute_objective_function(model *m) {
180  const std::vector<Layer*>& layers = m->get_layers();
181  objective_function* obj_fn = m->get_objective_function();
182  for (size_t l = 1; l < layers.size(); l++) {
183  layers[l]->forward_prop();
184  }
185  return obj_fn->evaluate(m->get_execution_mode());
186 }
