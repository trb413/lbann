40 void lbann_callback_gradient_check::on_test_begin(model *m) {
41 
42  // Get model members
43  lbann_comm *comm = m->get_comm();
44  const std::vector<Layer*>& layers = m->get_layers();
45 
46  // Initialize network for testing
47  for (auto&& w : m->get_weights()) {
48  auto&& opt = w->get_optimizer();
49  if (opt != nullptr) { opt->clear_gradient(); }
50  }
51  layers[0]->forward_prop();
52 
53  // Compute objective function
54  const DataType objective = compute_objective_function(m);
55 
56  // Choose finite difference step
57  // Note: Consider a central difference scheme:
58  // f'(x) ~ ( - f(x+2h) + 8 f(x+h) - 8 f(x-h) + f(x-2h) ) / 12h
59  // By Taylor's theorem, the truncation error is bounded by
60  // E_trunc <= | f'''''(xi) | / 18 * h^4
61  // Assuming f can be computed to a relative accuracy of epsilon,
62  // E_fl <= epsilon * | f(chi) | / h
63  // For simplicity, we assume f(chi) ~ f(x), and | f'''''(xi) | ~ 1.
64  // If step size is not specified, then we choose h so that
65  // E_fl <= sqrt(epsilon)
66  const DataType epsilon = std::pow(std::numeric_limits<DataType>::epsilon(), 0.9);
67  DataType step_size = m_step_size;
68  if (m_step_size <= DataType(0)) {
69  step_size = std::fabs(objective) * std::sqrt(epsilon);
70  }
71  DataType expected_error = (epsilon * objective / step_size
72  + std::pow(step_size, 4) / 18);
73  expected_error = std::pow(expected_error, 0.9);
74 
75  // Compute gradients
76  for (auto&& l : layers) {
77  l->clear_error_signals(m->get_current_mini_batch_size());
78  }
79  m->get_objective_function()->differentiate();
80  m->get_objective_function()->compute_weight_regularization();
81  for (int l = layers.size() - 1; l > 0; --l) {
82  layers[l]->back_prop();
83  }
84 
85  // Print objective function value
86  if (comm->am_world_master()) {
87  std::cout << "--------------------------------------------------------------------------------" << std::endl
88  << "Gradient checking..." << std::endl
89  << " Objective function value = " << objective << std::endl
90  << " Step size = " << step_size << std::endl
91  << " Expected gradient error = " << expected_error << std::endl;
92  }
93 
94  for (weights *w : m->get_weights()) {
95  if (w->get_optimizer() == nullptr) {
96  continue;
97  }
98  if (comm->am_world_master()) {
99  std::cout << "Checking " << w->get_name() << std::endl;
100  }
101 
102  // Get weights matrix and gradient
103  const AbsDistMat& weights_matrix = w->get_values();
104  const AbsDistMat& gradient = w->get_optimizer()->get_gradient();
105 
106  // Iterate through weights matrix entries
107  for (El::Int col = 0; col < weights_matrix.Width(); ++col) {
108  for (El::Int row = 0; row < weights_matrix.Height(); ++row) {
109  const bool weight_is_local = weights_matrix.IsLocal(row, col);
110  const El::Int local_row = (weight_is_local ?
111  weights_matrix.LocalRow(row) :
112  0);
113  const El::Int local_col = (weight_is_local ?
114  weights_matrix.LocalCol(col) :
115  0);
116  const DataType initial_weight = (weight_is_local ?
117  weights_matrix.GetLocal(local_row,
118  local_col) :
119  DataType(0));
120 
121  // Compute objective function values
122  // Note: matrix entry is reset after computing objective
123  // function values
124  w->set_value(initial_weight + 2 * step_size, row, col);
125  const DataType f_2h = compute_objective_function(m);
126  w->set_value(initial_weight + step_size, row, col);
127  const DataType f_h = compute_objective_function(m);
128  w->set_value(initial_weight - step_size, row, col);
129  const DataType f_nh = compute_objective_function(m);
130  w->set_value(initial_weight - 2 * step_size, row, col);
131  const DataType f_n2h = compute_objective_function(m);
132  w->set_value(initial_weight, row, col);
133 
134  // Compute relative error in gradient.
135  // Note: only weight owner participates
136  if (weight_is_local && weights_matrix.RedundantRank() == 0) {
137  const DataType analytical_gradient
138  = gradient.GetLocal(local_row, local_col);
139  const DataType numerical_gradient
140  = (- f_2h + 8 * f_h - 8 * f_nh + f_n2h) / (12 * step_size);
141  const DataType error = std::fabs(analytical_gradient - numerical_gradient);
142  auto relative_error = DataType(0);
143  if (error != DataType(0)) {
144  relative_error = error / std::max(std::fabs(analytical_gradient),
145  std::fabs(numerical_gradient));
146  }
147 
148  // Print warning if relative error is large
149  if (error > expected_error || std::isnan(error) || std::isinf(error)) {
150  std::cout << " GRADIENT ERROR: " << w->get_name() << ", "
151  << "entry (" << row << "," << col << ")" << std::endl;
152  std::cout << " Weight = " << initial_weight << std::endl
153  << " Analytical gradient = " << analytical_gradient << std::endl
154  << " Numerical gradient = " << numerical_gradient << std::endl
155  << " Error = " << error << std::endl
156  << " Relative error = " << relative_error << std::endl;
157  if (m_fail_on_error) {
158  throw lbann_exception("callback_gradient_check: found large error in gradient");
159  }
160  } else if (m_verbose) {
161  std::cout << " " << w->get_name() << ", "
162  << "entry (" << row << "," << col << ")" << std::endl;
163  std::cout << " Weight = " << initial_weight << std::endl
164  << " Analytical gradient = " << analytical_gradient << std::endl
165  << " Numerical gradient = " << numerical_gradient << std::endl
166  << " Error = " << error << std::endl
167  << " Relative error = " << relative_error << std::endl;
168  }
169  }
170 
171  }
172  }
173 
174  }
175 
176  if (comm->am_world_master()) {
177  std::cout << "--------------------------------------------------------------------------------" << std::endl;
178  }
179 
180 }

182 DataType lbann_callback_gradient_check::compute_objective_function(model *m) {
183  const std::vector<Layer*>& layers = m->get_layers();
184  objective_function* obj_fn = m->get_objective_function();
185  for (size_t l = 1; l < layers.size(); l++) {
186  layers[l]->forward_prop();
187  }
188  return obj_fn->evaluate(m->get_execution_mode(),
189  m->get_current_mini_batch_size());
190 }
