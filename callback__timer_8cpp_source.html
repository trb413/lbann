52 void lbann_callback_timer::timing_end(model *m) {
53  lbann_comm *comm = m->get_comm();
54 
55  // Get run time
56  const EvalType run_time = get_time() - m_start_time;
57 
58  // Compute minibatch statistics
59  const int num_batches = m_batch_times.size();
60  EvalType batch_time_mean = std::nan("");
61  EvalType batch_time_min = std::nan("");
62  EvalType batch_time_max = std::nan("");
63  EvalType batch_time_stdev = std::nan("");
64  if (num_batches > 0) {
65  batch_time_mean = std::accumulate(m_batch_times.begin(),
66  m_batch_times.end(),
67  0.0);
68  batch_time_mean /= num_batches;
69  batch_time_min = *std::min_element(m_batch_times.begin(),
70  m_batch_times.end());
71  batch_time_max = *std::max_element(m_batch_times.begin(),
72  m_batch_times.end());
73  }
74  if (num_batches > 1) {
75  const EvalType sqsum = std::inner_product(m_batch_times.begin(),
76  m_batch_times.end(),
77  m_batch_times.begin(),
78  0.0);
79  EvalType var = sqsum / num_batches - batch_time_mean * batch_time_mean;
80  var = num_batches * var / (num_batches - 1);
81  batch_time_stdev = std::sqrt(std::max(var, 0.0));
82  }
83 
84  // Get string for execution mode
85  std::string mode_string;
86  switch(m->get_execution_mode()) {
87  case execution_mode::training:
88  mode_string = "training epoch " + std::to_string(m->get_cur_epoch()-1);
89  break;
90  case execution_mode::validation:
91  mode_string = "validation";
92  break;
93  case execution_mode::testing:
94  mode_string = "test";
95  break;
96  default:
97  std::stringstream err;
98  err << __FILE__ << " " << __LINE__ << " :: "
99  << "invalid execution mode for reporting results";
100  throw lbann_exception(err.str());
101  }
102 
103  // Report timing results
104  const int num_models = comm->get_num_models();
105  if (comm->am_model_master()) {
106 
107  // Gather timing results in world master
108  std::vector<EvalType> run_time_list(num_models);
109  std::vector<EvalType> mean_list(num_models);
110  std::vector<EvalType> min_list(num_models);
111  std::vector<EvalType> max_list(num_models);
112  std::vector<EvalType> stdev_list(num_models);
113  if (comm->am_world_master()) {
114  comm->intermodel_gather(run_time, run_time_list);
115  comm->intermodel_gather(batch_time_mean, mean_list);
116  comm->intermodel_gather(batch_time_min, min_list);
117  comm->intermodel_gather(batch_time_max, max_list);
118  comm->intermodel_gather(batch_time_stdev, stdev_list);
119  } else {
120  comm->intermodel_gather(run_time, comm->get_intermodel_master());
121  comm->intermodel_gather(batch_time_mean, comm->get_intermodel_master());
122  comm->intermodel_gather(batch_time_min, comm->get_intermodel_master());
123  comm->intermodel_gather(batch_time_max, comm->get_intermodel_master());
124  comm->intermodel_gather(batch_time_stdev, comm->get_intermodel_master());
125  }
126 
127  // Print results
128  if (comm->am_world_master()) {
129  for (int i = 0; i < num_models; ++i) {
130  std::cout << "Model " << i << " " << mode_string << " "
131  << "run time : " << run_time_list[i] << "s"
132  << std::endl;
133  }
134  for (int i = 0; i < num_models; ++i) {
135  std::cout << "Model " << i << " " << mode_string << " "
136  << "mini-batch time statistics : ";
137  if (std::isnan(mean_list[i])) {
138  std::cout << "N/A";
139  } else {
140  std::cout << mean_list[i] << "s";
141  }
142  std::cout << " mean, ";
143  if (std::isnan(max_list[i])) {
144  std::cout << "N/A";
145  } else {
146  std::cout << max_list[i] << "s";
147  }
148  std::cout << " max, ";
149  if (std::isnan(min_list[i])) {
150  std::cout << "N/A";
151  } else {
152  std::cout << min_list[i] << "s";
153  }
154  std::cout << " min, ";
155  if (std::isnan(stdev_list[i])) {
156  std::cout << "N/A";
157  } else {
158  std::cout << stdev_list[i] << "s";
159  }
160  std::cout << " stdev" << std::endl;
161  }
162 
163  }
164  }
