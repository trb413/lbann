89  void update_weight_bias_matrix(ElMat &WB_D, ElMat& WB) {
90  // update accumulator
91  // KERAS: for p, g, a, c in zip(params, grads, accumulators, constraints):
92  // KERAS: new_a = self.rho * a + (1 - self.rho) * K.square(g)
93  Scale(rho /*DecayRate*/, WB_D_Cache);
94  Copy(WB_D, WB_D_Temp);
95  EntrywiseMap(WB_D_Temp, std::function<DataType(const DataType&)>(_sq));
96  Scale(1 - rho /*DecayRate*/, WB_D_Temp);
97  Axpy((DataType) 1, WB_D_Temp, WB_D_Cache);
98 
99  // update parameters
100  // KERAS: new_p = p - self.lr * g / K.sqrt(new_a + self.epsilon)
101  Copy(WB_D_Cache, WB_D_Temp);
102  EntrywiseMap(WB_D_Temp, std::function<DataType(const DataType&)>(_sqrt));
103  Copy(WB_D, WB_D_Temp2);
104  Hadamard(WB_D_Temp2, WB_D_Temp, WB_D);
105 
106  // WBL2NormSum = 0.0;
107  Axpy((DataType)-LearnRate, WB_D, WB);
108  }
