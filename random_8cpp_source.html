163 void init_random(int seed, lbann_comm *comm) {
164  if (seed != -1) {
165  // Seed every OpenMP thread, if present.
166  // Note: Threadprivate OMP variables don't work with dynamic threads.
167 #ifdef _OPENMP
168  #pragma omp parallel
169  {
170  get_generator().seed((seed << 8) | omp_get_thread_num());
171  get_fast_generator().seed((seed << 8) | omp_get_thread_num());
172  }
173 #else
174  get_generator().seed(seed);
175  get_fast_generator().seed(seed);
176 #endif
177 #ifdef LBANN_SET_EL_RNG
178  if (comm != nullptr) {
179  El::Generator().seed(seed ^ comm->get_rank_in_model());
180  } else {
181  El::Generator().seed(seed ^ El::mpi::Rank(El::mpi::COMM_WORLD));
182  }
183 #endif
184  } else {
185  // Seed with a random value.
186  std::random_device rd;
187  unsigned rand_val = rd();
188 #ifdef _OPENMP
189  #pragma omp parallel
190  {
191  get_generator().seed((rand_val << 8) | omp_get_thread_num());
192  get_fast_generator().seed((rand_val << 8) | omp_get_thread_num());
193  }
194 #else
195  get_generator().seed(rand_val);
196  get_fast_generator().seed(rand_val);
197 #endif
198 #ifdef LBANN_SET_EL_RNG
199  El::Generator().seed(rand_val);
200 #endif
201  }
202 }

249 void gaussian_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType mean,
250  DataType stddev) {
251  El::Zeros(mat, m, n);
252  if (mat.Grid().Rank() == 0) {
253  mat.Reserve(n * m);
254  auto& gen = get_generator();
255  std::normal_distribution<DataType> dist(mean, stddev);
256  for (El::Int col = 0; col < n; ++col) {
257  for (El::Int row = 0; row < m; ++row) {
258  mat.QueueUpdate(row, col, dist(gen));
259  }
260  }
261  }
262  mat.ProcessQueues();
263 }

280 void uniform_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType center,
281  DataType radius) {
282  El::Zeros(mat, m, n);
283  if (mat.Grid().Rank() == 0) {
284  mat.Reserve(n * m);
285  auto& gen = get_generator();
286  std::uniform_real_distribution<DataType> dist(center - radius,
287  center + radius);
288  for (El::Int col = 0; col < n; ++col) {
289  for (El::Int row = 0; row < m; ++row) {
290  mat.QueueUpdate(row, col, dist(gen));
291  }
292  }
293  }
294  mat.ProcessQueues();
295 }
