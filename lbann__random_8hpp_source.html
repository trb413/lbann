61 inline T fast_rand_int(Generator& g, T max) {
62  typename Generator::result_type x;
63  do {
64  x = g();
65  } while (x >= (Generator::max() - Generator::max() % max));
66  return x % max;
67 }

74 inline T fast_rand_int_pow2(Generator& g, T max) {
75  typename Generator::result_type x;
76  max -= 1;
77  const typename Generator::result_type upper = Generator::max() -
78  (Generator::max() & (typename Generator::result_type) max);
