63  //cout << "Setting up input layer, with " << Layer::m_comm->get_num_models() << " models and " << m_num_parallel_readers_training << " parallel readers and " << Layer::m_mini_batch_size << " mb size, which gives a stride of " << stride << endl;
64  io_layer::setup_data_readers_for_training(base_offset,
65  stride,
66  model_offset);
67  distributed_minibatch_parallel_io::calculate_num_iterations_per_epoch(m_training_dataset.data_reader);
69  io_layer::setup_data_readers_for_evaluation(Layer::m_comm->get_rank_in_model() * Layer::m_mini_batch_size,
70  m_num_parallel_readers_training * Layer::m_mini_batch_size);
71  } else {
72  io_layer::setup_data_readers_for_training(Layer::m_comm->get_rank_in_model() * Layer::m_mini_batch_size,
73  m_num_parallel_readers_training * Layer::m_mini_batch_size);
74  io_layer::setup_data_readers_for_evaluation(Layer::m_comm->get_rank_in_model() * Layer::m_mini_batch_size,
75  m_num_parallel_readers_training * Layer::m_mini_batch_size);

99  void fp_linearity() {
100  int num_samples_in_batch = fetch_to_local_matrix(Y_local);
101  if(is_current_root()) {
103  target_layer::update_num_samples_processed(num_samples_in_batch);
104  }
105 
106  int64_t curr_mini_batch_size = m_neural_network_model->get_current_mini_batch_size();
107  if(is_current_root() && num_samples_in_batch != curr_mini_batch_size) {
108  throw lbann_exception("lbann_target_layer_distributed_minibatch_parallel_io: number of labels does not match the current mini-batch size.");
109  }
111  distribute_from_local_matrix(Y_local, Ys);
112  Copy(Ys, *m_activations);
113 
115  DataType avg_error = m_neural_network_model->obj_fn->compute_obj_fn(*m_prev_activations_v, *m_activations_v);
116  m_neural_network_model->obj_fn->record_obj_fn(m_execution_mode, avg_error);
117 
118  for (auto&& m : m_neural_network_model->metrics) {
119  double num_errors = (int) m->compute_metric(*m_prev_activations_v, *m_activations_v);
120  m->record_error(num_errors, curr_mini_batch_size);
121  }
122 
123  return;
124  }
