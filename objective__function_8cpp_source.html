75 EvalType objective_function::evaluate(execution_mode mode,
76  int mini_batch_size) {
77  const auto start_time = get_time();
78  EvalType value = EvalType(0);
79  for (const auto& term : m_terms) {
80  value += term->evaluate();
81  }
82  m_statistics[mode].add_value(mini_batch_size * value,
83  mini_batch_size);
84  m_evaluation_time += get_time() - start_time;
85  return value;
86 }

132 void objective_function::set_layer_pointers(std::vector<Layer*> layers) {
133  auto it = layers.begin();
134  for (objective_function_term *term : m_terms) {
135  const size_t num_layers = term->get_layer_pointers().size();
136  std::vector<Layer*> term_layers(it, it + num_layers);
137  term->set_layer_pointers(term_layers);
138  it += num_layers;

140  if (it != layers.end()) {
141  std::stringstream err;

143  << "attempted to set an invalid number of layer pointers "

146  throw lbann_exception(err.str());
