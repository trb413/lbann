100 void csv_reader::load() {
101  bool master = m_comm->am_world_master();
102  setup_ifstreams();
103  std::ifstream& ifs = *m_ifstreams[0];
104  const El::mpi::Comm world_comm = m_comm->get_world_comm();
105  // Parse the header to determine how many columns there are.
106  // Skip rows if needed.
107  if (master) {
108  skip_rows(ifs, m_skip_rows);
109  }
110  m_comm->broadcast<int>(0, m_skip_rows, world_comm);
111 
112  //This will be broadcast from root to other procs, and will
113  //then be converted to std::vector<int> m_labels; this is because
114  //El::mpi::Broadcast<std::streampos> doesn't work
115  std::vector<long long> index;
116 
117  if (master) {
118  std::string line;
119  std::streampos header_start = ifs.tellg();
120  // TODO: Skip comment lines.
121  if (std::getline(ifs, line)) {
122  m_num_cols = std::count(line.begin(), line.end(), m_separator) + 1;
123  if (m_skip_cols >= m_num_cols) {
124  throw lbann_exception(
125  "csv_reader: asked to skip more columns than are present");
126  }
127 
128  if (!m_disable_labels) {
129  if (m_label_col < 0) {
130  // Last column becomes the label column.
131  m_label_col = m_num_cols - 1;
132  }
133  if (m_label_col >= m_num_cols) {
134  throw lbann_exception(
135  "csv_reader: label column" + std::to_string(m_label_col) +
136  " is not present");
137  }
138  }
139 
140  if (!m_disable_responses) {
141  if (m_response_col < 0) {
142  // Last column becomes the response column.
143  m_response_col = m_num_cols - 1;
144  }
145  if (m_response_col >= m_num_cols) {
146  throw lbann_exception(
147  "csv_reader: response column" + std::to_string(m_response_col) +
148  " is not present");
149  }
150  }
151 
152  } else {
153  throw lbann_exception(
154  "csv_reader: failed to read header in " + get_data_filename());
155  }
156  if (ifs.eof()) {
157  throw lbann_exception(
158  "csv_reader: reached EOF after reading header");
159  }
160  // If there was no header, skip back to the beginning.
161  if (!m_has_header) {
162  ifs.clear();
163  ifs.seekg(header_start, std::ios::beg);
164  }
165  // Construct an index mapping each line (sample) to its offset.
166  // TODO: Skip comment lines.
167  // Used to count the number of label classes.
168  std::unordered_set<int> label_classes;
169  index.push_back(ifs.tellg());
170 
171  int num_samples_to_use = get_absolute_sample_count();
172  int line_num = 0;
173  if (num_samples_to_use == 0) {
174  num_samples_to_use = -1;
175  }
176  while (std::getline(ifs, line)) {
177  if (line_num == num_samples_to_use) {
178  break;
179  }
180  ++line_num;
181 
182  // Verify the line has the right number of columns.
183  if (std::count(line.begin(), line.end(), m_separator) + 1 != m_num_cols) {
184  throw lbann_exception(
185  "csv_reader: line " + std::to_string(line_num) +
186  " does not have right number of entries");
187  }
188  index.push_back(ifs.tellg());
189  // Extract the label.
190  if (!m_disable_labels) {
191  size_t cur_pos = 0;
192  for (int col = 0; col < m_num_cols; ++col) {
193  size_t end_pos = line.find_first_of(m_separator, cur_pos);
194  if (col == m_label_col) {
195  int label = m_label_transform(line.substr(cur_pos, end_pos - cur_pos));
196  label_classes.insert(label);
197  m_labels.push_back(label);
198  break;
199  }
200  cur_pos = end_pos + 1;
201  }
202  }
203  // Possibly extract the response.
204  if (!m_disable_responses) {
205  size_t cur_pos = 0;
206  for (int col = 0; col < m_num_cols; ++col) {
207  size_t end_pos = line.find_first_of(m_separator, cur_pos);
208  if (col == m_response_col) {
209  DataType response = m_response_transform(
210  line.substr(cur_pos, end_pos - cur_pos));
211  m_responses.push_back(response);
212  break;
213  }
214  cur_pos = end_pos + 1;
215  }
216  }
217  }
218 
219  if (!ifs.eof() && num_samples_to_use == 0) {
220  //If we didn't get to EOF, something went wrong.
221  throw lbann_exception(
222  "csv_reader: did not reach EOF");
223  }
224  if (!m_disable_labels) {
225  // Do some simple validation checks on the classes.
226  // Ensure the elements begin with 0, and there are no gaps.
227  auto minmax = std::minmax_element(label_classes.begin(), label_classes.end());
228  if (*minmax.first != 0) {
229  throw lbann_exception(
230  "csv_reader: classes are not indexed from 0");
231  }
232  if (*minmax.second != (int) label_classes.size() - 1) {
233  throw lbann_exception(
234  "csv_reader: label classes are not contiguous");
235  }
236  m_num_labels = label_classes.size();
237  }
238  ifs.clear();
239  } // if (master)
240 
241  m_comm->broadcast<int>(0, m_num_cols, world_comm);
242  m_label_col = m_num_cols - 1;
243 
244  //bcast the index vector
245  m_comm->world_broadcast<long long>(0, index);
246  m_num_samples = index.size() - 1;
247  if (m_master) std::cerr << "num samples: " << m_num_samples << "\n";
248 
249  m_index.reserve(index.size());
250  for (auto t : index) {
251  m_index.push_back(t);
252  }
253 
254  //optionally bcast the response vector
255  if (!m_disable_responses) {
256  m_response_col = m_num_cols - 1;
257  m_comm->world_broadcast<DataType>(0, m_responses);
258  }
259 
260  //optionally bcast the label vector
261  if (!m_disable_labels) {
262  m_comm->world_broadcast<int>(0, m_labels);
263  m_num_labels = m_labels.size();
264  }
265 
266  // Reset indices.
267  m_shuffled_indices.resize(m_num_samples);
268  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
269  select_subset_of_data();
270 }

299 std::vector<DataType> csv_reader::fetch_line(int data_id) {
300  std::string line = fetch_raw_line(data_id);
301  std::vector<DataType> parsed_line;
302  // Note: load already verified that every line is properly formatted.
303  size_t cur_pos = 0; // Current *start* of a column.
304  for (int col = 0; col < m_num_cols; ++col) {
305  // Note for last column, this returns std::npos, which substr handles.
306  size_t end_pos = line.find_first_of(m_separator, cur_pos);
307  // Skip columns if needed.
308  if (col < m_skip_cols) {
309  cur_pos = end_pos + 1;
310  continue;
311  }
312  // @TODO Note: This results in a copy, switch to string_view when we can.
313  std::string str_val = line.substr(cur_pos, end_pos - cur_pos);
314  cur_pos = end_pos + 1;
315  DataType val;
316  if (m_col_transforms.count(col)) {
317  val = m_col_transforms[col](str_val);
318  } else {
319  // No easy way to parameterize based on DataType, so always use double.
320  val = std::stod(str_val);
321  }
322  parsed_line.push_back(val);
323  }
324  return parsed_line;
325 }

327 std::vector<DataType> csv_reader::fetch_line_label_response(
328  int data_id) {
329  std::string line = fetch_raw_line(data_id);
330  std::vector<DataType> parsed_line;
331  // Note: load already verified that every line is properly formatted.
332  size_t cur_pos = 0; // Current *start* of a column.
333  for (int col = 0; col < m_num_cols; ++col) {
334  // Note for last column, this returns std::npos, which substr handles.
335  size_t end_pos = line.find_first_of(m_separator, cur_pos);
336  // Skip the label, response, and any columns if needed.
337  if ((!m_disable_labels && col == m_label_col) ||
338  (!m_disable_responses && col == m_response_col) ||
339  col < m_skip_cols) {
340  cur_pos = end_pos + 1;
341  continue;
342  }
343  // Note: This results in a copy, switch to string_view when we can.
344  std::string str_val = line.substr(cur_pos, end_pos - cur_pos);
345  cur_pos = end_pos + 1;
346  DataType val;
347  if (m_col_transforms.count(col)) {
348  val = m_col_transforms[col](str_val);
349  } else {
350  // No easy way to parameterize based on DataType, so always use double.
351  try {
352  val = std::stod(str_val);
353  } catch (std::invalid_argument& e) {
354  throw lbann_exception(
355  "csv_reader: could not convert '" + str_val + "'");
356  }
357  }
358  parsed_line.push_back(val);
359  }
360  return parsed_line;
361 }

363 std::string csv_reader::fetch_raw_line(int data_id) {
364 static int n = 0;
365  std::ifstream& ifs = *m_ifstreams[omp_get_thread_num()];
366  // Seek to the start of this datum's line.
367  ifs.seekg(m_index[data_id], std::ios::beg);
368  // Compute the length of the line to read, excluding newline.
369  std::streamsize cnt = m_index[data_id+1] - m_index[data_id] - 1;
370  // Read directly into a string buffer.
371  std::string line;
372  line.resize(cnt);
373  if (!ifs.read(&line[0], cnt)) {
374  std::stringstream err;
375  err << __FILE__ << " " << __LINE__ << " :: "
376  << "csv_reader: error on reading data_id: " << data_id << "\n"
377  <<"index.size(): " << m_index.size() << " m_index["<< data_id
378  << ")=" <<m_index[data_id] << "; index[" << data_id+1
379  << ")= " << m_index[data_id+1] << "\ncnt: " << cnt << " role: "
380  << get_role() << " gcount: " << ifs.gcount() << " n: " << n;
381  throw lbann_exception(err.str());
382  }
383 ++n;
384  return line;
385 }
