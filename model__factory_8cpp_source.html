36 model* instantiate_model(lbann_comm* comm,
37  objective_function* obj,
38  const lbann_data::Optimizer& proto_opt,
39  const lbann_data::Model& proto_model) {
40  std::stringstream err;
41 
42  // Default optimizer
43  auto&& opt = construct_optimizer(comm, proto_opt);
44 
45  // Construct model
46  const auto& type = proto_model.name();
47  const auto& mini_batch_size = proto_model.mini_batch_size();
48  if (type == "sequential_model" || type == "") {
49  return new sequential_model(comm, mini_batch_size, obj, opt);
50  }
51  if (type == "directed_acyclic_graph_model") {
52  return new directed_acyclic_graph_model(comm, mini_batch_size, obj, opt);
53  }
54  if (type == "recurrent_model") {
55  const auto& params = proto_model.recurrent();
56  return new recurrent_model(comm,
57  mini_batch_size,
58  obj,
59  opt,
60  params.unroll_depth());
61  }
62  if (type == "siamese_model") {
63  const auto& params = proto_model.siamese();
64  return new siamese_model(comm,
65  mini_batch_size,
66  obj,
67  opt,
68  params.num_heads());
69  }
70 
71  // Throw error if model type is not supported
72  err << "unknown model type (" << type << ")";
73  LBANN_ERROR(err.str());
74  return nullptr;
75 
76 }

78 void assign_layers_to_objective_function(std::vector<Layer*>& layer_list,
79  objective_function* obj,
80  const lbann_data::ObjectiveFunction& proto_obj) {
81  std::stringstream err;
82 
83  // Construct map from layer names to layers
84  std::unordered_map<std::string, Layer*> names_to_layers;
85  for (auto&& l : layer_list) {
86  const auto& name = l->get_name();
87  if (names_to_layers.count(name) > 0) {
88  err << "layer name \"" << name << "\" is not unique";
89  LBANN_ERROR(err.str());
90  }
91  names_to_layers[name] = l;
92  }
93 
94  // Assign evaluation layers to layer terms in objective function
95  auto&& obj_terms = obj->get_terms();
96  int num_layer_terms = 0;
97  for (size_t i = 0; i < obj_terms.size(); ++i) {
98  auto&& term = dynamic_cast<layer_term*>(obj_terms[i]);
99  if (term != nullptr) {
100  ++num_layer_terms;
101  if (num_layer_terms > proto_obj.layer_term_size()) { continue; }
102  const auto& params = proto_obj.layer_term(num_layer_terms-1);
103  auto&& eval = names_to_layers[params.layer()];
104  term->set_evaluation_layer(eval);
105  }
106  }
107 
108  // Check that layer terms in objective function match prototext
109  if (num_layer_terms != proto_obj.layer_term_size()) {
110  err << "number of layer terms in objective function does not match prototext "
111  << "(expected " << proto_obj.layer_term_size() << ", "
112  << "found " << num_layer_terms << ")";
113  LBANN_ERROR(err.str());
114  }
115 
116 }

139  auto&& eval = names_to_layers[params.layer()];

147 void assign_weights_to_layers(std::vector<Layer*>& layer_list,
148  std::vector<weights*>& weights_list,
149  const lbann_data::Model& proto_model) {
150  std::stringstream err;
151 
152  // Construct map from weights names to weights
153  std::unordered_map<std::string, weights*> names_to_weights;
154  for (auto&& w : weights_list) {
155  const auto& name = w->get_name();
156  if (names_to_weights.count(name) > 0) {
157  err << "weights name \"" << name << "\" is not unique";
158  LBANN_ERROR(err.str());
159  }
160  names_to_weights[name] = w;
161  }
162 
163  // Find weights assigned to each layer
164  for (int i=0; i<proto_model.layer_size(); ++i) {
165  const auto& proto_layer = proto_model.layer(i);
166  auto& layer_weights = layer_list[i]->get_weights();
167  for (auto&& name : parse_list<std::string>(proto_layer.weights())) {
168  auto&& w = names_to_weights[name];
169  if (w == nullptr) {
170  err << "could not find weights named \"" << name << "\", "
171  << "which are expected by layer " << layer_list[i]->get_name();
172  LBANN_ERROR(err.str());
173  }
174  layer_weights.push_back(w);
175  }
176  }

182 model* construct_model(lbann_comm* comm,
183  cudnn::cudnn_manager* cudnn,
184  std::map<execution_mode, generic_data_reader*>& data_readers,
185  const lbann_data::Optimizer& proto_opt,
186  const lbann_data::Model& proto_model) {
187 
188  // Add layer graph
189  auto&& layer_list = construct_layer_graph(comm,
190  data_readers,
191  cudnn,
192  proto_model);
193 
194  // Construct objective function
195  const auto& proto_obj = proto_model.objective_function();
196  auto&& obj = construct_objective_function(proto_obj);
197  assign_layers_to_objective_function(layer_list, obj, proto_obj);
198 
199  // Instantiate model
200  auto&& m = instantiate_model(comm, obj, proto_opt, proto_model);
201  for (auto&& l : layer_list) { m->add_layer(l); }
202 
203  // Add weights and assign to layers
204  for (int i=0; i<proto_model.weights_size(); i++) {
205  m->add_weights(construct_weights(comm,
206  cudnn,
207  proto_opt,
208  proto_model.weights(i)));
209  }
210  auto weights_list = m->get_weights();
211  assign_weights_to_layers(layer_list, weights_list, proto_model);
212 
213  // Add metrics
214  for (int i=0; i<proto_model.metric_size(); ++i) {
215  m->add_metric(construct_metric(comm, proto_model.metric(i)));
216  }
217  auto metric_list = m->get_metrics();
218  assign_layers_to_metrics(layer_list, metric_list, proto_model);
219 
220  // Add callbacks
221  auto&& summarizer = construct_summarizer(comm, proto_model);
222  for (int i=0; i<proto_model.callback_size(); i++) {
223  m->add_callback(construct_callback(comm,
224  proto_model.callback(i),
225  data_readers,
226  layer_list,
227  weights_list,
228  summarizer));
229  }
230 
231  return m;
