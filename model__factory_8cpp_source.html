35 model* instantiate_model(lbann_comm* comm,
36  objective_function* obj,
37  optimizer* opt,
38  const lbann_data::Model& proto_model) {
39  std::stringstream err;
40 
41  // Construct model
42  const auto& type = proto_model.name();
43  const auto& mini_batch_size = proto_model.mini_batch_size();
44  if (type == "sequential_model" || type == "") {
45  return new sequential_model(comm, mini_batch_size, obj, opt);
46  }
47  if (type == "directed_acyclic_graph_model") {
48  return new directed_acyclic_graph_model(comm, mini_batch_size, obj, opt);
49  }
50  if (type == "recurrent_model") {
51  const auto& params = proto_model.recurrent();
52  return new recurrent_model(comm,
53  mini_batch_size,
54  obj,
55  opt,
56  params.unroll_depth());
57  }
58  if (type == "siamese_model") {
59  const auto& params = proto_model.siamese();
60  return new siamese_model(comm,
61  mini_batch_size,
62  obj,
63  opt,
64  params.num_heads());
65  }
66 
67  // Throw error if model type is not supported
68  err << "unknown model type (" << type << ")";
69  LBANN_ERROR(err.str());
70  return nullptr;
71 
72 }

75 void assign_weights_to_layers(std::vector<Layer*>& layer_list,
76  std::vector<weights*>& weights_list,
77  const lbann_data::Model& proto_model) {
78  std::stringstream err;
79 
80  // Construct map from weights names to weights
81  std::unordered_map<std::string, weights*> names_to_weights;
82  for (auto&& w : weights_list) {
83  const auto& name = w->get_name();
84  if (names_to_weights.count(name) > 0) {
85  err << "weights name \"" << name << "\" is not unique";
86  LBANN_ERROR(err.str());
87  }
88  names_to_weights[name] = w;
89  }
90 
91  // Find weights assigned to each layer
92  for (int i=0; i<proto_model.layer_size(); ++i) {
93  const auto& proto_layer = proto_model.layer(i);
94  auto& layer_weights = layer_list[i]->get_weights();
95  for (auto&& name : parse_list<std::string>(proto_layer.weights())) {
96  auto&& w = names_to_weights[name];
97  if (w == nullptr) {
98  err << "could not find weights named \"" << name << "\", "
99  << "which are expected by layer " << layer_list[i]->get_name();
100  LBANN_ERROR(err.str());
101  }
102  layer_weights.push_back(w);
103  }
104  }
