73 void lbann_callback_imcomm::setup(model *m) {
74  for (weights *w : m->get_weights()) {
75 
76  // Add weights if not already in list
77  if (m_weights_params.find(w) == m_weights_params.end()) {
78  m_weights_params[w] = {};
79  m_weights_params[w].ct = (w->get_optimizer() != nullptr ?
80  m_default_ct :
81  NONE);
82  }
83 
84  // Setup imcomm parameters if needed
85  imcomm_params& params = m_weights_params[w];
86  if (params.ct != NONE) {
87  optimizer *opt = w->get_optimizer();
88  if (opt == nullptr) {
89  std::stringstream err;
90  err << __FILE__ << " " << __LINE__ << " :: "
91  << "imcomm: trying to do inter-model gradient communication on "
92  << w->get_name() << ", which has no optimizer";
93  throw(err.str());
94  }
95  if (ct_needs_reshape(params.ct)) {
96  // Currently, no weights need reshaping.
97  }
98  if (ct_does_quantization(params.ct)) {
99  const AbsDistMat& gradients = opt->get_gradient();
100  if (params.reshape_height > 0) {
101  El::Zeros(params.error, params.reshape_height, params.reshape_width);
102  } else {
103  El::Zeros(params.error, gradients.LocalHeight(), gradients.LocalWidth());
104  }
105  }
106  }
107 
108  }
109 }

111 void lbann_callback_imcomm::on_train_begin(model *m) {
112  lbann_comm *comm = m->get_comm();
113  if (comm->get_num_models() == 1) {
114  return; // No point with only one model.
115  }
116  for (weights *w : m->get_weights()) {
117  AbsDistMat *values = w->get_values().Copy();
118  comm->intermodel_broadcast_matrix(*values, 0);
119  w->set_values(*values);
120  delete values;
121  }
122 }

124 void lbann_callback_imcomm::on_epoch_end(model *m) {
125  lbann_comm *comm = m->get_comm();
126  if (comm->get_num_models() == 1 ||
127  m->get_execution_mode() != execution_mode::training) {
128  return; // No point with only one model.
129  }
130  for (weights *w : m->get_weights()) {
131  imcomm_params& params = m_weights_params[w];
132  optimizer *opt = w->get_optimizer();
133  if (ct_does_quantization(params.ct)) {
134  comm->intermodel_sum_matrix(params.error);
135  opt->clear_gradient();
136  auto gradient = opt->get_gradient().Copy();
137  Mat *local_gradients = nullptr;
138  Mat reshaped;
139  if (params.reshape_height > 0) {
140  reshape_mat(gradient->Matrix(),
141  reshaped, params.reshape_height, params.reshape_width);
142  local_gradients = &reshaped;
143  } else {
144  local_gradients = &(gradient->Matrix());
145  }
146  *local_gradients = params.error;
147  opt->add_to_gradient(*gradient);
148  delete gradient;
149  // Apply optimizer update with accumulated gradient error.
150  opt->step();
151  El::Zero(params.error);
152  }
153  }
154 }

156 void lbann_callback_imcomm::on_backward_prop_end(model *m) {
157  lbann_comm *comm = m->get_comm();
158  if (comm->get_num_models() == 1 ||
159  m->get_execution_mode() != execution_mode::training) {
160  return; // No point with only one model.
161  }
162  for (weights *w : m->get_weights()) {
163  EvalType start_time = get_time();
164  imcomm_params& params = m_weights_params[w];
165  if (params.ct == NONE) {
166  continue;
167  }
168  optimizer *opt = w->get_optimizer();
169  auto gradient = opt->get_gradient().Copy();
170  Mat* local_gradients = nullptr;
171  Mat reshaped;
172  if (params.reshape_height > 0) {
173  reshape_mat(gradient->Matrix(),
174  reshaped, params.reshape_height, params.reshape_width);
175  local_gradients = &reshaped;
176  } else {
177  local_gradients = &(gradient->Matrix());
178  }
179  switch (params.ct) {
180  case NORMAL:
181  comm->intermodel_sum_matrix(*local_gradients);
182  break;
183  case ONEBIT_QUANTIZATION:
184  m_quantizer.intermodel_sum_onebit_quantized(
185  comm, *local_gradients, params.error);
186  break;
187  case THRESH_QUANTIZATION:
188  m_quantizer.intermodel_sum_threshold_quantized(
189  comm, *local_gradients, params.error, params.pos_thresh, params.neg_thresh);
190  break;
191  case ADAPTIVE_QUANTIZATION:
192  m_quantizer.intermodel_sum_adaptive_quantized(
193  comm, *local_gradients, params.error, params.proportion);
194  break;
195  default:
196  throw(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " :: "
197  + "imcomm: unknown comm type");
198  }
199  opt->clear_gradient();
200  opt->add_to_gradient(*gradient);
201  delete gradient;
202  EvalType im_time = get_time() - start_time;
203  do_summary(m, w, im_time);
204  }

207 void lbann_callback_imcomm::do_summary(model *m, weights *w,
208  EvalType im_time) {
209  if (m_summarizer == nullptr) {
210  return;
211  }
212  lbann_comm *comm = m->get_comm();
213  std::string prefix = w->get_name() + "/imcomm_";
214  m_summarizer->reduce_scalar(prefix + "time",
215  im_time, m->get_cur_step());
216  size_t bytes_sent = 0;
217  size_t bytes_received = 0;
218  if (ct_does_quantization(m_weights_params[w].ct)) {
219  bytes_sent = comm->get_ar_bytes_sent();
220  bytes_received = comm->get_ar_bytes_received();
221  } else {
222  // Use the same approximation the comm layer does.
223  const Mat& local_gradients =
224  w->get_optimizer()->get_gradient().LockedMatrix();
225  bytes_sent =
226  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
227  bytes_received =
228  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
229  }
230  m_summarizer->reduce_scalar(prefix + "bytes_sent",
231  bytes_sent, m->get_cur_step());
232  m_summarizer->reduce_scalar(prefix + "bytes_received",
233  bytes_received, m->get_cur_step());
234  if (ct_does_quantization(m_weights_params[w].ct)) {
235  m_summarizer->reduce_scalar(prefix + "rs_bytes_sent",
236  comm->get_ar_rs_bytes_sent(),
237  m->get_cur_step());
238  m_summarizer->reduce_scalar(prefix + "ag_bytes_sent",
239  comm->get_ar_ag_bytes_sent(),
240  m->get_cur_step());
241  m_summarizer->reduce_scalar(prefix + "rs_bytes_received",
242  comm->get_ar_rs_bytes_received(),
243  m->get_cur_step());
244  m_summarizer->reduce_scalar(prefix + "ag_bytes_received",
245  comm->get_ar_ag_bytes_received(),
246  m->get_cur_step());
247  m_summarizer->reduce_scalar(prefix + "ar_send_trans_time",
248  comm->get_ar_send_transform_time(),
249  m->get_cur_step());
250  m_summarizer->reduce_scalar(prefix + "ar_recv_trans_time",
251  comm->get_ar_recv_transform_time(),
252  m->get_cur_step());
253  m_summarizer->reduce_scalar(prefix + "ar_recv_apply_trans_time",
254  comm->get_ar_recv_apply_transform_time(),
255  m->get_cur_step());
256  if (m_weights_params[w].ct == ADAPTIVE_QUANTIZATION) {
257  m_summarizer->reduce_scalar(prefix + "quantized_count",
258  m_quantizer.get_quantized_count(),
259  m->get_cur_step());
260  }
261  m_quantizer.reset_counters();
262  comm->reset_stats_counters();
263  }
264 }
