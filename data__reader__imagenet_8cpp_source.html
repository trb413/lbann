88 bool imagenet_reader::replicate_processor(const cv_process& pp) {
89  const int nthreads = omp_get_max_threads();
90  m_pps.resize(nthreads);
91 
92  // Construct thread private preprocessing objects out of a shared pointer
93  #pragma omp parallel for schedule(static, 1)
94  for (int i = 0; i < nthreads; ++i) {
95  //auto ppu = std::make_unique<cv_process>(pp); // c++14
96  std::unique_ptr<cv_process> ppu(new cv_process(pp));
97  m_pps[i] = std::move(ppu);
98  }
99 
100  bool ok = true;
101  for (int i = 0; ok && (i < nthreads); ++i) {
102  if (!m_pps[i]) ok = false;
103  }
104 
105  if (!ok || (nthreads <= 0)) {
106  std::stringstream err;
107  err << __FILE__<<" "<<__LINE__<< " :: " << get_type() << " cannot replicate image processor";
108  throw lbann_exception(err.str());
109  return false;
110  }
111 
112  const std::vector<unsigned int> dims = pp.get_data_dims();
113  if ((dims.size() == 2u) && (dims[0] != 0u) && (dims[1] != 0u)) {
114  m_image_width = static_cast<int>(dims[0]);
115  m_image_height = static_cast<int>(dims[1]);
116  set_linearized_image_size();
117  }
118 
119  return true;
120 }

126 bool imagenet_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
127  const std::string imagepath = get_file_dir() + m_image_list[data_id].first;
128 
129  int width=0, height=0, img_type=0;
130 
131  std::vector<unsigned char> *image_buf;
132 
133  ::Mat X_v = create_datum_view(X, mb_idx);
134 
135  bool ret;
136  if (m_data_store != nullptr) {
137  m_data_store->get_data_buf(data_id, image_buf, 0);
138  ret = lbann::image_utils::load_image(*image_buf, width, height, img_type, *(m_pps[tid]), X_v);
139  } else {
140  ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v);
141  }
142 
143  if(!ret) {
144  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
145  + get_type() + ": image_utils::load_image failed to load - "
146  + imagepath);
147  }
148  if((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
149  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
150  + get_type() + ": mismatch data size -- either width, height or channel - "
151  + imagepath + "[w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
152  + "x" + std::to_string(CV_MAT_CN(img_type)) + "]");
153  }
154  return true;
155 }
