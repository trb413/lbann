64 void data_reader_merge_features::load() {
65  // Load each data reader separately.
66  for (auto&& reader : m_data_readers) {
67  double tm1 = get_time();
68  reader->set_comm(m_comm);
69  reader->load();
70  m_data_size += reader->get_linearized_data_size();
71  if (is_master()) {
72  std::cerr << "time to set up subsidiary reader: " << get_time() - tm1 << "\n";
73  }
74  }
75  // Verify the readers have the same number of samples.
76  int num_samples = m_data_readers[0]->get_num_data();
77  for (auto&& reader : m_data_readers) {
78  if (num_samples != reader->get_num_data()) {
79  throw lbann_exception(
80  "data_reader_merge_features: data readers do not have the same amount of data");
81  }
82  }
83  m_label_reader->load();
84  // Reset indices.
85  m_shuffled_indices.resize(num_samples);
86  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
87  select_subset_of_data();
88 }

90 bool data_reader_merge_features::fetch_datum(Mat& X, int data_id, int mb_idx,
91  int tid) {
92  int start = 0;
93  for (auto&& reader : m_data_readers) {
94  auto X_view = X(El::IR(start, start + reader->get_linearized_data_size()),
95  El::ALL);
96  reader->fetch_datum(X_view, data_id, mb_idx, tid);
97  start += reader->get_linearized_data_size();
98  }
99  return true;
100 }
