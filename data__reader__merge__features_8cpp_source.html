61 void data_reader_merge_features::load() {
62  // Load each data reader separately.
63  for (auto&& reader : m_data_readers) {
64  reader->load();
65  m_data_size += reader->get_linearized_data_size();
66  }
67  // Verify the readers have the same number of samples.
68  int num_samples = m_data_readers[0]->get_num_data();
69  for (auto&& reader : m_data_readers) {
70  if (num_samples != reader->get_num_data()) {
71  throw lbann_exception(
72  "data_reader_merge_features: data readers do not have the same amount of data");
73  }
74  }
75  m_label_reader->load();
76  // Reset indices.
77  m_shuffled_indices.resize(num_samples);
78  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
79  select_subset_of_data();
80 }

82 bool data_reader_merge_features::fetch_datum(Mat& X, int data_id, int mb_idx,
83  int tid) {
84  int start = 0;
85  for (auto&& reader : m_data_readers) {
86  auto X_view = X(El::IR(start, start + reader->get_linearized_data_size()),
87  El::ALL);
88  reader->fetch_datum(X_view, data_id, mb_idx, tid);
89  start += reader->get_linearized_data_size();
90  }
91  return true;
