31 EvalType boolean_false_positives_metric::evaluate_compute(
32  const AbsDistMat& prediction, const AbsDistMat& ground_truth) {
33  const El::Int height = prediction.Height();
34  const El::Int local_height = prediction.LocalHeight();
35  const El::Int local_width = prediction.LocalWidth();
36  const Mat& prediction_local = prediction.LockedMatrix();
37  const Mat& ground_truth_local = ground_truth.LockedMatrix();
38 
39  // Compute sum of predictions.
40  EvalType sum = 0;
41  #pragma omp parallel for reduction(+:sum) collapse(2)
42  for(El::Int col = 0; col < local_width; ++col) {
43  for(El::Int row = 0; row < local_height; ++row) {
44  const bool true_val = ground_truth_local(row, col) > DataType(0.5);
45  const bool pred_val = prediction_local(row, col) > DataType(0.5);
46  sum += EvalType((true_val != pred_val) && pred_val);
47  }
48  }
49 
50  // Compute mean value across mini-batch.
51  return get_comm().allreduce(sum / height, prediction.DistComm()) *
52  EvalType(100);
53 }
