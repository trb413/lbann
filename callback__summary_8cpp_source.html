47 void lbann_callback_summary::on_batch_end(model *m) {
48  m->summarize_stats(*m_summarizer);
49  if (m->get_cur_step() % m_mat_interval == 0) {
50  m->summarize_matrices(*m_summarizer);
51  }
52  lbann_comm *comm = m->get_comm();
53  size_t bytes_sent = comm->get_bytes_sent();
54  size_t bytes_received = comm->get_bytes_received();
55  size_t model_barriers = comm->get_num_model_barriers();
56  size_t intermodel_barriers = comm->get_num_intermodel_barriers();
57  size_t global_barriers = comm->get_num_global_barriers();
58  comm->reset_stats_counters();
59  m_summarizer->sum_reduce_scalar("bytes_sent", bytes_sent, m->get_cur_step());
60  m_summarizer->sum_reduce_scalar("bytes_received", bytes_received,
61  m->get_cur_step());
62  m_summarizer->reduce_scalar("model_barriers", model_barriers,
63  m->get_cur_step());
64  m_summarizer->reduce_scalar("intermodel_barriers", intermodel_barriers,
65  m->get_cur_step());
66  m_summarizer->reduce_scalar("global_barriers", global_barriers,
67  m->get_cur_step());
68 }

70 void lbann_callback_summary::on_epoch_end(model *m) {
71  for (auto&& metric : m->get_metrics()) {
72  double train_score = metric->report_metric(execution_mode::training);
73  // Replace spaces with _ for consistency.
74  std::string metric_name = metric->name();
75  std::transform(metric_name.begin(), metric_name.end(), metric_name.begin(),
76  [] (char c) { return c == ' ' ? '_' : c; });
77  std::string phase = "train_" + metric_name;
78  m_summarizer->reduce_scalar(phase, train_score, m->get_cur_step());
79  }
80  save_histograms(m);
81  m_summarizer->flush();
82 }

102 void lbann_callback_summary::save_histograms(model *m) {
103  for (const auto& layer : m->get_layers()) {
104  std::string prefix = "layer" + std::to_string(layer->get_index()) +
105  "/";
106  m_summarizer->reduce_histogram(prefix + "activations",
107  layer->get_activations(),
108  m->get_cur_step());
109  learning *learning_layer = (learning *) dynamic_cast<learning *> (layer);
110  if(learning_layer != nullptr) {
111  m_summarizer->reduce_histogram(prefix + "weights",
112  learning_layer->get_weights(),
113  m->get_cur_step());
114  m_summarizer->reduce_histogram(
115  prefix + "weights_gradient",
116  learning_layer->get_weights_gradient(),
117  m->get_cur_step());
118  }
119  }
