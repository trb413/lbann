239  void bp_compute_gpu() {
240  #ifndef LBANN_HAS_CUDNN
241  std::stringstream err;
242  err << __FILE__ << " " << __LINE__ << " :: slice_layer: cuDNN not detected";
243  throw lbann_exception(err.str());
244  #else
245 
246  // Split the error signal tensor into slices of width 1 along the
247  // slice axis
248  const int num_slices
249  = std::accumulate(this->m_prev_neuron_dims.begin(),
250  this->m_prev_neuron_dims.begin() + m_slice_axis,
251  1,
252  std::multiplies<int>());
253  const int slice_unit_size
254  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
255  this->m_prev_neuron_dims.end(),
256  1,
257  std::multiplies<int>());
258  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
259  const int output_slice_size = output_slice_dim * slice_unit_size;
260 
261  // Copy entries in each child to error signal tensor
262  const int num_gpus = this->m_cudnn->get_num_gpus();
263  for(size_t child_index = 0; child_index < this->m_child_layers.size(); ++child_index) {
264  const Layer* child = this->m_child_layers[child_index];
265 
266  // Get child error signal on GPUs
267  std::vector<DataType*> input;
268  if(child_index == 0) {
269  input = this->m_prev_error_signal_dv;
270  }
271  else {
272  std::vector<DataType*> work_spaces;
273  for(int i=0; i<num_gpus; ++i) {
274  work_spaces.push_back((DataType*)this->m_cudnn->get_work_space(i));
275  }
276  if(child->using_gpus()) {
277  child->get_gpu_bp_output(input, work_spaces, this);
278  }
279  else {
280  child->get_bp_output(*this->m_prev_error_signal_v, this);
281  this->m_cudnn->scatter_to_gpus(work_spaces,
282  this->m_prev_error_signal_v->LockedMatrix(),
283  this->m_mini_batch_size_per_gpu);
284  input = work_spaces;
285  }
286  }
287 
288  // Split previous error signal tensor into slices
289  const int input_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
290  const int input_slice_size = input_slice_dim * slice_unit_size;
291  const int input_size = num_slices * input_slice_size;
292  const int slice_offset = m_slice_points[child_index] * slice_unit_size;
293 
294  // Copy slices from previous error signal tensor into error signal tensor
295  for(int slice = 0; slice < num_slices; ++slice) {
296  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
297  for(int i = 0; i < num_gpus; ++i) {
298  input_slice[i] = input[i] + slice * input_slice_size;
299  output_slice[i] = this->m_error_signal_d[i] + slice * output_slice_size + slice_offset;
300  }
301  this->m_cudnn->copy_on_gpus(output_slice,
302  input_slice,
303  input_slice_size,
304  this->m_mini_batch_size_per_gpu,
305  input_size,
306  this->m_num_prev_neurons);
307  }
308 
309  }
310 
311  #endif // #ifndef LBANN_HAS_CUDNN

314  void bp_compute_cpu() {
315 
316  // Clear error signal if needed
317  if(m_slice_points.front() != 0
318  || m_slice_points.back() != this->m_neuron_dims[m_slice_axis]) {
319  El::Zero(*this->m_error_signal_v);
320  }
321 
322  // Split the error signal tensor into slices of width 1 along the
323  // slice axis
324  const int num_slices
325  = std::accumulate(this->m_prev_neuron_dims.begin(),
326  this->m_prev_neuron_dims.begin() + m_slice_axis,
327  1,
328  std::multiplies<int>());
329  const int slice_unit_size
330  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
331  this->m_prev_neuron_dims.end(),
332  1,
333  std::multiplies<int>());
334  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
335  const int output_slice_size = output_slice_dim * slice_unit_size;
336 
337  // Copy entries in each child to error signal tensor
338  for(size_t i = 0; i < this->m_child_layers.size(); ++i) {
339 
340  // Split previous error signal tensor into slices
341  this->m_child_layers[i]->get_bp_output(*this->m_prev_error_signal_v, this);
342  const int input_slice_dim = m_slice_points[i+1] - m_slice_points[i];
343  const int input_slice_size = input_slice_dim * slice_unit_size;
344  const int slice_offset_start = m_slice_points[i] * slice_unit_size;
345  const int slice_offset_end = m_slice_points[i+1] * slice_unit_size;
346 
347  // Copy slices from previous error signal tensor into error signal tensor
348  for(int slice = 0; slice < num_slices; ++slice) {
349  El::LockedView(*m_input_slice_v,
350  *this->m_prev_error_signal_v,
351  El::IR(slice * input_slice_size,
352  (slice+1) * input_slice_size),
353  El::ALL);
354  El::View(*m_output_slice_v,
355  *this->m_error_signal_v,
356  El::IR(slice * output_slice_size + slice_offset_start,
357  slice * output_slice_size + slice_offset_end),
358  El::ALL);
359  El::Copy(*m_input_slice_v, *m_output_slice_v);
360  }
361 
362  }

366  void get_fp_output(AbsDistMat& fp_output, const Layer* next_layer) const override {
367 
368  // Check if input is in the list of child layers
369  const int child_index = (std::find(this->m_child_layers.begin(),
370  this->m_child_layers.end(),
371  next_layer)
372  - this->m_child_layers.begin());
373  if(child_index >= (int) this->m_child_layers.size()) {
374  transform_layer::get_fp_output(fp_output, next_layer);
375  }
376 
377  // Split the activations tensor into slices of width 1 along the
378  // slice axis
379  const int num_slices
380  = std::accumulate(this->m_prev_neuron_dims.begin(),
381  this->m_prev_neuron_dims.begin() + m_slice_axis,
382  1,
383  std::multiplies<int>());
384  const int slice_unit_size
385  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
386  this->m_prev_neuron_dims.end(),
387  1,
388  std::multiplies<int>());
389  const int input_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
390  const int output_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
391  const int input_slice_size = input_slice_dim * slice_unit_size;
392  const int output_slice_size = output_slice_dim * slice_unit_size;
393  const int slice_offset_start = m_slice_points[child_index] * slice_unit_size;
394  const int slice_offset_end = m_slice_points[child_index+1] * slice_unit_size;
395 
396  if(num_slices == 1
397  && m_activations_v->DistData() == fp_output.DistData()) {
398  // Return view of activations tensor slice
399  El::LockedView(fp_output,
400  *this->m_activations_v,
401  El::IR(slice_offset_start, slice_offset_end),
402  El::ALL);
403  }
404  else {
405  // Copy slices from activations tensor into output
406  fp_output.Empty(false);
407  fp_output.Resize(output_slice_size, m_activations_v->Width());
408  AbsDistMat* output_slice_v
409  = fp_output.Construct(fp_output.Grid(), fp_output.Root());
410  for(int slice = 0; slice < num_slices; ++slice) {
411  El::LockedView(*m_input_slice_v,
412  *this->m_activations_v,
413  El::IR(slice * input_slice_size + slice_offset_start,
414  slice * input_slice_size + slice_offset_end),
415  El::ALL);
416  El::View(*output_slice_v,
417  fp_output,
418  El::IR(slice * output_slice_size,
419  (slice+1) * output_slice_size),
420  El::ALL);
421  El::Copy(*m_input_slice_v, *output_slice_v);
422  }
423  delete output_slice_v;
424  }
