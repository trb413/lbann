215 void data_reader_jag::load() {
216  const std::string data_dir = add_delimiter(get_file_dir());
217  const std::string namestr = get_data_filename();
218  std::vector<std::string> file_names = get_tokens(namestr);
219  if (file_names.size() != 3u) {
220  throw lbann_exception(
221  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
222  " :: unexpected number of files " + std::to_string(file_names.size()));
223  }
224 
225  for(auto& str : file_names) {
226  str = data_dir + str;
227  }
228  load(file_names[0], file_names[1], file_names[2], m_first_n);
229 
230  size_t num_samples = get_num_samples();
231 
232  if (m_first_n > 0) {
233  num_samples = (static_cast<size_t>(m_first_n) <= num_samples)?
234  static_cast<size_t>(m_first_n) : num_samples;
235 
236  m_first_n = num_samples;
237  set_use_percent(1.0);
238  set_absolute_sample_count(0u);
239  }
240 
241  // reset indices
242  m_shuffled_indices.clear();
243 
244  m_shuffled_indices.resize(num_samples);
245  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
246 
247  select_subset_of_data();
248 }

251 void data_reader_jag::load(const std::string image_file,
252  const std::string scalar_file,
253  const std::string input_file,
254  const size_t first_n) {
255  if ((m_independent == Undefined) && (m_dependent == Undefined)) {
256  throw lbann_exception("data_reader_jag: no type of variables to load is defined.");
257  }
258  if ((m_independent == JAG_Image || m_dependent == JAG_Image) &&
259  !image_file.empty() && !check_if_file_exists(image_file)) {
260  throw lbann_exception("data_reader_jag: failed to load " + image_file);
261  }
262  if ((m_independent == JAG_Scalar || m_dependent == JAG_Scalar) &&
263  !scalar_file.empty() && !check_if_file_exists(scalar_file)) {
264  throw lbann_exception("data_reader_jag: failed to load " + scalar_file);
265  }
266  if ((m_independent == JAG_Input || m_dependent == JAG_Input) &&
267  !input_file.empty() && !check_if_file_exists(input_file)) {
268  throw lbann_exception("data_reader_jag: failed to load " + input_file);
269  }
270 
271  m_num_samples = 0u;
272 
273  // read in only those that will be used
274  if ((m_independent == JAG_Image || m_dependent == JAG_Image) &&
275  !image_file.empty()) {
276  m_images = cnpy::npy_load(image_file);
277  if (first_n > 0u) { // to use only first_n samples
278  cnpy_utils::shrink_to_fit(m_images, first_n);
279  }
280  m_image_loaded = true;
281  set_linearized_image_size();
282  }
283  if ((m_independent == JAG_Scalar || m_dependent == JAG_Scalar) &&
284  !scalar_file.empty()) {
285  m_scalars = cnpy::npy_load(scalar_file);
286  if (first_n > 0u) { // to use only first_n samples
287  cnpy_utils::shrink_to_fit(m_scalars, first_n);
288  }
289  m_scalar_loaded = true;
290  set_linearized_scalar_size();
291  }
292  if ((m_independent == JAG_Input || m_dependent == JAG_Input) &&
293  !input_file.empty()) {
294  m_inputs = cnpy::npy_load(input_file);
295  if (first_n > 0u) { // to use only first_n samples
296  cnpy_utils::shrink_to_fit(m_inputs, first_n);
297  }
298  m_input_loaded = true;
299  set_linearized_input_size();
300  }
301 
302  size_t num_samples = 0u;
303  bool ok = check_data(num_samples);
304 
305  if (!ok) {
306  get_description();
307  throw lbann_exception("data_reader_jag: loaded data format not consistent");
308  }
309  m_num_samples = num_samples;
310  if (m_image_loaded) {
311  m_img_min = get_image_min();
312  m_img_max = get_image_max();
313  if (m_img_min == m_img_max) {
314  throw lbann_exception("data_reader_jag: no_variation in data");
315  }
316  normalize_image();
317  }
318 }

321 bool data_reader_jag::check_data(size_t& num_samples) const {
322  bool ok = true;
323  num_samples = 0u;
324  if (ok && m_image_loaded) {
325  ok = (m_linearized_image_size > 0u) && (m_images.word_size == sizeof(data_t));
326  if (!ok) {
327  std::cerr << "m_images.shape.size() = " << m_images.shape.size() << std::endl
328  << "m_linearized_image_size = " << m_linearized_image_size << std::endl
329  << "m_images.word_size = " << m_images.word_size << std::endl
330  << "sizeof(data_t) = " << sizeof(data_t) << std::endl;
331  return false;
332  }
333  num_samples = m_images.shape[0];
334  }
335  if (ok && m_scalar_loaded) {
336  ok = (m_linearized_scalar_size > 0u) && (m_scalars.word_size == sizeof(scalar_t));
337  if (!ok) {
338  std::cerr << "m_scalars.shape.size() = " << m_scalars.shape.size() << std::endl
339  << "m_linearized_scalar_size = " << m_linearized_scalar_size << std::endl
340  << "m_scalars.word_size = " << m_scalars.word_size << std::endl
341  << "sizeof(scalar_t) = " << sizeof(scalar_t) << std::endl;
342  return false;
343  }
344  if (num_samples > 0u) {
345  ok = ok && (num_samples == m_scalars.shape[0]);
346  } else {
347  num_samples = m_scalars.shape[0];
348  }
349  }
350  if (ok && m_input_loaded) {
351  ok = (m_linearized_input_size > 0u) && (m_inputs.word_size == sizeof(input_t));
352  if (!ok) {
353  std::cerr << "m_inputs.shape.size() = " << m_inputs.shape.size() << std::endl
354  << "m_linearized_input_size = " << m_linearized_input_size << std::endl
355  << "m_inputs.word_size = " << m_inputs.word_size << std::endl
356  << "sizeof(input_t) = " << sizeof(input_t) << std::endl;
357  return false;
358  }
359  if (num_samples > 0u) {
360  ok = ok && (num_samples == m_inputs.shape[0]);
361  } else {
362  num_samples = m_inputs.shape[0];
363  }
364  }
365  if (!ok) {
366  num_samples = 0u;
367  } else {
368  if (m_independent == JAG_Image || m_dependent == JAG_Image) {
369  ok = ok && m_image_loaded;
370  }
371  if (m_independent == JAG_Scalar || m_dependent == JAG_Scalar) {
372  ok = ok && m_scalar_loaded;
373  }
374  if (m_independent == JAG_Input || m_dependent == JAG_Input) {
375  ok = ok && m_input_loaded;
376  }
377  }
378 
379  return ok;
380 }

403 void data_reader_jag::normalize_image() {
404  if (!m_image_loaded) {
405  return;
406  }
407  using depth_t = cv_image_type<data_t>;
408  const int type_code = depth_t::T(1u);
409 
410  if (m_image_normalization == 1) {
411  data_t* const ptr = get_image_ptr(0);
412  // Present the entire image data as a single image
413  // and normalize it once and for all
414  cv::Mat img(m_num_samples, m_linearized_image_size,
415  type_code, reinterpret_cast<void*>(ptr));
416  cv::normalize(img, img, 0.0, 1.0, cv::NORM_MINMAX);
417  } else if (m_image_normalization == 2) {
418  // normalize each image independently
419  for (size_t i=0u; i < m_num_samples; ++i) {
420  data_t* const ptr = get_image_ptr(i);
421  cv::Mat img(1, m_linearized_image_size,
422  type_code, reinterpret_cast<void*>(ptr));
423  cv::normalize(img, img, 0.0, 1.0, cv::NORM_MINMAX);
424  }
425  } else if (m_image_normalization != 0) {
426  throw lbann_exception("data_reader_jag: invalid normalization mode " +
427  std::to_string(m_image_normalization));
428  }
429 }

435 cv::Mat data_reader_jag::get_image(const size_t i) const {
436  using InputBuf_T = cv_image_type<data_t>;
437 
438  data_t* const ptr = get_image_ptr(i);
439  if (ptr == nullptr) {
440  return cv::Mat();
441  }
442  // Construct a zero copying view to data
443  const cv::Mat img_org(m_linearized_image_size, 1, InputBuf_T::T(1u),
444  reinterpret_cast<void*>(ptr));
445 
446  cv::Mat img;
447  if (std::is_same<data_t, DataType>::value) {
448  img = img_org.clone();
449  } else {

451  }
452  return img.reshape(0, m_image_height);
