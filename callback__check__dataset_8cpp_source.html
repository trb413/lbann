35 void lbann_callback_check_dataset::add_to_set(model *m, Layer *l, int64_t step, std::set<long>& set) {
36  if (!dynamic_cast<io_layer*>(l)) {
37  return;
38  }
39 
40  El::Matrix<El::Int>* indices = l->get_sample_indices_per_mb();
41 
42  std::set<long>::iterator it;
43 
44  for(El::Int i = 0; i < indices->Height(); i++) {
45  for(El::Int j = 0; j < indices->Width(); j++) {
46  El::Int idx = indices->Get(i,j);
47  it = set.find(idx);
48  if(it != set.end()) {
49  throw lbann_exception(
50  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
51  + " :: @" + std::to_string(step)
52  + " :: found a duplicate index in being loaded: " + std::to_string(idx));
53  }else {
54  set.insert(idx);
55  }
56  }
57  }
58 }
