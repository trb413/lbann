38 void lbann_callback_early_stopping::on_validation_end(model *m) {
39  double score = m->obj_fn->report_aggregate_avg_obj_fn(execution_mode::validation);
40  if (score < m_last_score) {
41  if (m->get_comm()->am_model_master()) {
42  std::cout << "Model " << m->get_comm()->get_model_rank() <<
43  " early stopping: score is improving " << m_last_score << " >> " <<
44  score << std::endl;
45  }
46  m_last_score = score;
47  m_wait = 0;
48  } else {
49  if (m_wait >= m_patience) {
50  m->set_terminate_training(true);
51  if (m->get_comm()->am_model_master()) {
52  std::cout << "Model " << m->get_comm()->get_model_rank() <<
53  " terminating training due to early stopping: " << score <<
54  " score and " << m_last_score << " last score" << std::endl;
55  }
56  } else {
57  ++m_wait;
58  }
59  }
60 }
