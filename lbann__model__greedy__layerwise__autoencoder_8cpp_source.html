75 bool lbann::greedy_layerwise_autoencoder::load_from_checkpoint_shared(lbann::persist& p)
76 {
77  // have rank 0 read whether we have a mirror layer inserted
78  struct lbann_model_greedy_layerwise_autoencoder_header header;
79  if (p.m_rank == 0) {
80  p.read_uint32(persist_type::train, "gla_phase_index", &header.phase_index);
81  p.read_uint32(persist_type::train, "gla_have_mirror", &header.have_mirror);
82  }
83 
84  // TODO: this assumes homogeneous processors
85  // broadcast state from rank 0
86  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
87 
88  // insert the mirror layer if needed
89  uint32_t phase_index = header.phase_index;
90  uint32_t have_mirror = header.have_mirror;
91  if (have_mirror) {
92  // note that this calls setup on the layers,
93  // and setup reinitializes a bunch of values like data reader positions
94  // and optimization layer cache values that we'll overwrite
95  // in load_from_checkpoint_shared below
96  insert_mirror(phase_index);
97  }
98 
99  // read parameters from base class first
100  sequential_model::load_from_checkpoint_shared(p);
101 
102  return true;
103 }

112 void lbann::greedy_layerwise_autoencoder::insert_mirror(uint32_t layer_index)
113 {
114  // compute layer index for mirrror
115  size_t mirror_index = layer_index + 2;
116 
117  // build mirror layer
118  Layer* original_layer = m_layers[layer_index];
119  optimizer *opt = create_optimizer();
120  reconstruction_layer* mirror_layer = new reconstruction_layer(original_layer->m_data_layout, mirror_index, comm, opt, m_mini_batch_size, original_layer);
121 
122  // insert mirror layer into model
123  insert(mirror_index, mirror_layer);
124 
125  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
126  //assume that necessary layer parameters are set e.g., NumNeurons when layers were constructed
127  setup(layer_index, mirror_index+1); //set up all active layers
128 
129  // set flag to indicate that we have a mirror layer inserted
130  m_have_mirror = 1;
131 }

155 void lbann::greedy_layerwise_autoencoder::train(int num_epochs, int evaluation_frequency)
156 {
157  size_t num_phases = m_layers.size()-1;
158  // get to training, layer by layer
159  while(m_current_phase < num_phases){
160  //m_current_phase = phase_index;
161  m_phase_end = m_current_phase+2;
162  Layer* original_layer = m_layers[m_current_phase];
163  optimizer *opt = create_optimizer();
164  reconstruction_layer* mirror_layer = new reconstruction_layer(original_layer->m_data_layout, m_phase_end, comm, opt, m_mini_batch_size,original_layer);
165  Layer* tmp;
166  //if not at the last layer/phase, swap otherwise insert new
167  if(m_current_phase < num_phases-1) tmp = swap(m_phase_end,mirror_layer);
168  else insert(m_phase_end,mirror_layer);
169  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
170  //assume that necessary layer parameters are set e.g., NumNeurons when layers were constructed
171  setup(m_phase_end,m_phase_end+1); //set up just the added (new) layers
172  train_phase(num_epochs,evaluation_frequency);
173 
174  if (comm->am_world_master()) {
175  //end of phase cbs e.g., save a number of image to file
176  do_phase_end_cbs();
177  }
178  m_reconstruction_layers.insert(m_reconstruction_layers.begin(),mirror_layer);
179  //swap back
180  if(m_current_phase < num_phases-1) swap(m_phase_end,tmp);
181 
182  // move on to the next phase
183  m_current_phase++;
184  }
185  //evaluate and save all layers for i.e., (1) global cost (2) image reconstruction
186  evaluate(execution_mode::testing);
187 
188 }

223 void lbann::greedy_layerwise_autoencoder::train_phase(int num_epochs, int evaluation_frequency)
224 {
225  do_train_begin_cbs();
226 
227  // Epoch main loop
228  while (get_cur_epoch() < num_epochs) {
229  // Check if training has been terminated
230  if (get_terminate_training()) break;
231 
232  // due to restart, may not always be at start of epoch
233  // use mini batch index in data reader to signify start of epoch
234  if (at_epoch_start()) {
235  ++m_current_epoch;
236  do_epoch_begin_cbs(); // needed for selected callback e.g., dump matrices
237  }
238 
239  //Overide default print callback
240  if (comm->am_world_master()) {
241  //std::cout << "-----------------------------------------------------------" << std::endl;
242  //std::cout << "Phase [" << m_current_phase << "] Epoch [" << m_current_epoch << "]" << std::endl;
243  std::cout << "\n Training hidden layer [" << m_current_phase+1 << "] at layer-wise epoch [" << m_current_epoch << "]" << std::endl;
244  std::cout << "-----------------------------------------------------------" << std::endl;
245  }
246 
248  m_execution_mode = execution_mode::training;
249  for (size_t l =0; l < m_layers.size(); l++) {
250  m_layers[l]->m_execution_mode = execution_mode::training;
251  }
252 
253  // Train on mini-batches until data set is traversed
254  // Note: The data reader shuffles the data after each epoch
255  for (auto&& m : metrics) { m->reset_metric(); }
256  bool finished_epoch;
257  do {
258  finished_epoch = train_mini_batch();
259 
260  // save a checkpoint if needed
261  if (need_checkpoint()) {
262  checkpointShared();
263  }
264  } while(!finished_epoch);
265 
266 
267  //print training reconstruction cost
268  if (comm->am_world_master()) std::cout << "Layer-wise training ";
269  m_layers[m_phase_end]->epoch_print();
270 
271 
272  do_epoch_end_cbs(); //needed for selected callback e.g., dump matrices
273 
274  for (Layer* layer : m_layers) {
275  layer->epoch_reset();
276  } // train epoch end, this reset cost
277 
278  evaluate_phase(execution_mode::validation);
279 
280  //print validation reconstruction cost
281  if (comm->am_world_master()) std::cout << "Layer-wise validation ";
282  m_layers[m_phase_end]->epoch_print();
283 
284  //Reset cost again
285  for (Layer* layer : m_layers) {
286  layer->epoch_reset();
287  } // train epoch
288 
289  // Reset execution mode back to training
290  m_execution_mode = execution_mode::training;
291  for (Layer* layer : m_layers) {
292  layer->m_execution_mode = execution_mode::training;
293  }
294 
295  // save checkpoint after epoch
296  if (need_checkpoint()) {
297  checkpointShared();
298  }
299  }
300 
301  do_train_end_cbs();
302  m_current_epoch = 0; //reset epoch counter
303 }

305 bool lbann::greedy_layerwise_autoencoder::train_mini_batch()
306 {
307  do_batch_begin_cbs();
308 
309  // Forward propagation
310  do_model_forward_prop_begin_cbs();
311  //@todo; optimize this? change start index from 0 to phase_index
312  for (size_t l = 0; l <= m_phase_end; ++l) {
313  do_layer_forward_prop_begin_cbs(m_layers[l]);
314  m_layers[l]->forwardProp();
315  do_layer_forward_prop_end_cbs(m_layers[l]);
316  }
317  do_model_forward_prop_end_cbs();
318 
319  ++m_current_step;
320 
321  // Backward propagation
322  do_model_backward_prop_begin_cbs();
323  //@todo; optimize to backprop up to phase_index and not 0
324  for (size_t l = m_phase_end+1; l-- > 0;) {
325  do_layer_backward_prop_begin_cbs(m_layers[l]);
326  m_layers[l]->backProp();
327  do_layer_backward_prop_end_cbs(m_layers[l]);
328  }
329  do_model_backward_prop_end_cbs();
330 
333  for (size_t l = m_phase_end; l > m_current_phase; --l) {
334  m_layers[l]->update();
335  }
336  const bool data_set_processed = m_layers[0]->update();
337 
338  do_batch_end_cbs();
339  return data_set_processed;
