102  void fp_compute(void) {
103  int num_samples_in_batch = fetch_to_local_matrix(this->m_activations->Matrix());
104 
105  target_layer::update_num_samples_processed(num_samples_in_batch);
106 
107  int curr_mini_batch_size = this->m_neural_network_model->get_current_mini_batch_size();
108 
110  DataType avg_error = this->m_neural_network_model->m_obj_fn->compute_obj_fn(*this->m_prev_activations_v, *this->m_activations_v);
111  this->m_neural_network_model->m_obj_fn->record_obj_fn(this->m_execution_mode, avg_error);
112 
113  for (auto&& m : this->m_neural_network_model->m_metrics) {
114  double num_errors = m->compute_metric(*this->m_prev_activations_v, *this->m_activations_v);
115  m->record_error(num_errors, curr_mini_batch_size);
116  }
117 
118  return;
119  }
