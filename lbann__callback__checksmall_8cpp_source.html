34 void lbann_callback_checksmall::on_forward_prop_end(model* m, Layer* l) {
35  // Skip output layer.
36  if (l->get_index() == m->get_layers().size() - 1) {
37  return;
38  }
39  ElMat& acts = l->get_activations();
40  if (!is_good(acts)) {
41  lbann_comm* comm = m->get_comm();
42  std::cout << "[" << comm->get_rank_in_world() << "]: error in layer " <<
43  l->get_index() << " activations (step=" << m->get_cur_step() << ")" <<
44  std::endl;
45  throw lbann_exception("checksmall: error in activations");
46  }
47 }

49 void lbann_callback_checksmall::on_backward_prop_end(model* m, Layer* l) {
50  // Skip input/output layers.
51  if (l->get_index() == 0 || l->get_index() == m->get_layers().size() - 1) {
52  return;
53  }
54  ElMat& grad = l->get_weights_biases_gradient();
55  if (!is_good(grad)) {
56  lbann_comm* comm = m->get_comm();
57  std::cout << "[" << comm->get_rank_in_world() << "]: error in layer " <<
58  l->get_index() << " gradients (shape=" << grad.Height() << "x" <<
59  grad.Width() << " step=" << m->get_cur_step() << ")" <<
60  std::endl;
61  throw lbann_exception("checksmall: error in gradients");
62  }
63 }

65 void lbann_callback_checksmall::on_batch_end(model* m) {
66  std::vector<Layer*>& layers = m->get_layers();
67  // Skip input/output layers-- they don't have weights.
68  for (size_t i = 1; i < layers.size() - 1; ++i) {
69  Layer* l = layers[i];
70  ElMat& weights = l->get_weights_biases();
71  if (!is_good(weights)) {
72  lbann_comm* comm = m->get_comm();
73  std::cout << "[" << comm->get_rank_in_world() << "]: error in layer " <<
74  l->get_index() << " weights (step=" << m->get_cur_step() << ")" <<
75  std::endl;
76  throw lbann_exception("checksmall: error in weights");
77  }
78  }
79 }

81 bool lbann_callback_checksmall::is_good(const ElMat& m) {
82  const Mat& local_mat = m.LockedMatrix();
83  const Int height = local_mat.Height();
84  const Int width = local_mat.Width();
85  for (Int col = 0; col < width; ++col) {
86  for (Int row = 0; row < height; ++row) {
87  const DataType val = Abs(local_mat(row, col));
88  if (val > 0 && val <= threshold) {
89  std::cout << "Found small value " << val << " at (" << row << "," <<
90  col << ")!" << std::endl;
91  return false;
92  }
93  }
94  }
95  return true;
96 }
