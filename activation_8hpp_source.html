99  virtual void fp_compute_cpu() {
100 
101  // Get local matrices
102  const Mat& prev_activations_local = this->m_prev_activations->LockedMatrix();
103  Mat& activations_local = this->m_activations_v->Matrix();
104 
105  // Local matrix parameters
106  const int local_height = prev_activations_local.Height();
107  const int local_width = prev_activations_local.Width();
108  const int prev_activations_ldim = prev_activations_local.LDim();
109  const int activations_ldim = activations_local.LDim();
110  const DataType* __restrict__ prev_activations_buffer
111  = prev_activations_local.LockedBuffer();
112  DataType* __restrict__ activations_buffer = activations_local.Buffer();
113 
114  // Apply activation function
115  if(prev_activations_ldim == local_height
116  && activations_ldim == local_height) {
117  // Contiguous data
118  #pragma omp parallel for
119  for(int i = 0; i < local_height * local_width; ++i) {
120  const DataType prev_activations_entry = prev_activations_buffer[i];
121  DataType& activations_entry = activations_buffer[i];
122  activations_entry = activation_function(prev_activations_entry);
123  }
124  } else {
125  // Non-contiguous data
126  #pragma omp parallel for collapse(2)
127  for(int col = 0; col < local_width; ++col) {
128  for(int row = 0; row < local_height; ++row) {
129  const DataType prev_activations_entry
130  = prev_activations_buffer[row + col * prev_activations_ldim];
131  DataType& activations_entry
132  = activations_buffer[row + col * activations_ldim];
133  activations_entry = activation_function(prev_activations_entry);
134  }
135  }
136  }
