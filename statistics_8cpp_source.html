32 void entrywise_mean_and_stdev(const Mat& data,
33  DataType& mean,
34  DataType& stdev) {
35  // Note: This routine is primarily called in an OpenMP-parallelized
36  // loop in data_readers/image_preprocessor.hpp. If a more
37  // significant use-case is found, it may be worthwhile parallelizing
38  // the loop.
39 
40  // Matrix dimensions
41  const El::Int height = data.Height();
42  const El::Int width = data.Width();
43  const El::Int size = height * width;
44 
45  // Compute sums over matrix entries
46  const DataType shift = data(0, 0);
47  DataType shifted_sum = 0;
48  DataType shifted_sqsum = 0;
49  for(El::Int col = 0; col < width; ++col) {
50  for(El::Int row = 0; row < height; ++row) {
51  const DataType shifted_val = data(row, col) - shift;
52  shifted_sum += shifted_val;
53  shifted_sqsum += shifted_val * shifted_val;
54  }
55  }
56 
57  // Compute mean and standard deviation
58  const DataType shifted_mean = shifted_sum / size;
59  const DataType shifted_sqmean = shifted_sqsum / size;
60  mean = shifted_mean + shift;
61  const DataType var = std::max(shifted_sqmean - shifted_mean * shifted_mean,
62  DataType(0));
63  stdev = std::sqrt(var);
64 
65 }

67 void entrywise_mean_and_stdev(const AbsDistMat& data,
68  DataType& mean,
69  DataType& stdev) {
70 
71  // Matrix dimensions
72  const El::Int size = data.Height() * data.Width();
73  const El::Int local_height = data.LocalHeight();
74  const El::Int local_width = data.LocalWidth();
75 
76  // Local matrices
77  const Mat& local_data = data.LockedMatrix();
78 
79  // Compute sums over matrix entries
80  DataType sum = 0;
81  DataType sqsum = 0;
82  #pragma omp parallel for reduction(+:sum,sqsum) collapse(2)
83  for(El::Int col = 0; col < local_width; ++col) {
84  for(El::Int row = 0; row < local_height; ++row) {
85  const DataType val = local_data(row, col);
86  sum += val;
87  sqsum += val * val;
88  }
89  }
90  DataType sum_sqsum[2] = {sum, sqsum}; // Pack to do one allreduce.
91  El::mpi::AllReduce(sum_sqsum, 2, data.DistComm());
92 
93  // Compute mean and standard deviation
94  mean = sum_sqsum[0] / size;
95  const DataType var = std::max(sum_sqsum[1] / size - mean * mean, DataType(0));
96  stdev = std::sqrt(var);
97 
98 }

144  || sqsum_dist.colDist != El::STAR

156  sums.Resize(1, width);
157  sqsums.Resize(1, width);

159  // Local matrices
160  const Mat& local_data = data.LockedMatrix();
161  Mat& local_sum = sums.Matrix();
162  Mat& local_sqsum = sqsums.Matrix();
163 
164  // Compute sum and sum of squares of each matrix column
165  #pragma omp parallel for
166  for(El::Int col = 0; col < local_width; ++col) {
167  DataType sum_val = 0;
168  DataType sqsum_val = 0;
169  for(El::Int row = 0; row < local_height; ++row) {
170  const DataType val = local_data(row, col);
171  sum_val += val;
172  sqsum_val += val * val;
173  }
174  local_sum(0, col) = sum_val;
175  local_sqsum(0, col) = sqsum_val;
176  }
177

179  AllReduce(sums, sums.RedundantComm(), El::mpi::SUM);

218 void rowwise_mean_and_stdev(const Mat& data,
219  Mat& means,
220  Mat& stdevs) {
221 
222  // Matrix dimensions
223  const El::Int height = data.Height();
224  const El::Int width = data.Width();
225 
226  // Initialize outputs
227  means.Resize(height, 1);
228  stdevs.Resize(height, 1);
229 
230  // Iterate through row blocks
231  const El::Int block_size = 16;
232  #pragma omp parallel for
233  for(El::Int row_start = 0; row_start < height; row_start += block_size) {
234  const El::Int row_end = std::min(row_start + block_size, height);
235 
236  // Initialize shift and sums for each row
237  auto *shifts = new DataType[block_size];
238  for(El::Int row = row_start; row < row_end; ++row) {
239  means(row, 0) = 0;
240  stdevs(row, 0) = 0;
241  shifts[row-row_start] = data(row, 0);
242  }
243 
244  // Iterate through blocks in row block
245  for(El::Int col_start = 0; col_start < width; col_start += block_size) {
246  const El::Int col_end = std::min(col_start + block_size, width);
247 
248  // Compute sums by iterating through block entries
249  for(El::Int col = col_start; col < col_end; ++col) {
250  for(El::Int row = row_start; row < row_end; ++row) {
251  const DataType shift = shifts[row - row_start];
252  const DataType shifted_val = data(row, col) - shift;
253  means(row, 0) += shifted_val;
254  stdevs(row, 0) += shifted_val * shifted_val;
255  }
256  }
257 
258  }
259 
260  // Compute mean and standard deviation of each row
261  for(El::Int row = row_start; row < row_end; ++row) {
262  const DataType shifted_mean = means(row, 0) / width;
263  const DataType shifted_sqmean = stdevs(row, 0) / width;
264  const DataType mean = shifted_mean + shifts[row - row_start];
265  const DataType var = std::max(shifted_sqmean - shifted_mean * shifted_mean,
266  DataType(0));
267  const DataType stdev = std::sqrt(var);
268  means(row, 0) = mean;
269  stdevs(row, 0) = stdev;
270  }
271 
272  // Deallocate shifts
273  delete[] shifts;
274 
275  }

365 void columnwise_covariance(const AbsDistMat& data1,
366  const AbsDistMat& data2,
367  const AbsDistMat& means1,
368  const AbsDistMat& means2,
369  AbsDistMat& covs) {
370 
371  // Check matrix formats and dimensions are valid
372  El::DistData data1_dist(data1), data2_dist(data2),
373  means1_dist(means1), means2_dist(means2), covs_dist(covs);
374  if(data1_dist != data2_dist
375  || means1_dist.colDist != El::STAR
376  || means1_dist.rowDist != data1_dist.rowDist
377  || means1_dist != means2_dist
378  || means1_dist != covs_dist) {
379  throw lbann_exception("columnwise_covariance: invalid matrix format");
380  }
381  if(data1.Height() != data2.Height() || data1.Width() != data2.Width()) {
382  throw lbann_exception("columnwise_covariance: data matrix dimensions don't match");
383  }
384 
385  // Matrix dimensions
386  const El::Int height = data1.Height();
387  const El::Int width = data1.Width();
388  const El::Int local_height = data1.LocalHeight();
389  const El::Int local_width = data1.LocalWidth();
390 
391  // Initialize covariance
392  covs.Resize(1, width);
393 
394  // Local matrices
395  const Mat& local_data1 = data1.LockedMatrix();
396  const Mat& local_data2 = data2.LockedMatrix();
397  const Mat& local_means1 = means1.LockedMatrix();
398  const Mat& local_means2 = means2.LockedMatrix();
399  Mat& local_covs = covs.Matrix();
400 
401  // Accumulate sum and divide to get covariance
402  #pragma omp parallel for
403  for(El::Int col = 0; col < local_width; ++col) {
404  DataType sum = 0;
405  const DataType mean1 = local_means1(0, col);
406  const DataType mean2 = local_means2(0, col);
407  for(El::Int row = 0; row < local_height; ++row) {
408  const DataType val1 = local_data1(row, col);
409  const DataType val2 = local_data2(row, col);
410  sum += (val1 - mean1) * (val2 - mean2);
411  }
412  local_covs(0, col) = sum;
413  }
414  AllReduce(covs, covs.RedundantComm(), El::mpi::SUM);
415  local_covs *= DataType(1) / height;
416 
417 }
