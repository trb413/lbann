71 EvalType binary_cross_entropy::finish_evaluate_compute(
72  const AbsDistMat& predictions, const AbsDistMat& ground_truth) {
73 
74  // Local matrices
75  const Mat& predictions_local = predictions.LockedMatrix();
76  const Mat& ground_truth_local = ground_truth.LockedMatrix();
77 
78  // Matrix parameters
79  const int width = predictions.Width();
80  const int local_height = predictions_local.Height();
81  const int local_width = predictions_local.Width();
82 
83  // Compute sum of cross entropy terms
84  EvalType sum = 0;
85  #pragma omp parallel for reduction(+:sum) collapse(2)
86  for (int col = 0; col < local_width; ++col) {
87  for (int row = 0; row < local_height; ++row) {
88  const DataType true_val = ground_truth_local(row, col);
89  const DataType pred_val = predictions_local(row, col);
90  #ifdef LBANN_DEBUG
91  binary_cross_entropy_debug::check_entry(ground_truth.GlobalRow(row),
92  ground_truth.GlobalCol(col),
93  true_val,
94  pred_val);
95  #endif // LBANN_DEBUG
96  if (true_val > DataType(0)) {
97  sum += - true_val * std::log(pred_val);
98  }
99  if (true_val < DataType(1)) {
100  sum += - (EvalType(1) - true_val) * std::log(EvalType(1) - pred_val);
101  }
102  }
103  }
104 
105  // Compute mean objective function value across mini-batch
106  return get_comm().allreduce(sum / width, predictions.DistComm());
107 
108 }

118 
119  // Matrix parameters
120  const El::Int local_height = gradient_local.Height();
121  const El::Int local_width = gradient_local.Width();
122 
123  // Compute gradient
124  #pragma omp parallel for collapse(2)
125  for (El::Int col = 0; col < local_width; ++col) {
126  for (El::Int row = 0; row < local_height; ++row) {
127  const DataType true_val = ground_truth_local(row, col);
128  const DataType pred_val = predictions_local(row, col);
129  #ifdef LBANN_DEBUG
130  binary_cross_entropy_debug::check_entry(ground_truth.GlobalRow(row),
131  ground_truth.GlobalCol(col),
132  true_val,
133  pred_val);
134  #endif // LBANN_DEBUG
135  DataType grad_val = DataType(0);
136  if (true_val > DataType(0)) {
137  grad_val += - true_val / pred_val;
138  }
139  if (true_val < DataType(1)) {
140  grad_val += (DataType(1) - true_val) / (DataType(1) - pred_val);
141  }
142  gradient_local(row, col) = grad_val;
143  }
144  }
