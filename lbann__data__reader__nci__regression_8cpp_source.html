139 bool lbann::data_reader_nci_regression::load(const std::string infile)
140 {
141  ifstream ifs(infile.c_str());
142  if (!ifs) { std::cout << "\n In load: can't open file : " << infile; exit(1); }
143  m_infile = infile;
144  string line;
145  int i;
146 
147  m_index_map.push_back(ifs.tellg());
148 #if 1
149  std::getline (ifs, line); // to skip the header
150  m_index_map.pop_back();
151  m_index_map.push_back(ifs.tellg());
152 #endif
153 
154  while(std::getline (ifs, line) ) {
155 #if 0 // enable to skip empty or comment lines
156  static const std::string whitespaces(" \t\f\v\n\r");
157  std::size_t pos = line.find_first_not_of(whitespaces);
158  if ((pos == std::string::npos) || (line[pos] == '#')) {
159  m_index_map.pop_back();
160  m_index_map.push_back(ifs.tellg());
161  continue;
162  }
163 #endif
164 
165  m_index_map.push_back(ifs.tellg()); // The beginning of the next data to read
166  m_num_samples++;
167 
168  string field;
169  istringstream lstream(line);
170  i=0;
171  m_num_features = 0;
172  while(getline(lstream, field, ' ')) {
173  i++;
174  if (i > 5) {
175  m_num_features++;
176  }
177  }
178  }
179  ifs.close();
180  m_index_map.pop_back();
181  m_index_map.shrink_to_fit();
182  m_responses.resize(m_num_samples);
183  // reset indices
184  ShuffledIndices.clear();
185  ShuffledIndices.resize(m_num_samples);
186  for (size_t n = 0; n < ShuffledIndices.size(); ++n) {
187  ShuffledIndices[n] = n;
188  }
189  return true;
190 }
