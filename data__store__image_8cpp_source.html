153 void data_store_image::exchange_data() {
154  double tm1 = get_time();
155  std::stringstream err;
156 
157  //build map: proc -> global indices that proc needs for this epoch, and
158  // which I own
159  std::unordered_map<int, std::unordered_set<int>> proc_to_indices;
160  for (size_t p=0; p<m_all_minibatch_indices.size(); p++) {
161  for (auto idx : m_all_minibatch_indices[p]) {
162  int index = (*m_shuffled_indices)[idx];
163  if (m_my_datastore_indices.find(index) != m_my_datastore_indices.end()) {
164  proc_to_indices[p].insert(index);
165  }
166  }
167  }
168 
169  //start sends
170  std::vector<std::vector<El::mpi::Request<unsigned char>>> send_req(m_np);
171  for (int p=0; p<m_np; p++) {
172  send_req[p].resize(proc_to_indices[p].size()*m_num_img_srcs);
173  size_t jj = 0;
174  for (auto idx : proc_to_indices[p]) {
175  for (size_t k=0; k<m_num_img_srcs; k++) {
176  int index = idx*m_num_img_srcs+k;
177  if (m_file_sizes.find(index) == m_file_sizes.end()) {
178  err << __FILE__ << " " << __LINE__ << " :: "
179  << " m_file_sizes.find(" << index << ") failed";
180  throw lbann_exception(err.str());
181  }
182  int len = m_file_sizes[index];
183  m_comm->nb_tagged_send<unsigned char>(
184  m_data[index].data(), len, p, index,
185  send_req[p][jj++], m_comm->get_model_comm());
186 
187  }
188  }
189  if (jj != send_req[p].size()) throw lbann_exception("ERROR 1");
190  } //start sends
191 
192  //build map: proc -> global indices that proc owns that I need
193  proc_to_indices.clear();
194  for (auto idx : m_my_minibatch_indices_v) {
195  int index = (*m_shuffled_indices)[idx];
196  int owner = get_index_owner(index);
197  proc_to_indices[owner].insert(index);
198  }
199 
200  //start recvs
201  m_my_minibatch_data.clear();
202  std::vector<std::vector<El::mpi::Request<unsigned char>>> recv_req(m_np);
203  for (auto t : proc_to_indices) {
204  int owner = t.first;
205  size_t jj = 0;
206  const std::unordered_set<int> &s = t.second;
207  recv_req[owner].resize(s.size()*m_num_img_srcs);
208  for (auto idx : s) {
209  for (size_t k=0; k<m_num_img_srcs; k++) {
210  size_t index = idx*m_num_img_srcs+k;
211  if (m_file_sizes.find(index) == m_file_sizes.end()) {
212  err << __FILE__ << " " << __LINE__ << " :: "
213  << " m_file_sizes.find(" << index << ") failed"
214  << " m_file_sizes.size(): " << m_file_sizes.size()
215  << " m_my_minibatch_indices_v.size(): " << m_my_minibatch_indices_v.size();
216  }
217  size_t len = m_file_sizes[index];
218  m_my_minibatch_data[index].resize(len);
219  m_comm->nb_tagged_recv<unsigned char>(
220  m_my_minibatch_data[index].data(), len, owner,
221  index, recv_req[owner][jj++], m_comm->get_model_comm());
222  }
223  }
224  }
225 
226  //wait for sends to finish
227  for (size_t i=0; i<send_req.size(); i++) {
228  m_comm->wait_all<unsigned char>(send_req[i]);
229  }
230 
231  //wait for recvs to finish
232  for (size_t i=0; i<recv_req.size(); i++) {
233  m_comm->wait_all<unsigned char>(recv_req[i]);
234  }
235 
236  if (m_master) {
237  std::cerr << "TIME for exchange_data: " << get_time() - tm1
238  << "; role: " << m_reader->get_role() << "\n";
239  }
240 }

243 void data_store_image::exchange_file_sizes(
244  std::vector<int> &my_global_indices,
245  std::vector<int> &my_num_bytes) {
246 
247  if (my_global_indices.size() == 0) {
248  my_global_indices.push_back(-1);
249  my_num_bytes.push_back(-1);
250  }
251 
252  std::vector<int> rcv_counts(m_np);
253  int nbytes = my_global_indices.size();
254  m_comm->model_all_gather<int>(nbytes, rcv_counts);
255  int num_global_indices = std::accumulate(rcv_counts.begin(), rcv_counts.end(), 0);
256 
257  std::vector<int> disp(m_np); //@todo: fix for model
258  disp[0] = 0;
259  for (int h=1; h<m_np; h++) {
260  disp[h] = disp[h-1] + rcv_counts[h-1];
261  }
262  std::vector<int> all_global_indices(num_global_indices);
263  std::vector<int> all_num_bytes(num_global_indices);
264 
265  m_comm->all_gather<int>(my_global_indices, all_global_indices, rcv_counts, disp, m_comm->get_world_comm());
266 
267  m_comm->all_gather<int>(my_num_bytes, all_num_bytes, rcv_counts, disp, m_comm->get_world_comm());
268 
269  for (size_t j=0; j<all_global_indices.size(); j++) {
270  if (all_global_indices[j] != -1) {
271  m_file_sizes[all_global_indices[j]] = all_num_bytes[j];
272  }
273  }

305 size_t data_store_image::get_available_memory() {
306  std::ifstream in("/proc/meminfo");
307  std::string line;
308  size_t size;
309  bool found = false;
310  std::string name;
311  std::string units;
312  while (! in.eof()) {
313  getline(in, line);
314  std::stringstream s(line);
315  s >> name >> size >> units;
316  if (name.find("MemFree") != std::string::npos) {
317  found = true;

319  }
320  }
321  in.close();
322

329  "to contain all data files in memory\n";
330  }
331  return 0;
332  }

340 void data_store_image::report_memory_constraints() {
341  size_t count = get_my_num_file_bytes();
342 
343  std::vector<long long> counts(m_np);
344  if (m_master) {
345  m_comm->gather<long long>(count, counts.data(), m_comm->get_world_comm());
346  } else {
347  m_comm->gather<long long>(count, 0, m_comm->get_world_comm());
348  }
349 
350  double global = get_global_num_file_bytes()/1000000;
351 
352  if (!m_master) {
353  return;
354  }
355 
358  double required = 0;
359  for (int p=0; p<m_np; p++) {
360  if (m_comm->is_rank_node_local(p, m_comm->get_world_comm())) {
361  required += counts[p];
362  }
363  }
364  required /= 1000000;
365 
366  double available = get_available_memory();
367  if (available == 0) {
368  std::cerr << required << " kB of memory are required for files on this node\n";
369  return;
370  }
371  available /= 1000;
372 
373  double percent = required / available * 100.0;
374  std::cerr << "\n"
375  << "===============================================\n"
376  << "Memory Constraints for: " << m_reader->get_role() << "\n"
377  << "Global data set size: " << global << " MB\n"
378  << "Required for data set on this node: " << required << " MB\n"
379  << "Available memory on this node: " << available << " MB\n"
380  << "Required is " << percent << " % of Available\n"
381  << "===============================================\n\n";
382 
383  double limit = 0.8;
384  if (options::get()->has_float("mem_limit")) {
385  limit = options::get()->get_float("mem_limit");
386  }
387  if (required > limit*available) {
388  std::stringstream err;
389  err << __FILE__ << " " << __LINE__ << " :: "
390  << "You have insufficient memory to hold all required files;\n"
391  << "required is > 80% of available\n"
392  << "quitting now, so you don't waste your time\n";
393  }
394 }

401 void data_store_image::create_dirs(const std::string &s) {
402  if (s.size() == 0) {
403  return;
404  }
405  size_t idx;
406  size_t last = s[0] == '/' ? 1 : 0;
407  while ((idx = s.find('/', last)) != std::string::npos) {
408  last = idx+1;
409  std::string d = s.substr(0, idx);
410  std::ifstream in(d.c_str());
411  if (! in.good()) {
412  const int dir_err = mkdir(d.c_str(), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH);
413  //note: there can be race conditions where two procs attempt to create the
414  // same directory, which can cause mkdir to fail with "File Exists"
415  // error, which is errno=17. Need to guard against this!
416  if (dir_err == -1 && errno != 17) {
417  std::stringstream err;
418  err << __FILE__ << " " << __LINE__ << " :: "
419  << "failed to create directory: " << d << "\n"
420  << "error code is: " << errno << " -> " << std::strerror(errno)
421  << "\n" << getenv("SLURMD_NODENAME");
422  throw lbann_exception(err.str());
423 
424  }
425  } else {
426  in.close();
427  }
428  }

431 void data_store_image::stage_files() {
432  std::stringstream err;
433 
434  //create directory structure on local file store
435  std::string local_dir = m_reader->get_local_file_dir();
436  create_dirs(local_dir);
437  m_comm->global_barrier();
438  std::unordered_set<std::string> make_dirs;
439  for (auto t : m_data_filepaths) {
440  size_t j = t.second.rfind('/');
441  if (j != std::string::npos) {
442  make_dirs.insert(t.second.substr(0, j+1));
443  }
444  }
445 
446  std::string dir = m_reader->get_file_dir();
447  std::stringstream ss;
448  for (auto t : make_dirs) {
449  ss.clear();
450  ss.str("");
451  ss << local_dir << "/" << t;
452  create_dirs(ss.str());
453  }
454  m_comm->global_barrier();
455 
456  size_t j = 0;
457  struct stat stat_buf;
458  double tm = get_time();
459  std::stringstream s;
460  int write_fd;
461 
462  for (auto t : m_data_filepaths) {
463  s.clear();
464  s.str("");
465  s << local_dir << '/' << t.second;
466  ++j;
467  if (j % 100 == 0 and m_master) {
468  double e = get_time() - tm;
469  double time_per_file = e / j;
470  int remaining_files = m_data_filepaths.size()-j;
471  double estimated_remaining_time = time_per_file * remaining_files;
472  std::cerr << "P_0: staged " << j << " of " << m_data_filepaths.size()
473  << " files; elapsed time: " << get_time() - tm
474  << "s est. remaining time: " << estimated_remaining_time << "s\n";
475  }
476  if (access(s.str().c_str(), F_OK | R_OK) == -1 ) {
477  write_fd = open(s.str().c_str(), O_RDWR | O_CREAT, S_IRWXU);
478  //write_fd = open(s.str().c_str(), O_RDWR | O_CREAT, stat_buf.st_mode);
479  if (write_fd == -1) {
480  err << __FILE__ << " " << __LINE__ << " :: "
481  << "failed to open " << s.str() << " for writing;\n"
482  << "error code is: " << std::strerror(errno) << "\n"
483  << "local_dir: " << local_dir << " m_cur_minibatch: " << 1+m_cur_minibatch;
484  throw lbann_exception(err.str());
485  }
486  off_t offset = 0;
487  s.clear();
488  s.str("");
489  s << dir << '/' << t.second;
490  int read_fd = open(s.str().c_str(), O_RDONLY);
491  if (read_fd == -1) {
492  err << __FILE__ << " " << __LINE__ << " :: "
493  << "failed to open " << s.str() << " for reading;\n"
494  << "error code is: " << std::strerror(errno);
495  throw lbann_exception(err.str());
496  }
497  int e2 = fstat(read_fd, &stat_buf);
498  if (e2 == -1) {
499  err << __FILE__ << " " << __LINE__ << " :: "
500  << "fstat failed for file: " << s.str();
501  throw lbann_exception(err.str());
502  }
503  ssize_t e = sendfile(write_fd, read_fd, &offset, stat_buf.st_size);
504  if (e == -1) {
505  err << __FILE__ << " " << __LINE__ << " :: "
506  << "failed to copy file to location: " << s.str()
507  << ";\nerror code is: " << std::strerror(errno);
508  throw lbann_exception(err.str());
509 
510  }
511  close(read_fd);
512  close(write_fd);
513  }
514  }

517 void data_store_image::fetch_data() {
518  if (!m_is_setup) {
519  return;
520  }
521  std::stringstream err;
522  double tm1 = get_time();
523  ++m_cur_minibatch;
524  //if (m_cur_minibatch >= m_num_minibatches) {
525  if (m_cur_minibatch >= m_all_partitioned_indices[0].size()) {
526  m_cur_minibatch = 0;
527  }
528 
529  //build map: proc -> global indices that proc needs for this epoch, and
530  // which I own
531  std::unordered_map<int, std::unordered_set<int>> proc_to_indices;
532 
533  for (int p = 0; p<m_np; p++) {
534  if (m_cur_minibatch > m_all_partitioned_indices[p].size() -1) {
535  err << __FILE__ << " " << __LINE__ << " :: "
536  << "send to: P_" << p << " m_cur_minibatch: " << m_cur_minibatch
537  << " m_all_partitioned_indices[p].size(): " << m_all_partitioned_indices[p].size();
538  throw lbann_exception(err.str());
539  }
540  const std::vector<int> &v = m_all_partitioned_indices[p][m_cur_minibatch];
541  for (auto idx : v) {
542  int index = (*m_shuffled_indices)[idx];
543  if (m_my_datastore_indices.find(index) != m_my_datastore_indices.end()) {
544  proc_to_indices[p].insert(index);
545  }
546  }
547  }
548 
549  //read required files and start sends
550  m_data.clear();
551 
552  //compute number of sends, and allocate Request vector
553  size_t num_sends = 0;
554  for (auto t : proc_to_indices) {
555  num_sends += t.second.size();
556  }
557  num_sends *= m_num_img_srcs;
558  std::vector<El::mpi::Request<unsigned char>> send_req(num_sends);
559 
560  size_t req_idx = 0;
561  for (int p=0; p<m_np; p++) {
562  if (m_all_partitioned_indices[p].size() >= m_cur_minibatch
563  && proc_to_indices.find(p) != proc_to_indices.end()) {
564  const std::unordered_set<int> &s = proc_to_indices[p];
565  read_files(s);
566  for (auto idx : s) {
