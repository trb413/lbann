223  void fp_compute_cpu() {
224 
225  // Local matrices
226  const auto& local_input = get_local_prev_activations();
227  auto& local_output = get_local_activations();
228 
229  // Matrix parameters
230  const int local_width = local_input.Width();
231  const DataType* input_buffer = local_input.LockedBuffer();
232  const int input_ldim = local_input.LDim();
233  DataType* output_buffer = local_output.Buffer();
234  const int output_ldim = local_output.LDim();
235 
236  // Get LRN parameters
237  const int num_channels = this->m_neuron_dims[0];
238  const int num_per_channel = this->m_num_neurons / num_channels;
239 
240  // Check if LRN is using default beta parameter
241  const bool default_beta = (std::fabs((m_lrn_beta - 0.75) / 0.75)
242  < 2 * std::numeric_limits<DataType>::epsilon());
243 
245  // activations(i) = prev_activations(i) / scale_factor(i) ^ beta
246  // scale_factor(i)
247  // = k + alpha * sum( prev_activations(j) ^ 2 )
248  // Note: The sum is over entries in the normalization window.
250 
251  // Iterate through blocks in channels of each data sample
252  const int max_block_size = 16;
253  #pragma omp parallel for collapse(2)
254  for (int sample = 0; sample < local_width; ++sample) {
255  for (int block_start = 0;
256  block_start < num_per_channel;
257  block_start += max_block_size) {
258  const int block_size = std::min(max_block_size,
259  num_per_channel - block_start);
260  DataType workspace[max_block_size];
261 
262  // Iterate through channels
263  for (int channel = 0; channel < num_channels; ++channel) {
264  const int window_start = std::max(channel - m_window_width / 2, 0);
265  const int window_end = std::min(channel + m_window_width / 2, num_channels - 1);
266 
267  // Compute sum of squares in workspace
268  std::fill(workspace, workspace + block_size, DataType(0));
269  for (int window_pos = window_start; window_pos <= window_end; ++window_pos) {
270  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
271  const int index = block_start + block_pos + window_pos * num_per_channel;
272  const DataType input_entry = input_buffer[index + sample * input_ldim];
273  workspace[block_pos] += input_entry * input_entry;
274  }
275  }
276 
277  // Compute 1 / (k + alpha * sum(x^2) ) in workspace
278  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
279  workspace[block_pos] = 1 / (m_lrn_k + m_lrn_alpha * workspace[block_pos]);
280  }
281 
282  // Compute output
283  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
284  const int index = block_start + block_pos + channel * num_per_channel;
285  const DataType scale_factor = workspace[block_pos];
286  const DataType input_entry = input_buffer[index + sample * input_ldim];
287  DataType& output_entry = output_buffer[index + sample * output_ldim];
288  if (default_beta) { // Special case when beta = 0.75
289  output_entry = (input_entry
290  * std::sqrt(scale_factor * std::sqrt(scale_factor)));
291  }
292  else {
293  output_entry = input_entry * std::pow(scale_factor, m_lrn_beta);
294  }
295  }
296 
297  }
298 
299  }
300  }

305  void bp_compute_cpu() {
306 
307  // Get local matrices
308  const auto& local_input = get_local_prev_activations();
309  const auto& local_output = get_local_activations();
310  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
311  auto& local_gradient_wrt_input = get_local_error_signals();
312 
313  // Get matrix buffers
314  const int local_width = local_input.Width();
315  const DataType* input_buffer = local_input.LockedBuffer();
316  const int input_ldim = local_input.LDim();
317  const DataType* output_buffer = local_output.LockedBuffer();
318  const int output_ldim = local_output.LDim();
319  const DataType* gradient_wrt_output_buffer = local_gradient_wrt_output.LockedBuffer();
320  const int gradient_wrt_output_ldim = local_gradient_wrt_output.LDim();
321  DataType* gradient_wrt_input_buffer = local_gradient_wrt_input.Buffer();
322  const int gradient_wrt_input_ldim = local_gradient_wrt_input.LDim();
323 
324  // Get LRN parameters
325  const int num_channels = this->m_neuron_dims[0];
326  const int num_per_channel = this->m_num_neurons / num_channels;
327 
328  // Check if LRN is using default beta parameter
329  const bool default_beta = (std::fabs((m_lrn_beta - 0.75) / 0.75)
330  < 2 * std::numeric_limits<DataType>::epsilon());
331 
333  // error_signal(i)
334  // = prev_error_signal(i) / scale_factor(i) ^ beta
335  // - 2 * alpha * beta / window_width * prev_activations(i)
336  // * sum( prev_error_signal(j) * activations(j)
337  // / scale_factor(j) )
338  // Note: See comments in fp_linearity_cpu for a definition of
339  // scale_factor. The sum is over entries in the normalization
340  // window.
342 
343  // Iterate through blocks in channels of each data sample
344  const int max_block_size = 16;
345  #pragma omp parallel for collapse(2)
346  for (int sample = 0; sample < local_width; ++sample) {
347  for (int block_start = 0;
348  block_start < num_per_channel;
349  block_start += max_block_size) {
350  const int block_size = std::min(max_block_size,
351  num_per_channel - block_start);
352  DataType workspace[max_block_size];
353 
354  // Iterate through channels
355  for (int channel = 0; channel < num_channels; ++channel) {
356  const int window_start = std::max(channel - m_window_width / 2, 0);
357  const int window_end = std::min(channel + m_window_width / 2, num_channels - 1);
358 
359  // Compute sum of squares in workspace
360  std::fill(workspace, workspace + block_size, DataType(0));
361  for (int window_pos = window_start; window_pos <= window_end; ++window_pos) {
362  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
363  const int index = block_start + block_pos + window_pos * num_per_channel;
364  const DataType input_entry = input_buffer[index + sample * input_ldim];
365  workspace[block_pos] += input_entry * input_entry;
366  }
367  }
368 
369  // Compute 1 / (k + alpha * sum(x^2) ) in workspace
370  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
371  workspace[block_pos] = 1 / (m_lrn_k + m_lrn_alpha * workspace[block_pos]);
372  }
373 
374  // Compute error signal contribution for current entry
375  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
376  const int index = block_start + block_pos + channel * num_per_channel;
377  const DataType scale_factor = workspace[block_pos];
378  const DataType gradient_wrt_output_entry
379  = gradient_wrt_output_buffer[index + sample * gradient_wrt_output_ldim];
380  DataType& gradient_wrt_input_entry
381  = gradient_wrt_input_buffer[index + sample * gradient_wrt_input_ldim];
382  if (default_beta) { // Special case when beta = 0.75
383  gradient_wrt_input_entry
384  += gradient_wrt_output_entry * std::sqrt(scale_factor * std::sqrt(scale_factor));
385  }
386  else {
387  gradient_wrt_input_entry
388  += gradient_wrt_output_entry * std::pow(scale_factor, m_lrn_beta);
389  }
390  }
391 
392  // Compute y * dy / (k + alpha * sum(x^2) ) in workspace
393  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
394  const int index = block_start + block_pos + channel * num_per_channel;
395  const DataType output_entry = output_buffer[index + sample * output_ldim];
396  const DataType gradient_wrt_output_entry
397  = gradient_wrt_output_buffer[index + sample * gradient_wrt_output_ldim];
398  workspace[block_pos] = (-2 * m_lrn_alpha * m_lrn_beta * workspace[block_pos]
399  * output_entry * gradient_wrt_output_entry);
400  }
401 
402  // Compute error signal contribution for entries in window
403  for (int window_pos = window_start; window_pos <= window_end; ++window_pos) {
404  for (int block_pos = 0; block_pos < block_size; ++block_pos) {
405  const int index = block_start + block_pos + window_pos * num_per_channel;
406  const DataType input_entry = input_buffer[index + sample * input_ldim];
407  gradient_wrt_input_buffer[index + sample * gradient_wrt_input_ldim]
408  += workspace[block_pos] * input_entry;
409  }
410  }
411 
412  }
413 
414  }
415  }
