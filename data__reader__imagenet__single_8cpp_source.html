69 void imagenet_reader_single::load() {
70  const std::string image_dir = get_file_dir();
71  const std::string base_filename = get_data_filename();
72 
73  //open offsets file, with error checking
74  std::stringstream b;
75  b << image_dir << "/" << base_filename << "_offsets.txt";
76  if (is_master()) {
77  std::cout << "opening: " << b.str() << " " << std::endl;
78  }
79  std::ifstream in(b.str().c_str());
80  if (not in.is_open() and in.good()) {
81  std::stringstream err;
82  err << __FILE__ << " " << __LINE__
83  << " :: failed to open " << b.str() << " for reading";
84  throw lbann_exception(err.str());
85  }
86 
87  //read the offsets file
88  size_t num_images = 0u;
89  in >> num_images;
90 
91  if (is_master()) {
92  std::cout << "num images: " << num_images << std::endl;
93  }
94 
95  m_offsets.reserve(num_images+1);
96  m_offsets.emplace_back(0,0);
97  size_t last_offset = 0u;
98  size_t offset = 0u;
99  int label = 0;
100 
101  while (in >> offset >> label) {
102  m_offsets.emplace_back(offset + last_offset, label);
103  last_offset = m_offsets.back().first;
104  }
105 
106  if (num_images+1 != m_offsets.size()) {
107  std::stringstream err;
108  err << __FILE__ << " " << __LINE__
109  << " :: we read " << m_offsets.size() << " offsets, but should have read " << num_images;
110  throw lbann_exception(err.str());
111  }
112  in.close();
113 
114  open_data_stream();
115 
116  m_shuffled_indices.resize(num_images);
117  for (size_t n = 0; n < num_images; n++) {
118  m_shuffled_indices[n] = n;
119  }
120 
121  select_subset_of_data();
122 }

125 bool imagenet_reader_single::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
126  std::stringstream err;
127 
128  if (static_cast<size_t>(data_id+1) >= m_offsets.size()) {
129  err << __FILE__ << " " << __LINE__ << " :: data_id= " << data_id << " is larger than m_offsets.size()= " << m_offsets.size() << " -2";
130  throw lbann_exception(err.str());
131  }
132  const size_t start = m_offsets[data_id].first;
133  const size_t end = m_offsets[data_id+1].first;
134 
135  if (end > m_file_size) {
136  err << __FILE__ << " " << __LINE__ << " :: end= " << end << " is larger than m_file_size= " << m_file_size << " for P_" << get_rank() << " with role: " << get_role() << " m_offsets.size(): " << m_offsets.size() << " mb_idx: " << mb_idx << " data_id: " << data_id;
137  throw lbann_exception(err.str());
138  }
139 
140  const int ssz = end - start;
141 
142  if (ssz <= 0) {
143  err << "P_" << get_rank() << " start: " << start << " end: " << end << " ssz= " << ssz << " is <= 0";
144  throw lbann_exception(err.str());
145  }
146 
147  m_work_buffer[tid].resize(ssz);
148  m_data_filestream[tid]->seekg(start);
149  m_data_filestream[tid]->read((char *)&m_work_buffer[tid][0], ssz);
150 
151  int width=0, height=0, img_type=0;
152  ::Mat X_v = create_datum_view(X, mb_idx);
153 
154  const bool ret = image_utils::import_image(m_work_buffer[tid], width, height, img_type, *(m_pps[tid]), X_v);
155 
156  if(!ret) {
157  err << __FILE__ << " " << __LINE__ << " :: " << get_type() << ": image_utils::import_image failed to load index: " << data_id;
158  throw lbann_exception(err.str());
159  }
160  if ((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
161  err << __FILE__ << " " << __LINE__ << " :: " << get_type() << ": mismatch data size -- either width, height or channel";
162  throw lbann_exception(err.str());
163  }
164 
165  return true;
166 }

174 void imagenet_reader_single::open_data_stream() {
175  const std::string image_dir = get_file_dir();
176  const std::string base_filename = get_data_filename();
177 
178  std::stringstream b;
179  b << image_dir << "/" << base_filename << "_data.bin";
180  if (is_master()) {
181  std::cout << "opening: " << b.str() << " " << std::endl;
182  }
183 
184  const int nthreads = omp_get_max_threads();
185  m_work_buffer.resize(nthreads);
186 
187  for(auto & i : m_data_filestream) {
188  if (i) delete i;
189  }
190  m_data_filestream.clear();
191  m_data_filestream.resize(nthreads);
192 
193  #pragma omp parallel for schedule(static, 1)
194  for(int i=0; i < nthreads; ++i) {
195  m_data_filestream[i] = new std::ifstream(b.str().c_str(), std::ios::in | std::ios::binary);
196  if (m_data_filestream[i] && (not m_data_filestream[i]->is_open() or not m_data_filestream[i]->good())) {
197  std::stringstream err;
198  err << __FILE__ << " " << __LINE__
199  << " :: failed to open " << b.str() << " for reading";
200  throw lbann_exception(err.str());
201  }
202  m_data_filestream[i]->unsetf(std::ios::skipws);
203  if (i==0) {
204  m_data_filestream[i]->seekg(0, m_data_filestream[i]->end);
205  m_file_size = m_data_filestream[i]->tellg();
206  }
207  m_data_filestream[i]->seekg(0, m_data_filestream[i]->beg);
208  }
209 }
