59 void categorical_accuracy_metric::setup(model& m) {
60  metric::setup(m);
61  const El::DistData dist_data(get_target_layer().get_prediction());
62  if (dist_data.colDist == El::MC
63  && dist_data.rowDist == El::MR) {
64  m_prediction_values = new StarMRMat(*dist_data.grid, dist_data.root);
65  } else if (dist_data.colDist == El::STAR
66  && dist_data.rowDist == El::VC) {
67  m_prediction_values = new StarVCMat(*dist_data.grid, dist_data.root);
68  } else {
69  std::stringstream err;
70  err << __FILE__ << " " << __LINE__ << " :: "
71  << "prediction matrix has invalid distribution "
72  << "(colDist=" << dist_data.colDist << ","
73  << "rowDist=" << dist_data.rowDist << ")";
74  throw lbann_exception(err.str());
75  }
76  m_prediction_values->Resize(1, m.get_max_mini_batch_size());
77  m_prediction_indices.resize(m_prediction_values->LocalWidth());
78 }

80 EvalType categorical_accuracy_metric::evaluate_compute(const AbsDistMat& prediction,
81  const AbsDistMat& ground_truth) {
82 
83  // Get matrix dimensions
84  const int height = prediction.Height();
85  const int width = prediction.Width();
86  const int local_height = prediction.LocalHeight();
87  const int local_width = prediction.LocalWidth();
88 
89  // Get local matrices
90  const Mat& prediction_local = prediction.LockedMatrix();
91  const Mat& ground_truth_local = ground_truth.LockedMatrix();
92 
93  // Initialize workspace matrices
94  m_prediction_values->Resize(1, width);
95  Mat& prediction_values_local = m_prediction_values->Matrix();
96  m_prediction_indices.resize(local_width);
97 
98  // Find largest value in each column of prediction matrix
99  #pragma omp parallel for
100  for (int col = 0; col < local_width; ++col) {
101  DataType max_val;
102  int max_index;
103  if (local_height > 0) {
104  max_val = prediction_local(0, col);
105  max_index = 0;
106  } else {
107  max_val = std::numeric_limits<DataType>::lowest();
108  max_index = -1;
109  }
110  for (int row = 1; row < local_height; ++row) {
111  const DataType pred_val = prediction_local(row, col);
112  if (pred_val > max_val) {
113  max_val = pred_val;
114  max_index = row;
115  }
116  }
117  prediction_values_local(0, col) = max_val;
118  m_prediction_indices[col] = max_index;
119  }
120  get_comm().allreduce(*m_prediction_values,
121  m_prediction_values->RedundantComm(),
122  El::mpi::MAX);
123 
124  // Find first index corresponding to maximum prediction matrix values
125  #pragma omp parallel for
126  for (int col = 0; col < local_width; ++col) {
127  const int row = m_prediction_indices[col];
128  if (local_height > 0
129  && prediction_local(row, col) >= prediction_values_local(0, col)) {
130  m_prediction_indices[col] = prediction.GlobalRow(row);
131  } else {
132  m_prediction_indices[col] = height;
133  }
134  }
135  get_comm().allreduce(m_prediction_indices.data(),
136  local_width,
137  m_prediction_values->RedundantComm(),
138  El::mpi::MIN);
139 
140  // Count number of correct predictions
141  int correct_predictions = 0;
142  for (int col = 0; col < local_width; ++col) {
143  const int global_row = m_prediction_indices[col];
144  if (ground_truth.IsLocalRow(global_row)) {
145  const int row = ground_truth.LocalRow(global_row);
146  if (ground_truth_local(row, col) != DataType(0)) {
147  ++correct_predictions;
148  }
149  }
150  }
151  correct_predictions = get_comm().model_allreduce(correct_predictions);
152 
153  // Return percentage of correct predictions
154  return EvalType(100) * correct_predictions;
155 
156 }
