95  // Compute entrywise product

123  for (const auto& input : this->m_prev_activations) {
124  local_inputs.push_back(&input->LockedMatrix());
125  }
126  auto& local_gradient_wrt_output = get_local_prev_error_signals();
127  std::vector<Mat*> local_gradient_wrt_inputs;
128  for (auto& gradient_wrt_input : this->m_error_signals) {
129  local_gradient_wrt_inputs.push_back(&gradient_wrt_input->Matrix());
130  }
131 
132  // Compute derivative of entrywise product
133  const int local_height = local_gradient_wrt_output.Height();
134  const int local_width = local_gradient_wrt_output.Width();
135  const int num_parents = get_num_parents();
136  #pragma omp parallel for collapse(2)
137  for (int col = 0; col < local_width; ++col) {
138  for (int row = 0; row < local_height; ++row) {
139  const DataType dy = local_gradient_wrt_output(row, col);
140  for (int parent = 0; parent < num_parents; ++parent) {
141  DataType dx = dy;
142  for (int i = 0; i < num_parents; ++i) {
143  if (i != parent) {
144  dx *= (*local_inputs[i])(row, col);
145  }
146  }
147  (*local_gradient_wrt_inputs[parent])(row, col) += dx;
148  }
149  }
150  }

152  }
