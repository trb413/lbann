57  if (! m_in_memory) {
58  err << __FILE__ << " " << __LINE__ << " :: "

60  throw lbann_exception(err.str());
61  }
62 
63  else {
64  //sanity check

66  if (reader == nullptr) {
67  err << __FILE__ << " " << __LINE__ << " :: "
68  << "dynamic_cast<data_reader_pilot2_molecular*>(m_reader) failed";
69  throw lbann_exception(err.str());
70  }
71  m_pilot2_reader = reader;
72 
73  // get list of indices used in calls to generic_data_reader::fetch_data
74  // for this processor
75  get_minibatch_index_vector();
76 
77  // get list of indices used in calls to generic_data_reader::fetch_data
78  // for all processors
79  if (m_master) std::cerr << "calling exchange_mb_indices\n";
80  exchange_mb_indices();
81 
82  // allocate storage for the data that will be passed to the data reader's

89  }

92  std::cerr << "calling construct_data_store()\n";

130 void data_store_pilot2_molecular::fill_in_data(
131  const int data_id,
132  const int num_samples_per_frame,
133  const int num_features,
134  double *features) {
135  const int frame = m_pilot2_reader->get_frame(data_id);
136  const int frame_offset = frame * num_features * num_samples_per_frame;
137  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
138  double *data = features + frame_offset + intra_frame_data_id * num_features;
139  if (m_data.find(data_id) != m_data.end()) {
140  std::stringstream err;
141  err << __FILE__ << " :: " << __LINE__ << " :: "
142  << " duplicate data_id: " << data_id;
143  throw lbann_exception(err.str());
144  }
145  m_data[data_id].resize(num_features);
146  for (int i=0; i<num_features; i++) {
147  m_data[data_id][i] = m_pilot2_reader->scale_data<double>(i, data[i]);
148  }
149 }

154 void data_store_pilot2_molecular::build_nabor_map() {
155  //bcast neighbor data
156  size_t sz;
157  if (m_owner) {
158  sz = m_pilot2_reader->get_neighbors_data_size();
159  }
160  m_comm->world_broadcast<size_t>(0, sz);
161 
162  double *neighbors_8;
163  std::vector<double> work;
164  if (m_owner) {
165  neighbors_8 = m_pilot2_reader->get_neighbors_8();
166  } else {
167  work.resize(sz);
168  neighbors_8 = work.data();
169  }
170  m_comm->world_broadcast<double>(0, neighbors_8, sz);
171 
172  //fill in the nabors map
173  for (auto data_id : (*m_shuffled_indices)) {
174  int frame = m_pilot2_reader->get_frame(data_id);
175  int max_neighborhood = m_pilot2_reader->get_max_neighborhood();
176  int num_samples_per_frame = m_pilot2_reader->get_num_samples_per_frame();
177  const int neighbor_frame_offset = frame * num_samples_per_frame * (2 * max_neighborhood);
178  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
179  int num_neighbors = m_pilot2_reader->get_num_neighbors();
180  m_neighbors[data_id].reserve(num_neighbors);
181  double *neighbor_data = neighbors_8 + neighbor_frame_offset + intra_frame_data_id * (2 * max_neighborhood);
182  for (int i=1; i<num_neighbors + 1; ++i) {
183  int neighbor_id = neighbor_data[i];
184  m_neighbors[data_id].push_back(neighbor_id);
185  }
186  }
187 }

189 void data_store_pilot2_molecular::get_data_buf(int data_id, int tid, std::vector<double> *&buf) {
190  std::stringstream err;
191  std::vector<double> &v = m_data_buffer[tid];
192  std::fill(v.begin(), v.end(), 0.0);
193  if (m_neighbors.find(data_id) == m_neighbors.end()) {
194  err << __FILE__ << " " << __LINE__ << " :: "
195  << data_id << " not found in m_neighbors (primary molecule)";
196  throw lbann_exception(err.str());
197  }
198  if (m_my_molecules.find(data_id) == m_my_molecules.end()) {
199  err << __FILE__ << " " << __LINE__ << " :: "
200  << data_id << " not found in m_my_molecules";
201  throw lbann_exception(err.str());
202  }
203 
204  //fill in data for the primary molecule
205  size_t jj = 0;
206  std::vector<double> &d1 = m_my_molecules[data_id];
207  for (size_t j=0; j<d1.size(); j++) {
208  v[jj++] = d1[j];
209  }
210 
211  //fill in data for the primary molecule's neighbor
212  std::vector<int> &nabors = m_neighbors[data_id];
213  int num_features = m_pilot2_reader->get_num_features();
214  for (size_t h=0; h<nabors.size(); h++) {
215  if (nabors[h] != -1) {
216  if (m_my_molecules.find(data_id) == m_my_molecules.end()) {
217  err << __FILE__ << " " << __LINE__ << " :: "
218  << nabors[h] << " not found in m_my_molecules (neighbor)";
219  throw lbann_exception(err.str());
220  }
221  std::vector<double> &d2 = m_my_molecules[nabors[h]];
222  for (size_t i=0; i<d2.size(); i++) {
223  v[jj++] = d2[i];
224  }
225  } else {
226  jj += num_features;
227  }
228  }
229  buf = &v;
230 }

232 void data_store_pilot2_molecular::get_required_molecules(std::unordered_set<int> &required_molecules, int p) {
233  required_molecules.clear();
234  std::vector<int> &v = m_all_minibatch_indices[p];
235  for (auto t : v) {
236  int data_id = (*m_shuffled_indices)[t];
237  required_molecules.insert(data_id);
238  if (m_neighbors.find(data_id) == m_neighbors.end()) {
239  std::stringstream err;
240  err << __FILE__ << " :: " << __LINE__ << " :: "
241  << " m_neighbors.find(" << data_id << " failed";
242  throw lbann_exception(err.str());
243  }
244  for (auto t2 : m_neighbors[data_id]) {
245  if (t2 != -1) {
246  required_molecules.insert(t2);
247  }
248  }
249  }

252 void data_store_pilot2_molecular::exchange_data() {
253  double tm1 = get_time();
254  std::stringstream err;
255 
256  //get set of molecules required for the next epoch for myself
257  std::unordered_set<int> required_molecules;
258  get_required_molecules(required_molecules, m_rank);
259 
260  //start receives for my required molecules
261  m_my_molecules.clear();
262  int num_features = m_pilot2_reader->get_num_features();
263 
264  std::vector<El::mpi::Request<double>> recv_req(required_molecules.size());
265  size_t jj = 0;
266  for (auto data_id : required_molecules) {
267  m_my_molecules[data_id].resize(num_features);
268  m_comm->nb_tagged_recv<double>(
269  m_my_molecules[data_id].data(), num_features, m_owner_rank,
270  data_id, recv_req[jj++], m_comm->get_world_comm());
271  }
272 
273  //owner starts sends
274  std::vector<std::vector<El::mpi::Request<double>>> send_req;
275  if (m_owner) {
276  send_req.resize(m_np);
277  for (int p = 0; p<m_np; p++) {
278  jj = 0;
279  get_required_molecules(required_molecules, p);
280  send_req[p].resize(required_molecules.size());
281  for (auto data_id : required_molecules) {
282  m_comm->nb_tagged_send<double>(
283  m_data[data_id].data(), num_features, p,
284  data_id, send_req[p][jj++], m_comm->get_world_comm());
285  }
286  }
287  }
288 
289  //wait for sends to finish
290  if (m_owner) {
291  for (size_t i=0; i<send_req.size(); i++) {
292  m_comm->wait_all<double>(send_req[i]);
293  }
294  }
295 
296  //wait for recvs to finish
297  m_comm->wait_all<double>(recv_req);
298 
299  if (m_owner) {
300  std::cout << "TIME for data_store_pilot2_molecular::exchange_data(): "
301  << get_time() - tm1 << "; role: " << m_reader->get_role() << "\n";
302  }
303 }
