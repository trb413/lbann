39 bool get_single_channel_stats(const cv::Mat& _img, image_stats& stats) {
40  if (_img.channels() != 1) {
41  return false;
42  }
43 
44  cv::Mat img; // pw_fp_t matrix
45  _img.convertTo(img, _PW_CV_FP_);
46 
47  cv::MatIterator_<pw_fp_t> itBegin = img.begin<pw_fp_t>();
48  cv::MatIterator_<pw_fp_t> itEnd = img.end<pw_fp_t>();
49 
50  const auto typeZero = static_cast<pw_fp_t>(0);
51 
52  double sum = 0.0;
53 
54  std::vector<pw_fp_t> data(itBegin, itEnd);
55  stats.cnt = data.size();
56  if (stats.cnt == 0u) {
57  return false;
58  }
59 
60  std::sort(data.begin(), data.end());
61  if (data[0] < typeZero) {
62  return false;
63  }
64 
65  stats.max = data.back();
66  stats.min = data[0];
67 
68  std::vector<pw_fp_t>::const_iterator itbeg = data.begin();
69  std::vector<pw_fp_t>::const_iterator itend = data.end();
70  std::vector<pw_fp_t>::const_iterator itbegNZ = std::upper_bound(data.begin(), data.end(), static_cast<pw_fp_t>(0));
71 
72  stats.cntZeros = std::distance(itbeg, itbegNZ);
73  stats.minNZ = *itbegNZ;
74 
75  const size_t nnz = stats.cnt - stats.cntZeros;
76  const size_t halfPointNZ = nnz/2;
77  const size_t halfPoint = stats.cnt/2;
78  auto itMedNZ = itbegNZ;
79  auto itMed = itbeg;
80  std::advance(itMedNZ, halfPointNZ);
81  std::advance(itMed, halfPoint);
82 
83  stats.medianNZ = *itMedNZ;
84  stats.median = *itMed;
85 
86  auto it = itbegNZ;
87  for( ; it != itend; ++it) {
88  sum += *it;
89  }
90 
91  stats.avg = sum/stats.cnt;
92  if (nnz == 0u) {
93  stats.avgNZ = stats.avg;
94  } else {
95  stats.avgNZ = sum/nnz;
96  }
97 
98  double var = 0.0;
99  double varNZ = 0.0;
100  it = itbegNZ;
101 
102  for(it = itbeg; it != itbegNZ; ++it) {
103  const double dev = (*it-stats.avg);
104  var += dev*dev;
105  }
106 
107  for( ; it != itend; ++it) {
108  const double dev = (*it-stats.avg);
109  var += dev*dev;
110  const double devNZ = (*it-stats.avgNZ);
111  varNZ += devNZ*devNZ;
112  }
113 
114  stats.stdev = sqrt(var/stats.cnt);
115  stats.stdevNZ = sqrt(varNZ/nnz);
116 
117  return true;
118 }

120 bool get_channel_stats(const cv::Mat& img, std::vector<image_stats>& stats) {
121  if (img.data == nullptr) {

123  return false;

129
