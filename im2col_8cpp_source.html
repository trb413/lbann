35  const int im_num_dims,
36  const int * im_dims,
37  const int * im_pads,
38  const int * window_dims,
39  const int * window_strides) {
40 
41  // Input and output parameters
42  const int col_height = col.Height();
43  const int col_width = col.Width();
44  const DataType *__restrict__ im_buffer = im.LockedBuffer();
45  DataType *__restrict__ col_buffer = col.Buffer();

47  // im2col parameters
48  std::vector<int> offset_start(im_num_dims);
49  std::vector<int> offset_end(im_num_dims);
50  std::vector<int> offset_stride(im_num_dims);
51  std::vector<int> offset_num(im_num_dims);
52  for(int d = 0; d < im_num_dims; ++d) {
53  offset_start[d] = -im_pads[d];
54  offset_end[d] = im_dims[d] + im_pads[d] - window_dims[d] + 1;
55  offset_stride[d] = window_strides[d];
56  offset_num[d] = (offset_end[d] - offset_start[d] + offset_stride[d] - 1) / offset_stride[d];
57  }
58 
59  #ifdef LBANN_DEBUG
60  const int im_size = im.Height();
61  // Check matrix dimensions
62  const int expected_im_size = std::accumulate(im_dims,
63  im_dims + im_num_dims,
64  num_channels,
65  std::multiplies<int>());
66  const int expected_col_height = std::accumulate(window_dims,
67  window_dims + im_num_dims,
68  num_channels,
69  std::multiplies<int>());
70  const int expected_col_width = std::accumulate(offset_num.begin(),
71  offset_num.end(),
72  1,
73  std::multiplies<int>());
74  if(im_size != expected_im_size || im.Width() != 1) {
75  std::stringstream ss;
76  ss << "im2col: im matrix has invalid dimensions "
77  << "(expected " << expected_im_size << " x " << 1 << ", "
78  << "found " << im_size << " x " << im.Width() << ")";
79  throw lbann_exception(ss.str());
80  }
81  if(col_height != expected_col_height
82  || col_width != expected_col_width) {
83  std::stringstream ss;
84  ss << "im2col: col matrix has invalid dimensions "
85  << "(expected " << expected_col_height << " x " << expected_col_width << ", "
86  << "found " << col_height << " x " << col_width << ")";
87  throw lbann_exception(ss.str());
88  }
89  #endif // LBANN_DEBUG
90 
91  // Call optimized routine for 1x1 im2col
92  std::vector<int> zeros(im_num_dims, 0), ones(im_num_dims, 1);
93  if(std::equal(im_pads, im_pads + im_num_dims, zeros.begin())
94  && std::equal(window_dims, window_dims + im_num_dims, ones.begin())
95  && std::equal(window_strides, window_strides + im_num_dims, ones.begin())) {
96  im2col_1x1(im_buffer, col_buffer, num_channels, im_num_dims, im_dims);
97  return;
98  }
99 
100  // Call optimized routine for 2D data
101  if(im_num_dims == 2) {
102  im2col_2d(im_buffer, col_buffer,
103  im_dims[1], im_dims[0], im_pads[1], im_pads[0], num_channels,
104  window_dims[1], window_dims[0],
105  window_strides[1], window_strides[0]);
106  return;
107  }
108 
109  // Iterate through col matrix columns
110  #pragma omp parallel for
111  for(int col_col = 0; col_col < col_width; ++col_col) {
112 
113  // Initialize arrays
114  std::vector<int> offset_pos(im_num_dims);
115  std::vector<int> window_pos(im_num_dims);
116 
117  // Get position of current offset
118  int col_col_remainder = col_col;
119  for(int d = im_num_dims-1; d >= 0; --d) {
120  const int offset = col_col_remainder % offset_num[d];
121  offset_pos[d] = offset_start[d] + offset * offset_stride[d];
122  col_col_remainder /= offset_num[d];
123  }
124 
125  // Iterate through col matrix entries
126  for(int col_row = 0; col_row < col_height; ++col_row) {
127  const int col_index = col_row + col_col * col_height;
128 
129  // Get position in window and channel
130  int col_row_remainder = col_row;
131  for(int d = im_num_dims-1; d >= 0; --d) {
132  window_pos[d] = col_row_remainder % window_dims[d];
133  col_row_remainder /= window_dims[d];
134  }
135  const int channel = col_row_remainder;
136 
137  // Get im matrix entry
138  bool im_pos_valid = true;
139  int im_index = channel;
140  for(int d = 0; d < im_num_dims; ++d) {
141  const int im_pos = offset_pos[d] + window_pos[d];
142  im_pos_valid = im_pos_valid && 0 <= im_pos && im_pos < im_dims[d];
143  im_index = im_pos + im_index * im_dims[d];
144  }
145 
146  // Copy im matrix entry to col matrix if valid
147  col_buffer[col_index] = (im_pos_valid ?
148  im_buffer[im_index] : DataType(0));
149 
150  }
151  }

239 void col2im(const Mat& col,
240  Mat& im,
241  const int num_channels,
242  const int im_num_dims,
243  const int * im_dims,
244  const int * im_pads,
245  const int * window_dims,
246  const int * window_strides,
247  std::function<DataType(const DataType&,const DataType&)> reduction_op) {
248 
249  // Input and output parameters
250  const int col_height = col.Height();
251  const int im_size = im.Height();
252  const DataType *__restrict__ col_buffer = col.LockedBuffer();
253  DataType *__restrict__ im_buffer = im.Buffer();
254 
255  // im2col parameters
256  std::vector<int> offset_start(im_num_dims);
257  std::vector<int> offset_end(im_num_dims);
258  std::vector<int> offset_stride(im_num_dims);
259  std::vector<int> offset_num(im_num_dims);
260  for(int d = 0; d < im_num_dims; ++d) {
261  offset_start[d] = -im_pads[d];
262  offset_end[d] = im_dims[d] + im_pads[d] - window_dims[d] + 1;
263  offset_stride[d] = window_strides[d];
264  offset_num[d] = (offset_end[d] - offset_start[d] + offset_stride[d] - 1) / offset_stride[d];
265  }
266 
267  // Call optimized routine for 1x1 col2im
268  std::vector<int> zeros(im_num_dims, 0), ones(im_num_dims, 1);
269  if(std::equal(im_pads, im_pads + im_num_dims, zeros.begin())
270  && std::equal(window_dims, window_dims + im_num_dims, ones.begin())
271  && std::equal(window_strides, window_strides + im_num_dims, ones.begin())) {
272  col2im_1x1(col_buffer, im_buffer, num_channels, im_num_dims, im_dims);
273  return;
274  }
275 
276  // Iterate through im matrix entries
277  #pragma omp parallel for
278  for(int im_index = 0; im_index < im_size; ++im_index) {
279 
280  // Initialize arrays
281  std::vector<int> im_pos(im_num_dims);
282  std::vector<int> first_offset(im_num_dims);
283  std::vector<int> last_offset(im_num_dims);
284  std::vector<int> offset(im_num_dims);
285 
286  // Get position of im matrix entry
287  int im_index_remainder = im_index;
288  for(int d = im_num_dims-1; d >= 0; --d) {
289  im_pos[d] = im_index_remainder % im_dims[d];
290  im_index_remainder /= im_dims[d];
291  }
292  const int channel = im_index_remainder;
293 
294  // Initialize im matrix entry
295  DataType im_entry = 0;
296  bool im_entry_initialized = false;
297  bool offsets_finished = false;
298 
299  // Get window offsets containing im matrix entry
300  for(int d = 0; d < im_num_dims; ++d) {
301  first_offset[d] = (im_pos[d] - offset_start[d] - window_dims[d] + offset_stride[d]) / offset_stride[d];
302  first_offset[d] = std::max(first_offset[d], 0);
303  last_offset[d] = (im_pos[d] - offset_start[d]) / offset_stride[d];
304  last_offset[d] = std::min(last_offset[d], offset_num[d] - 1);
305  offset[d] = first_offset[d];
306  if(first_offset[d] > last_offset[d]) {
307  offsets_finished = true;
308  }
309  }
310 
311  // Iterate through window offsets containing im matrix entry
312  while(!offsets_finished) {
313 
314  // Get col matrix entry corresponding to im matrix entry
315  int col_row = channel;
316  int col_col = 0;
317  for(int d = 0; d < im_num_dims; ++d) {
318  const int window_pos = im_pos[d] - (offset_start[d] + offset[d] * offset_stride[d]);
319  col_row = window_pos + col_row * window_dims[d];
320  col_col = offset[d] + col_col * offset_num[d];
321  }
322  const int col_index = col_row + col_col * col_height;
323 
324  // Add col matrix entry to im matrix entry
325  const DataType col_entry = col_buffer[col_index];
326  im_entry = (im_entry_initialized ?
327  reduction_op(im_entry, col_entry) :
328  col_entry);
329  im_entry_initialized = true;
330 
331  // Move to next window offset
332  ++offset[im_num_dims-1];
333  for(int d = im_num_dims-1; d >= 1; --d) {
334  if(offset[d] > last_offset[d]) {
335  offset[d] = first_offset[d];
336  ++offset[d-1];
337  }
338  }
339  offsets_finished = offset[0] > last_offset[0];
340 
341  }
342 
343  // Update output entry
344  im_buffer[im_index] = im_entry;
345 
346  }

406  && input_pos_y < input_dim_y
407  && 0 <= input_pos_x
408  && input_pos_x < input_dim_x);
409 
410  // Get output entry
411  const int output_row = (window_pos_x
412  + window_pos_y * window_dim_x
413  + channel * window_dim_x * window_dim_y);

417  // Copy input entry to output entry if valid
