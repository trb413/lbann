527  bool pack_scalars(persist& p, const char *name) {
528  char fieldname[1024];
529  lbann::persist_type persist_value;
530  std::string s_name(name);
531  if(s_name.compare("data_reader_validation") == 0){
532  persist_value = persist_type::validate;
533  } else {
534  persist_value= persist_type::train;
535  }
536 
537 
538  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
539  p.write_uint64(persist_value, fieldname, (uint64_t) m_current_mini_batch_idx);
540 
541  int size = m_shuffled_indices.size();
542  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
543  p.write_uint64(persist_value, fieldname, (uint64_t) size);
544 
545  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
546  p.write_uint64(persist_value, fieldname, (uint64_t) m_current_pos);
547 
548  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
549  p.write_int32_contig(persist_value, fieldname, &m_shuffled_indices[0], (uint64_t) size);
550 
551  return true;
552  }

554  bool unpack_scalars(persist& p, struct packing_header *header, const char *name){
555  char fieldname[1024];
556  lbann::persist_type persist_value;
557  std::string s_name(name);
558  if(s_name.compare("data_reader_validation") == 0){
559  persist_value = persist_type::validate;
560  } else {
561  persist_value= persist_type::train;
562  }
563  // Closest to non checkpoint run only loads m_current_pos
564 
565  // record minibatch index
566  uint64_t val;
567 
568  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
569  p.read_uint64(persist_value, fieldname, &val);
570  m_current_mini_batch_idx = (int) val;
571 
572  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
573  p.read_uint64(persist_value, fieldname, &val);
574  auto size = (int) val;
575 
576  // get current position within data
577  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
578  p.read_uint64(persist_value, fieldname, &val);
579  m_current_pos = (int) val;
580  //resize shuffled index array to hold values
581  m_shuffled_indices.resize(size);
582 
583  //read list of indices
584  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
585  p.read_int32_contig(persist_value, fieldname, &m_shuffled_indices[0], (uint64_t) size);
586 
587  if(header != nullptr){
588  //shuffled data indices array size, used for resize after broadcast. Not unpacked.
589  header->data_size = size;
590  // all else, unpacked and set in unpack header.
591  header->current_pos = m_current_pos;
592  header->current_mini_batch_idx = m_current_mini_batch_idx;
593  }
594 
595  return true;
596  }
