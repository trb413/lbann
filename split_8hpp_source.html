137  void bp_compute_cudnn() {
138  #ifndef LBANN_HAS_CUDNN
139  throw lbann_exception("split_layer: cuDNN not detected");
140  #else
141 
142  // Useful constant
143  const DataType one = 1;
144 
145  // Copy error signal from first child layer
146  this->m_cudnn->copy_on_gpus(this->m_error_signal_d,
147  this->m_prev_error_signal_dv,
148  this->m_num_neurons,
149  this->m_mini_batch_size_per_gpu);
150 
151  // Iterate through child layers
152  const int num_gpus = this->m_cudnn->get_num_gpus();
153  for(size_t child_index = 1;
154  child_index < this->m_child_layers.size();
155  ++child_index) {
156  const Layer* child = this->m_child_layers[child_index];
157 
158  // Get child error signal on GPUs
159  if(child->using_gpus()) {
160  child->get_gpu_bp_output(this->m_prev_error_signal_dv,
161  this->m_prev_error_signal_d,
162  this);
163  }
164  else {
165  child->get_bp_output(*this->m_prev_error_signal_v, this);
166  if(m_prev_error_signal_d.empty()) {
167  m_cudnn->allocate_on_gpus(m_prev_error_signal_d,
168  m_num_neurons,
169  m_max_mini_batch_size_per_gpu);
170  }
171  this->m_cudnn->scatter_to_gpus(this->m_prev_error_signal_d,
172  this->m_prev_error_signal_v->LockedMatrix(),
173  this->m_mini_batch_size_per_gpu);
174  m_prev_error_signal_dv = m_prev_error_signal_d;
175  }
176 
177  // Add child error signal to this layer's error signal
178  for(int i=0; i<num_gpus; ++i) {
179  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
180  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
181  this->m_cudnn->get_stream(i)));
182  CHECK_CUDNN(cudnnAddTensor(this->m_cudnn->get_handle(i),
183  &one,
184  this->m_neurons_cudnn_desc,
185  this->m_prev_error_signal_dv[i],
186  &one,
187  this->m_prev_neurons_cudnn_desc,
188  this->m_error_signal_d[i]));
189  }
190 
191  }
192 
193  #endif // #ifndef LBANN_HAS_CUDNN
194  }

196  void bp_compute_cpu() {
197  El::Copy(*this->m_prev_error_signal_v, *this->m_error_signal_v);
198  for(size_t i=1; i<this->m_child_layers.size(); ++i) {
199  this->m_child_layers[i]->get_bp_output(*this->m_prev_error_signal_v, this);
200  El::Axpy(DataType(1),
201  *this->m_prev_error_signal_v,
202  *this->m_error_signal_v);
203  }
204  }
