54 void data_store_merge_features::setup() {
55  double tm1 = get_time();
56  if (m_master) {
57  std::cerr << "starting data_store_merge_features::setup() for data reader with role: " << m_reader->get_role() << std::endl;
58  }
59 
60  generic_data_store::setup();
61 
62  if (! m_in_memory) {
63  std::stringstream err;
64  err << __FILE__ << " " << __LINE__ << " :: "
65  << "not yet implemented";
66  throw lbann_exception(err.str());
67  }
68 
69  else {
70  //sanity check
71  data_reader_merge_features *reader = dynamic_cast<data_reader_merge_features*>(m_reader);
72  if (reader == nullptr) {
73  std::stringstream err;
74  err << __FILE__ << " " << __LINE__ << " :: "
75  << "dynamic_cast<merge_features_reader*>(m_reader) failed";
76  throw lbann_exception(err.str());
77  }
78 
79  // get list of indices used in calls to generic_data_reader::fetch_data
80  if (m_master) std::cerr << "calling get_minibatch_index_vector\n";
81  get_minibatch_index_vector();
82 
83  if (m_master) std::cerr << "calling get_my_datastore_indices\n";
84  get_my_datastore_indices();
85 
86  if (m_master) std::cerr << "calling exchange_mb_indices()\n";
87  exchange_mb_indices();
88 
89  std::vector<generic_data_reader*> &readers = reader->get_data_readers();
90  m_subsidiary_stores.reserve(readers.size());
91  for (auto r : readers) {
92  data_store_csv *store = new data_store_csv(r, m_model);
93  m_subsidiary_stores.push_back(store);
94  r->set_data_store(store);
95  store->set_is_subsidiary_store();
96  store->set_minibatch_indices(get_minibatch_indices());
97  store->set_all_minibatch_indices(get_all_minibatch_indices());
98  store->set_minibatch_indices_v(get_minibatch_indices_v());
99  store->set_datastore_indices(get_datastore_indices());
100  store->setup();
101  store->set_shuffled_indices(m_shuffled_indices, false);
102  store->populate_datastore();
103  store->exchange_data();
104  }
105  }
106  if (m_master) {
107  std::cerr << "TIME for data_store_merge_features setup: " << get_time() - tm1 << "\n";
108  }
109 }
