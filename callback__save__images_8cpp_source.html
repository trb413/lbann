36 void lbann_callback_save_images::on_epoch_end(model *m) {

60 void lbann_callback_save_images::save_image(model& m,
61  AbsDistMat& input,
62  AbsDistMat& output,
63  int index) {
64 
65  // Save input image
66  AbsDistMat* input_col = input.Construct(input.Grid(),
67  input.Root());
68  El::View(*input_col, input, El::ALL, El::IR(0));
69  CircMat input_circ = *input_col;
70  delete input_col;
71  if(m.get_comm()->am_world_master()) {
72  m_reader->save_image(input_circ.Matrix(), m_image_dir+"input_"+ std::to_string(index)+"."+m_extension);
73  }
74 
75  // Save output image if it is a reconstruction
76  if(output.Height() == input.Height()) {
77  AbsDistMat* output_col = output.Construct(output.Grid(),
78  output.Root());
79  El::View(*output_col, output, El::ALL, El::IR(0));
80  CircMat output_circ = *output_col;
81  delete output_col;
82  if(m.get_comm()->am_world_master()) {
83  m_reader->save_image(output_circ.Matrix(), m_image_dir+"output_"+ std::to_string(index)+"."+m_extension);
84  }
85  }
86 
87  // Output message
88  if(m.get_comm()->am_world_master()) {
89  std::cout << "Saved images to " << m_image_dir << std::endl;
90  }
91 
92 }
