76 double lbann::metrics::categorical_accuracy::compute_metric(ElMat& predictions_v, ElMat& groundtruth_v) {
77 
78  // Clear the contents of the intermediate matrices
79  Zeros(YsColMax, m_max_mini_batch_size, 1);
80  Zeros(YsColMaxStar, m_max_mini_batch_size, 1);
81 
83  ColumnMaxNorms((DistMat) predictions_v, YsColMax_v);
84  Copy(YsColMax_v, YsColMaxStar_v);
85 
86  Zeros(m_max_index, m_max_mini_batch_size, 1); // Clear the entire matrix
87 
89  for(Int mb_index = 0; mb_index < predictions_v.LocalWidth(); mb_index++) {
90  int mb_global_index = predictions_v.GlobalCol(mb_index);
91  DataType sample_max = YsColMaxStar_v.GetLocal(mb_global_index, 0);
92  for(Int f_index = 0; f_index < predictions_v.LocalHeight(); f_index++) {
93  if(predictions_v.GetLocal(f_index, mb_index) == sample_max) {
94  m_max_index_v.Set(mb_global_index, 0, predictions_v.GlobalRow(f_index));
95  }
96  }
97  }
98 
99  Zeros(m_reduced_max_indices, m_max_mini_batch_size, 1); // Clear the entire matrix
102  comm->model_allreduce(m_max_index.Buffer(), m_max_index.Height() * m_max_index.Width(), m_reduced_max_indices.Buffer(), mpi::MAX);
103 
105  int num_errors = 0;
106 
107  // Copy(groundtruth_v, Y_local);
108 
112 
115  for(Int mb_index= 0; mb_index < groundtruth_v.LocalWidth(); mb_index++) {
116  Int targetidx = -1;
117  for(Int f_index= 0; f_index < groundtruth_v.LocalHeight(); f_index++) {
118  if(groundtruth_v.GetLocal(f_index, mb_index) == (DataType) 1.) {
119  targetidx = groundtruth_v.GlobalRow(f_index);
120  }
121  }
122  if(targetidx != -1) {
123  Int global_mb_index = groundtruth_v.GlobalCol(mb_index);
124  if(m_reduced_max_indices_v.Get(global_mb_index, 0) != targetidx) {
125  num_errors++;
126  }
127  }
128  }
129 
130  num_errors = comm->model_allreduce(num_errors);
131  return num_errors;
132 }
