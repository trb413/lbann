209  // Free allocated memory and copy data from other matrix
210 #define COPY_MATRIX(src, dst) \
211  do { \
212  if(src != nullptr && dst != nullptr) { \
213  El::Copy(*src, *dst); \
214  } \
215  if(src != nullptr && dst == nullptr) { \
216  dst = src->Copy(); \
217  } \
218  if(src == nullptr && dst != nullptr) { \
219  delete dst; \
220  dst = nullptr; \
221  } \
222  } while(false)
223  COPY_MATRIX(other.m_activations, m_activations);
224  COPY_MATRIX(other.m_error_signal, m_error_signal);
225  COPY_MATRIX(other.m_prev_activations_v, m_prev_activations_v);
226  COPY_MATRIX(other.m_prev_error_signal_v, m_prev_error_signal_v);
227  COPY_MATRIX(other.m_activations_v, m_activations_v);
228  COPY_MATRIX(other.m_error_signal_v, m_error_signal_v);
229 #undef COPY_MATRIX
230 
231 #ifdef LBANN_HAS_CUDNN
232  if(m_cudnn != nullptr) {
233  m_cudnn->deallocate_on_gpus(m_prev_activations_d);
234  m_cudnn->deallocate_on_gpus(m_activations_d);
235  m_cudnn->deallocate_on_gpus(m_prev_error_signal_d);
236  m_cudnn->deallocate_on_gpus(m_error_signal_d);
237  m_prev_activations_d = m_cudnn->copy(other.m_prev_activations_d,

239  m_max_mini_batch_size_per_gpu);
240  m_activations_d = m_cudnn->copy(other.m_activations_d,
241  m_num_neurons,

243  m_prev_error_signal_d = m_cudnn->copy(other.m_prev_error_signal_d,

249  }
250  cudnn::copy_tensor_cudnn_desc(other.m_prev_neurons_cudnn_desc,
251  m_prev_neurons_cudnn_desc);

287 void Layer::forward_prop() {
288  double fp_start = get_time();
289 
290  // Set matrix views based on current mini-batch size
291  fp_set_std_matrix_view();
292 
293  // Get incoming activations and convert matrix distribution if necessary
294  if(!m_parent_layers.empty()) {
295  m_parent_layers.front()->get_fp_output(*m_prev_activations_v, this);
296  } else {
297  const int mini_batch_size = m_model->get_current_mini_batch_size();
298  El::Zeros(*m_prev_activations_v, m_num_prev_neurons, mini_batch_size);
299  }
300 
301 #ifdef LBANN_HAS_CUDNN
302  if(m_using_gpus) {
303 
304  // Determine mini-batch size per GPU
305  const int num_gpus = m_cudnn->get_num_gpus();
306  const int local_mini_batch_size = m_activations_v->LocalWidth();
307  m_mini_batch_size_per_gpu = (local_mini_batch_size + num_gpus - 1) / num_gpus;
308 
309  // Set tensor descriptors
310  cudnn::set_tensor_cudnn_desc(m_prev_neurons_cudnn_desc,
311  m_mini_batch_size_per_gpu,
312  m_prev_neuron_dims);
313  cudnn::set_tensor_cudnn_desc(m_neurons_cudnn_desc,
314  m_mini_batch_size_per_gpu,
315  m_neuron_dims);
316 
317  // Transfer inputs from CPU to GPUs if needed
318  if(m_copy_fp_input_to_gpus) {
319  if(m_prev_activations_d.empty()) {
320  m_cudnn->allocate_on_gpus(m_prev_activations_d,
321  m_num_prev_neurons,
322  m_max_mini_batch_size_per_gpu);
323  }
324  m_cudnn->scatter_to_gpus(m_prev_activations_d,
325  m_prev_activations_v->LockedMatrix(),
326  m_mini_batch_size_per_gpu);
327  m_prev_activations_dv = m_prev_activations_d;
328  } else {
329  if(!m_parent_layers.empty()) {
330  m_parent_layers.front()->get_gpu_fp_output(m_prev_activations_dv,
331  m_prev_activations_d,
332  this);
333  }
334  }
335 
336  }
337 #endif // LBANN_HAS_CUDNN
338 
339  // Apply layer's compute function
340  double fp_compute_start = get_time();
341  fp_compute();
342  fp_compute_time += get_time() - fp_compute_start;
343 
344 #ifdef LBANN_HAS_CUDNN
345  // Transfer outputs from GPUs to CPU if needed
346  if(m_using_gpus && m_copy_fp_output_from_gpus) {
347  m_cudnn->gather_from_gpus(m_activations_v->Matrix(),
348  m_activations_d,
349  m_mini_batch_size_per_gpu);
350  m_cudnn->synchronize();
351  }
352 #endif // LBANN_HAS_CUDNN
353 
354  fp_time += get_time() - fp_start;
355 }

436 void Layer::summarize_stats(lbann_summary& summarizer, int step) {
437  std::string prefix = m_name + "/";
438  summarizer.reduce_scalar(prefix + "fp_time", fp_time, step);
439  summarizer.reduce_scalar(prefix + "bp_time", bp_time, step);
440  summarizer.reduce_scalar(prefix + "update_time", update_time, step);
441  reset_counters();
442  // Combine the optimizer step time from all the weights.
443  double step_time = 0.0;
444  for (weights *w : get_weights()) {
445  optimizer *opt = w->get_optimizer();
446  if (opt) {
447  step_time += opt->get_step_time();
448  opt->reset_counters();
449  }
450  }
451  summarizer.reduce_scalar(prefix + "opt_time", step_time, step);
452 }

492 void Layer::setup_pointers() {
493  // Check if the number of parents/children are valid
494  if(m_max_num_parent_layers >= 0
495  && (int)m_parent_layers.size() > m_max_num_parent_layers) {
496  std::stringstream err;
497  err << __FILE__ << " " << __LINE__ << " :: "
498  << "layer " << m_name << " has too many parent layers "
499  << "(found ";
500  for (const auto& parent : m_parent_layers) {
501  err << parent->get_name() << ", ";
502  }
503  err << "but only " << m_max_num_parent_layers << " parent layers are allowed)";
504  throw lbann_exception(err.str());
505  }
506  if(m_max_num_child_layers >= 0
507  && (int)m_child_layers.size() > m_max_num_child_layers) {
508  std::stringstream err;
509  err << __FILE__ << " " << __LINE__ << " :: "
510  << "layer " << m_name << " has too many child layers "
511  << "(found ";
512  for (const auto& child : m_child_layers) {
513  err << child->get_name() << ", ";
514  }
515  err << "but only " << m_max_num_child_layers << " child layers are allowed)";
516  throw lbann_exception(err.str());
517  }

700 void Layer::pin_data() {
701 
702  // Get maximum mini-batch size
703  const int max_mini_batch_size = m_model->get_max_mini_batch_size();
704 
705  // Flags to determine whether to pin memory
706  bool pin_fp_input = false;
707  bool pin_fp_output = false;
708  bool pin_bp_input = false;
709  bool pin_bp_output = false;
710 
711  // Pin fp input if there is no input layer and this layer uses GPUs
712  if(m_parent_layers.empty() && m_using_gpus) {
713  pin_fp_input = true;
714  }
715 
716  // Pin fp input if input layer does not use GPUs, this layer uses
717  // GPUs, and input layer has different distribution
718  if(!m_parent_layers.empty()
719  && !m_parent_layers.front()->using_gpus()
720  && m_using_gpus) {
721  if(m_parent_layers.front()->m_activations->DistData()
722  != m_prev_activations_v->DistData()) {
723  pin_fp_input = true;
724  }
725  }
726 
727  // Pin fp output if this layer does not use GPUs, output layer uses
728  // GPUs, and output layer has same distribution
729  if(!m_using_gpus
730  && !m_parent_layers.empty()
731  && m_parent_layers.front()->using_gpus()) {
732  if(m_parent_layers.front()->m_prev_activations_v->DistData()
733  != m_activations->DistData()) {
734  pin_fp_output = true;
735  }
736  }
737 
738  // Pin fp output if this layer uses GPUs and output layer does not
739  // use GPUs
740  if(m_using_gpus
741  && !m_child_layers.empty()
742  && !m_child_layers.front()->using_gpus()) {
743  pin_fp_output = true;
744  }
745

747  if(m_using_gpus && m_child_layers.empty()) {
748  pin_fp_output = true;
749  }
750 
751  // Pin bp input if there is no input layer and this layer uses GPUs
752  if(m_child_layers.empty() && m_using_gpus) {
753  pin_bp_input = true;
754  }
755 
756  // Pin bp input if input layer does not use GPUs, this layer uses
757  // GPUs, and input layer has different distribution
758  if(!m_child_layers.empty()
