247 void Layer::forward_prop() {
248  const auto fp_start = get_time();
249 
250  // Setup matrix data, e.g. input matrices
251  fp_setup_data(m_model->get_current_mini_batch_size());
252 
253  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
254  // Synchronize GPUs and check for errors
255  if (m_using_gpus) { this->m_cudnn->check_error(); }
256  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
257 
258  // Apply layer's compute function
259  const auto fp_compute_start = get_time();
260  fp_compute();
261  m_fp_compute_time += get_time() - fp_compute_start;
262 
263  // Add this layer as a gradient source for weight optimizers
264  for (auto&& w : m_weights) {
265  optimizer* opt = w->get_optimizer();
266  if (opt != nullptr) { opt->add_gradient_source(this); }
267  }
268 
269  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
270  // Synchronize GPUs and check for errors
271  if (m_using_gpus) { this->m_cudnn->check_error(); }
272  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
273 
274  m_fp_time += get_time() - fp_start;
275 }

332 void Layer::summarize_stats(lbann_summary& summarizer, int step) {
333  std::string prefix = m_name + "/";
334  summarizer.reduce_scalar(prefix + "fp_time", m_fp_time, step);
335  summarizer.reduce_scalar(prefix + "bp_time", m_bp_time, step);
336  summarizer.reduce_scalar(prefix + "update_time", m_update_time, step);
337  reset_counters();
338  // Combine the optimizer step time from all the weights.
339  double step_time = 0.0;
340  for (weights *w : get_weights()) {
341  optimizer *opt = w->get_optimizer();
342  if (opt) {
343  step_time += opt->get_step_time();
344  opt->reset_counters();
345  }
346  }
347  summarizer.reduce_scalar(prefix + "opt_time", step_time, step);
348 }
