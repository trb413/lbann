37 lbann::lbann_comm::lbann_comm(int _procs_per_model) :
38  procs_per_model(_procs_per_model), num_model_barriers(0),
39  num_intermodel_barriers(0), num_global_barriers(0), bytes_sent(0),
40  bytes_received(0) {
41 
42  // Initialize parameters
43  int world_size = mpi::Size(mpi::COMM_WORLD);
44  if (procs_per_model == 0) {
45  procs_per_model = world_size;
46  }
47  num_models = world_size / procs_per_model;
48  model_rank = mpi::Rank(mpi::COMM_WORLD) / procs_per_model;
49  rank_in_model = mpi::Rank(mpi::COMM_WORLD) % procs_per_model;
50 
51  // Check if parameters are valid
52  if (procs_per_model > world_size) {
53  stringstream err;
54  err << __FILE__ << " " << __LINE__
55  << " :: Not enough processes to create one model; procs_per_model: "
56  << procs_per_model << " is larger than world_size: " << world_size;
57  throw lbann_exception(err.str());
58  }
59  if (world_size % procs_per_model != 0) {
60  stringstream err;
61  err << __FILE__ << " " << __LINE__
62  << " :: Procs per model does not divide total number of procs; procs_per_model: "
63  << procs_per_model << " total number of procs (world size): " << world_size;
64  throw lbann_exception(err.str());
65  }
66 
67  // Initialize model and intermodel communicators
68  mpi::Split(mpi::COMM_WORLD, model_rank, rank_in_model, model_comm);
69  mpi::Split(mpi::COMM_WORLD, rank_in_model, model_rank, intermodel_comm);
70 
71  // Initialize Elemental grid
72  grid = new Grid(model_comm);
73 
74  // Initialize node communicators
75  setup_node_comm();
76  procs_per_node = mpi::Size(node_comm);
77  rank_in_node = mpi::Rank(node_comm);
78 
79 }

212 void lbann::lbann_comm::setup_node_comm() {
213 
214  // Get string specifying compute node
215  char node_name[MPI_MAX_PROCESSOR_NAME];
216  int node_name_len;
217  int status = MPI_Get_processor_name(node_name, &node_name_len);
218  if(status != MPI_SUCCESS) {
219  throw lbann_exception("lbann_comm: error in MPI_Get_processor_name");
220  }
221 
222  // Hash the node names and split MPI processes
223  // TODO: Generate random salt. The salt must be shared across all ranks.
224  std::string node_string = "hXFgQFNrqyL1mIsq";
225  node_string += node_name;
226  int hash = std::hash<std::string>()(node_string);
227  mpi::Comm hash_comm;
228  mpi::Split(mpi::COMM_WORLD, hash, mpi::Rank(mpi::COMM_WORLD), hash_comm);
229 
230  // Hash the node names again and split MPI processes
231  // TODO: Generate random salt. The salt must be shared across all ranks.
232  node_string = "BGqXbNZqxuXBV5lm";
233  node_string += node_name;
234  hash = std::hash<std::string>()(node_string);
235  mpi::Split(hash_comm, hash, mpi::Rank(mpi::COMM_WORLD), node_comm);
