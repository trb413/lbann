55 lbann::lbann_comm::lbann_comm(int _procs_per_model) :
56  procs_per_model(_procs_per_model), num_model_barriers(0),
57  num_intermodel_barriers(0), num_global_barriers(0), bytes_sent(0),
58  bytes_received(0) {
59 
60  // Initialize parameters
61  int world_size = mpi::Size(mpi::COMM_WORLD);
62  if (procs_per_model == 0) {
63  procs_per_model = world_size;
64  }
65  num_models = world_size / procs_per_model;
66  model_rank = mpi::Rank(mpi::COMM_WORLD) / procs_per_model;
67  rank_in_model = mpi::Rank(mpi::COMM_WORLD) % procs_per_model;
68 
69  // Check if parameters are valid
70  if (procs_per_model > world_size) {
71  stringstream err;
72  err << __FILE__ << " " << __LINE__
73  << " :: Not enough processes to create one model; procs_per_model: "
74  << procs_per_model << " is larger than world_size: " << world_size;
75  throw lbann_exception(err.str());
76  }
77  if (world_size % procs_per_model != 0) {
78  stringstream err;
79  err << __FILE__ << " " << __LINE__
80  << " :: Procs per model does not divide total number of procs; procs_per_model: "
81  << procs_per_model << " total number of procs (world size): " << world_size;
82  throw lbann_exception(err.str());
83  }
84 
85  // Initialize model and intermodel communicators
86  mpi::Split(mpi::COMM_WORLD, model_rank, rank_in_model, model_comm);
87  mpi::Split(mpi::COMM_WORLD, rank_in_model, model_rank, intermodel_comm);
88 
89  // Initialize Elemental grid
90  grid = new Grid(model_comm);
91 
92  // Initialize node communicators
93  setup_node_comm();
94  procs_per_node = mpi::Size(node_comm);
95  rank_in_node = mpi::Rank(node_comm);
96 
97 }

199 void lbann::lbann_comm::intermodel_allreduce(
200  Mat& mat, int max_recv_count,
201  std::function<uint8_t *(Mat&, IR, IR, int&, bool, int)> send_transform,
202  std::function<int(uint8_t *, Mat&)> recv_transform,
203  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
204  const lbann::lbann_comm::allreduce_options opts) {
205  // Determine which algorithm to actually use.
206  lbann::lbann_comm::allreduce_algorithm algo = opts.algo;
207  if (algo == allreduce_algorithm::DEFAULT) {
208  algo = get_default_allreduce_algorithm();
209  }
210  const int nprocs = get_num_models();
211  const Int small_message_threshold = 64*64;
212  if (algo == allreduce_algorithm::DYNAMIC) {
213  // For small messages and power-of-2 number of processes, use RD.
214  if (!(nprocs & (nprocs - 1)) &&
215  mat.Height() * mat.Width() <= small_message_threshold) {
216  algo = allreduce_algorithm::RECURSIVE_DOUBLING;
217  } else {
218  algo = allreduce_algorithm::PAIRWISE_EXCHANGE_RING;
219  }
220  }
221  switch (algo) {
222  case allreduce_algorithm::RECURSIVE_DOUBLING:
223  recursive_doubling_allreduce_pow2(
224  intermodel_comm, mat, max_recv_count,
225  send_transform, recv_apply_transform, opts);
226  break;
227  case allreduce_algorithm::PAIRWISE_EXCHANGE_RING:
228  pe_ring_allreduce(
229  intermodel_comm, mat, max_recv_count, send_transform,
230  recv_transform, recv_apply_transform, opts);
231  break;
232  case allreduce_algorithm::RING:
233  ring_allreduce(
234  intermodel_comm, mat, max_recv_count, send_transform,
235  recv_transform, recv_apply_transform, opts);
236  break;
237  case allreduce_algorithm::RABENSEIFNER:
238  rabenseifner_allreduce(
239  intermodel_comm, mat, max_recv_count, send_transform,
240  recv_transform, recv_apply_transform, opts);
241  break;
242  case allreduce_algorithm::DEFAULT:
243  case allreduce_algorithm::DYNAMIC:
244  default:
245  throw lbann_exception("intermodel_allreduce: bad algorithm");
246  break;
247  }
248 }

250 void lbann::lbann_comm::recursive_doubling_allreduce_pow2(
251  mpi::Comm comm, Mat& mat, int max_recv_count,
252  std::function<uint8_t *(Mat&, IR, IR, int&, bool, int)> send_transform,
253  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
254  const lbann::lbann_comm::allreduce_options opts) {
255  double ar_start = get_time();
256  const int rank = mpi::Rank(comm);
257  const unsigned int nprocs = mpi::Size(comm);
258  if (nprocs == 1) {
259  return; // Nothing to do.
260  }
261  // This implementation requires a power-of-2 number of processes.
262  if (nprocs & (nprocs - 1)) {
263  throw lbann_exception("lbann_comm: recursive doubling allreduce requires"
264  " a power-of-2 number of participating processes");
265  }
266  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
267  uint8_t *recv_buf = max_recv_buf;
268  unsigned int mask = 1;
269  while (mask < nprocs) {
270  int partner = rank ^ mask; // The rank we exchange with this step.
271  const bool is_local = opts.no_local_trans &&
272  is_rank_node_local(partner, comm);
273  // Transform the data we want to send.
274  double send_trans_start = get_time();
275  int send_size;
276  int recv_size = max_recv_count;
277  uint8_t *send_buf = nullptr;
278  if (is_local) {
279  send_buf = (uint8_t *) mat.Buffer();
280  send_size = sizeof(DataType) * mat.Height() * mat.Width();
281  recv_size = send_size;
282  recv_buf = get_collective_buffer(recv_size);
283  } else {
284  send_buf = send_transform(mat, ALL, ALL, send_size, false, 0);
285  recv_buf = max_recv_buf;
286  }
287  ar_send_transform_time += get_time() - send_trans_start;
288  bytes_sent += send_size;
289  ar_bytes_sent += send_size;
290  double sendrecv_start = get_time();
291  mpi::SendRecv(send_buf, send_size, partner,
292  recv_buf, recv_size, partner, comm);
293  double sendrecv_tot = get_time() - sendrecv_start;
294  ar_send_time += sendrecv_tot;
295  ar_recv_time += sendrecv_tot;
296  // Transform and reduce the received data.
297  double recv_apply_trans_start = get_time();
298  recv_size = recv_apply_transform(recv_buf, mat, is_local);
299  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
300  bytes_received += recv_size;
301  ar_bytes_received += recv_size;
302  mask <<= 1;
303  }
304  ar_time += get_time() - ar_start;
305 }

307 void lbann::lbann_comm::pe_ring_allreduce(
308  mpi::Comm comm, Mat& mat, int max_recv_count,
309  std::function<uint8_t *(Mat&, IR, IR, int&, bool, int)> send_transform,
310  std::function<int(uint8_t *, Mat&)> recv_transform,
311  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
312  const lbann::lbann_comm::allreduce_options opts) {
313  double ar_start = get_time();
314  const int rank = mpi::Rank(comm);
315  const int nprocs = mpi::Size(comm);
316  if (nprocs == 1) {
317  return; // Nothing to do.
318  }
319  // Compute the number of columns each processor sends.
320  // If it doesn't divide evenly, give one extra to the earlier ranks.
321  const Int cols_per_proc = mat.Width() / nprocs;
322  const Int cols_remainder = mat.Width() % nprocs;
323  // Compute the lengths/ends of the slices.
324  std::vector<Int> slice_lengths(nprocs, cols_per_proc);
325  for (int i = 0; i < cols_remainder; ++i) {
326  slice_lengths[i] += 1;
327  }
328  std::vector<Int> slice_ends(nprocs);
329  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
330  slice_ends.begin());
331  std::vector<uint8_t *> max_recv_buffers(opts.max_reduces, nullptr);
332  for (size_t i = 0; i < max_recv_buffers.size(); ++i) {
333  max_recv_buffers[i] = get_collective_buffer(max_recv_count, i);
334  }
335  // Local slice of our accumulated data.
336  auto accum_view = mat(ALL, IR(slice_ends[rank] - slice_lengths[rank],
337  slice_ends[rank]));
338  // Do a pairwise-exchange reduce-scatter.
339  double rs_start = get_time();
340  for (int outer_step = 1;
341  outer_step < nprocs;
342  outer_step += opts.max_reduces) {
343  const int reduces_this_step = std::min(opts.max_reduces,
344  nprocs - outer_step);
345  std::vector<mpi::Request<uint8_t>> send_reqs(reduces_this_step);
346  std::vector<mpi::Request<uint8_t>> recv_reqs(reduces_this_step);
347  std::vector<uint8_t *> recv_buffers(max_recv_buffers);
348  int num_local_recvs = 0;
349  std::vector<bool> local_recvs(reduces_this_step, false);
350  for (int step = outer_step; step < outer_step + reduces_this_step; ++step) {
351  const int reduce_idx = step - outer_step;
352  // Compute where we send to/receive from.
353  const int dst = (rank + step) % nprocs;
354  const int src = (rank - step + nprocs) % nprocs;
355  const bool is_send_local = opts.no_local_trans &&
356  is_rank_node_local(dst, comm);
357  const bool is_recv_local = opts.no_local_trans &&
358  is_rank_node_local(src, comm);
359  // Post the receive.
360  double recv_start = get_time();
361  int recv_size = max_recv_count;
362  if (is_recv_local) {
363  recv_size = sizeof(DataType) * accum_view.Height() * accum_view.Width();
364  recv_buffers[reduce_idx] = get_collective_buffer(recv_size,
365  num_local_recvs);
366  ++num_local_recvs;
367  local_recvs[reduce_idx] = is_recv_local;
368  }
369  mpi::IRecv(recv_buffers[reduce_idx], recv_size, src, comm,
370  recv_reqs[reduce_idx]);
371  double recv_tot = get_time() - recv_start;
372  ar_recv_time += recv_tot;
373  ar_rs_recv_time += recv_tot;
374  // Transform the data we send. We do not look at the same chunk of data
375  // twice.
376  double send_trans_start = get_time();
377  int send_size;
378  uint8_t *send_buf = nullptr;
379  if (is_send_local) {
380  auto send_view = mat(
381  ALL, IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
382  send_buf = (uint8_t *) send_view.Buffer();
383  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
384  } else {
385  send_buf = send_transform(
386  mat, ALL, IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]),
387  send_size, true, reduce_idx);
388  }
389  ar_send_transform_time += get_time() - send_trans_start;
390  bytes_sent += send_size;
391  ar_bytes_sent += send_size;
392  ar_rs_bytes_sent += send_size;
393  // Post the send.
394  double send_start = get_time();
395  mpi::ISend(send_buf, send_size, dst, comm, send_reqs[reduce_idx]);
396  double send_tot = get_time() - send_start;
397  ar_send_time += send_tot;
398  ar_rs_send_time += send_tot;
399  }
400  // Complete the receives (in any order).
401  // We need to extract the raw MPI_Request because Elemental does not support
402  // MPI_Waitany.
403  std::vector<MPI_Request> raw_reqs(reduces_this_step);
404  for (int i = 0; i < reduces_this_step; ++i) {
405  raw_reqs[i] = recv_reqs[i].backend;
406  }
407  for (int i = 0; i < reduces_this_step; ++i) {
408  int completed_idx;
409  double recv_start = get_time();
410  MPI_Waitany(reduces_this_step, raw_reqs.data(), &completed_idx,
411  MPI_STATUS_IGNORE);
412  double recv_tot = get_time() - recv_start;
413  ar_recv_time += recv_tot;
414  ar_rs_recv_time += recv_tot;
415  double recv_apply_trans_start = get_time();
416  int recv_size = recv_apply_transform(
417  recv_buffers[completed_idx], accum_view, local_recvs[completed_idx]);
418  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
419  bytes_received += recv_size;
420  ar_bytes_received += recv_size;
421  ar_rs_bytes_received += recv_size;
422  }
423  // Complete all the sends.
424  double send_start = get_time();
425  mpi::WaitAll(reduces_this_step, send_reqs.data(), MPI_STATUSES_IGNORE);
426  double send_tot = get_time() - send_start;
427  ar_send_time += send_tot;
428  ar_rs_send_time += send_tot;
429  }
430  uint8_t *recv_buf = max_recv_buffers[0]; // Get a regular recv buffer.
431  ar_rs_time += get_time() - rs_start;
432  // Do a ring allgather.
433  double ag_start = get_time();
434  const int src = (rank - 1 + nprocs) % nprocs;
435  const int dst = (rank + 1) % nprocs;
436  // Apply the transform to our locally-accumulated slice of the data.
437  // Since the same data is cycled to every process, we do not do the
438  // no_local_trans here.
439  int send_size;
440  // Do the first step where we forward our local data.
441  {
442  double send_trans_start = get_time();
443  uint8_t *send_buf = send_transform(
444  mat, ALL, IR(slice_ends[rank] - slice_lengths[rank], slice_ends[rank]),
445  send_size, false, 0);
446  ar_send_transform_time += get_time() - send_trans_start;
447  const int data_src = (rank - 1 + nprocs) % nprocs;
448  bytes_sent += send_size;
449  ar_bytes_sent += send_size;
450  ar_ag_bytes_sent += send_size;
451  auto recv_view = mat(ALL,
452  IR(slice_ends[data_src] - slice_lengths[data_src],
453  slice_ends[data_src]));
454  // If we can, receive directly into the destination matrix.
455  if (opts.id_recv) {
456  recv_buf = (uint8_t *) recv_view.Buffer();
457  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
458  }
459  double sendrecv_start = get_time();
460  mpi::SendRecv(send_buf, send_size, dst,
461  recv_buf, max_recv_count, src, comm);
462  double sendrecv_tot = get_time() - sendrecv_start;
463  ar_send_time += sendrecv_tot;
464  ar_recv_time += sendrecv_tot;
465  ar_ag_send_time += sendrecv_tot;
466  ar_ag_recv_time += sendrecv_tot;
467  double recv_trans_start = get_time();
468  int recv_size = 0;
469  if (opts.id_recv) {
470  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
471  } else {
472  recv_size = recv_transform(recv_buf, recv_view);
473  }
474  ar_recv_transform_time += get_time() - recv_trans_start;
475  bytes_received += recv_size;
476  ar_bytes_received += recv_size;
477  ar_ag_bytes_received += send_size;
478  send_size = recv_size;
479  }
480  // Now do the remaining nprocs - 2 steps.
481  // We always send from recv_buf and receive to recv_buf2, swapping
482  // pointers to avoid copying.
483  uint8_t *recv_buf2 = nullptr;
484  if (!opts.id_recv) {
485  recv_buf2 = get_collective_buffer(max_recv_count, 1);
486  }
487  for (int step = 1; step < nprocs - 1; ++step) {
488  // Compute where the data we get is coming from.
489  const int data_src = (rank - step - 1 + nprocs) % nprocs;
490  auto recv_view = mat(ALL,
491  IR(slice_ends[data_src] - slice_lengths[data_src],
492  slice_ends[data_src]));
493  if (opts.id_recv) {
494  recv_buf2 = (uint8_t *) recv_view.Buffer();
495  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
496  }
497  bytes_sent += send_size;
498  ar_bytes_sent += send_size;
499  ar_ag_bytes_sent += send_size;
500  double sendrecv_start = get_time();
501  mpi::SendRecv(recv_buf, send_size, dst,
502  recv_buf2, max_recv_count, src, comm);
503  double sendrecv_tot = get_time() - sendrecv_start;
504  ar_send_time += sendrecv_tot;
505  ar_recv_time += sendrecv_tot;
506  ar_ag_send_time += sendrecv_tot;
507  ar_ag_recv_time += sendrecv_tot;
508  double recv_trans_start = get_time();
509  int recv_size = 0;
510  if (opts.id_recv) {
511  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
512  } else {
513  recv_size = recv_transform(recv_buf2, recv_view);
514  }
515  ar_recv_transform_time += get_time() - recv_trans_start;
516  bytes_received += recv_size;
517  // Swap the send and receive buffers.
518  std::swap(recv_buf, recv_buf2);
519  send_size = recv_size;
520  ar_bytes_received += recv_size;
521  ar_ag_bytes_received += send_size;
522  }

527 void lbann::lbann_comm::ring_allreduce(
528  mpi::Comm comm, Mat& mat, int max_recv_count,
529  std::function<uint8_t *(Mat&, IR, IR, int&, bool, int)> send_transform,
530  std::function<int(uint8_t *, Mat&)> recv_transform,
531  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
532  const lbann::lbann_comm::allreduce_options opts) {
533  double ar_start = get_time();
534  const int rank = mpi::Rank(comm);
535  const int nprocs = mpi::Size(comm);
536  if (nprocs == 1) {
537  return; // Nothing to do.
538  }
539  // Compute the number of columns each processor sends.
540  const Int cols_per_proc = mat.Width() / nprocs;
541  const Int cols_remainder = mat.Width() % nprocs;
542  // Compute the lengths/ends of the slices.
543  std::vector<Int> slice_lengths(nprocs, cols_per_proc);
544  for (int i = 0; i < cols_remainder; ++i) {
545  slice_lengths[i] += 1;
546  }
547  std::vector<Int> slice_ends(nprocs);
548  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
549  slice_ends.begin());
550  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
551  uint8_t *recv_buf = max_recv_buf;
552  // Compute source/destination in the ring.
553  const int src = (rank - 1 + nprocs) % nprocs;
554  const int dst = (rank + 1) % nprocs;
555  const bool is_send_local = opts.no_local_trans &&
556  is_rank_node_local(dst, comm);
557  const bool is_recv_local = opts.no_local_trans &&
558  is_rank_node_local(src, comm);
559  // Do a ring-based reduce-scatter.
560  // This is like the pairwise-exchange reduce-scatter except instead of
561  // rank i accumulating only slice i, the slices are cycled around and
562  // each node accumulates its portion into the slice when it passes
563  // through. After the nprocs-1 steps slice k will be on rank
564  // (k + nprocs - 1) % nprocs.
565  double rs_start = get_time();
566  for (int step = 0; step < nprocs - 1; ++step) {
567  // Compute the slices to send/recv.
568  const int send_slice = (rank - step + nprocs) % nprocs;
569  const int recv_slice = (rank - step - 1 + nprocs) % nprocs;
570  // Transform the data to send.
571  double send_trans_start = get_time();
572  int send_size;
573  int recv_size = max_recv_count;
574  uint8_t *send_buf = nullptr;
575  if (is_send_local) {
576  auto send_view = mat(
577  ALL, IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
578  send_buf = (uint8_t *) send_view.Buffer();
579  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
580  } else {
581  send_buf = send_transform(
582  mat, ALL, IR(slice_ends[send_slice] - slice_lengths[send_slice],
583  slice_ends[send_slice]), send_size, false, 0);
584  }
585  auto recv_view = mat(
586  ALL, IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
587  slice_ends[recv_slice]));
588  if (is_recv_local) {
589  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
590  recv_buf = get_collective_buffer(recv_size);
591  } else {
592  recv_buf = max_recv_buf;
593  }
594  ar_send_transform_time += get_time() - send_trans_start;
595  bytes_sent += send_size;
596  ar_bytes_sent += send_size;
597  ar_rs_bytes_sent += send_size;
598  double sendrecv_start = get_time();
599  mpi::SendRecv(send_buf, send_size, dst,
600  recv_buf, recv_size, src, comm);
601  double sendrecv_tot = get_time() - sendrecv_start;
602  ar_send_time += sendrecv_tot;
603  ar_recv_time += sendrecv_tot;
604  ar_rs_send_time += sendrecv_tot;
605  ar_rs_recv_time += sendrecv_tot;
606  double recv_apply_trans_start = get_time();
607  recv_size = recv_apply_transform(recv_buf, recv_view, is_recv_local);
608  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
609  bytes_received += recv_size;
610  ar_bytes_received += recv_size;
611  ar_rs_bytes_received += recv_size;
612  }
613  recv_buf = max_recv_buf; // Ensure we're back to the original.
614  ar_rs_time += get_time() - rs_start;
615  // Do a ring allgather, first applying the transform to local data.
616  double ag_start = get_time();
617  int send_size;
618  {
619  const int send_slice = (rank + 1) % nprocs;
620  const int recv_slice = rank;
621  double send_trans_start = get_time();
622  uint8_t *send_buf = send_transform(
623  mat, ALL, IR(slice_ends[send_slice] - slice_lengths[send_slice],
624  slice_ends[send_slice]), send_size, false, 0);
625  ar_send_transform_time += get_time() - send_trans_start;
626  bytes_sent += send_size;
627  ar_bytes_sent += send_size;
628  ar_ag_bytes_sent += send_size;
629  auto recv_view = mat(ALL,
630  IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
631  slice_ends[recv_slice]));
632  // If we can, receive directly into the destination matrix.
633  if (opts.id_recv) {
634  recv_buf = (uint8_t *) recv_view.Buffer();
635  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
636  }
637  double sendrecv_start = get_time();
638  mpi::SendRecv(send_buf, send_size, dst,
639  recv_buf, max_recv_count, src, comm);
640  double sendrecv_tot = get_time() - sendrecv_start;
641  ar_send_time += sendrecv_tot;
642  ar_recv_time += sendrecv_tot;
643  ar_ag_send_time += sendrecv_tot;
644  ar_ag_recv_time += sendrecv_tot;
645  double recv_trans_start = get_time();
646  int recv_size = 0;
647  if (opts.id_recv) {
648  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
649  } else {
650  recv_size = recv_transform(recv_buf, recv_view);
651  }
652  ar_recv_transform_time += get_time() - recv_trans_start;
653  send_size = recv_size;
654  bytes_received += recv_size;
655  ar_bytes_received += recv_size;
656  ar_ag_bytes_received += recv_size;
657  }
658  uint8_t *recv_buf2 = nullptr;
659  if (!opts.id_recv) {
660  recv_buf2 = get_collective_buffer(max_recv_count, 1);
661  }
662  for (int step = 1; step < nprocs - 1; ++step) {
663  const int recv_slice = (rank - step + nprocs) % nprocs;
664  auto recv_view = mat(ALL,
665  IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
666  slice_ends[recv_slice]));
667  if (opts.id_recv) {
668  recv_buf2 = (uint8_t *) recv_view.Buffer();
669  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
670  }
671  bytes_sent += send_size;
672  ar_bytes_sent += send_size;
673  ar_ag_bytes_sent += send_size;
674  double sendrecv_start = get_time();
675  mpi::SendRecv(recv_buf, send_size, dst,
676  recv_buf2, max_recv_count, src, comm);
677  double sendrecv_tot = get_time() - sendrecv_start;
678  ar_send_time += sendrecv_tot;
679  ar_recv_time += sendrecv_tot;
680  ar_ag_send_time += sendrecv_tot;
681  ar_ag_recv_time += sendrecv_tot;
682  double recv_trans_start = get_time();
683  int recv_size = 0;
684  if (opts.id_recv) {
685  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
686  } else {
687  recv_size = recv_transform(recv_buf2, recv_view);
688  }
689  ar_recv_transform_time += get_time() - recv_trans_start;
690  // Swap the send and receive buffers.
691  std::swap(recv_buf, recv_buf2);
692  send_size = recv_size;
693  bytes_received += recv_size;
694  ar_bytes_received += recv_size;
695  ar_ag_bytes_received += recv_size;
696  }
697  ar_ag_time += get_time() - ag_start;
698  ar_time += get_time() - ar_start;
699 }

701 void lbann::lbann_comm::rabenseifner_allreduce(
702  mpi::Comm comm, Mat& mat, int max_recv_count,
703  std::function<uint8_t *(Mat&, IR, IR, int&, bool, int)> send_transform,
704  std::function<int(uint8_t *, Mat&)> recv_transform,
705  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
706  const lbann::lbann_comm::allreduce_options opts) {
707  double ar_start = get_time();
708  const int rank = mpi::Rank(comm);
709  const unsigned int nprocs = mpi::Size(comm);
710  if (nprocs == 1) {
711  return; // Nothing to do.
712  }
713  // This implementation requires a power-of-2 number of processes.
714  if (nprocs & (nprocs - 1)) {
715  throw lbann_exception("lbann_comm: Rabenseifner allreduce requires"
716  " a power-of-2 number of participating processes");
717  }
718  // Compute the slices on each processor.
719  const Int cols_per_proc = mat.Width() / nprocs;
720  const Int cols_remainder = mat.Width() % nprocs;
721  // Compute the lengths/ends of the slices.
722  std::vector<Int> slice_lengths(nprocs, cols_per_proc);
723  for (int i = 0; i < cols_remainder; ++i) {
724  slice_lengths[i] += 1;
725  }
726  std::vector<Int> slice_ends(nprocs);
727  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
728  slice_ends.begin());
729  // Do a recursive-halving reduce-scatter.
730  // In each step here a process sends all the data needed for the other
731  // "half" of the processes. i.e. each process sends half their data in the
732  // first step, a quarter in the second step, etc.
733  double rs_start = get_time();
734  unsigned int partner_mask = nprocs >> 1;
735  unsigned int slice_mask = 1;
736  unsigned int send_idx = 0;
737  unsigned int recv_idx = 0;
738  unsigned int last_idx = nprocs;
739  uint8_t *recv_buf = get_collective_buffer(max_recv_count);
740  while (partner_mask > 0) {
741  int partner = rank ^ partner_mask; // The rank we exchange with this step.
742  const bool is_local = opts.no_local_trans &&
743  is_rank_node_local(partner, comm);
744  // Determine the range of data to send/recv.
745  IR send_range, recv_range;
746  if (rank < partner) {
747  send_idx = recv_idx + nprocs / (slice_mask*2);
748  send_range = IR(slice_ends[send_idx] - slice_lengths[send_idx],
749  slice_ends[last_idx-1]);
750  recv_range = IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
751  slice_ends[send_idx-1]);
752  } else {
753  recv_idx = send_idx + nprocs / (slice_mask*2);
754  send_range = IR(slice_ends[send_idx] - slice_lengths[send_idx],
755  slice_ends[recv_idx-1]);
756  recv_range = IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
757  slice_ends[last_idx-1]);
758  }
759  auto recv_view = mat(ALL, recv_range);
760  // Transform the data to send.
761  double send_trans_start = get_time();
762  int send_size;
763  int recv_size = max_recv_count;
764  uint8_t *send_buf = nullptr;
765  if (is_local) {
766  auto send_view = mat(ALL, send_range);
767  send_buf = (uint8_t *) send_view.Buffer();
768  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
769  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
770  } else {
771  send_buf = send_transform(mat, ALL, send_range, send_size, false, 0);
772  }
773  ar_send_transform_time += get_time() - send_trans_start;
774  bytes_sent += send_size;
775  ar_bytes_sent += send_size;
776  ar_rs_bytes_sent += send_size;
777  double sendrecv_start = get_time();
778  mpi::SendRecv(send_buf, send_size, partner,
779  recv_buf, recv_size, partner, comm);
780  double sendrecv_tot = get_time() - sendrecv_start;
781  ar_send_time += sendrecv_tot;
782  ar_recv_time += sendrecv_tot;
783  ar_rs_send_time += sendrecv_tot;
784  ar_rs_recv_time += sendrecv_tot;
785  // Transform the received data.
786  double recv_apply_trans_start = get_time();
787  recv_size = recv_apply_transform(recv_buf, recv_view, is_local);
788  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
789  bytes_received += recv_size;
790  ar_bytes_received += recv_size;
791  ar_rs_bytes_received += send_size;
792  // Update info for next iteration.
793  // Except last_idx when needed for the allgather.
794  send_idx = recv_idx;
795  partner_mask >>= 1;
796  slice_mask <<= 1;
797  if (partner_mask > 0) {
798  last_idx = recv_idx + nprocs / (slice_mask);
799  }
800  }
801  ar_rs_time += get_time() - rs_start;
802  // Do a recursive-doubling algather.
803  double ag_start = get_time();
804  slice_mask >>= 1;
805  partner_mask = 1;
806  // Now do the remaining steps.
807  while (partner_mask < nprocs) {
808  int partner = rank ^ partner_mask;
809  const bool is_local = opts.no_local_trans &&
810  is_rank_node_local(partner, comm);
811  // Determine range to send/recv.
812  IR send_range, recv_range;
813  if (rank < partner) {
814  if (slice_mask != nprocs / 2) {
815  last_idx = last_idx + nprocs / (slice_mask*2);
816  }
817  recv_idx = send_idx + nprocs / (slice_mask*2);
818  send_range = IR(slice_ends[send_idx] - slice_lengths[send_idx],
819  slice_ends[recv_idx-1]);
820  recv_range = IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
821  slice_ends[last_idx-1]);
822  } else {
823  recv_idx = send_idx - nprocs / (slice_mask*2);
824  send_range = IR(slice_ends[send_idx] - slice_lengths[send_idx],
825  slice_ends[last_idx-1]);
826  recv_range = IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
827  slice_ends[send_idx-1]);
828  }
829  auto recv_view = mat(ALL, recv_range);
830  // Transform the data to send.
831  double send_trans_start = get_time();
832  int send_size;
833  int recv_size = max_recv_count;
834  uint8_t *send_buf = nullptr;
835  if (is_local) {
836  auto send_view = mat(ALL, send_range);
837  send_buf = (uint8_t *) send_view.Buffer();
838  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
839  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
840  } else {
841  send_buf = send_transform(mat, ALL, send_range, send_size, false, 0);
842  }
843  ar_send_transform_time += get_time() - send_trans_start;
844  if (opts.id_recv || is_local) {
845  recv_buf = (uint8_t *) recv_view.Buffer();
846  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
847  }
848  bytes_sent += send_size;
849  ar_bytes_sent += send_size;
850  ar_ag_bytes_sent += send_size;
851  double sendrecv_start = get_time();
852  mpi::SendRecv(send_buf, send_size, partner,
853  recv_buf, recv_size, partner, comm);
854  double sendrecv_tot = get_time() - sendrecv_start;
855  ar_send_time += sendrecv_tot;
856  ar_recv_time += sendrecv_tot;
857  ar_ag_send_time += sendrecv_tot;
858  ar_ag_recv_time += sendrecv_tot;
859  double recv_trans_start = get_time();
860  if (opts.id_recv) {
861  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
862  } else {
863  recv_size = recv_transform(recv_buf, recv_view);
864  }
865  ar_recv_transform_time += get_time() - recv_trans_start;
866  bytes_received += recv_size;
867  ar_bytes_received += recv_size;
868  ar_ag_bytes_received += send_size;
869  // Update for the next iteration.
870  if (rank > partner) {
871  send_idx = recv_idx;
872  }
873  partner_mask <<= 1;
874  slice_mask >>= 1;
875  }
876  ar_ag_time += get_time() - ag_start;
877  ar_time += get_time() - ar_start;
878 }
