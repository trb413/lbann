66 void greedy_layerwise_autoencoder::setup_layer_topology() {
67  sequential_model::setup_layer_topology();
68 
69  // Divide model into sections
70  // Note: first half are encoder sections and second half are decoder
71  // sections
72  for (size_t i = 1; i < m_layers.size() - 1; ++i) {
73  if (dynamic_cast<learning_layer*>(m_layers[i]) != nullptr) {
74  m_sections.push_back(i);
75  }
76  }
77  m_sections.push_back(m_layers.size() - 1);
78  const int num_sections = m_sections.size() - 1;
79  m_num_phases = (num_sections + 1) / 2;
80 
81  // Check that section input and output dimensions are valid
82  for (int i = 0; i < m_num_phases; ++i) {
83 
84  // Beginning and end of encoder and decoder
85  const int encoder_start = m_sections[i];
86  const int encoder_end = m_sections[i+1];
87  const int decoder_start = m_sections[num_sections-i-1];
88  const int decoder_end = m_sections[num_sections-i];
89 
90  // Encoder input and decoder output should match
91  std::vector<int> input_dims, output_dims;
92  input_dims = m_layers[encoder_start]->get_prev_neuron_dims();
93  output_dims = m_layers[decoder_end-1]->get_neuron_dims();
94  if (input_dims != output_dims) {
95  std::stringstream err;
96  err << __FILE__ << " " << __LINE__ << " :: "
97  << "input dimensions of layer " << m_layers[encoder_start]->get_name()
98  << " (" << input_dims[0];
99  for (size_t j = 1; j < input_dims.size(); ++i) {
100  err << "x" << input_dims[j];
101  }
102  err << ") does not match "
103  << "output dimensions of layer " << m_layers[decoder_end-1]->get_name()
104  << " (" << output_dims[0];
105  for (size_t j = 1; j < output_dims.size(); ++i) {
106  err << "x" << output_dims[j];
107  }
108  err << ")";
109  throw lbann_exception(err.str());
110  }
111 
112  // Encoder output and decoder input should match
113  output_dims = m_layers[encoder_end-1]->get_neuron_dims();
114  input_dims = m_layers[decoder_start]->get_prev_neuron_dims();
115  if (input_dims != output_dims) {
116  std::stringstream err;
117  err << __FILE__ << " " << __LINE__ << " :: "
118  << "output dimensions of layer " << m_layers[encoder_end-1]->get_name()
119  << " (" << output_dims[0];
120  for (size_t j = 1; j < output_dims.size(); ++i) {
121  err << "x" << output_dims[j];
122  }
123  err << ") does not match "
124  << "input dimensions of layer " << m_layers[decoder_start]->get_name()
125  << " (" << input_dims[0];
126  for (size_t j = 1; j < input_dims.size(); ++i) {
127  err << "x" << input_dims[j];
128  }

137 void greedy_layerwise_autoencoder::train(int num_epochs) {
138  do_train_begin_cbs();
139 
140  // Train each autoencoder phase on several epochs
141  for (int phase = 0; phase < m_num_phases; ++phase) {
142  set_phase(phase);
143  for (int epoch = 0; epoch < num_epochs / m_num_phases; ++epoch) {
144  if (get_terminate_training()) { goto train_end; }
145  reset_mode_and_model(execution_mode::training);
146  do_epoch_begin_cbs();
147  while (!train_mini_batch()) {}
148  evaluate(execution_mode::validation);
149  m_current_epoch++;
150  do_epoch_end_cbs();
151  reset_epoch_statistics(execution_mode::training);
152  }
153  }
154 
155  train_end:
156  restore_sequential_model();
157  do_train_end_cbs();
158 }

160 void greedy_layerwise_autoencoder::set_phase(int phase) {
161 
162  // Restore sequential model first
163  restore_sequential_model();
164  m_phase = phase;
165  if (phase < 0) {
166  return;
167  }
168 
169  // Determine encoder and decoder being trained
170  const int num_sections = m_sections.size() - 1;
171  const int encoder_start = m_sections[phase];
172  const int encoder_end = m_sections[phase+1];
173  const int decoder_start = m_sections[num_sections-phase-1];
174  const int decoder_end = m_sections[num_sections-phase];
175  auto& encoder_parents = m_layers[encoder_start]->get_parent_layers();
176  auto& encoder_children = m_layers[encoder_end-1]->get_child_layers();
177  auto& decoder_parents = m_layers[decoder_start]->get_parent_layers();
178  auto& decoder_children = m_layers[decoder_end-1]->get_child_layers();
179 
180  // Initialize reconstruction layer
181  if (m_reconstruction != nullptr) delete m_reconstruction;
182  Layer* original_layer = m_layers[encoder_start-1];
183  switch (encoder_parents[0]->get_data_layout()) {
184  case data_layout::MODEL_PARALLEL:
185  m_reconstruction = new reconstruction_layer<data_layout::MODEL_PARALLEL>(m_comm, original_layer);
186  break;
187  case data_layout::DATA_PARALLEL:
188  m_reconstruction = new reconstruction_layer<data_layout::DATA_PARALLEL>(m_comm, original_layer);
189  break;
190  default:
191  std::stringstream err;
192  err << __FILE__ << " " << __LINE__ << " :: "
193  << "invalid data layout for reconstruction layer";
194  throw lbann_exception(err.str());
195  }
196  m_reconstruction->set_name("reconstruction_phase" + std::to_string(phase));
197 
198  // Setup layer pointers to train encoder and decoder
199  encoder_children[0] = m_layers[decoder_start];
200  decoder_parents[0] = m_layers[encoder_end-1];
201  decoder_children[0] = m_reconstruction;
202  m_reconstruction->add_parent_layer(m_layers[decoder_end-1]);
203 
204  // Set objective function to reconstruction layer
205  for (auto term : m_objective_function->get_terms()) {
206  auto* loss = dynamic_cast<loss_function*>(term);
207  if (loss != nullptr) {
208  loss->set_target_layer((generic_target_layer*) m_reconstruction);
209  }
210  }

214 void greedy_layerwise_autoencoder::restore_sequential_model() {
215 
216  // Return if model is already sequential
217  if (m_phase < 0) {
218  return;
219  }
220  m_phase = -1;
221 
222  // Restore sequential layer order
223  for (int m_section : m_sections) {
224  Layer* prev_layer = m_layers[m_section-1];
225  Layer* next_layer = m_layers[m_section];
226  auto& prev_layer_children = prev_layer->get_child_layers();
227  auto& next_layer_parents = next_layer->get_parent_layers();
228  prev_layer_children[0] = next_layer;
229  next_layer_parents[0] = prev_layer;
230  }
231 
232  // Delete reconstruction layer
233  if (m_reconstruction != nullptr) delete m_reconstruction;
234  m_reconstruction = nullptr;
235 
236  // Restore objective function to target layer
237  for (auto term : m_objective_function->get_terms()) {
238  auto* loss = dynamic_cast<loss_function*>(term);
239  if (loss != nullptr) {
240  loss->set_target_layer((generic_target_layer*) m_layers.back());
241  }
242  }

246 void greedy_layerwise_autoencoder::forward_prop(execution_mode mode) {
247 
248  // Use base implementation if model is sequential
249  if (m_phase < 0) {
250  sequential_model::forward_prop(mode);
251  return;
252  }
253  do_model_forward_prop_begin_cbs(mode);
254 
255  // Determine encoder and decoder to train
256  const int num_sections = m_sections.size() - 1;
257  const int encoder_start = m_sections[m_phase];
258  const int encoder_end = m_sections[m_phase+1];
259  const int decoder_start = m_sections[num_sections-m_phase-1];
260  const int decoder_end = m_sections[num_sections-m_phase];
261 
262  // Forward prop on layers that are already trained
263  for (int i = 0; i < encoder_start; ++i) {
264  Layer* layer = m_layers[i];
265  do_layer_forward_prop_begin_cbs(mode, layer);
266  layer->forward_prop();
267  do_layer_forward_prop_end_cbs(mode, layer);
268  }
269 
270  // Forward prop on encoder being trained
271  for (int i = encoder_start; i < encoder_end; ++i) {
272  Layer* layer = m_layers[i];
273  do_layer_forward_prop_begin_cbs(mode, layer);
274  layer->forward_prop();
275  do_layer_forward_prop_end_cbs(mode, layer);
276  }
277 
278  // Forward prop on decoder being trained
279  for (int i = decoder_start; i < decoder_end; ++i) {
280  Layer* layer = m_layers[i];
281  do_layer_forward_prop_begin_cbs(mode, layer);
282  layer->forward_prop();
283  do_layer_forward_prop_end_cbs(mode, layer);
284  }
285 
286  // Forward prop on reconstruction layer
287  do_layer_forward_prop_begin_cbs(mode, m_reconstruction);
288  m_reconstruction->forward_prop();
289  do_layer_forward_prop_end_cbs(mode, m_reconstruction);
290 
291  do_model_forward_prop_end_cbs(mode);
292 }

295

298  sequential_model::backward_prop();

305  const int encoder_start = m_sections[m_phase];
306  const int encoder_end = m_sections[m_phase+1];
307  const int decoder_start = m_sections[num_sections-m_phase-1];
308  const int decoder_end = m_sections[num_sections-m_phase];
309 
310  // Backward prop on reconstruction layer
311  do_layer_backward_prop_begin_cbs(m_reconstruction);
312  m_reconstruction->back_prop();
313  do_layer_backward_prop_end_cbs(m_reconstruction);
314 
315  // Backward prop on decoder being trained
316  for (int i = decoder_end; i < decoder_start; --i) {
317  Layer* layer = m_layers[i];
318  do_layer_backward_prop_begin_cbs(layer);
319  layer->back_prop();
320  do_layer_backward_prop_end_cbs(layer);
321  }
322 
323  // Backward prop on encoder being trained
324  for (int i = encoder_end; i >= encoder_start; --i) {
325  Layer* layer = m_layers[i];
326  do_layer_backward_prop_begin_cbs(layer);
327  layer->back_prop();
328  do_layer_backward_prop_end_cbs(layer);
329  }
330 
331  do_model_backward_prop_end_cbs();
332 }
