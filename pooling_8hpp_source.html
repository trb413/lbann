125  void setup_dims() {
126 
127  // Initialize previous neuron tensor dimensions
128  transform::setup_dims();
129 
130  // Initialize neuron tensor dimensions
131  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
132  const int effective_dim = (this->m_prev_neuron_dims[i+1]
133  + 2*m_pool_pads[i] - m_pool_dims[i] + 1);

260  void fp_compute_im2col() {
261 
262  // Throw exception if pooling mode is not max or average pooling
263  if(m_pool_mode != pool_mode::max
264  && m_pool_mode != pool_mode::average) {
265  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
266  }
267 
268  // Get local matrices
269  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
270  Mat& activations_local = this->m_activations_v->Matrix();
271 
272  // Output entries are divided amongst channels
273  const int num_channels = this->m_prev_neuron_dims[0];
274  const int num_per_output_channel = this->m_num_neurons / num_channels;
275 
276  // Initialize im2col matrix
277  Mat im2col_mat(m_pool_size * num_channels, num_per_output_channel);
278 
279  // Iterate through data samples
280  for(int sample = 0; sample < prev_activations_local.Width(); ++sample) {
281 
282  // Construct im2col matrix from input
283  const Mat input_mat = El::LockedView(prev_activations_local, El::ALL, El::IR(sample));
284  im2col(input_mat,
285  im2col_mat,
286  num_channels,
287  this->m_num_prev_neuron_dims - 1,
288  this->m_prev_neuron_dims.data() + 1,
289  m_pool_pads.data(),
290  m_pool_dims.data(),
291  m_pool_strides.data());
292 
293  // Apply max pooling
294  if(m_pool_mode == pool_mode::max) {
295  DataType *output_buffer = activations_local.Buffer(0, sample);
296  #pragma omp parallel for collapse(2)
297  for(int c = 0; c < num_channels; ++c) {
298  for(int j = 0; j < num_per_output_channel; ++j) {
299  DataType *im2col_buffer = im2col_mat.Buffer(c*m_pool_size, j);
300  DataType output_entry = -INFINITY;
301  for(int i = 0; i < m_pool_size; ++i) {
302  output_entry = std::max(output_entry, im2col_buffer[i]);
303  }
304  const int output_index = j + c * num_per_output_channel;
305  output_buffer[output_index] = output_entry;
306  }
307  }
308  }
309 
310  // Apply average pooling
311  if(m_pool_mode == pool_mode::average) {
312  DataType *output_buffer = activations_local.Buffer(0, sample);
313  #pragma omp parallel for collapse(2)
314  for(int c = 0; c < num_channels; ++c) {
315  for(int j = 0; j < num_per_output_channel; ++j) {
316  DataType *im2col_buffer = im2col_mat.Buffer(c*m_pool_size, j);
317  DataType output_entry = 0;
318  for(int i = 0; i < m_pool_size; ++i) {
319  output_entry += im2col_buffer[i];
320  }
321  output_entry /= m_pool_size;
322  const int output_index = j + c * num_per_output_channel;
323  output_buffer[output_index] = output_entry;
324  }
325  }
326  }
327 
328  }

333  void bp_compute_im2col() {
334 
335  // Throw exception if pooling mode is not max or average pooling
336  if(m_pool_mode != pool_mode::max
337  && m_pool_mode != pool_mode::average) {
338  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
339  }
340 
341  // Get local matrices
342  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
343  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->LockedMatrix();
344  Mat& error_signal_local = this->m_error_signal_v->Matrix();
345 
346  // Output entries are divided amongst channels
347  const int num_channels = this->m_prev_neuron_dims[0];
348  const int num_per_output_channel = this->m_num_neurons / num_channels;
349 
350  // Initialize im2col matrix
351  Mat im2col_mat(m_pool_size * num_channels, num_per_output_channel);
352 
353  // Iterate through data samples
354  for(int sample = 0; sample < prev_activations_local.Width(); ++sample) {
355 
356  // Compute gradient w.r.t. im2col matrix for max pooling
357  if(m_pool_mode == pool_mode::max) {
358 
359  // Construct im2col matrix from input
360  const Mat input_mat = El::LockedView(prev_activations_local, El::ALL, El::IR(sample));
361  im2col(input_mat,
362  im2col_mat,
363  num_channels,
364  this->m_num_prev_neuron_dims - 1,
365  this->m_prev_neuron_dims.data() + 1,
366  m_pool_pads.data(),
367  m_pool_dims.data(),
368  m_pool_strides.data());
369 
370  // Copy previous error signal to im2col matrix entries
371  // corresponding to max
372  const DataType *prev_error_signal_buffer
373  = prev_error_signal_local.LockedBuffer(0, sample);
374  #pragma omp parallel for collapse(2)
375  for(int j = 0; j < num_per_output_channel; ++j) {
376  for(int c = 0; c < num_channels; ++c) {
377  DataType *im2col_buffer = im2col_mat.Buffer(c*m_pool_size, j);
378  int max_index = 0;
379  DataType max_entry = -INFINITY;
380  for(int i = 0; i < m_pool_size; ++i) {
381  const DataType current_entry = im2col_buffer[i];
382  im2col_buffer[i] = 0;
383  if(current_entry > max_entry) {
384  max_index = i;
385  max_entry = current_entry;
386  }
387  }
388  const int prev_error_signal_index = j + c * num_per_output_channel;
389  im2col_buffer[max_index]
390  = prev_error_signal_buffer[prev_error_signal_index];
391  }
392  }
393 
394  }
395 
396  // Compute gradient w.r.t. im2col matrix for average pooling
397  if(m_pool_mode == pool_mode::average) {
398  #pragma omp parallel for collapse(2)
399  for(int j = 0; j < num_per_output_channel; ++j) {
400  for(int c = 0; c < num_channels; ++c) {
401  DataType *im2col_buffer = im2col_mat.Buffer(c*m_pool_size, j);
402  const int input_index = j + c * num_per_output_channel;
403  const DataType output_entry
404  = prev_error_signal_local(input_index, sample) / m_pool_size;
405  for(int i = 0; i < m_pool_size; ++i) {
406  im2col_buffer[i] = output_entry;
407  }
408  }
409  }
410 
411  }
412 
413  // Compute error signal (i.e. gradient w.r.t. input)
414  Mat output_mat = El::View(error_signal_local, El::ALL, El::IR(sample));
415  col2im(im2col_mat,
416  output_mat,
417  num_channels,
418  this->m_num_prev_neuron_dims - 1,
419  this->m_prev_neuron_dims.data() + 1,
420  m_pool_pads.data(),
421  m_pool_dims.data(),
422  m_pool_strides.data());
423 
424  }
425 
426  }
