124 const AbsDistMat& optimizer::get_gradient() {
125 
126  // Check if gradient is initialized
127  if (!is_initialized()) {
128  LBANN_ERROR("attempted to access gradients before they are set up");
129  }
130 
131  // Perform allreduce on staging matrix if needed
132  if (m_gradient_allreduce_needed && !m_gradient_allreduce_started) {
133  start_gradient_staging_allreduce();
134  }
135  if (m_gradient_allreduce_started && !m_gradient_allreduce_finished) {
136  m_comm->wait(m_gradient_allreduce_req);
137  m_gradient_allreduce_finished = true;
138  }
139  if (m_gradient_allreduce_needed) {
140  if (m_cudnn == nullptr) {
141  add_to_gradient(*m_gradient_staging);
142  } else {
143  #ifdef LBANN_HAS_CUDNN
144  add_to_gradient(m_gradient_staging_d);
145  #endif // LBANN_HAS_CUDNN
146  }
147  }
148  m_gradient_allreduce_needed = false;
149  m_gradient_allreduce_started = false;
150  m_gradient_allreduce_finished = false;
151 
152  // Return CPU gradient matrix
153  if (m_cudnn != nullptr) {
154  #ifdef LBANN_HAS_CUDNN
155  m_cudnn->copy_from_gpu(0,
156  m_gradient->Matrix(),
157  m_gradient_d.get_locked_data(0),
158  m_gradient_d.get_leading_dim());
159  m_cudnn->synchronize();
160  #endif // LBANN_HAS_CUDNN
161  }
162  return *m_gradient;
163 
164 }

208 void optimizer::start_gradient_staging_allreduce() {
209  if (!m_gradient_allreduce_needed || m_gradient_allreduce_started) {
210  return;
211  }
212 
213  m_gradient_allreduce_started = true;
214  if (m_cudnn == nullptr) {
215  m_comm->nb_allreduce(*m_gradient_staging,
216  m_gradient_staging->RedundantComm(),
217  m_gradient_allreduce_req,
218  El::mpi::SUM,
219  std::type_index(typeid(Al::mpi_backend)));
220  m_gradient_allreduce_finished = false;
221  } else {
222  #ifndef LBANN_HAS_CUDNN
223  LBANN_ERROR("cuDNN not detected");
224  #else
225  #if defined(LBANN_HAS_ALUMINUM) && defined(LBANN_HAS_NCCL2)
226  // Non-blocking GPU allreduce with NCCL
227  // Note: We assume each process has one GPU and that the gradient
228  // is in STAR,STAR distribution.
229  if (m_cudnn->get_num_gpus() != 1) {
230  LBANN_ERROR("non-blocking GPU allreduce with NCCL assumes one GPU per process");
231  }
232  StarMat gradient_staging_d;
233  gradient_staging_d.Attach(m_gradient_staging_d.get_height(),
234  m_gradient_staging_d.get_width_per_gpu(),
235  m_gradient_staging->Grid(),
236  m_gradient_staging->ColAlign(),
237  m_gradient_staging->RowAlign(),
238  m_gradient_staging_d.get_data(0),
239  m_gradient_staging_d.get_leading_dim(),
240  m_gradient_staging->Root());
241  m_cudnn->synchronize();
242  m_comm->nb_allreduce(gradient_staging_d,
243  gradient_staging_d.RedundantComm(),
244  m_gradient_allreduce_req,
245  El::mpi::SUM,
246  std::type_index(typeid(Al::nccl_backend)));
247  m_gradient_allreduce_finished = false;
248  #else
249  // Naive GPU allreduce
250  m_cudnn->global_allreduce_on_gpus(m_gradient_staging_d.get_data(),
251  m_gradient_staging_d.get_height(),
252  m_gradient_staging_d.get_width_per_gpu(),
253  m_gradient->RedundantComm());
254  m_gradient_allreduce_finished = true;
255  #endif // defined(LBANN_HAS_ALUMINUM) && defined(LBANN_HAS_NCCL2)
256  #endif // LBANN_HAS_CUDNN
257  }

279 void optimizer::add_to_gradient(const AbsDistMat& gradient,
280  DataType scale) {
281  if (!is_initialized()) {
282  LBANN_ERROR("attempted to access gradients before they are set up");
283  }
284  if (scale != DataType(0)) {
285  if (m_cudnn == nullptr) {
286  El::Axpy(scale, gradient, *m_gradient);
287  } else {
288  #ifndef LBANN_HAS_CUDNN
289  LBANN_ERROR("cuDNN not detected");
290  #else
291  cudnn::matrix gradient_d(m_cudnn);
292  gradient_d.attach_to_work_spaces(gradient.LocalHeight(),
293  gradient.LocalWidth());
294  m_cudnn->broadcast_to_gpus(gradient_d.get_data(),
295  gradient.LockedMatrix());
296  add_to_gradient(gradient_d, scale);
297  #endif // LBANN_HAS_CUDNN
298  }
299  }
300 }

324  if (!is_initialized()) {
325  LBANN_ERROR("attempted to access gradients before they are set up");
326  }
327  if (m_gradient_allreduce_started) {
328  LBANN_ERROR("attempted to add to staging matrix after gradient accumulation has started");
329  }
330  if (scale != DataType(0)) {
331 
332  // Clear staging matrix if needed
333  if (!m_gradient_allreduce_needed) {
334  if (m_cudnn == nullptr) {
335  El::Zero(*m_gradient_staging);
336  } else {
337  #ifndef LBANN_HAS_CUDNN

341  #endif // LBANN_HAS_CUDNN
342  }
343  }
344  m_gradient_allreduce_needed = true;
345 
346  // Add to staging matrix
347  if (m_cudnn == nullptr) {
348  El::Axpy(scale, gradient, *m_gradient_staging);
349  } else {
350  #ifndef LBANN_HAS_CUDNN
351  LBANN_ERROR("cuDNN not detected");
352  #else
353  cudnn::matrix gradient_d(m_cudnn,
354  gradient.LocalHeight(),
355  gradient.LocalWidth());
356  gradient_d.zero();
357  m_cudnn->copy_to_gpu(0,
358  gradient_d.get_data(0),
359  gradient.LockedMatrix(),

414 void optimizer::setup(weights& w) {
415  if (is_initialized()) {
416  LBANN_ERROR("attempted to setup an optimizer that is already set up");
417  }
418  set_weights(w);
419 
420  // Initialize matrices
421  const int height = m_weights->get_matrix_height();
422  const int width = m_weights->get_matrix_width();
423  const AbsDistMat& values = m_weights->get_values();
424 
425  m_gradient = values.Construct(values.Grid(), values.Root());
426  m_gradient_staging = values.Construct(values.Grid(), values.Root());
427  m_gradient->Resize(height, width);
428  m_gradient_staging->Resize(height, width);
429 
430  // Initialize GPU
431  m_cudnn = m_weights->m_cudnn;
432  if (m_cudnn != nullptr) {
433 #ifdef LBANN_HAS_CUDNN
434  m_gradient_d = cudnn::matrix(m_cudnn, height, width);
435  m_gradient_staging_d = cudnn::matrix(m_cudnn, height, width);
436 #endif // LBANN_HAS_CUDNN
437  }
438 
439  // Initialize with zero gradient
440  clear_gradient();
441

444 void optimizer::step() {
445  if (!is_initialized()) {
446  LBANN_ERROR("optimizer must be set up before performing optimization step");
447  }
448 
449  double step_start = get_time();
450  // Apply optimization step
451  if (m_cudnn != nullptr) {
452  #ifdef LBANN_HAS_CUDNN
453  cudnn::matrix values_d(m_cudnn);
454  values_d.attach(m_weights->m_values_d, m_weights->get_size());
455  const auto& gradient_d = get_gradient_gpu();
456  step_compute_gpu(values_d, gradient_d);
457  #endif // LBANN_HAS_CUDNN
458  } else {
459  m_weights->get_values(); // Move data to CPU
460  auto& values = *m_weights->m_values;
461  const auto& gradient = get_gradient();
462  step_compute(values, gradient);
463  }
464 
465  // Clear gradients
466  clear_gradient();
467 
468  m_step_time += get_time() - step_start;
469 
470 }

474  const cudnn::matrix& gradient_d) {
