60  // Store neuron tensor dimensions
61  const int num_neuron_dims = this->m_num_neuron_dims;
62  const std::vector<int> neuron_dims = this->m_neuron_dims;
63 
64  // Initialize previous neuron tensor dimensions

66 
67  // Initialize neuron tensor dimensions
68  this->m_num_neuron_dims = num_neuron_dims;
69  this->m_neuron_dims = neuron_dims;
70 
71  // Determine any unspecified dimensions
72  int unspecified_dim = -1;
73  for(int dim = 0; dim < this->m_num_neuron_dims; ++dim) {
74  if(this->m_neuron_dims[dim] <= 0) {
75  unspecified_dim = dim;
76  this->m_neuron_dims[dim] = 1;
77  }
78  }
79  if(unspecified_dim >= 0) {
80  const int specified_size = std::accumulate(this->m_neuron_dims.begin(),
81  this->m_neuron_dims.end(),
82  1,
83  std::multiplies<int>());
84  this->m_neuron_dims[unspecified_dim] = this->m_num_neurons / specified_size;
85  }
86 
87  // Check that reshape is valid
88  if(this->m_num_neurons != std::accumulate(this->m_neuron_dims.begin(),
89  this->m_neuron_dims.end(),
90  1,
91  std::multiplies<int>())) {
92  std::string num_neurons = " {";
93  for(int n: m_neuron_dims) num_neurons += " " + std::to_string(n);
94  num_neurons += " }";
95  throw lbann_exception("reshape_layer: invalid neuron dimensions, " +
96  std::to_string(m_num_neurons) + " != " + num_neurons);
97  }
