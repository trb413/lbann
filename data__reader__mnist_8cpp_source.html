51 bool mnist_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
52  int pixelcount = m_image_width * m_image_height;
53  std::vector<unsigned char>& tmp = m_image_data[data_id];
54 
55  for (int p = 0; p < pixelcount; p++) {
56  X.Set(p, mb_idx, tmp[p+1]);
57  }
58 
59  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
60  augment(pixel_col, m_image_height, m_image_width, 1);
61  normalize(pixel_col, 1);
62  pixel_noise(pixel_col); //add noise to image, disable by default
63  return true;
64 }

66 bool mnist_reader::fetch_label(Mat& Y, int data_id, int mb_idx, int tid) {
67  if(!m_gan_labelling) { //default
68  unsigned char label = m_image_data[data_id][0];

72  else { //fake sample (second half of minibatch is set to 0;discriminator model
73  //mb_idx < (m_mb_size/2) ? Y.Set(1,mb_idx,1) : Y.Set(m_gan_label_value,mb_idx,1);
74  mb_idx < (get_current_mini_batch_size()/2) ? Y.Set(1,mb_idx,1) : Y.Set(m_gan_label_value,mb_idx,1);
75  }
76  }
77  return true;
78 }

82 void load_mnist_data(const std::string imagepath, const std::string labelpath,
83  const int m_first_n, std::vector<std::vector<unsigned char> >& m_image_data) {
84 
85  // read labels
86  FILE *fplbl = fopen(labelpath.c_str(), "rb");
87  if (!fplbl) {
88  throw lbann_exception(
89  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
90  " :: load_mnist_data: failed to open file: " + labelpath);
91  }
92 
93  int magicnum1, numitems1;
94  fread(&magicnum1, 4, 1, fplbl);
95  fread(&numitems1, 4, 1, fplbl);
96  __swapEndianInt((unsigned int&)magicnum1);
97  __swapEndianInt((unsigned int&)numitems1);
98 
99  // read images
100  FILE *fpimg = fopen(imagepath.c_str(), "rb");
101  if (!fpimg) {
102  throw lbann_exception(
103  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
104  " :: load_mnist_data: failed to open file: " + imagepath);
105  }
106 
107  int magicnum2, numitems2, imgwidth, imgheight;
108  fread(&magicnum2, 4, 1, fpimg);
109  fread(&numitems2, 4, 1, fpimg);
110  fread(&imgwidth, 4, 1, fpimg);
111  fread(&imgheight, 4, 1, fpimg);
112  __swapEndianInt((unsigned int&)magicnum2);
113  __swapEndianInt((unsigned int&)numitems2);
114  __swapEndianInt((unsigned int&)imgwidth);
115  __swapEndianInt((unsigned int&)imgheight);
116 
117  if (numitems1 != numitems2) {
118  fclose(fplbl);
119  fclose(fpimg);
120  throw lbann_exception(
121  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
122  " :: load_mnist_data: numitems1 != numitems2");
123  }
124 
125  if (m_first_n > 0) {
126  numitems1 = m_first_n > numitems1 ? numitems1 : m_first_n;
127  }
128 
129  // set to array
130  m_image_data.resize(numitems1);
131  for (int n = 0; n < numitems1; n++) {
132  m_image_data[n].resize(1+(imgwidth * imgheight));
133  fread(&m_image_data[n][0], 1, 1, fplbl);
134  fread(&m_image_data[n][1], imgwidth * imgheight, 1, fpimg);
135  }
136  fclose(fpimg);
137  fclose(fplbl);
138 }

140 void mnist_reader::load() {
141  if (is_master()) {

143  }
144  m_image_data.clear();

146  if(m_gan_labelling) m_num_labels=2;
147  std::cout << "MNIST load GAN m_gan_labelling : label_value "
148  << m_gan_labelling <<" : " << m_gan_label_value << std::endl;
149 
150  const std::string FileDir = get_file_dir();
151  const std::string ImageFile = get_data_filename();
152  const std::string LabelFile = get_label_filename();
153 
154  // set filepath
155  const std::string imagepath = FileDir + "/" + ImageFile;
156  const std::string labelpath = FileDir + "/" + LabelFile;
157 
158  if (is_master()) {
159  std::cerr << "read labels!\n";
160  }
161 
162  load_mnist_data(imagepath, labelpath, m_first_n, m_image_data);
163 
164  if (m_first_n > 0) {
165  set_use_percent(1.0);
166  set_absolute_sample_count(0u);
167  }
168 
169  // reset indices
170  m_shuffled_indices.clear();
171  m_shuffled_indices.resize(m_image_data.size());
172  for (size_t n = 0; n < m_shuffled_indices.size(); n++) {
173  m_shuffled_indices[n] = n;
174  }
175  if (is_master()) {
176  std::cerr << "calling select_subset_of_data; m_shuffled_indices.size: " <<
177  m_shuffled_indices.size() << std::endl;
178  }
179  select_subset_of_data();
180 }
