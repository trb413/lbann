39 std::unordered_set<T*> select_from_list(std::string names,
40  std::vector<T*> list) {
41  std::unordered_set<T*> selected;
42  for (const auto& name : parse_list<std::string>(names)) {
43  for (auto&& t : list) {
44  if (name == t->get_name()) {
45  selected.insert(t);
46  }
47  }
48  }
49  return selected;
50 }

107  type = lbann_callback_imcomm::comm_type::THRESH_QUANTIZATION;
108  } else if (type_str == "adaptive_quantization") {
109  type = lbann_callback_imcomm::comm_type::ADAPTIVE_QUANTIZATION;

171  params.max_iter(),
172  selected_weights);
173  }
174

179  const auto& params = proto_cb.step_minibatch();
180  return new lbann_callback_step_minibatch(params.starting_mbsize(),

182  params.ramp_time());

184  if (proto_cb.has_minibatch_schedule()) {
185  const auto& params = proto_cb.minibatch_schedule();
186  std::vector<lbann_callback_minibatch_schedule::minibatch_step> steps;

194  return new lbann_callback_minibatch_schedule(params.starting_mbsize(),

201  if (proto_cb.has_checkpoint()) {
202  const auto& params = proto_cb.checkpoint();
203  return new lbann_callback_checkpoint(params.checkpoint_dir(),
204  params.checkpoint_epochs(),
205  params.checkpoint_steps(),
206  params.checkpoint_secs(),

209  params.ckpt_dist_steps());
210  }
211  if (proto_cb.has_save_model()) {
212  const auto& params = proto_cb.save_model();
213  return new lbann_callback_save_model(params.dir(),
214  params.extension());
215  }
216 
218  // Profiling
220  if (proto_cb.has_summary()) {
221  const auto& params = proto_cb.summary();
222  return new lbann_callback_summary(summarizer,
223  params.batch_interval(),
224  params.mat_interval());
225  }
226  if (proto_cb.has_profiler()) {
227  return new lbann_callback_profiler();
228  }
229 
231  // Debugging
233  if (proto_cb.has_debug()) {
234  const auto& phase = proto_cb.debug().phase();
235  if (phase == "train" || phase == "training") {
236  return new lbann_callback_debug(execution_mode::training, summarizer);
237  } else if (phase == "validate" || phase == "validation") {

239  } else if (phase == "test" || phase == "testing") {

243  }
244  }

248  const auto& lvl = params.lvl();
249  if (phase == "train" || phase == "training") {
250  return new lbann_callback_debug_io(execution_mode::training, lvl);
251  } else if (phase == "validate" || phase == "validation") {
252  return new lbann_callback_debug_io(execution_mode::validation, lvl);
253  } else if (phase == "test" || phase == "testing") {
254  return new lbann_callback_debug_io(execution_mode::testing, lvl);
255  } else {
256  return new lbann_callback_debug_io();
257  }
258  }
259  if (proto_cb.has_dump_weights()) {
260  const auto& params = proto_cb.dump_weights();
261  return new lbann_callback_dump_weights(params.basename());
262  }
263  if (proto_cb.has_dump_activations()) {
264  const auto& params = proto_cb.dump_activations();
265  const auto& layer_names = parse_list<>(params.layer_names());
266  return new lbann_callback_dump_activations(params.basename(),
267  params.interval(),
268  layer_names);
269  }
270  if (proto_cb.has_dump_gradients()) {
271  const auto& params = proto_cb.dump_gradients();
272  return new lbann_callback_dump_gradients(params.basename(),
273  params.interval());
274  }
275  if (proto_cb.has_dump_mb_indices()) {
276  const auto& params = proto_cb.dump_mb_indices();
277  return new lbann_callback_dump_minibatch_sample_indices(params.basename(),
278  params.interval());
279  }
280  if (proto_cb.has_check_dataset()) {
281  return new lbann_callback_check_dataset();
282  }
283  if (proto_cb.has_check_small()) {
284  return new lbann_callback_checksmall();
285  }
286  if (proto_cb.has_check_nan()) {
287  return new lbann_callback_checknan();
288  }
289  if (proto_cb.has_hang()) {
290  const auto& rank_to_hang = proto_cb.hang().rank();
291  if (comm->am_world_master()) {

317 lbann_summary* construct_summarizer(lbann_comm* comm,
318  const lbann_data::Model& m) {
319  lbann_summary *summary = nullptr;
320  bool master = comm->am_world_master();
321  int size = m.callback_size();
322  for (int j=0; j<size; j++) {
323  const lbann_data::Callback& callback = m.callback(j);
324  if (callback.has_summary()) {
325  const lbann_data::CallbackSummary& c = callback.summary();
326  if (master) {
327  std::cout << "constructing summarizer with dir: " << c.dir() << std::endl;
328  }
329 
330  //check to see if directory exists
331  struct stat sb;
332  if (! ( stat(c.dir().c_str(), &sb) == 0 && S_ISDIR(sb.st_mode) )) {
333  if (master) {
334  throw lbann_exception(
335  std::string {} + __FILE__ + " " + std::to_string(__LINE__) + " :: " +
336  "summary directory " + c.dir() + " does not exist");
337  }
338  }
339  summary = new lbann_summary(c.dir(), comm);
340  }
341  }
342  return summary;
343 }
