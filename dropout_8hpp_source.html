180  void fp_compute_cpu() {
181 
182  // Matrices
183  const auto& input = get_prev_activations();
184  auto& output = get_activations();
185 
186  // Do nothing if dropout is disabled
187  const auto& mode = this->m_model->get_execution_mode();
188  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
189  El::LockedView(output, input);
190  return;
191  }
192 
193  // Construct mask matrix
194  const DataType scale = 1 / m_keep_prob;
195  const auto& height = input.Height();
196  const auto& width = input.Width();
197  m_mask->Resize(height, width);
198 #ifdef LBANN_SEQUENTIAL_CONSISTENCY
199  bernoulli_fill_procdet(*m_mask, height, width, DataType(m_keep_prob));
200  *m_mask *= scale;
201 #else
202  El::EntrywiseMap(*m_mask,
203  (std::function<DataType(const DataType&)>)
204  ([this,scale](const DataType& z)->DataType {
205  auto& gen = get_fast_generator();
206  std::bernoulli_distribution dist(m_keep_prob);
207  return dist(gen) ? scale : DataType(0);
208  }));
209 #endif // LBANN_SEQUENTIAL_CONSISTENCY
210 
211  // Apply mask matrix to get activations
212  El::Hadamard(input, *m_mask, output);
213 
214  }

217  void bp_compute_cpu() {
218  const auto& gradient_wrt_output = get_prev_error_signals();
219  auto& gradient_wrt_input = get_error_signals();
220  const auto& mode = this->m_model->get_execution_mode();
221  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
222  El::Axpy(DataType(1), gradient_wrt_output, gradient_wrt_input);
223  } else {
224  El::Hadamard(gradient_wrt_output, *m_mask, *m_mask);
225  El::Axpy(DataType(1), *m_mask, gradient_wrt_input);
226  }
227  }

229  void fp_compute_gpu() {
230  #ifndef LBANN_HAS_CUDNN
231  LBANN_ERROR("cuDNN not detected");
232  #else
233  const auto& input_d = this->m_prev_activations_d[0];
234  auto& output_d = this->m_activations_d[0];
235  const auto& num_gpus = this->m_cudnn->get_num_gpus();
236 
237  // Do nothing if dropout is disabled
238  const auto& mode = this->m_model->get_execution_mode();
239  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
240  output_d.locked_view(input_d);
241  return;
242  }
243 
244  // Apply dropout on each GPU
245  for (int i = 0; i < num_gpus; ++i) {
246  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
247  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
248  this->m_cudnn->get_stream(i)));
249  CHECK_CUDNN(cudnnDropoutForward(this->m_cudnn->get_handle(i),
250  m_dropout_cudnn_desc[i],
251  this->m_prev_activations_cudnn_desc,
252  input_d.get_locked_data(i),
253  this->m_activations_cudnn_desc,
254  output_d.get_data(i),
255  m_reserve_space_d.get_data(i),
256  m_reserve_space_d.get_height() * sizeof(DataType)));
257  }
258 
259  #endif // LBANN_HAS_CUDNN
260  }

262  void bp_compute_gpu() {
263  #ifndef LBANN_HAS_CUDNN
264  LBANN_ERROR("cuDNN not detected");
265  #else
266  const auto& gradient_wrt_output_d = this->m_prev_error_signals_d[0];
267  auto& gradient_wrt_input_d = this->m_error_signals_d[0];
268  const auto& num_gpus = this->m_cudnn->get_num_gpus();
269 
270  // Add to error signal if dropout is disabled
271  const auto& mode = this->m_model->get_execution_mode();
272  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
273  for (int i=0; i<num_gpus; ++i) {
274  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
275  cublas::geam(this->m_cudnn->get_cublas_handle(i),
276  CUBLAS_OP_N, CUBLAS_OP_N,
277  gradient_wrt_input_d.get_height(),
278  this->m_mini_batch_size_per_gpu,
279  DataType(1),
280  gradient_wrt_output_d.get_locked_data(i),
281  gradient_wrt_output_d.get_leading_dim(),
282  DataType(1),
283  gradient_wrt_input_d.get_locked_data(i),
284  gradient_wrt_input_d.get_leading_dim(),
285  gradient_wrt_input_d.get_data(i),
286  gradient_wrt_input_d.get_leading_dim());
287  }
288  return;
289  }
290 
291  // Apply dropout backprop on each GPU
293  for (int i = 0; i < num_gpus; ++i) {
294  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
295  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
296  this->m_cudnn->get_stream(i)));
297  CHECK_CUDNN(cudnnDropoutBackward(this->m_cudnn->get_handle(i),
298  m_dropout_cudnn_desc[i],
299  this->m_prev_error_signals_cudnn_desc,
300  gradient_wrt_output_d.get_locked_data(i),
301  this->m_error_signals_cudnn_desc,
302  gradient_wrt_input_d.get_data(i),
303  m_reserve_space_d.get_data(i),
304  m_reserve_space_d.get_height() * sizeof(DataType)));
305  }
306 
307  #endif // LBANN_HAS_CUDNN
308  }
