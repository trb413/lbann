94  if (this->m_model->get_execution_mode() != execution_mode::training

96  // Do nothing if dropout is disabled
97  El::LockedView(*this->m_activations_v, *this->m_prev_activations_v);
98  } else {
99 
100  // Construct mask matrix
101  const DataType scale = DataType(1) / m_keep_prob;
102  const int height = this->m_activations_v->Height();
103  const int width = this->m_activations_v->Width();
104  m_mask->Resize(height, width);
105 #ifdef LBANN_SEQUENTIAL_CONSISTENCY
106  bernoulli_fill_procdet(*m_mask, height, width, m_keep_prob);
107  *m_mask *= scale;
108 #else
109  El::EntrywiseMap(*m_mask,
110  (std::function<DataType(const DataType&)>)
111  ([this,scale](const DataType& z)->DataType {
112  auto& gen = get_fast_generator();
113  std::bernoulli_distribution dist(m_keep_prob);
114  return dist(gen) ? scale : DataType(0);
115  }));
116 #endif // LBANN_SEQUENTIAL_CONSISTENCY
117 
118  // Apply mask matrix to get activations
