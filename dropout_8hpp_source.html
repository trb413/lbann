94  void fp_compute() {
95 
96  // Copy previous activations if dropout is disabled
97  if (this->get_execution_mode() != execution_mode::training
98  || m_keep_prob < 0.0f) {
99  El::Copy(*this->m_prev_activations, *this->m_activations_v);
100  return;

102 
103  // Construct mask matrix
104  const DataType scale = DataType(1) / m_keep_prob;
105  const int height = this->m_activations_v->Height();
106  const int width = this->m_activations_v->Width();
107  m_mask->Resize(height, width);
108 #ifdef LBANN_PROCDET_DROPOUT
109  bernoulli_fill_procdet(*m_mask, height, width, m_keep_prob);
110  *m_mask *= scale;
111 #else
112  El::EntrywiseMap(*m_mask,
113  (std::function<DataType(const DataType&)>)
114  ([this,scale](const DataType& z)->DataType {
115  auto& gen = get_fast_generator();
116  std::bernoulli_distribution dist(m_keep_prob);
117  return dist(gen) ? scale : DataType(0);
118  }));
119 #endif // LBANN_PROCDET_DROPOUT
120 
121  // Apply mask matrix to get activations
