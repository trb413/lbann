72 int lbann::DataReader_MNIST::fetch_data(Mat& X)
73 {
74  if(!DataReader::position_valid()) {
75  stringstream err;
76  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader load error: !position_valid";
77  throw lbann_exception(err.str());
78  }
79 
80  int pixelcount = m_image_width * m_image_height;
81  int current_batch_size = getBatchSize();
82 
83  int n = 0;
84  std::vector<float> pixels(pixelcount);
85  for (n = CurrentPos; n < CurrentPos + current_batch_size; n++) {
86  if (n >= (int)ShuffledIndices.size())
87  break;
88 
89  int k = n - CurrentPos;
90  int index = ShuffledIndices[n];
91  vector<unsigned char> &tmp = m_image_data[index];
92 
93  for (int p = 0; p < pixelcount; p++) {
94  X.Set(p, k, tmp[p+1]);
95  }
96 
97  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
98  augment(pixel_col, m_image_height, m_image_width, 1);
99  normalize(pixel_col, 1);
100  }
101 
102  return (n - CurrentPos);
103 }

133 void lbann::DataReader_MNIST::load()
134 {
135  if (is_master()) cerr << "starting lbann::DataReader_MNIST::load\n";
136  m_image_data.clear();
137 
138  string FileDir = get_file_dir();
139  string ImageFile = get_data_filename();
140  string LabelFile = get_label_filename();
141 
142  // set filepath
143  string imagepath = FileDir + __DIR_DELIMITER + ImageFile;
144  string labelpath = FileDir + __DIR_DELIMITER + LabelFile;
145 
146  // read labels
147  FILE* fplbl = fopen(labelpath.c_str(), "rb");
148  if (!fplbl) {
149  stringstream err;
150  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << labelpath;
151  throw lbann_exception(err.str());
152  }
153 
154  int magicnum1, numitems1;
155  fread(&magicnum1, 4, 1, fplbl);
156  fread(&numitems1, 4, 1, fplbl);
157  __swapEndianInt((unsigned int&)magicnum1);
158  __swapEndianInt((unsigned int&)numitems1);
159 
160  // read images
161  FILE* fpimg = fopen(imagepath.c_str(), "rb");
162  if (!fpimg) {
163  stringstream err;
164  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << imagepath;
165  throw lbann_exception(err.str());
166  }
167 
168  int magicnum2, numitems2, imgwidth, imgheight;
169  fread(&magicnum2, 4, 1, fpimg);
170  fread(&numitems2, 4, 1, fpimg);
171  fread(&imgwidth, 4, 1, fpimg);
172  fread(&imgheight, 4, 1, fpimg);
173  __swapEndianInt((unsigned int&)magicnum2);
174  __swapEndianInt((unsigned int&)numitems2);
175  __swapEndianInt((unsigned int&)imgwidth);
176  __swapEndianInt((unsigned int&)imgheight);
177 
178  if (numitems1 != numitems2) {
179  fclose(fplbl);
180  fclose(fpimg);
181  stringstream err;
182  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: numitems1 != numitems2";
183  throw lbann_exception(err.str());
184  }
185 
186  // set to array
187  m_image_data.resize(numitems1);
188  for (int n = 0; n < numitems1; n++) {
189  m_image_data[n].resize(1+(imgwidth * imgheight));
190  fread(&m_image_data[n][0], 1, 1, fplbl);
191  fread(&m_image_data[n][1], imgwidth * imgheight, 1, fpimg);
192  }
193  fclose(fpimg);
194  fclose(fplbl);
195 
196  // reset indices
197  ShuffledIndices.clear();
198  ShuffledIndices.resize(m_image_data.size());
199  for (size_t n = 0; n < ShuffledIndices.size(); n++) {
200  ShuffledIndices[n] = n;
201  }
202  if (is_master()) cerr << "calling select_subset_of_data; ShuffledIndices.size: " << ShuffledIndices.size() << endl;
203  select_subset_of_data();
204 }
