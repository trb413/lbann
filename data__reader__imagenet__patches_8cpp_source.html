93 bool imagenet_reader_patches::replicate_processor(const cv_process_patches& pp) {
94  const int nthreads = omp_get_max_threads();
95  m_pps.resize(nthreads);
96 
97  // Construct thread private preprocessing objects out of a shared pointer
98  #pragma omp parallel for schedule(static, 1)
99  for (int i = 0; i < nthreads; ++i) {
100  //auto ppu = std::make_unique<cv_process_patches>(pp); // c++14
101  std::unique_ptr<cv_process_patches> ppu(new cv_process_patches(pp));
102  m_pps[i] = std::move(ppu);
103  }
104 
105  bool ok = true;
106  for (int i = 0; ok && (i < nthreads); ++i) {
107  if (!m_pps[i]) ok = false;
108  }
109 
110  if (!ok || (nthreads <= 0)) {
111  std::stringstream err;
112  err << __FILE__<<" "<<__LINE__<< " :: " << get_type() << " construction error: cannot replicate image processor";
113  throw lbann_exception(err.str());
114  return false;
115  }
116  const std::vector<unsigned int> dims = pp.get_data_dims();
117  if ((dims.size() == 3u) && (dims[0] != 0u) && (dims[1] != 0u) && (dims[2] != 0u)) {
118  m_num_patches = static_cast<int>(dims[0]);
119  m_image_width = static_cast<int>(dims[1]);
120  m_image_height = static_cast<int>(dims[2]);
121  set_linearized_image_size();
122  }
123  if (pp.is_self_labeling()) {
124  m_num_labels = pp.get_num_labels();
125  }
126 
127  return true;
128 }

148 bool imagenet_reader_patches::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
149  const std::string imagepath = get_file_dir() + m_image_list[data_id].first;
150 
151  int width=0, height=0, img_type=0;
152  std::vector<::Mat> X_v = create_datum_views(X, mb_idx);
153  bool ret;
154  if (m_data_store != nullptr) {
155  std::vector<unsigned char> *image_buf;
156  m_data_store->get_data_buf(data_id, image_buf, 0);
157  ret = lbann::image_utils::load_image(*image_buf, width, height, img_type, *(m_pps[tid]), X_v);
158  } else {
159  ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v);
160  }
161  //ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v);
162 
163  if (m_pps[tid]->is_self_labeling()) {
164  m_image_list[data_id].second = m_pps[tid]->get_patch_label();
165  }
166 
167  if(!ret) {
168  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
169  + get_type() + ": image_utils::load_image failed to load - "
170  + imagepath);
171  }
172  if((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
173  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
174  + get_type() + ": mismatch data size -- either width, height or channel - "
175  + imagepath + " [w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
176  + "x" + std::to_string(CV_MAT_CN(img_type)) + "] != " + std::to_string(m_image_linearized_size));
177  }
178  return true;
179 }
