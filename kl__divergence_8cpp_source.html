63 EvalType kl_divergence::finish_evaluation() {
64  if (m_scale_factor == EvalType(0)) { return EvalType(0); }
65 
66  // Matrices
67  const auto& global_z_mean = m_z_mean_layer->get_activations();
68  const auto& local_z_mean = global_z_mean.LockedMatrix();
69  const auto& local_z_log_sigma = m_z_log_sigma_layer->get_local_activations();
70 
71  // Matrix dimensions
72  const int height = global_z_mean.Height();
73  const int width = global_z_mean.Width();
74  const int local_height = local_z_mean.Height();
75  const int local_width = local_z_mean.Width();
76 
77  // Compute KL divergence
78  EvalType sum = 0;
79  #pragma omp parallel for reduction(+:sum) collapse(2)
80  for (int col = 0; col < local_width; ++col) {
81  for (int row = 0; row < local_height; ++row) {
82  const auto z_mean = local_z_mean(row, col);
83  const auto z_log_sigma = local_z_log_sigma(row, col);
84  sum += (std::exp(z_log_sigma) + z_mean * z_mean
85  - z_log_sigma - DataType(1)) / (2 * height);
86  }
87  }
88  const EvalType val = get_comm().allreduce(sum / width,
89  global_z_mean.DistComm());
90  return m_scale_factor * val;
91 
92 }

94 void kl_divergence::differentiate() {
95  if (m_scale_factor == EvalType(0)) { return; }
96 
97  // Matrices
98  const auto& global_z_mean = m_z_mean_layer->get_activations();
99  const auto& global_z_log_sigma = m_z_log_sigma_layer->get_activations();
100  auto&& global_dz_mean = global_z_mean.Copy();
101  auto&& global_dz_log_sigma = global_z_log_sigma.Copy();
102  const auto& local_z_mean = global_z_mean.LockedMatrix();
103  const auto& local_z_log_sigma = global_z_log_sigma.LockedMatrix();
104  auto& local_dz_mean = global_dz_mean->Matrix();
105  auto& local_dz_log_sigma = global_dz_log_sigma->Matrix();
106 
107  // Matrix dimensions
108  const int height = global_z_mean.Height();
109  const int local_height = local_z_mean.Height();
110  const int local_width = local_z_mean.Width();
111 
112  // Compute gradient of KL divergence
113  #pragma omp parallel for collapse(2)
114  for (int col = 0; col < local_width; ++col) {
115  for (int row = 0; row < local_height; ++row) {
116  const auto z_mean = local_z_mean(row, col);
117  const auto z_log_sigma = local_z_log_sigma(row, col);
118  local_dz_mean(row, col) = z_mean / height;
119  local_dz_log_sigma(row, col) = (std::exp(z_log_sigma) - DataType(1)) / (2 * height);
120  }
121  }
122 
123  // Add to error signal
124  auto&& z_mean_child_layer = const_cast<Layer*>(m_z_mean_layer->get_child_layers().front());
125  auto&& z_log_sigma_child_layer = const_cast<Layer*>(m_z_log_sigma_layer->get_child_layers().front());
126  z_mean_child_layer->add_to_error_signal(*global_dz_mean,
127  DataType(m_scale_factor));
128  z_log_sigma_child_layer->add_to_error_signal(*global_dz_log_sigma,
129  DataType(m_scale_factor));
130 
131  // Clean up
132  delete global_dz_mean;
133  delete global_dz_log_sigma;
134 
135 }
