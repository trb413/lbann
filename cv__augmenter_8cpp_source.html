114 bool cv_augmenter::determine_transform(const cv::Mat& image) {
115  reset();
116 
117  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
118 
119  if (!check_to_enable()) {
120  return false;
121  }
122 
123  rng_gen& gen = get_generator();
124 
125  std::uniform_int_distribution<int> bool_dist(0, 1);
126 
127  // Flips
128 #ifdef _COMPAT_WITH_EL_AUGMENT_
129  const bool horiz_flip = bool_dist(gen) && m_do_horizontal_flip;
130  const bool vert_flip = bool_dist(gen) && m_do_vertical_flip;
131 #else
132  const bool horiz_flip = m_do_horizontal_flip && bool_dist(gen);
133  const bool vert_flip = m_do_vertical_flip && bool_dist(gen);
134 #endif
135 
136  if (horiz_flip && vert_flip) {
137  m_flip = _both_axes_;
138  } else if (horiz_flip) {
139  m_flip = _horizontal_;
140  } else if (vert_flip) {
141  m_flip = _vertical_;
142  } else {
143  m_flip = _no_flip_;
144  }
145 
146  // Shift (Translate)
147  float x_shift = 0.0f;
148  float y_shift = 0.0f;
149  if (m_horizontal_shift_range != 0.0f) {
150  std::uniform_real_distribution<float> dist(-m_horizontal_shift_range,
151  m_horizontal_shift_range);
152  x_shift = dist(gen) * image.cols;
153  }
154  if (m_vertical_shift_range != 0.0f) {
155  std::uniform_real_distribution<float> dist(-m_vertical_shift_range,
156  m_vertical_shift_range);
157  y_shift = dist(gen) * image.rows;
158  }
159  cv::Mat_<float> shift_mat = cv::Mat_<float>::eye(3,3);
160  shift_mat(0, 2) = x_shift;
161  shift_mat(1, 2) = y_shift;
162  //std::cout << "x_shift " << x_shift << ", y_shift " << y_shift << std::endl;
163 
164  // Shearing
165  float shear = 0.0f;
166  if (m_shear_range != 0.0f) {
167  std::uniform_real_distribution<float> dist(-m_shear_range,
168  m_shear_range);
169  shear = dist(gen);
170  }
171  cv::Mat_<float> shear_mat = cv::Mat_<float>::zeros(3,3);
172  shear_mat(0, 0) = 1.0f;
173  shear_mat(2, 2) = 1.0f;
174  shear_mat(0, 1) = -std::sin(shear);
175  shear_mat(1, 1) = std::cos(shear);
176  //std::cout << "shear " << shear << std::endl;
177 
178  // Rotation
179  float rotate = 0.0f;
180  if (m_rotation_range != 0.0f) {
181  std::uniform_real_distribution<float> dist(-m_rotation_range,
182  m_rotation_range);
183  rotate = pi / 180.0f * dist(gen);
184  }
185  cv::Mat_<float> rot_mat = cv::Mat_<float>::zeros(3,3);
186  rot_mat(2, 2) = 1.0f;
187  rot_mat(0, 0) = std::cos(rotate);
188  rot_mat(0, 1) = -std::sin(rotate);
189  rot_mat(1, 0) = std::sin(rotate);
190  rot_mat(1, 1) = std::cos(rotate);
191  //std::cout << "rotate " << rotate << std::endl;
192 
193  // Compute the final transformation.
194 #if 0
195  cv::Mat_<float> tmp_mat = cv::Mat_<float>::zeros(3, 3);
196  cv::gemm(shift_mat, shear_mat, 1.0f, tmp_mat, 0.0f, tmp_mat, 0);
197  cv::gemm(tmp_mat, rot_mat, 1.0f, m_trans, 0.0f, m_trans, 0);
198 #else
199  //m_trans = (shift_mat * shear_mat) * rot_mat;
200  m_trans = shear_mat * rot_mat;
201  m_trans(0,2) = x_shift;
202  m_trans(1,2) = y_shift;
203 #endif
204 
205  return (m_enabled = true);
206 }

209 bool cv_augmenter::apply(cv::Mat& image) {
210  m_enabled = false; // turn off as it is applied
211 
212  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
213 
214  cv::Mat image_copy;
215 
216  if (m_flip == _no_flip_) {
217  image_copy = image.clone();
218  } else {
219  cv::flip(image, image_copy, static_cast<int>(m_flip));
220  }
221 
222  cv::Mat_<float> _trans(m_trans, cv::Rect_<float>(0,0,3,2));
223 
224  cv::warpAffine(image_copy, image, _trans, image.size(),
225  cv::INTER_LINEAR, cv::BORDER_REPLICATE);
226 
227  return true;
228 }
