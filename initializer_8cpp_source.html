34 AbsDistMat* weights_initializer::construct_matrix(int height,
35  int width,
36  El::Distribution col_dist,
37  El::Distribution row_dist) const {
38 
39  // Construct distributed matrix with desired matrix distribution
40  AbsDistMat* weights_matrix = nullptr;
41  const El::Grid& grid = m_comm->get_model_grid();
42  if (col_dist == El::MC && row_dist == El::MR) {
43  weights_matrix = new DistMat(grid);
44  }
45  if (col_dist == El::STAR && row_dist == El::STAR) {
46  weights_matrix = new StarMat(grid);
47  }
48  if (col_dist == El::CIRC && row_dist == El::CIRC) {
49  weights_matrix = new CircMat(grid);
50  }
51  if (col_dist == El::MR && row_dist == El::STAR) {
52  weights_matrix = new ColSumMat(grid);
53  }
54  if (col_dist == El::MC && row_dist == El::STAR) {
55  weights_matrix = new RowSumMat(grid);
56  }
57 
58  // Check that weights has been constructed
59  if (weights_matrix == nullptr) {
60  std::stringstream err;
61  err << __FILE__ << " " << __LINE__ << " :: "
62  << "could not construct weights matrix with specified distribution "
63  << "(col_dist=" << col_dist << ",row_dist=" << row_dist << ")";
64  throw lbann_exception(err.str());
65  }
66 
67  // Resize weights matrix
68  weights_matrix->Resize(height, width);
69 
70  // Initialize weights matrix entries and return
71  initialize_entries(*weights_matrix);
72  return weights_matrix;
73 
74 }
