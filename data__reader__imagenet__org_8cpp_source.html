63 bool imagenet_reader_org::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
64  const std::string imagepath = get_file_dir() + m_image_list[data_id].first;
65 
66  int width, height;
67  unsigned char *pixels = m_pixel_bufs[tid].data();
68  bool ret = lbann::image_utils::loadIMG(imagepath, width, height, false, pixels);
69  if(!ret) {
70  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
71  + get_type() + ": image_utils::loadIMG failed to load - "
72  + imagepath);
73  }
74  if(width != m_image_width || height != m_image_height) {
75  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
76  + get_type() + ": mismatch data size -- either width or height - "
77  + imagepath + "[w,h]=[" + std::to_string(width) + "x" + std::to_string(height) +"]");
78  }
79 
80  for (int p = 0; p < m_image_linearized_size; p++) {
81  X(p, mb_idx) = pixels[p];
82  }
83 
84  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
85  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
86  normalize(pixel_col, m_image_num_channels);
87  pixel_noise(pixel_col); //add noise to image, disable by default
88 
89  return true;
90 }
